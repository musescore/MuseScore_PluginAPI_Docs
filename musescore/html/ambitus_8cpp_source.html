119 void Ambitus::setTopPitch(int val)
120  {
121  int deltaPitch = val - topPitch();
122  // if deltaPitch is not an integer number of octaves, adjust tpc
123  // (to avoid 'wild' tpc changes with octave changes)
124  if (deltaPitch % PITCH_DELTA_OCTAVE != 0) {
125  int newTpc = topTpc() + deltaPitch * TPC_DELTA_SEMITONE;
126  // reduce newTpc into acceptable range via enharmonic
127  while (newTpc < Tpc::TPC_MIN)
128  newTpc += TPC_DELTA_ENHARMONIC;
129  while (newTpc > Tpc::TPC_MAX)
130  newTpc -= TPC_DELTA_ENHARMONIC;
131  _topTpc = newTpc;
132  }
133  _topPitch = val;
134  normalize();
135  }

156 // setTop/BottomTpc
157 //
158 // setting either tpc requires to adjust the corresponding pitch
159 // (but remaining in the same octave)
160 //---------------------------------------------------------
161 
162 void Ambitus::setTopTpc(int val)
163  {
164  int octave = topPitch() / PITCH_DELTA_OCTAVE;
165  int deltaTpc = val - topTpc();
166  // get new pitch according to tpc change
167  int newPitch = topPitch() + deltaTpc * TPC_DELTA_SEMITONE;
168  // reduce pitch to the same octave as original pitch
169  newPitch = (octave * PITCH_DELTA_OCTAVE) + (newPitch % PITCH_DELTA_OCTAVE);
170  _topPitch = newPitch;
171  _topTpc = val;
172  normalize();
173  }

234 bool Ambitus::readProperties(XmlReader& e)
235  {
236  const QStringRef& tag(e.name());
237  if (tag == "head")
238  readProperty(e, Pid::HEAD_GROUP);
239  else if (tag == "headType")
240  readProperty(e, Pid::HEAD_TYPE);
241  else if (tag == "mirror")
242  readProperty(e, Pid::MIRROR_HEAD);
243  else if (tag == "hasLine")
244  setHasLine(e.readInt());
245  else if (tag == "lineWidth")
246  readProperty(e, Pid::LINE_WIDTH_SPATIUM);
247  else if (tag == "topPitch")
248  _topPitch = e.readInt();
249  else if (tag == "bottomPitch")
250  _bottomPitch = e.readInt();
251  else if (tag == "topTpc")
252  _topTpc = e.readInt();
253  else if (tag == "bottomTpc")
254  _bottomTpc = e.readInt();
255  else if (tag == "topAccidental") {
256  while (e.readNextStartElement()) {
257  if (e.name() == "Accidental") {
258  if (score()->mscVersion() < 301)
259  readAccidental206(&_topAccid, e);
260  else
261  _topAccid.read(e);
262  }
263  else
264  e.skipCurrentElement();
265  }
266  }
267  else if (tag == "bottomAccidental") {
268  while (e.readNextStartElement()) {
269  if (e.name() == "Accidental") {
270  if (score()->mscVersion() < 301)
271  readAccidental206(&_bottomAccid, e);
272  else
273  _bottomAccid.read(e);
274  }
275  else
276  e.skipCurrentElement();
277  }
278  }
279  else if (Element::readProperties(e))
280  ;
281  else
282  return false;
283  return true;
284  }

290 void Ambitus::layout()
291  {
292  int bottomLine, topLine;
293  ClefType clf;
294  qreal headWdt = headWidth();
295  Key key;
296  qreal lineDist;
297  int numOfLines;
298  Segment* segm = segment();
299  qreal _spatium = spatium();
300  Staff* stf = nullptr;
301  if (segm && track() > -1) {
302  Fraction tick = segm->tick();
303  stf = score()->staff(staffIdx());
304  lineDist = stf->lineDistance(tick) * _spatium;
305  numOfLines = stf->lines(tick);
306  clf = stf->clef(tick);
307  }
308  else { // for use in palettes
309  lineDist = _spatium;
310  numOfLines = 3;
311  clf = ClefType::G;
312  }
313 
314  //
315  // NOTEHEADS Y POS
316  //
317  // if pitch == INVALID_PITCH oor tpc == INALID_TPC, set to some default:
318  // for use in palettes and when actual range cannot be calculated (new ambitus or no notes in staff)
319  //
320  qreal xAccidOffTop = 0;
321  qreal xAccidOffBottom = 0;
322  if (stf)
323  key = stf->key(segm->tick());
324  else
325  key = Key::C;
326 
327  // top notehead
328  if (_topPitch == INVALID_PITCH || _topTpc == Tpc::TPC_INVALID)
329  _topPos.setY(0); // if uninitialized, set to top staff line
330  else {
331  topLine = absStep(_topTpc, _topPitch);
332  topLine = relStep(topLine, clf);
333  _topPos.setY(topLine * lineDist * 0.5);
334  // compute accidental
335  AccidentalType accidType;
336  // if (13 <= (tpc - key) <= 19) there is no accidental)
337  if (_topTpc - int(key) >= 13 && _topTpc - int(key) <= 19)
338  accidType = AccidentalType::NONE;
339  else {
340  AccidentalVal accidVal = tpc2alter(_topTpc);
341  accidType = Accidental::value2subtype(accidVal);
342  if (accidType == AccidentalType::NONE)
343  accidType = AccidentalType::NATURAL;
344  }
345  _topAccid.setAccidentalType(accidType);
346  if (accidType != AccidentalType::NONE)
347  _topAccid.layout();
348  else
349  _topAccid.setbbox(QRect());
350  _topAccid.rypos() = _topPos.y();
351  }
352 
353  // bottom notehead
354  if (_bottomPitch == INVALID_PITCH || _bottomTpc == Tpc::TPC_INVALID)
355  _bottomPos.setY( (numOfLines-1) * lineDist); // if uninitialized, set to last staff line
356  else {
357  bottomLine = absStep(_bottomTpc, _bottomPitch);
358  bottomLine = relStep(bottomLine, clf);
359  _bottomPos.setY(bottomLine * lineDist * 0.5);
360  // compute accidental
361  AccidentalType accidType;
362  if (_bottomTpc - int(key) >= 13 && _bottomTpc - int(key) <= 19)
363  accidType = AccidentalType::NONE;
364  else {
365  AccidentalVal accidVal = tpc2alter(_bottomTpc);
366  accidType = Accidental::value2subtype(accidVal);
367  if (accidType == AccidentalType::NONE)
368  accidType = AccidentalType::NATURAL;
369  }
370  _bottomAccid.setAccidentalType(accidType);
371  if (accidType != AccidentalType::NONE)
372  _bottomAccid.layout();
373  else
374  _bottomAccid.setbbox(QRect());
375  _bottomAccid.rypos() = _bottomPos.y();
376  }
377 
378  //
379  // NOTEHEAD X POS
380  //
381  // Note: manages colliding accidentals
382  //
383  qreal accNoteDist = point(score()->styleS(Sid::accidentalNoteDistance));
384  xAccidOffTop = _topAccid.width() + accNoteDist;
385  xAccidOffBottom = _bottomAccid.width() + accNoteDist;
386 
387  // if top accidental extends down more than bottom accidental extends up,
388  // AND ambitus is not leaning right, bottom accidental needs to be displaced
389  bool collision =
390  (_topAccid.ipos().y() + _topAccid.bbox().y() + _topAccid.height()
391  > _bottomAccid.ipos().y() + _bottomAccid.bbox().y() )
392  && _dir != MScore::DirectionH::RIGHT;
393  if (collision) {
394  // displace bottom accidental (also attempting to 'undercut' flats)
395  xAccidOffBottom = xAccidOffTop +
396  ((_bottomAccid.accidentalType() == AccidentalType::FLAT
397  || _bottomAccid.accidentalType() == AccidentalType::FLAT2
398  || _bottomAccid.accidentalType() == AccidentalType::NATURAL)
399  ? _bottomAccid.width() * 0.5 : _bottomAccid.width());
400  }
401 
402  switch (_dir) {
403  case MScore::DirectionH::AUTO: // noteheads one above the other
404  // left align noteheads and right align accidentals 'hanging' on the left
405  _topPos.setX(0.0);
406  _bottomPos.setX(0.0);
407  _topAccid.rxpos() = - xAccidOffTop;
408  _bottomAccid.rxpos() = - xAccidOffBottom;
409  break;
410  case MScore::DirectionH::LEFT: // top notehead at the left of bottom notehead
411  // place top notehead at left margin; bottom notehead at right of top head;
412  // top accid. 'hanging' on left of top head and bottom accid. 'hanging' at left of bottom head
413  _topPos.setX(0.0);
414  _bottomPos.setX(headWdt);
415  _topAccid.rxpos() = - xAccidOffTop;
416  _bottomAccid.rxpos() = collision ? - xAccidOffBottom : headWdt - xAccidOffBottom;
417  break;
418  case MScore::DirectionH::RIGHT: // top notehead at the right of bottom notehead
419  // bottom notehead at left margin; top notehead at right of bottomnotehead
420  // top accid. 'hanging' on left of top head and bottom accid. 'hanging' at left of bottom head
421  _bottomPos.setX(0.0);
422  _topPos.setX(headWdt);
423  _bottomAccid.rxpos() = - xAccidOffBottom;
424  _topAccid.rxpos() = headWdt - xAccidOffTop;
425  break;
426  }
427 
428  // compute line from top note centre to bottom note centre
429  QLineF fullLine(_topPos.x() + headWdt*0.5, _topPos.y(),
430  _bottomPos.x() + headWdt*0.5, _bottomPos.y());
431  // shorten line on each side by offsets
432  qreal yDelta = _bottomPos.y() - _topPos.y();
433  if (yDelta != 0.0) {
434  qreal off = _spatium * LINEOFFSET_DEFAULT;
435  QPointF p1 = fullLine.pointAt(off / yDelta);
436  QPointF p2 = fullLine.pointAt(1 - (off / yDelta));
437  _line = QLineF(p1, p2);
438  }
439  else
440  _line = fullLine;
441 
442  QRectF headRect = QRectF(0, -0.5*_spatium, headWdt, 1*_spatium);
443  setbbox(headRect.translated(_topPos).united(headRect.translated(_bottomPos))
444  .united(_topAccid.bbox().translated(_topAccid.ipos()))
445  .united(_bottomAccid.bbox().translated(_bottomAccid.ipos()))
446  );
447  }

453 void Ambitus::draw(QPainter* p) const
454  {
455  qreal _spatium = spatium();
456  qreal lw = lineWidth().val() * _spatium;
457  p->setPen(QPen(curColor(), lw, Qt::SolidLine, Qt::FlatCap));
458  drawSymbol(noteHead(), p, _topPos);
459  drawSymbol(noteHead(), p, _bottomPos);
460  if (_hasLine)
461  p->drawLine(_line);
462 
463  // draw ledger lines (if not in a palette)
464  if (segment() && track() > -1) {
465  Fraction tick = segment()->tick();
466  Staff* stf = score()->staff(staffIdx());
467  qreal lineDist = stf->lineDistance(tick);
468  int numOfLines = stf->lines(tick);
469  qreal step = lineDist * _spatium;
470  qreal stepTolerance = step * 0.1;
471  qreal ledgerOffset = score()->styleS(Sid::ledgerLineLength).val() * _spatium;
472  p->setPen(QPen(curColor(), score()->styleS(Sid::ledgerLineWidth).val() * _spatium,
473  Qt::SolidLine, Qt::FlatCap) );
474  if (_topPos.y()-stepTolerance <= -step) {
475  qreal xMin = _topPos.x() - ledgerOffset;
476  qreal xMax = _topPos.x() + headWidth() + ledgerOffset;
477  for (qreal y = -step; y >= _topPos.y()-stepTolerance; y -= step)
478  p->drawLine(QPointF(xMin, y), QPointF(xMax, y));
479  }
480  if (_bottomPos.y()+stepTolerance >= numOfLines * step) {
481  qreal xMin = _bottomPos.x() - ledgerOffset;
482  qreal xMax = _bottomPos.x() + headWidth() + ledgerOffset;
483  for (qreal y = numOfLines*step; y <= _bottomPos.y()+stepTolerance; y += step)
484  p->drawLine(QPointF(xMin, y), QPointF(xMax, y));
485  }
486  }
487  }

569 //---------------------------------------------------------
570 // updateRange
571 //
572 // scans the staff contents up to next section break to update the range pitches/tpc's
573 //---------------------------------------------------------
574 
575 void Ambitus::updateRange()
576  {
577  if (!segment())
578  return;
579  Chord* chord;
580  int firstTrack = track();
581  int lastTrack = firstTrack + VOICES-1;
582  int pitchTop = -1000;
583  int pitchBottom = 1000;
584  int tpcTop = 0; // Initialized to prevent warning
585  int tpcBottom = 0; // Initialized to prevent warning
586  int trk;
587  Measure* meas = segment()->measure();
588  Segment* segm = meas->findSegment(SegmentType::ChordRest, segment()->tick());
589  bool stop = meas->sectionBreak();
590  while (segm) {
591  // moved to another measure?
592  if (segm->measure() != meas) {
593  // if section break has been found, stop here
594  if (stop)
595  break;
596  // update meas and stop condition
597  meas = segm->measure();
598  stop = meas->sectionBreak();
599  }
600  // scan all relevant tracks of this segment for chords
601  for (trk = firstTrack; trk <= lastTrack; trk++) {
602  Element* e = segm->element(trk);
603  if (!e || !e->isChord())
604  continue;
605  chord = toChord(e);
606  // update pitch range (with associated tpc's)
607  for (Note* n : chord->notes()) {
608  if (!n->play()) // skip notes which are not to be played
609  continue;
610  int pitch = n->epitch();
611  if (pitch > pitchTop) {
612  pitchTop = pitch;
613  tpcTop = n->tpc();
614  }
615  if (pitch < pitchBottom) {
616  pitchBottom = pitch;
617  tpcBottom = n->tpc();
618  }
619  }
620  }
621  segm = segm->nextCR();
622  }
623 
624  if (pitchTop > -1000) { // if something has been found, update this
625  _topPitch = pitchTop;
626  _bottomPitch = pitchBottom;
627  _topTpc = tpcTop;
628  _bottomTpc = tpcBottom;
629  }
630  }

636 QVariant Ambitus::getProperty(Pid propertyId) const
637  {
638  switch(propertyId) {
639  case Pid::HEAD_GROUP:
640  return int(noteHeadGroup());
641  case Pid::HEAD_TYPE:
642  return int(noteHeadType());
643  case Pid::MIRROR_HEAD:
644  return int(direction());
645  case Pid::GHOST: // recycled property = _hasLine
646  return hasLine();
647  case Pid::LINE_WIDTH_SPATIUM:
648  return lineWidth();
649  case Pid::TPC1:
650  return topTpc();
651  case Pid::FBPARENTHESIS1: // recycled property = _bottomTpc
652  return bottomTpc();
