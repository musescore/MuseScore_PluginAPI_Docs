65 void Arpeggio::write(XmlWriter& xml) const
66  {
67  if (!xml.canWrite(this))
68  return;
69  xml.stag(this);
70  Element::writeProperties(xml);
71  writeProperty(xml, Pid::ARPEGGIO_TYPE);
72  if (_userLen1 != 0.0)
73  xml.tag("userLen1", _userLen1 / spatium());
74  if (_userLen2 != 0.0)
75  xml.tag("userLen2", _userLen2 / spatium());
76  if (_span != 1)

78  writeProperty(xml, Pid::PLAY);
79  writeProperty(xml, Pid::TIME_STRETCH);
80  xml.etag();

87 void Arpeggio::read(XmlReader& e)
88  {
89  while (e.readNextStartElement()) {
90  const QStringRef& tag(e.name());
91  if (tag == "subtype")
92  _arpeggioType = ArpeggioType(e.readInt());
93  else if (tag == "userLen1")
94  _userLen1 = e.readDouble() * spatium();
95  else if (tag == "userLen2")
96  _userLen2 = e.readDouble() * spatium();
97  else if (tag == "span")
98  _span = e.readInt();
99  else if (tag == "play")
100  _playArpeggio = e.readBool();
101  else if (tag == "timeStretch")
102  _stretch = e.readDouble();
103  else if (!Element::readProperties(e))
104  e.unknown();
105  }
106  }

113 void Arpeggio::symbolLine(SymId end, SymId fill)
114  {
115  qreal y1 = -_userLen1;
116  qreal y2 = _height + _userLen2;
117  qreal w = y2 - y1;
118  qreal mag = magS();
119  ScoreFont* f = score()->scoreFont();
120 
121  symbols.clear();
122  qreal w1 = f->advance(end, mag);
123  qreal w2 = f->advance(fill, mag);
124  int n = lrint((w - w1) / w2);
125  for (int i = 0; i < n; ++i)
126  symbols.push_back(fill);
127  symbols.push_back(end);
128  }

134 void Arpeggio::layout()
135  {
136  qreal y1 = -_userLen1;
137  qreal y2 = _height + _userLen2;
138  _hidden = false;
139  if (score()->styleB(Sid::ArpeggioHiddenInStdIfTab)) {
140  if (staff() && staff()->isPitchedStaff(tick())) {
141  for (Staff* s : staff()->staffList()) {
142  if (s->score() == score() && s->isTabStaff(tick())) {
143  _hidden = true;
144  setbbox(QRect());
145  return;
146  }
147  }
148  }
149  }
150  if (staff())
151  setMag(staff()->mag(tick()));
152  switch (arpeggioType()) {
153  case ArpeggioType::NORMAL: {
154  symbolLine(SymId::wiggleArpeggiatoUp, SymId::wiggleArpeggiatoUp);
155  // string is rotated -90 degrees
156  QRectF r(symBbox(symbols));
157  setbbox(QRectF(0.0, -r.x() + y1, r.height(), r.width()));
158  }
159  break;
160 
161  case ArpeggioType::UP: {
162  symbolLine(SymId::wiggleArpeggiatoUpArrow, SymId::wiggleArpeggiatoUp);
163  // string is rotated -90 degrees
164  QRectF r(symBbox(symbols));
165  setbbox(QRectF(0.0, -r.x() + y1, r.height(), r.width()));
166  }
167  break;
168 
169  case ArpeggioType::DOWN: {
170  symbolLine(SymId::wiggleArpeggiatoUpArrow, SymId::wiggleArpeggiatoUp);
171  // string is rotated +90 degrees (so that UpArrow turns into a DownArrow)
172  QRectF r(symBbox(symbols));
173  setbbox(QRectF(0.0, r.x() + y1, r.height(), r.width()));
174  }
175  break;
176 
177  case ArpeggioType::UP_STRAIGHT: {
178  qreal _spatium = spatium();
179  qreal x1 = _spatium * .5;
180  qreal w = symBbox(SymId::arrowheadBlackUp).width();
181  qreal lw = score()->styleS(Sid::ArpeggioLineWidth).val() * _spatium;
182  setbbox(QRectF(x1 - w * .5, y1, w, y2 - y1 + lw * .5));
183  }
184  break;
185 
186  case ArpeggioType::DOWN_STRAIGHT: {
187  qreal _spatium = spatium();
188  qreal x1 = _spatium * .5;
189  qreal w = symBbox(SymId::arrowheadBlackDown).width();
190  qreal lw = score()->styleS(Sid::ArpeggioLineWidth).val() * _spatium;
191  setbbox(QRectF(x1 - w * .5, y1 - lw * .5, w, y2 - y1 + lw * .5));
192  }
193  break;
194 
195  case ArpeggioType::BRACKET: {
196  qreal _spatium = spatium();
197  qreal lw = score()->styleS(Sid::ArpeggioLineWidth).val() * _spatium * .5;
198  qreal w = score()->styleS(Sid::ArpeggioHookLen).val() * _spatium;
199  setbbox(QRectF(0.0, y1, w, y2-y1).adjusted(-lw, -lw, lw, lw));
200  break;
201  }
202  }
203  }

209 void Arpeggio::draw(QPainter* p) const
210  {
211  if (_hidden)
212  return;
213  qreal _spatium = spatium();
214 
215  qreal y1 = -_userLen1;
216  qreal y2 = _height + _userLen2;
217 
218  p->setPen(QPen(curColor(),
219  score()->styleS(Sid::ArpeggioLineWidth).val() * _spatium,
220  Qt::SolidLine, Qt::RoundCap));
221 
222  p->save();
223  switch (arpeggioType()) {
224  case ArpeggioType::NORMAL:
225  case ArpeggioType::UP:
226  {
227  QRectF r(symBbox(symbols));
228  qreal scale = p->worldTransform().m11();
229  p->rotate(-90.0);
230  score()->scoreFont()->draw(symbols, p, magS(), QPointF(-r.right() - y1, -r.bottom() + r.height()), scale);
231  }
232  break;
233 
234  case ArpeggioType::DOWN:
235  {
236  QRectF r(symBbox(symbols));
237  qreal scale = p->worldTransform().m11();
238  p->rotate(90.0);
239  score()->scoreFont()->draw(symbols, p, magS(), QPointF(-r.left() + y1, -r.top() - r.height()), scale);
240  }
241  break;
242 
243  case ArpeggioType::UP_STRAIGHT:
244  {
245  QRectF r(symBbox(SymId::arrowheadBlackUp));
246  qreal x1 = _spatium * .5;
247  drawSymbol(SymId::arrowheadBlackUp, p, QPointF(x1 - r.width() * .5, y1 - r.top()));
248  y1 -= r.top() * .5;
249  p->drawLine(QLineF(x1, y1, x1, y2));
250  }
251  break;
252 
253  case ArpeggioType::DOWN_STRAIGHT:
254  {
255  QRectF r(symBbox(SymId::arrowheadBlackDown));
256  qreal x1 = _spatium * .5;
257 
258  drawSymbol(SymId::arrowheadBlackDown, p, QPointF(x1 - r.width() * .5, y2 - r.bottom()));
259  y2 += r.top() * .5;
260  p->drawLine(QLineF(x1, y1, x1, y2));
261  }
262  break;
263 
264  case ArpeggioType::BRACKET:
265  {
266  qreal w = score()->styleS(Sid::ArpeggioHookLen).val() * _spatium;
267  p->drawLine(QLineF(0.0, y1, 0.0, y2));
268  p->drawLine(QLineF(0.0, y1, w, y1));
269  p->drawLine(QLineF(0.0, y2, w, y2));
270  }
271  break;
272  }
273  p->restore();
274  }

319 
320 QVector<QLineF> Arpeggio::gripAnchorLines(Grip grip) const
321  {
322  QVector<QLineF> result;
323 
324  Chord* _chord = chord();
325  if (!_chord)
326  return result;
327 
328  const Page* p = toPage(findAncestor(ElementType::PAGE));
329  const QPointF pageOffset = p ? p->pos() : QPointF();
330 
331  const QPointF gripCanvasPos = gripsPositions()[static_cast<int>(grip)] + pageOffset;
332 
333  if (grip == Grip::START)
334  result << QLineF(_chord->upNote()->canvasPos(), gripCanvasPos);
335  else if (grip == Grip::END) {
336  Note* downNote = _chord->downNote();
337  int btrack = track() + (_span - 1) * VOICES;
338  Element* e = _chord->segment()->element(btrack);
339  if (e && e->isChord())
340  downNote = toChord(e)->downNote();
341  result << QLineF(downNote->canvasPos(), gripCanvasPos);
342  }
343  return result;
344  }

361 
362 bool Arpeggio::edit(EditData& ed)
363  {
364  if (ed.curGrip != Grip::END || !(ed.modifiers & Qt::ShiftModifier))
365  return false;
366 
367  if (ed.key == Qt::Key_Down) {
368  Staff* s = staff();
369  Part* part = s->part();
370  int n = part->nstaves();
371  int ridx = part->staves()->indexOf(s);
372  if (ridx >= 0) {
373  if (_span + ridx < n)
374  ++_span;
375  }
376  }
377  else if (ed.key == Qt::Key_Up) {
378  if (_span > 1)
379  --_span;
380  }
381  else
382  return false;
383  layout();
384  Chord* c = chord();
385  rxpos() = -(width() + spatium() * .5);
386  c->layoutArpeggio2();
387  return true;
388  }
