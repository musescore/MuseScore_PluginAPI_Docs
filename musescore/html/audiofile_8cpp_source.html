73 sf_count_t AudioFile::readData(short* data, sf_count_t frames)
74  {
75  //see https://musescore.org/en/node/22086#comment-83671
76  //see https://github.com/erikd/libsndfile/issues/16
77  //this code fixes the bug in libsndfile: float values are not normalized when reading .ogg
78  //this leads to overflowing signed short values, reverting a sign and clicking noise
79  sf_count_t resFrames = 0;
80  if (s16p == _type)
81  resFrames = sf_readf_short(sf, data, frames);
82  else {
83  //read native float values
84  int totalFrames = frames * channels();
85  std::vector<float> dataF;
86  dataF.resize(totalFrames);
87  resFrames = sf_readf_float(sf, dataF.data(), frames);
88  //find the maximum signal value
89  float maxSignal = 0.f;
90  for (int i = 0; i < totalFrames; ++i) {
91  if (fabs(dataF[i]) > maxSignal)
92  maxSignal = dataF[i] > 0 ? dataF[i] : -dataF[i];
93  }
94  //normalize values if and only if sample values range is incorrect
95  //which means having at least one sample value more than 1.0
96  float adjScale = maxSignal > 1.f ? 1.f/maxSignal : 1.f;
97  //convert normalized floats to signed short values
98  for (int i = 0; i < totalFrames; ++i)
99  data[i] = adjScale * lrintf(dataF[i] * (dataF[i] > 0 ? SHRT_MAX : -SHRT_MIN));
100  }
101 
102  return resFrames;
103  }
