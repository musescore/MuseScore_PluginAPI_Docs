47 static Score::FileError doImportMSMR(MasterScore* score
48  , QIODevice *data
49  , const QString& filePath
50  , bool created)
51  {
52  IF_ASSERT(score) {
53  return Score::FileError::FILE_ERROR;
54  }
55 
56  IF_ASSERT(data) {
57  return Score::FileError::FILE_ERROR;
58  }
59 
60  QString baseName = QFileInfo(filePath).baseName();
61  auto msmr = std::make_shared<Avs::MsmrFile>(data->readAll(), baseName);
62 
63  // import score
64  QByteArray msczData = msmr->readMscz();
65  if (!msczData.isEmpty()) {
66  // read mscz
67  QBuffer msczBuf(&msczData);
68  msczBuf.open(QIODevice::ReadOnly);
69  Score::FileError err = score->loadMsc("score.mscz", &msczBuf, true);
70  if (err != Score::FileError::FILE_NO_ERROR)
71  return err;
72  }
73  else {
74  // import MusicXml
75  QByteArray muzicXmlData = msmr->readMuzicXml();
76  if (muzicXmlData.isEmpty())
77  return Score::FileError::FILE_BAD_FORMAT;
78 
79  QBuffer muzicXmlBuf(&muzicXmlData);
80  muzicXmlBuf.open(QIODevice::ReadOnly);
81 
82  Score::FileError err = importMusicXml(score, &muzicXmlBuf, baseName);
83  if (err != Score::FileError::FILE_NO_ERROR)
84  return err;
85 
86  // corrections after import
87 
88  //! NOTE Audiveris sets the title "[Audiveris detected movement]",
89  //! I decided it was not convenient for the user,
90  //! the file base name as title is more convenient.
91  QString titleText = baseName;
92  Text* title = score->getText(Tid::TITLE);
93  if (title)
94  title->setPlainText(titleText);
95 
96  score->setMetaTag("movementNumber", "");
97  score->setMetaTag("movementTitle", "");
98  score->setMetaTag("source", QFileInfo(filePath).fileName());
99  score->setMetaTag("workTitle", title ? title->plainText() : titleText);
100  }
101 
102  // import omr
103  QByteArray ormData = msmr->readOmr();
104  if (ormData.isEmpty()) {
105  return Score::FileError::FILE_BAD_FORMAT;
106  }
107 
108  QBuffer ormBuf(&ormData);
109  Avs::AvsOmrReader reader;
110  std::shared_ptr<Avs::AvsOmr> omr = reader.read(&ormBuf);
111  if (!omr)
112  return Score::FileError::FILE_BAD_FORMAT;
113 
114  omr->setMsmrFile(msmr);
115  score->setAvsOmr(omr);
116 
117  // set format
118  score->setCreated(created);
119 
120  //! NOTE format of file determined by .ext,
121  //! therefore setting .msmr for correctly saving
122  score->fileInfo()->setFile(QFileInfo(filePath).path() + "/" + baseName + ".msmr");
123 
124  return Score::FileError::FILE_NO_ERROR;
125  }

132 Score::FileError importMSMR(MasterScore* score, const QString& filePath)
133  {
134  QFile file(filePath);
135  if (!file.exists()) {
136  LOGE() << "not exists file: " << filePath;
137  return Score::FileError::FILE_NOT_FOUND;
138  }
139 
140  if (!file.open(QIODevice::ReadOnly)) {
141  LOGE() << "failed open file: " << filePath;
142  return Score::FileError::FILE_OPEN_ERROR;
143  }
144 
145  return doImportMSMR(score, &file, filePath, false);
146  }

153 Score::FileError loadAndImportMSMR(MasterScore* score, const QString& filePath)
154  {
155  Avs::RecognitionProccessDialog progDialog;

157  auto onStep = [&progDialog](const Avs::AvsOmrNetRecognizer::Step& step) {
158  progDialog.onStep(step);
159  };
160 
161  std::shared_ptr<Avs::IAvsOmrRecognizer> recognizer;
162  bool useLocal = Avs::AvsOmrLocal::instance()->isUseLocal();
163  if (useLocal)
164  recognizer = std::make_shared<Avs::AvsOmrLocalRecognizer>();
165  else
166  recognizer = std::make_shared<Avs::AvsOmrNetRecognizer>();
167 
168  LOGI() << "try use avs omr recognizer: " << recognizer->type();
169  progDialog.setType(recognizer->type());
170  progDialog.show();
171 
172  QByteArray data;
173  bool ok = recognizer->recognize(filePath, &data, onStep);
174 
175  progDialog.onFinished(ok);
176 
177  if (!ok) {
178  //! NOTE A message has already been shown to the user
179  return Score::FileError::FILE_IGNORE_ERROR;
180  }
181 
182  QBuffer buf(&data);
183  buf.open(QIODevice::ReadOnly);
184 
185  return doImportMSMR(score, &buf, filePath, true);
186  }
