73 void AvsOmrDrawer::draw(std::shared_ptr<Context> ctx, const QList<const Ms::Measure *> &ml)
74  {
75  IF_ASSERT(ctx) {
76  return;
77  }
78 
79  if (ctx->omr->config().isHiddenAll())
80  return;
81 
82  ctx->p->save();
83 
84  for (const Ms::Measure* m : ml) {
85  ctx->ormMeasureIdx = ctx->measureIdxs.value(m);
86  ctx->ormSheetNum = ctx->omr->sheetNumByMeausereIdx(ctx->ormMeasureIdx);
87 
88  MMetrics mm;
89  mm.bbox = m->bbox();
90 
91  auto setMSegment = [](MSegment& ms, const Ms::Segment* s) {
92  ms.pos = s->pos();
93  ms.w = s->width();
94  };
95 
96  for (Ms::Segment* s = m->first(); s; s = s->next()) {
97  if (Ms::SegmentType::Clef == s->segmentType() || Ms::SegmentType::HeaderClef == s->segmentType()) {
98  setMSegment(mm.clef, s);
99  }
100  else if (Ms::SegmentType::KeySig == s->segmentType()) {
101  setMSegment(mm.key, s);
102  }
103  else if (Ms::SegmentType::TimeSig == s->segmentType()) {
104  setMSegment(mm.time, s);
105  }
106  else if (Ms::SegmentType::ChordRest == s->segmentType()) {
107  setMSegment(mm.fchord, s);
108  break; //! NOTE The first chord and we donâ€™t need anything else
109  }
110  }
111 
112  QPointF ppos = m->pagePos();
113  ctx->p->translate(ppos);
114  drawMeasure(ctx, mm);
115  ctx->p->translate(-ppos);
116  }
117 
118  ctx->p->restore();
119  }

125 void AvsOmrDrawer::drawMeasure(std::shared_ptr<Context> ctx, const MMetrics &scoreMM) const
126  {
127  AvsOmr::MMetrics omm = ctx->omr->mmetrics(ctx->ormSheetNum, ctx->ormMeasureIdx);
128 
129  QRectF scoreMH = scoreMM.headerBBox();
130  QRectF scoreMCH = scoreMM.chordBBox();
131 
132  QRect ormMH = omm.headerBBox();
133  ormMH.setRight(ormMH.right() + 30);
134  QRect ormMCH = omm.chordBBox();
135 
136  int topGap = omm.ebbox.top() - omm.bbox.top();
137  int bottomGap = omm.ebbox.bottom() - omm.bbox.bottom();
138 
139  drawGlyphs(ctx, ormMH, topGap, bottomGap, scoreMH);
140  ctx->p->translate(scoreMCH.left(), 0);
141  drawGlyphs(ctx, ormMCH, topGap, bottomGap, scoreMCH);
142  ctx->p->translate(-scoreMCH.left(), 0);
143  }

149 void AvsOmrDrawer::drawGlyphs(std::shared_ptr<Context> ctx
150  , const QRect omrBB
151  , int topGap, int bottomGap
152  , const QRectF scoreBB) const
153  {
154  QTransform originTransform = ctx->p->transform();
155 
156  // Scale
157  qreal sx = scoreBB.width() / static_cast<qreal>(omrBB.width());
158  qreal sy = scoreBB.height() / static_cast<qreal>(omrBB.height());
159 
160  QTransform transform = originTransform;
161  transform.scale(sx, sy);
162  ctx->p->setTransform(transform);
163 
164  QRect abb = omrBB.adjusted(0, topGap, 0, bottomGap);
165 
166  QList<AvsOmr::GlyphUsed> accepted;
167  if (ctx->omr->config().isShowRecognized())
168  accepted << AvsOmr::GlyphUsed::Used;
169 
170  if (ctx->omr->config().isShowNotRecognized())
171  accepted << AvsOmr::GlyphUsed::Free;
172 
173  QList<const AvsOmr::Glyph*> glys = ctx->omr->glyphsByBBox(ctx->ormSheetNum, abb, accepted);
174  for (const auto& g : glys) {
175  QRect gb = g->bbox.translated(-omrBB.left(), -omrBB.top());
176  ctx->p->drawImage(gb.topLeft(), g->img);
177  }
178 
179  ctx->p->setTransform(originTransform);
180  }
