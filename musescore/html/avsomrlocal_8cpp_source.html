218 Ret AvsOmrLocal::checkInstallOrUpdate(bool isWait)
219  {
220  const State& st = state();
221  if (State::NotInstalled == st) {
222  installBackground();
223  }
224  else if (State::Instaling == st) {
225  // noop, only waiting
226  }
227  else {
228  if (isNeedUpdate())
229  installBackground();
230  }
231 
232  if (isWait)
233  waitInstallOrUpdate();
234 
235  if (State::Ready != state())
236  return stateToRet(state());
237 
238  return Ret::Ok;
239  }

311 Ret AvsOmrLocal::execAvs(const QString &cmd, const QString& filePath, const QString& buildDir) const
312  {
313  QString home = avsHomePath();
314  if (home.isEmpty()) {
315  LOGE() << "not found avs plugin";
316  return Ret::LocalNotInstalled;
317  }
318 
319  QString make_avs = avsHomePath() + "/" + AVS_MAKE_SCRIPT;
320  if (!QFileInfo(make_avs).exists()) {
321  LOGE() << "not found make script: " << make_avs;
322  return Ret::LocalNotInstalled;
323  }
324 
325  QStringList arguments;
326  if (!SHELL_ARGS.isEmpty()) {
327  arguments << SHELL_ARGS;
328  }
329 
330  arguments << make_avs << cmd << filePath << buildDir;
331 
332  QEventLoop loop;
333  QProcess proc;
334 
335  int exitCode = -1;
336  auto onFinished = [&loop, &exitCode](int code, QProcess::ExitStatus exitStatus) {
337  LOGI() << "onFinished exitCode: " << code << ", exitStatus: " << exitStatus;
338  exitCode = code;
339  loop.quit();
340  };
341 
342  QObject::connect(&proc, static_cast<void(QProcess::*)(int, QProcess::ExitStatus)>(&QProcess::finished), onFinished);
343 
344  LOGI() << "try start: " << SHELL << arguments;
345  proc.start(SHELL, arguments);
346  loop.exec();
347 
348  LOGD() << QString::fromLatin1(proc.readAll());
349 
350  if (exitCode != 0)
351  return Ret::LocalFailedExec;
352 
353  return Ret::Ok;
354  }
