150 bool AvsOmrLocalInstaller::doLoadReleaseInfo(ReleaseInfo* info, const QString& url) const
151  {
152  IF_ASSERT(info) {
153  return false;
154  }
155 
156  QByteArray json_data;
157  bool ok = getData(&json_data, url, "application/json");
158  if (!ok) {
159  LOGE() << "failed get data, url: " << url;
160  return false;
161  }
162 
163  QJsonParseError err;
164  QJsonDocument doc = QJsonDocument::fromJson(json_data, &err);
165  if (err.error != QJsonParseError::NoError) {
166  LOGE() << "failed parse json, err: " << err.errorString() << ", json: " << json_data;
167  return false;
168  }
169 
170  QJsonObject infoObj = doc.object();
171  info->tag = infoObj.value("tag_name").toString();
172  QJsonArray assetsArr = infoObj.value("assets").toArray();
173  for (const auto& av : assetsArr) {
174  QJsonObject ao = av.toObject();
175  QString name = ao.value("name").toString();
176  if (!name.contains(RELEASE_PREFIX))
177  continue;
178 
179  info->url = ao.value("browser_download_url").toString();
180  break;
181  }
182 
183  if (info->tag.isEmpty() || info->url.isEmpty())
184  return false;
185 
186  return true;
187  }

196 static QNetworkReply* execReq(QNetworkAccessManager& net, QNetworkRequest req)
197  {
198  QNetworkReply* reply = net.get(req);
199 
200  QEventLoop loop;
201  QObject::connect(reply, &QNetworkReply::finished, &loop, &QEventLoop::quit);
202  loop.exec();
203 
204  int status = reply->attribute(QNetworkRequest::HttpStatusCodeAttribute).toInt();
205  if (status >= 300 && status < 399) { // redirect
206  QString newurl = reply->rawHeader("Location");
207  req.setUrl(newurl);
208  return execReq(net, req); // recursion
209  }
210 
211  return reply;
212  }
213 }

219 bool AvsOmrLocalInstaller::getData(QByteArray* data, const QString& url, const QByteArray &mime) const
220  {
221  QNetworkRequest req;
222  req.setUrl(QUrl(url));
223  req.setRawHeader("accept", mime);
224  QNetworkAccessManager net;
225 
226  QNetworkReply* reply = execReq(net, req);
227  if (reply->error() != QNetworkReply::NoError) {
228  LOGE() << "reply error: " << reply->errorString();
229  reply->close();
230  return false;

232 
233  if (data)
