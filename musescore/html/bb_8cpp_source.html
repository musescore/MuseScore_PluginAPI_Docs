92 bool BBFile::read(const QString& name)
93  {
94  _siglist.clear();
95  _siglist.add(0, Fraction(4, 4)); // debug
96 
97  _path = name;
98  QFile f(name);
99  if (!f.open(QIODevice::ReadOnly)) {
100  return false;
101  }
102  ba = f.readAll();
103  f.close();
104 
105  a = (const unsigned char*) ba.data();
106  size = ba.size();
107 
108  //---------------------------------------------------
109  // read version
110  //---------------------------------------------------
111 
112  int idx = 0;
113  _version = a[idx++];
114  switch(_version) {
115  case 0x43:
116  case 0x44:
117  case 0x45:
118  case 0x46:
119  case 0x47:
120  case 0x48:
121  case 0x49:
122  break;
123  default:
124  qDebug("BB: unknown file version %02x", _version);
125  return false;
126  }
127 
128  qDebug("read <%s> version 0x%02x", qPrintable(name), _version);
129 
130  //---------------------------------------------------
131  // read title
132  //---------------------------------------------------
133 
134  int len = a[idx++];
135  _title = new char[len+1];
136  for (int i = 0; i < len; ++i)
137  _title[i] = a[idx++];
138  _title[len] = 0;
139 
140  //---------------------------------------------------
141  // read style(timesig), key and bpm
142  //---------------------------------------------------
143 
144  ++idx;
145  ++idx;
146  _style = a[idx++] - 1;
147  if (_style < 0 || _style >= int(sizeof(styles)/sizeof(*styles))) {
148  qDebug("Import bb: unknown style %d", _style + 1);
149  return false;
150  }
151  _key = a[idx++];
152 
153  // map D# G# A# to Eb Ab Db
154  // major C, Db, D, Eb, E, F, Gb, G, Ab, A, Bb, B, C#, D#, F#, G#, A#
155  // minor C, Db, D, Eb, E, F, Gb, G, Ab, A, Bb, B, C#, D#, F#, G#, A#
156  static int kt[] = {
157  0, 0, -5, 2, -3, 4, -1, -6, 1, -4, 3, -2, 5, 7, -3, 6, -4, -2,
158  -3, 4, -1, -6, 1, -4, 3, -2, 5, 0, -5, 2, 4, 6, 3, 5, 7
159  };
160  if (_key >= int (sizeof(kt)/sizeof(*kt))) {
161  qDebug("bad key %d", _key);
162  return false;
163  }
164  _key = kt[_key];
165 
166  _bpm = a[idx] + (a[idx+1] << 8);
167  idx += 2;
168 
169  qDebug("Title <%s>", _title);
170  qDebug("style %d", _style);
171  qDebug("key %d", _key);
172  qDebug("sig %d/%d", timesigZ(), timesigN());
173  qDebug("bpm %d", _bpm);
174 
175  //---------------------------------------------------
176  // read bar types
177  //---------------------------------------------------
178 
179  int bar = a[idx++]; // starting bar number
180  while (bar < 255) {
181  int val = a[idx++];
182  if (val == 0)
183  bar += a[idx++];
184  else {
185  qDebug("bar type: bar %d val %d", bar, val);
186  _barType[bar++] = val;
187  }
188  }
189 
190  //---------------------------------------------------
191  // read chord extensions
192  //---------------------------------------------------
193 
194  int beat;
195  for (beat = 0; beat < MAX_BARS * 4;) {
196  int val = a[idx++];
197  if (val == 0)
198  beat += a[idx++];
199  else {
200  BBChord c;
201  c.extension = val;
202  c.beat = beat * (timesigZ() / timesigN());
203  ++beat;
204  _chords.append(c);
205  }
206  }
207 
208  //---------------------------------------------------
209  // read chord root
210  //---------------------------------------------------
211 
212  int roots = 0;
213  int maxbeat = 0;
214  for (beat = 0; beat < MAX_BARS * 4;) {
215  int val = a[idx++];
216  if (val == 0)
217  beat += a[idx++];
218  else {
219  int root = val % 18;
220  int bass = (root - 1 + val / 18) % 18 + 1;
221  if (root == bass)
222  bass = 0;
223  int ibeat = beat * (timesigZ() / timesigN());
224  if (ibeat != _chords[roots].beat) {
225  qDebug("import bb: inconsistent chord type and root beat");
226  return false;
227  }
228  _chords[roots].root = root;
229  _chords[roots].bass = bass;
230  if (maxbeat < beat)
231  maxbeat = beat;
232  ++roots;
233  ++beat;
234  }
235  }
236 
237  _measures = ((maxbeat + timesigZ() - 1) / timesigZ()) + 1;
238 
239  if (roots != _chords.size()) {
240  qDebug("import bb: roots %d != extensions %d", roots, _chords.size());
241  return false;
242  }
243  qDebug("Measures %d", _measures);
244 
245 #if 0
246  qDebug("================chords=======================");
247  foreach(BBChord c, _chords) {
248  qDebug("chord beat %3d bass %d root %d extension %d",
249  c.beat, c.bass, c.root, c.extension);
250  }
251  qDebug("================chords=======================");
252 #endif
253 
254  if (a[idx] == 1) { //??
255  qDebug("Skip 0x%02x at 0x%04x", a[idx], idx);
256  ++idx;
257  }
258 
259  _startChorus = a[idx++];
260  _endChorus = a[idx++];
261  _repeats = a[idx++];
262 
263  qDebug("start chorus %d end chorus %d repeats %d, pos now 0x%x",
264  _startChorus, _endChorus, _repeats, idx);
265 
266  if (_startChorus >= _endChorus) {
267  _startChorus = 0;
268  _endChorus = 0;
269  _repeats = 1;
270  }
271 
272  //---------------------------------------------------
273  // read style file
274  //---------------------------------------------------
275 
276  bool found = false;
277  for (int i = idx; i < size; ++i) {
278  if (a[i] == 0x42) {
279  if (a[i+1] < 16) {
280  for (int k = i+2; k < (i+18); ++k) {
281  if (a[k] == '.' && a[k+1] == 'S' && a[k+2] == 'T' && a[k+3] == 'Y') {
282  found = true;
283  break;
284  }
285  }
286  }
287  if (found) {
288  idx = i + 1;
289  break;
290  }
291  }
292  }
293  if (!found) {
294  qDebug("import bb: style file not found");
295  return false;
296  }
297 
298  qDebug("read styleName at 0x%x", idx);
299  len = a[idx++];
300  _styleName = new char[len+1];
301 
302  for (int i = 0; i < len; ++i)
303  _styleName[i] = a[idx++];
304  _styleName[len] = 0;
305 
306  qDebug("style name <%s>", _styleName);
307 
308  // read midi events
309  int eventStart = a[size-4] + a[size-3] * 256;
310  int eventCount = a[size-2] + a[size-1] * 256;
311 
312  Fraction endTick = Fraction::fromTicks(_measures * bbDivision * 4 * timesigZ() / timesigN());
313 
314  if (eventCount == 0) {
315  qDebug("no melody");
316  return true;
317  }
318  else {
319  idx = eventStart;
320  qDebug("melody found at 0x%x", idx);
321  int i = 0;
322  int lastLen = 0;
323  for (i = 0; i < eventCount; ++i, idx+=12) {
324  int type = a[idx + 4] & 0xf0;
325  if (type == 0x90) {
326  int channel = a[idx + 7];
327  BBTrack* track = 0;
328  foreach (BBTrack* t, _tracks) {
329  if (t->outChannel() == channel) {
330  track = t;
331  break;
332  }
333  }
334  if (track == 0) {
335  track = new BBTrack(this);
336  track->setOutChannel(channel);
337  _tracks.append(track);
338  }
339  Fraction tick = Fraction::fromTicks(a[idx] + (a[idx+1]<<8) + (a[idx+2]<<16) + (a[idx+3]<<24));
340  tick -= Fraction::fromTicks(4 * bbDivision);
341  if (tick >= endTick) {
342  qDebug("event tick %d > %d", tick.ticks(), endTick.ticks());
343  continue;
344  }
345  Event note(ME_NOTE);
346  note.setOntime((tick.ticks() * MScore::division) / bbDivision);
347  note.setPitch(a[idx + 5]);
348  note.setVelo(a[idx + 6]);
349  note.setChannel(channel);
350  int len1 = a[idx+8] + (a[idx+9]<<8) + (a[idx+10]<<16) + (a[idx+11]<<24);
351  if (len1 == 0) {
352  if (lastLen == 0) {
353  qDebug("note event of len 0 at idx %04x", idx);
354  continue;
355  }
356  len1 = lastLen;
357  }
358  lastLen = len1;
359  note.setDuration((len1 * MScore::division) / bbDivision);
360  track->append(note);
361  }
362  else if (type == 0xb0 || type == 0xc0) {
363  // ignore controller
364  }
365  else if (type == 0)
366  break;
367  else {
368  qDebug("unknown event type 0x%02x at x%04x", a[idx + 4], idx);
369  break;
370  }
371  }
372  qDebug("Events found x%02x (%d)", i, i);
373  }
374  return true;
375  }

382 Score::FileError importBB(MasterScore* score, const QString& name)
383  {
384  BBFile bb;
385  if(!QFileInfo::exists(name))
386  return Score::FileError::FILE_NOT_FOUND;
387  if (!bb.read(name)) {
388  qDebug("Cannot open file <%s>", qPrintable(name));
389  return Score::FileError::FILE_OPEN_ERROR;
390  }
391  score->style().set(Sid::chordsXmlFile, true);
392  score->style().chordList()->read("chords.xml");
393  *(score->sigmap()) = bb.siglist();
394 
395  QList<BBTrack*>* tracks = bb.tracks();
396  int ntracks = tracks->size();
397  if (ntracks == 0) // no events?
398  ntracks = 1;
399  for (int i = 0; i < ntracks; ++i) {
400  Part* part = new Part(score);
401  Staff* s = new Staff(score);
402  s->setPart(part);
403  part->insertStaff(s, 0);
404  score->staves().append(s);
405  score->appendPart(part);
406  }
407 
408  //---------------------------------------------------
409  // create measures
410  //---------------------------------------------------
411 
412  for (int i = 0; i < bb.measures(); ++i) {
413  Measure* measure = new Measure(score);
414  Fraction tick = Fraction::fromTicks(score->sigmap()->bar2tick(i, 0));
415  measure->setTick(tick);
416  Fraction ts = score->sigmap()->timesig(tick.ticks()).timesig();
417  measure->setTimesig(ts);
418  measure->setTicks(ts);
419  score->measures()->add(measure);
420  }
421 
422  //---------------------------------------------------
423  // create notes
424  //---------------------------------------------------
425 
426  foreach (BBTrack* track, *tracks)
427  track->cleanup();
428 
429  if (tracks->isEmpty()) {
430  for (MeasureBase* mb = score->first(); mb; mb = mb->next()) {
431  if (mb->type() != ElementType::MEASURE)
432  continue;
433  Measure* measure = (Measure*)mb;
434  Rest* rest = new Rest(score, TDuration(TDuration::DurationType::V_MEASURE));
435  rest->setTicks(measure->ticks());
436  rest->setTrack(0);
437  Segment* s = measure->getSegment(SegmentType::ChordRest, measure->tick());
438  s->add(rest);
439  }
440  }
441  else {
442  int staffIdx = 0;
443  foreach (BBTrack* track, *tracks)
444  bb.convertTrack(score, track, staffIdx++);
445  }
446 
447  for (MeasureBase* mb = score->first(); mb; mb = mb->next()) {
448  if (mb->type() != ElementType::MEASURE)
449  continue;
450  Measure* measure = (Measure*)mb;
451  Segment* s = measure->findSegment(SegmentType::ChordRest, measure->tick());
452  if (s == 0) {
453  Rest* rest = new Rest(score, TDuration(TDuration::DurationType::V_MEASURE));
454  rest->setTicks(measure->ticks());
455  rest->setTrack(0);
456  Segment* s1 = measure->getSegment(SegmentType::ChordRest, measure->tick());
457  s1->add(rest);
458  }
459  }
460 
461  score->spell();
462 
463  //---------------------------------------------------
464  // create title
465  //---------------------------------------------------
466 
467  Text* text = new Text(score, Tid::TITLE);
468  text->setPlainText(bb.title());
469 
470  MeasureBase* measureB = score->first();
471  if (measureB->type() != ElementType::VBOX) {
472  measureB = new VBox(score);
473  measureB->setTick(Fraction(0,1));
474  measureB->setNext(score->first());
475  score->measures()->add(measureB);
476  }
477  measureB->add(text);
478 
479  //---------------------------------------------------
480  // create chord symbols
481  //---------------------------------------------------
482 
483  static const int table[] = {
484  //C Db, D, Eb, E, F, Gb, G, Ab, A, Bb, B, C#, D#, F# G# A#
485  14, 9, 16, 11, 18, 13, 8, 15, 10, 17, 12, 19, 21, 23, 20, 22, 24
486  };
487  foreach(const BBChord& c, bb.chords()) {
488  Fraction tick = Fraction(c.beat, 4); // c.beat * MScore::division;
489 // qDebug("CHORD %d %d", c.beat, tick);
490  Measure* m = score->tick2measure(tick);
491  if (m == 0) {
492  qDebug("import BB: measure for tick %d not found", tick.ticks());
493  continue;
494  }
495  Segment* s = m->getSegment(SegmentType::ChordRest, tick);
496  Harmony* h = new Harmony(score);
497  h->setTrack(0);
498  h->setRootTpc(table[c.root-1]);
499  if (c.bass > 0)
500  h->setBaseTpc(table[c.bass-1]);
501  else
502  h->setBaseTpc(Tpc::TPC_INVALID);
503  h->setId(c.extension);
504  h->getDescription();
505  h->render();
506  s->add(h);
507  }
508 
509  //---------------------------------------------------
510  // insert layout breaks
511  // add chorus repeat
512  //---------------------------------------------------
513 
514  int startChorus = bb.startChorus() - 1;
515  int endChorus = bb.endChorus() - 1;
516 
517  int n = 0;
518  for (MeasureBase* mb = score->first(); mb; mb = mb->next()) {
519  if (mb->type() != ElementType::MEASURE)
520  continue;
521  Measure* measure = (Measure*)mb;
522  if (n && (n % 4) == 0) {
523  LayoutBreak* lb = new LayoutBreak(score);
524  lb->setLayoutBreakType(LayoutBreak::Type::LINE);
525  measure->add(lb);
526  }
527  if (startChorus == n)
528  measure->setRepeatStart(true);
529  else if (endChorus == n) {
530  measure->setRepeatEnd(true);
531  measure->setRepeatCount(bb.repeats());
532  }
533  ++n;
534  }
535 
536  foreach(Staff* staff, score->staves()) {
537  Fraction tick = Fraction(0,1);
538  KeySigEvent ke;
539  ke.setKey(Key(bb.key()));
540  staff->setKey(tick, ke);
541  KeySig* keysig = new KeySig(score);
542  keysig->setTrack((score->staffIdx(staff->part()) + staff->rstaff()) * VOICES);
543  keysig->setKey(Key(bb.key()));
544  Measure* mks = score->tick2measure(tick);
545  Segment* sks = mks->getSegment(SegmentType::KeySig, tick);
546  sks->add(keysig);
547  }
548  score->fixTicks();
549  return Score::FileError::FILE_NO_ERROR;
550  }

556 Fraction BBFile::processPendingNotes(Score* score, QList<MNote*>* notes, const Fraction& l, int track)
557  {
558  Fraction len(l);
559  Staff* cstaff = score->staff(track/VOICES);
560  const Instrument* instrument = cstaff->part()->instrument();
561  const Drumset* drumset = instrument->drumset();
562  bool useDrumset = instrument->useDrumset();
563  Fraction tick = Fraction::fromTicks(notes->at(0)->mc.ontime());
564 
565  //
566  // look for len of shortest note
567  //
568  foreach (const MNote* n, *notes) {
569  if (n->mc.duration() < len.ticks())
570  len = Fraction::fromTicks(n->mc.duration());
571  }
572 
573  //
574  // split notes on measure boundary
575  //
576  Measure* measure = score->tick2measure(tick);
577  if (measure == 0 || (tick >= measure->endTick())) {
578  qDebug("no measure found for tick %d", tick.ticks());
579  notes->clear();
580  return len;
581  }
582  if ((tick + len) > measure->endTick())
583  len = measure->endTick() - tick;
584 
585  Chord* chord = new Chord(score);
586  chord->setTrack(track);
587  TDuration d;
588  d.setVal(len.ticks());
589  chord->setDurationType(d);
590  chord->setTicks(d.fraction());
591  Segment* s = measure->getSegment(SegmentType::ChordRest, tick);
592  s->add(chord);
593 
594  foreach (MNote* n, *notes) {
595  QList<Event>& nl = n->mc.notes();
596  for (int i = 0; i < nl.size(); ++i) {
597  const Event& mn = nl[i];
598  Note* note = new Note(score);
599  note->setPitch(mn.pitch(), mn.tpc(), mn.tpc());
600  note->setTrack(track);
601  chord->add(note);
602 
603  if (useDrumset) {
604  if (!drumset->isValid(mn.pitch())) {
605  qDebug("unmapped drum note 0x%02x %d", mn.pitch(), mn.pitch());
606  }
607  else {
608  chord->setStemDirection(drumset->stemDirection(mn.pitch()));
609  }
610  }
611  if (n->ties[i]) {
612  n->ties[i]->setEndNote(note);
613  n->ties[i]->setTrack(note->track());
614  note->setTieBack(n->ties[i]);
615  }
616  }
617  if (n->mc.duration() <= len.ticks()) {
618  notes->removeAt(notes->indexOf(n));
619  continue;
620  }
621  for (int i = 0; i < nl.size(); ++i) {
622  const Event& mn = nl[i];
623  Note* note = chord->findNote(mn.pitch());
624  n->ties[i] = new Tie(score);
625  n->ties[i]->setStartNote(note);
626  note->setTieFor(n->ties[i]);
627  }
628  n->mc.setOntime(n->mc.ontime() + len.ticks());
629  n->mc.setLen(n->mc.duration() - len.ticks());
630  }
631  return len;
632  }

638 static ciEvent collectNotes(const Fraction& tick, int voice, ciEvent i, const EventList* el, QList<MNote*>* notes)
639  {
640  for (;i != el->end(); ++i) {
641  const Event& e = *i;
642  if (e.type() != ME_CHORD)
643  continue;
644  if (e.voice() != voice)
645  continue;
646  if (e.ontime() > tick.ticks())
647  break;
648  if (e.ontime() < tick.ticks())
649  continue;
650  MNote* n = new MNote(e);
651  notes->append(n);
652  }
653  return i;

660 void BBFile::convertTrack(Score* score, BBTrack* track, int staffIdx)
661  {
662  track->findChords();
663  int voices = track->separateVoices(2);
664  const EventList el = track->events();
665 
666  for (int voice = 0; voice < voices; ++voice) {
667  int tr = staffIdx * VOICES + voice;
668  QList<MNote*> notes;
669 
670  Fraction ctick = Fraction(0,1);
671  ciEvent i = collectNotes(ctick, voice, el.begin(), &el, &notes);
672 
673  for (; i != el.end();) {
674  const Event& e = *i;
675  if (e.type() != ME_CHORD || e.voice() != voice) {
676  ++i;
677  continue;
678  }
679  //
680  // process pending notes
681  //
682  Fraction restLen = Fraction::fromTicks(e.ontime()) - ctick;
683 // qDebug("ctick %d rest %d ontick %d size %d", ctick, restLen, e.ontime(), notes.size());
684 
685  if (restLen <= Fraction(0,1))
686  qFatal("bad restlen ontime %d - ctick %d", e.ontime(), ctick.ticks());
687 
688  while (!notes.isEmpty()) {
689  Fraction len = processPendingNotes(score, &notes, restLen, tr);
690  if (len.isZero()) {
691  qDebug("processPendingNotes returns zero, restlen %d, track %d", restLen.ticks(), tr);
692  ctick += restLen;
693  restLen = Fraction(0,1);
694  break;
695  }
696  ctick += len;
697  restLen -= len;
698  }
699 // qDebug(" 1.ctick %d rest %d", ctick, restLen);
700  //
701  // check for gap and fill with rest
702  //
703  if (voice == 0) {
704  while (restLen > Fraction(0,1)) {
705  Fraction len = restLen;
706  Measure* measure = score->tick2measure(ctick);
707  if (measure == 0 || (ctick >= measure->endTick())) { // at end?
708  ctick += len;
709  restLen -= len;
710  break;
711  }
712  // split rest on measure boundary
713  if ((ctick + len) > measure->endTick()) {
714  len = measure->endTick() - ctick;
715  if (len <= Fraction(0,1)) {

717  break;
718  }
719  }

726  s->add(rest);
727 // qDebug(" add rest %d", len);
728 
729  ctick += len;
730  restLen -= len;
731  }
732  }
733  else
734  ctick += restLen;

737  //
738  // collect all notes at ctick
739  //
740  i = collectNotes(ctick, voice, i, &el, &notes);
741  }

744  // process pending notes
745  //
746  while (!notes.isEmpty()) {
747  Fraction len = processPendingNotes(score, &notes, Fraction(0x7fffffff,1), tr);
748  ctick += len;
749  }
750  if (voice == 0) {
751  Measure* measure = score->tick2measure(ctick);
752  if (measure && (ctick < measure->endTick())) { // at end?
753  TDuration d;
754  d.setVal((measure->endTick() - ctick).ticks());
755  Rest* rest = new Rest(score, d);
756  rest->setTicks(d.fraction());
757  rest->setTrack(staffIdx * VOICES);
758  Segment* s = measure->getSegment(SegmentType::ChordRest, ctick);
759  s->add(rest);
760  }
761  }

770 void BBTrack::quantize(int startTick, int endTick, EventList* dst)
771  {
772  int mintick = MScore::division * 64;
773  iEvent i = _events.begin();
774  for (; i != _events.end(); ++i) {
775  if (i->ontime() >= startTick)
776  break;
777  }
778  iEvent si = i;
779  for (; i != _events.end(); ++i) {
780  const Event& e = *i;
781  if (e.ontime() >= endTick)
782  break;
783  if (e.type() == ME_NOTE && (e.duration() < mintick))
784  mintick = e.duration();
785  }
786  if (mintick <= MScore::division / 16) // minimum duration is 1/64
787  mintick = MScore::division / 16;
788  else if (mintick <= MScore::division / 8)
789  mintick = MScore::division / 8;
790  else if (mintick <= MScore::division / 4)
791  mintick = MScore::division / 4;
792  else if (mintick <= MScore::division / 2)
793  mintick = MScore::division / 2;
794  else if (mintick <= MScore::division)
795  mintick = MScore::division;
796  else if (mintick <= MScore::division * 2)
797  mintick = MScore::division * 2;
798  else if (mintick <= MScore::division * 4)
799  mintick = MScore::division * 4;
800  else if (mintick <= MScore::division * 8)
801  mintick = MScore::division * 8;
802  int raster;
803  if (mintick > MScore::division)
804  raster = MScore::division;
805  else
806  raster = mintick;
807 
808  //
809  // quantize onset
810  //
811  for (i = si; i != _events.end(); ++i) {
812  Event e = *i;
813  if (e.ontime() >= endTick)
814  break;
815  if (e.type() == ME_NOTE) {
816  // prefer moving note to the right
817  int tick = ((e.ontime() + raster/2) / raster) * raster;
818  int diff = tick - e.ontime();
819  int len = e.duration() - diff;
820  e.setOntime(tick);
821  e.setLen(len);
822  }
823  dst->insert(e);
824  }
825  //
826  // quantize duration
827  //
828  for (i = dst->begin(); i != dst->end(); ++i) {
829  Event& e = *i;
830  if (e.type() != ME_NOTE)
831  continue;
832  int tick = e.ontime();
833  int len = e.duration();
834  int ntick = tick + len;
835  int nntick = -1;
836  for (iEvent ii = (i+1); ii != dst->end(); ++ii) {
837  if (ii->type() == ME_NOTE) {
838  const Event& ee = *ii;
839  if (ee.ontime() == tick)
840  continue;
841  nntick = ee.ontime();
842  break;
843  }
844  }
845  if (nntick == -1)
846  len = quantizeLen(len, raster);
847  else {
848  int diff = nntick - ntick;
849  if (diff > 0) {
850  // insert rest?
851  if (diff <= raster)
852  len = nntick - tick;
853  else
854  len = quantizeLen(len, raster);
855  }
856  else {
857  if (diff > -raster)
858  len = nntick - tick;
859  else
860  len = quantizeLen(len, raster);
861  }
862  }
863  e.setLen(len);
864  }
865  }

873 void BBTrack::cleanup()
874  {
875  EventList dl;
876 
877  //
878  // quantize
879  //
880  int lastTick = 0;
881  foreach (const Event& e, _events) {
882  if (e.type() != ME_NOTE)
883  continue;
884  int offtime = e.offtime();
885  if (offtime > lastTick)
886  lastTick = offtime;
887  }
888  int startTick = 0;
889  for (int i = 1;; ++i) {
890  int endTick = bb->siglist().bar2tick(i, 0);
891  quantize(startTick, endTick, &dl);
892  if (endTick > lastTick)
893  break;
894  startTick = endTick;
895  }
896 
897  //
898  //
899  //
900  _events.clear();
901 
902  for(iEvent i = dl.begin(); i != dl.end(); ++i) {
903  Event& e = *i;
904  if (e.type() == ME_NOTE) {
905  iEvent ii = i;
906  ++ii;
907  for (; ii != dl.end(); ++ii) {
908  const Event& ee = *ii;
909  if (ee.type() != ME_NOTE || ee.pitch() != e.pitch())
910  continue;
911  if (ee.ontime() >= e.ontime() + e.duration())
912  break;
913  e.setLen(ee.ontime() - e.ontime());
914  break;
915  }
916  if (e.duration() <= 0)
917  continue;
918  }
919  _events.insert(e);
920  }
921  }

927 void BBTrack::findChords()
928  {

930  int n = _events.size();
931 
932  Drumset* drumset;
933  if (_drumTrack)
934  drumset = smDrumset;
935  else
936  drumset = 0;
937  int jitter = 3; // tick tolerance for note on/off
938 
939  for (int i = 0; i < n; ++i) {
940  Event e = _events[i];
941  if (e.type() == ME_INVALID)
942  continue;
943  if (e.type() != ME_NOTE) {
944  dl.append(e);
945  continue;
946  }
947 
948  Event note(e);
949  int ontime = note.ontime();
950  int offtime = note.offtime();

952  chord.setOntime(ontime);
953  chord.setLen(note.duration());
954  chord.notes().append(note);
955  int voice = 0;
956  chord.setVoice(voice);
957  dl.append(chord);
958  _events[i].setType(ME_INVALID);
959 
960  bool useDrumset = false;
961  if (drumset) {
962  int pitch = note.pitch();
963  if (drumset->isValid(pitch)) {
964  useDrumset = true;
965  voice = drumset->voice(pitch);
966  chord.setVoice(voice);
967  }
968  }
969  for (int k = i + 1; k < n; ++k) {
970  if (_events[k].type() != ME_NOTE)
971  continue;
972  Event nn = _events[k];
973  if (nn.ontime() - jitter > ontime)
974  break;
975  if (qAbs(nn.ontime() - ontime) > jitter || qAbs(nn.offtime() - offtime) > jitter)
976  continue;
977  int pitch = nn.pitch();
978  if (useDrumset) {
979  if (drumset->isValid(pitch) && drumset->voice(pitch) == voice) {
980  chord.notes().append(nn);
981  _events[k].setType(ME_INVALID);
982  }
983  }
984  else {
985  chord.notes().append(nn);
986  _events[k].setType(ME_INVALID);
987  }
988  }
989  }
990  _events = dl;
991  }
