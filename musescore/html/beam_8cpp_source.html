256 bool Beam::twoBeamedNotes()
257  {
258  // if not two elements or elements are not chords or chords have more than 1 note, return failure
259  if ((_elements.size() != 2)
260  || !_elements[0]->isChord()
261  || !_elements[1]->isChord()) {
262  return false;
263  }
264  const Chord* c1 = toChord(_elements[0]);
265  const Chord* c2 = toChord(_elements[1]);
266  if (c1->notes().size() != 1 || c2->notes().size() != 1)
267  return false;
268 
269  int upDnLimit = staff()->lines(Fraction(0, 1)) - 1; // was '4' hard-coded in the next 2 lines
270  int dist1 = c1->upLine() - upDnLimit;
271  int dist2 = c2->upLine() - upDnLimit;
272  if ((dist1 == -dist2) || (-dist1 == dist2)) {
273  _up = false;
274 #if 0
275  // this code appears to implement a rule
276  // that says two middle-line beamed notes
277  // should follow the same beam direction as a previous beam group
278  // but we don't follow this rule for single notes or for larger beam groups
279  // also, it makes little sense to follow it if the previous group
280  // is in another measure, which may end up on another system or page
281  Segment* s = c1->segment();
282  s = s->prev1(SegmentType::ChordRest);
283  if (s) {
284  Element* e = s->element(c1->track());
285  if (e && e->isChord()) {
286  Chord* c = toChord(e);
287  if (c->beam())
288  _up = c->beam()->up();
289  }
290  }
291 #endif
292  }
293  else if (qAbs(dist1) > qAbs(dist2))
294  _up = dist1 > 0;
295  else
296  _up = dist2 > 0;
297  return true;
298  }

307 void Beam::layout1()
308  {
309  //delete old segments
310  qDeleteAll(beamSegments);
311  beamSegments.clear();
312 
313  setParent(nullptr); // parent is System
314 
315  maxDuration.setType(TDuration::DurationType::V_INVALID);
316  Chord* c1 = 0;
317  Chord* c2 = 0;
318 
319  // TAB's with stem beside staves have special layout
320  if (staff()->isTabStaff(Fraction(0,1)) && !staff()->staffType(Fraction(0,1))->stemThrough()) {
321  //TABULATURES: all beams (and related chords) are:
322  // UP or DOWN according to TAB duration position
323  // slope 0
324  _up = !staff()->staffType(Fraction(0,1))->stemsDown();
325  slope = 0.0;
326  _cross = false;
327  minMove = maxMove = 0; // no cross-beaming in TAB's!
328  for (ChordRest* cr : qAsConst(_elements)) {
329  if (cr->isChord()) {
330  // set members maxDuration, c1, c2
331  if (!maxDuration.isValid() || (maxDuration < cr->durationType()))
332  maxDuration = cr->durationType();
333  c2 = toChord(cr);
334  if (c1 == 0)
335  c1 = c2;
336  }
337  }
338  }
339  else if (staff()->isDrumStaff(Fraction(0,1))) {
340  if (_direction != Direction::AUTO)
341  _up = _direction == Direction::UP;
342  else {
343  for (ChordRest* cr :qAsConst(_elements)) {
344  if (cr->isChord()) {
345  c2 = toChord(cr);
346  _up = c2->up();
347  break;
348  }
349  }
350  }
351  for (ChordRest* cr : qAsConst(_elements)) {
352  cr->setUp(_up);
353  cr->layoutStem1();
354  }
355  }
356  else {
357  //PITCHED STAVES (and TAB's with stems through staves)
358  minMove = 1000;
359  maxMove = -1000;
360  _isGrace = false;
361  qreal mag = 0.0;
362 
363  int mUp = 0;
364  int mDown = 0;
365  int upDnLimit = staff()->lines(Fraction(0,1)) - 1; // was '4' hard-coded in following code
366 
367  int staffIdx = -1;
368  for (ChordRest* cr : qAsConst(_elements)) {
369  qreal m = cr->small() ? score()->styleD(Sid::smallNoteMag) : 1.0;
370  mag = qMax(mag, m);
371  if (cr->isChord()) {
372  c2 = toChord(cr);
373  staffIdx = c2->vStaffIdx();
374  if (c1 == 0)
375  c1 = c2;
376  int i = c2->staffMove();
377  if (i < minMove)
378  minMove = i;
379  if (i > maxMove)
380  maxMove = i;
381  int line = c2->upLine();
382  if ((upDnLimit - line) > mUp)
383  mUp = upDnLimit - line;
384  line = c2->downLine();
385  if (line - upDnLimit > mDown)
386  mDown = line - upDnLimit;
387  }
388  if (!maxDuration.isValid() || (maxDuration < cr->durationType()))
389  maxDuration = cr->durationType();
390  }
391  setMag(mag);
392  //
393  // determine beam stem direction
394  //
395  if (_direction != Direction::AUTO) {
396  _up = _direction == Direction::UP;
397  }
398  else {
399  if (maxMove > 0) // cross staff beaming down
400  _up = true;
401  else if (minMove < 0)
402  _up = false;
403  else if (c1) {
404  Measure* m = c1->measure();
405  if (c1->stemDirection() != Direction::AUTO)
406  _up = c1->stemDirection() == Direction::UP;
407  else if (m->hasVoices(c1->staffIdx(), tick(), ticks()))
408  _up = !(c1->voice() % 2);
409  else if (!twoBeamedNotes()) {
410  // highest or lowest note determines stem direction
411  // interval higher is bigger -> downstem
412  // interval lower is bigger -> upstem
413  // down-stems is preferred if equal
414  _up = mUp < mDown;
415  }
416  }
417  else
418  _up = true;
419  }
420 
421  _cross = minMove < maxMove;
422  if (minMove == 1 && maxMove == 1)
423  setTrack(staffIdx * VOICES + voice());
424  else if (c1)
425  setTrack(c1->staffIdx() * VOICES + voice());
426 
427  // int idx = (_direction == Direction::AUTO || _direction == Direction::DOWN) ? 0 : 1;
428  slope = 0.0;
429 
430  // leave initial guess alone for moved chords within a beam that crosses staves
431  // otherwise, assume beam direction is stem direction
432 
433  for (ChordRest* cr : qAsConst(_elements)) {
434  const bool staffMove = cr->isChord() ? toChord(cr)->staffMove() : false;
435  if (!_cross || !staffMove) {
436  if (cr->up() != _up) {
437  cr->setUp(_up);
438  cr->layoutStem1();
439  }
440  }
441  }
442  } // end of if/else(tablature)
443  }

594 bool Beam::slopeZero(const std::vector<ChordRest*>& cl)
595  {
596  if (hasNoSlope() || cl.size() < 2)
597  return true;
598 
599  //
600  // return true if beam spans a rest
601  //
602  if (cl.size() == 2 && (cl.front()->isRest() || cl.back()->isRest()))
603  return true;
604 
605 // for (const ChordRest* cr : cl) {
606 // if (!cr->isChord())
607 // return true;
608 // }
609  int l1 = cl.front()->line();
610  int le = cl.back()->line();
611 
612  // look for some pattern
613  if (cl.size() == 4) {
614  int l2 = cl[1]->line();
615  int l3 = cl[2]->line();
616 
617  if ((l1 < le) && (l2 > l1) && (l2 > l3) && (l3 > le))
618  return true;
619  if ((l1 == l3) && (l2 == le))
620  return true;
621  }
622  else if (cl.size() == 6) {
623  int l2 = cl[1]->line();
624  int l3 = cl[2]->line();
625  int l4 = cl[3]->line();
626  int l5 = cl[4]->line();
627  if ((l2 > l1) && (l3 > l2) && (l1 == l4) && (l2 == l5) && (l3 == le))
628  return true;
629  }
630  //
631  // concave beams have a slope of 0.0
632  //
633  bool sameLine = true;
634 
635  slope = 0.0;
636  if (cl.size() >= 3) {
637  int l4 = cl[1]->line(_up);
638  for (unsigned i = 1; i < cl.size()-1; ++i) {
639  // Don't consider interior rests
640  if (!cl[i]->isChord())
641  continue;
642  int l3 = cl[i]->line(_up);
643  if (_up) {
644  if (l3 < l1 && l3 < le)
645  return true;
646  }
647  else {
648  if (l3 > l1 && l3 > le)
649  return true;
650  }
651  sameLine = l3 == l4;
652  }
653  if (sameLine && (l1 == l4 || le == l4) && cl[1]->isChord()) {
654  if (_up) {
655  if (l1 == l4 && l1 < le)
656  return true;
657  if (le == l4 && le < l1)
658  return true;
659  }
660  else {
661  if (l1 == l4 && l1 > le)
662  return true;
663  else if (le == l4 && le > l1)
664  return true;
665  }
666  }
667  }
668  return l1 == le;
669  }

1089  {

1100  qreal yUp = p1.y() + (p3.x() - p1.x()) * slope;
1101  int l = lrint((yUp - p3.y()) / (_spatium4));
1102  ml = qMax(ml, l);
1103  }
1104  }
