71 void Bend::layout()
72  {
73  // during mtest, there may be no score. If so, exit.
74  if (!score())
75  return;
76 
77  qreal _spatium = spatium();
78 
79  if (staff() && !staff()->isTabStaff(tick())) {
80  if (!parent()) {
81  noteWidth = -_spatium*2;
82  notePos = QPointF(0.0, _spatium*3);
83  }
84  }
85 
86  qreal _lw = _lineWidth;
87  Note* note = toNote(parent());
88  if (note == 0) {
89  noteWidth = 0.0;
90  notePos = QPointF();
91  }
92  else {
93  notePos = note->pos();
94  notePos.ry() = qMax(notePos.y(), 0.0);
95  noteWidth = note->width();
96  }
97  QRectF bb;
98 
99  QFontMetricsF fm(font(_spatium), MScore::paintDevice());
100 
101  int n = _points.size();
102  qreal x = noteWidth;
103  qreal y = -_spatium * .8;
104  qreal x2, y2;
105 
106  qreal aw = _spatium * .5;
107  QPolygonF arrowUp;
108  arrowUp << QPointF(0, 0) << QPointF(aw*.5, aw) << QPointF(-aw*.5, aw);
109  QPolygonF arrowDown;
110  arrowDown << QPointF(0, 0) << QPointF(aw*.5, -aw) << QPointF(-aw*.5, -aw);
111 
112  for (int pt = 0; pt < n; ++pt) {
113  if (pt == (n-1))
114  break;
115  int pitch = _points[pt].pitch;
116  if (pt == 0 && pitch) {
117  y2 = -notePos.y() -_spatium * 2;
118  x2 = x;
119  bb |= QRectF(x, y, x2-x, y2-y);
120 
121  bb |= arrowUp.translated(x2, y2 + _spatium * .2).boundingRect();
122 
123  int idx = (pitch + 12)/25;
124  const char* l = label[idx];
125  bb |= fm.boundingRect(QRectF(x2, y2, 0, 0),
126  Qt::AlignHCenter | Qt::AlignBottom | Qt::TextDontClip, QString(l));
127  y = y2;
128  }
129  if (pitch == _points[pt+1].pitch) {
130  if (pt == (n-2))
131  break;
132  x2 = x + _spatium;
133  y2 = y;
134  bb |= QRectF(x, y, x2-x, y2-y);
135  }
136  else if (pitch < _points[pt+1].pitch) {
137  // up
138  x2 = x + _spatium*.5;
139  y2 = -notePos.y() -_spatium * 2;
140  qreal dx = x2 - x;
141  qreal dy = y2 - y;
142 
143  QPainterPath path;
144  path.moveTo(x, y);
145  path.cubicTo(x+dx/2, y, x2, y+dy/4, x2, y2);
146  bb |= path.boundingRect();
147  bb |= arrowUp.translated(x2, y2 + _spatium * .2).boundingRect();
148 
149  int idx = (_points[pt+1].pitch + 12)/25;
150  const char* l = label[idx];
151  bb |= fm.boundingRect(QRectF(x2, y2, 0, 0),
152  Qt::AlignHCenter | Qt::AlignBottom | Qt::TextDontClip, QString(l));
153  }
154  else {
155  // down
156  x2 = x + _spatium*.5;
157  y2 = y + _spatium * 3;
158  qreal dx = x2 - x;
159  qreal dy = y2 - y;
160 
161  QPainterPath path;
162  path.moveTo(x, y);
163  path.cubicTo(x+dx/2, y, x2, y+dy/4, x2, y2);
164  bb |= path.boundingRect();
165 
166  bb |= arrowDown.translated(x2, y2 - _spatium * .2).boundingRect();
167  }
168  x = x2;
169  y = y2;
170  }
171  bb.adjust(-_lw, -_lw, _lw, _lw);
172  setbbox(bb);
173  setPos(0.0, 0.0);
174  }

180 void Bend::draw(QPainter* painter) const
181  {
182  qreal _spatium = spatium();
183  qreal _lw = _lineWidth;
184 
185  QPen pen(curColor(), _lw, Qt::SolidLine, Qt::RoundCap, Qt::RoundJoin);
186  painter->setPen(pen);
187  painter->setBrush(QBrush(curColor()));
188 
189  QFont f = font(_spatium * MScore::pixelRatio);
190  painter->setFont(f);
191 
192  qreal x = noteWidth + _spatium * .2;
193  qreal y = -_spatium * .8;
194  qreal x2, y2;
195 
196  qreal aw = score()->styleP(Sid::bendArrowWidth);
197  QPolygonF arrowUp;
198  arrowUp << QPointF(0, 0) << QPointF(aw * .5, aw) << QPointF(-aw *.5, aw);
199  QPolygonF arrowDown;
200  arrowDown << QPointF(0, 0) << QPointF(aw * .5, -aw) << QPointF(-aw *.5, -aw);
201 
202  int n = _points.size();
203  for (int pt = 0; pt < n-1; ++pt) {
204  int pitch = _points[pt].pitch;
205  if (pt == 0 && pitch) {
206  y2 = -notePos.y() -_spatium * 2;
207  x2 = x;
208  painter->drawLine(QLineF(x, y, x2, y2));
209 
210  painter->setBrush(curColor());
211  painter->drawPolygon(arrowUp.translated(x2, y2));
212 
213  int idx = (pitch + 12)/25;
214  const char* l = label[idx];
215  QString s(l);
216 #if 1
217  painter->drawText(QRectF(x2, y2, .0, .0), Qt::AlignHCenter | Qt::AlignBottom | Qt::TextDontClip, s);
218 #else
219  // this is the code used originally,
220  // and it worked in 2.3.2, but fails currently - textWidth & textHeight are too large
221  // presumably they would need to be scaled accoridng to pixelRatio, DPI, and/or SPATIUM20
222  // now that the font & fontmetrics are also scaled
223  QFontMetrics fm(f, MScore::paintDevice());
224  qreal textWidth = fm.width(s);
225  qreal textHeight = fm.height();
226  painter->drawText(QRectF(x2 - textWidth / 2, y2 - textHeight / 2, .0, .0), Qt::AlignVCenter|Qt::TextDontClip, s);
227 #endif
228 
229  y = y2;
230  }
231  if (pitch == _points[pt+1].pitch) {
232  if (pt == (n-2))
233  break;
234  x2 = x + _spatium;
235  y2 = y;
236  painter->drawLine(QLineF(x, y, x2, y2));
237  }
238  else if (pitch < _points[pt+1].pitch) {
239  // up
240  x2 = x + _spatium*.5;
241  y2 = -notePos.y() -_spatium * 2;
242  qreal dx = x2 - x;
243  qreal dy = y2 - y;
244 
245  QPainterPath path;
246  path.moveTo(x, y);
247  path.cubicTo(x+dx/2, y, x2, y+dy/4, x2, y2);
248  painter->setBrush(Qt::NoBrush);
249  painter->drawPath(path);
250 
251  painter->setBrush(curColor());
252  painter->drawPolygon(arrowUp.translated(x2, y2 ));
253 
254  int idx = (_points[pt+1].pitch + 12)/25;
255  const char* l = label[idx];
256  qreal ty = y2; // - _spatium;
257  painter->drawText(QRectF(x2, ty, .0, .0),
258  Qt::AlignHCenter | Qt::AlignBottom | Qt::TextDontClip, QString(l));
259  }
260  else {
261  // down
262  x2 = x + _spatium*.5;
263  y2 = y + _spatium * 3;
264  qreal dx = x2 - x;
265  qreal dy = y2 - y;

286  {

302 void Bend::read(XmlReader& e)
303  {
304  while (e.readNextStartElement()) {
305  const QStringRef& tag(e.name());
306 
307  if (readStyledProperty(e, tag))
308  ;
309  else if (tag == "point") {
310  PitchValue pv;
311  pv.time = e.intAttribute("time");
312  pv.pitch = e.intAttribute("pitch");
313  pv.vibrato = e.intAttribute("vibrato");
314  _points.append(pv);
315  e.readNext();
316  }
317  else if (tag == "play")
318  setPlayBend(e.readBool());
319  else if (!Element::readProperties(e))
320  e.unknown();
321  }
322  }
