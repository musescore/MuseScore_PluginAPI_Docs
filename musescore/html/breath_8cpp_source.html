64 void Breath::layout()
65  {
66  bool palette = (!staff() || track() == -1);
67  if (!palette) {
68  int voiceOffset = placeBelow() * (staff()->lines(tick()) - 1) * spatium();
69  if (isCaesura())
70  setPos(rxpos(), spatium() + voiceOffset);
71  else if ((score()->styleSt(Sid::MusicalSymbolFont) == "Emmentaler") && (symId() == SymId::breathMarkComma))
72  setPos(rxpos(), 0.5 * spatium() + voiceOffset);
73  else
74  setPos(rxpos(), -0.5 * spatium() + voiceOffset);
75  }
76  setbbox(symBbox(_symId));
77  }

98 void Breath::read(XmlReader& e)
99  {
100  while (e.readNextStartElement()) {
101  const QStringRef& tag(e.name());
102  if (tag == "subtype") { // obsolete
103  switch (e.readInt()) {
104  case 0:
105  case 1:
106  _symId = SymId::breathMarkComma;
107  break;
108  case 2:
109  _symId = SymId::caesuraCurved;
110  break;
111  case 3:
112  _symId = SymId::caesura;
113  break;
114  }
115  }
116  else if (tag == "symbol")
117  _symId = Sym::name2id(e.readElementText());
118  else if (tag == "pause")
119  _pause = e.readDouble();
120  else if (!Element::readProperties(e))
121  e.unknown();
122  }
123  }
