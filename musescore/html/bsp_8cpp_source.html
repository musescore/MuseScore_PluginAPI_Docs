205 void BspTree::initialize(const QRectF& rec, int dep, int index)
206  {
207  Node* node = &nodes[index];
208  if (index == 0) {
209  node->type = Node::Type::HORIZONTAL;
210  node->offset = rec.center().x();
211  }
212 
213  if (dep) {
214  Node::Type type;
215  QRectF rect1, rect2;
216  qreal offset1, offset2;
217 
218  if (node->type == Node::Type::HORIZONTAL) {
219  type = Node::Type::VERTICAL;
220  rect1.setRect(rec.left(), rec.top(), rec.width(), rec.height() * .5);
221  rect2.setRect(rect1.left(), rect1.bottom(), rect1.width(), rec.height() - rect1.height());
222  offset1 = rect1.center().x();
223  offset2 = rect2.center().x();
224  }
225  else {
226  type = Node::Type::HORIZONTAL;
227  rect1.setRect(rec.left(), rec.top(), rec.width() * .5, rec.height());
228  rect2.setRect(rect1.right(), rect1.top(), rec.width() - rect1.width(), rect1.height());
229  offset1 = rect1.center().y();
230  offset2 = rect2.center().y();
231  }
232 
233  int childIndex = firstChildIndex(index);
234 
235  Node* child = &nodes[childIndex];
236  child->offset = offset1;
237  child->type = type;
238 
239  child = &nodes[childIndex + 1];
240  child->offset = offset2;
241  child->type = type;
242 
243  initialize(rect1, dep - 1, childIndex);
244  initialize(rect2, dep - 1, childIndex + 1);
245  }
246  else {
247  node->type = Node::Type::LEAF;
248  node->leafIndex = leafCnt++;
249  }
250  }

256 void BspTree::climbTree(BspTreeVisitor* visitor, const QPointF& pos, int index)
257  {
258  if (nodes.empty())
259  return;
260 
261  Node* node = &nodes[index];
262  int childIndex = firstChildIndex(index);
263 
264  switch (node->type) {
265  case Node::Type::LEAF:
266  visitor->visit(&leaves[node->leafIndex]);
267  break;
268  case Node::Type::VERTICAL:
269  if (pos.x() < node->offset)
270  climbTree(visitor, pos, childIndex);
271  else
272  climbTree(visitor, pos, childIndex + 1);
273  break;
274  case Node::Type::HORIZONTAL:
275  if (pos.y() < node->offset)
276  climbTree(visitor, pos, childIndex);
277  else
278  climbTree(visitor, pos, childIndex + 1);
279  break;
280  }
281  }

288  {
