60 bool BSymbol::readProperties(XmlReader& e)
61  {
62  const QStringRef& tag = e.name();
63 
64  if (Element::readProperties(e))
65  return true;
66  else if (tag == "systemFlag")
67  setSystemFlag(e.readInt());
68  else if (tag == "Symbol" || tag == "FSymbol") {
69  Element* element = name2Element(tag, score());
70  element->read(e);
71  add(element);
72  }
73  else if ( tag == "Image") {
74  if (MScore::noImages)
75  e.skipCurrentElement();
76  else {
77  Element* element = name2Element(tag, score());
78  element->read(e);
79  add(element);
80  }
81  }
82  else
83  return false;
84  return true;

176 QRectF BSymbol::drag(EditData& ed)
177  {
178  QRectF r(canvasBoundingRect());
179  foreach(const Element* e, _leafs)
180  r |= e->canvasBoundingRect();
181 
182  qreal x = ed.delta.x();
183  qreal y = ed.delta.y();
184 
185  qreal _spatium = spatium();
186  if (ed.hRaster) {
187  qreal hRaster = _spatium / MScore::hRaster();
188  int n = lrint(x / hRaster);
189  x = hRaster * n;
190  }
191  if (ed.vRaster) {
192  qreal vRaster = _spatium / MScore::vRaster();
193  int n = lrint(y / vRaster);
194  y = vRaster * n;
195  }
196 
197  setOffset(QPointF(x, y));
198 
199  r |= canvasBoundingRect();
200  foreach(const Element* e, _leafs)
201  r |= e->canvasBoundingRect();
202  return r;
203  }
