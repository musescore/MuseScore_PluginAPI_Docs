404 Fraction TupletFractionCap(int tupletCount, bool tuplettrp, bool tupletprol)
405  {
406  int dd = 0;
407  int nn = 0;
408  qreal exponent = 0;
409  qreal count = tupletCount;
410  Fraction f(3,2);
411 
412  if ((count > 0) && (count <= 15)) {
413  if (tuplettrp)
414  exponent = qFloor(qLn(count/3.0)/qLn(2.0));
415  else
416  exponent = qFloor(qLn(count)/qLn(2.0));
417  }
418  else {
419  qDebug("Unknown tuplet, count = %d",tupletCount);
420  return f;
421  }
422  if (tupletprol)
423  exponent += 1.0;
424  if (exponent < 0.0)
425  exponent = 0.0;
426  nn = tupletCount;
427  dd = static_cast<int>(qPow(2.0, exponent));
428  if (tuplettrp)
429  dd = dd * 3;
430  qDebug("Tuplet Fraction: %d / %d", nn, dd);
431  return Fraction(nn, dd);
432  }

439 static bool findChordRests(BasicDrawObj const* const o, Score* score, const int track, const Fraction& tick,
440  ChordRest*& cr1, ChordRest*& cr2, NoteObj* no, QList<NoteObj*> objects)
441  {
442  cr1 = 0; // ChordRest where BasicDrawObj o begins
443  cr2 = 0; // ChordRest where BasicDrawObj o ends
444 
445  // find the ChordRests where o begins and ends
446  int n = o->nNotes + 1; // # notes in BasicDrawObj (nNotes is # notes following the first note)
447  int graceNumber = 0;
448  int graceNumber1 = 0;
449  bool foundcr1 = false;
450  Fraction tick2 = tick;
451  foreach(NoteObj* nobj, objects) {
452  BasicDurationalObj* d = 0;
453  if (nobj->type() == CapellaNoteObjectType::REST) {
454  d = static_cast<BasicDurationalObj*>(static_cast<RestObj*>(nobj));
455  graceNumber = 0;
456  }
457  else if (nobj->type() == CapellaNoteObjectType::CHORD) {
458  ChordObj* cho = static_cast<ChordObj*>(nobj);
459  d = static_cast<BasicDurationalObj*>(cho);
460  if (!(cho->invisible) && (cho->ticks().isZero())) // grace note
461  ++graceNumber;
462  else
463  graceNumber = 0;
464  }
465  if (!d)
466  continue;
467  if (nobj == no) {
468  foundcr1 = true;
469  graceNumber1 = graceNumber;
470  }
471  Fraction ticks = Fraction(0,1);
472  if (foundcr1) {
473  --n; // found the object corresponding to cr1, count down to find the second one
474  ticks = d->ticks();
475  if (d->count) {
476  Fraction f = TupletFractionCap(d->count, d->tripartite, d->isProlonging);
477  ticks = ticks / f;
478  }
479  if (nobj->type() == CapellaNoteObjectType::REST) {
480  RestObj* ro = static_cast<RestObj*>(nobj);
481  if (ro->fullMeasures) {
482  Measure* m = score->getCreateMeasure(tick2);
483  Fraction ft = m->ticks();
484  ticks = ft * ro->fullMeasures;
485  }
486  }
487  if (n == 0)
488  break;
489  tick2 += ticks;
490  }
491  }
492  // Now we have the tick (tick) and the level of grace note (graceNumber1, if "no" is a grace note) for the first ChordRest
493  // and the tick (tick2) and the level of grace note (graceNumber, if the target is a grace note) for the 2nd ChordRest
494  for (Segment* seg = score->tick2segment(tick); seg; seg = seg->next1()) {
495  if (seg->segmentType() != SegmentType::ChordRest)
496  continue;
497  ChordRest* cr = static_cast<ChordRest*>(seg->element(track));
498  if (cr) {
499  if (graceNumber1 > 0) { // the spanner is starting from a grace note
500  Chord* chord = static_cast<Chord*>(cr);
501  foreach(Chord* cc, chord->graceNotes()) {
502  --graceNumber1;
503  if ((graceNumber1 == 0) && (!cr1))
504  cr1 = static_cast<ChordRest*>(cc); // found first ChordRest
505  }
506  }
507  if (!cr1) cr1 = cr; // found first ChordRest
508  break;
509  }
510  }
511  for (Segment* seg = score->tick2segment(tick2); seg; seg = seg->next1()) {
512  if (seg->segmentType() != SegmentType::ChordRest)
513  continue;
514  ChordRest* cr = static_cast<ChordRest*>(seg->element(track));
515  if (cr) {
516  if ((graceNumber > 0) && (cr->type() == ElementType::CHORD)) { // the spanner is ending on a grace note
517  Chord* chord = static_cast<Chord*>(cr);
518  foreach(Chord* cc, chord->graceNotes()) {
519  --graceNumber;
520  if ((graceNumber == 0) && (!cr2))
521  cr2 = static_cast<ChordRest*>(cc); // found 2nd ChordRest
522  }
523  }
524  if (!cr2) cr2 = cr; // found 2nd ChordRest
525  break;
526  }
527  }
528  qDebug("findChordRests o %p nNotes %d score %p track %d tick %d cr1 %p cr2 %p",
529  o, o->nNotes, score, track, tick.ticks(), cr1, cr2);
530 
531  if (!(cr1 && cr2)) {
532  qDebug("first or second anchor for BasicDrawObj not found (tick %d type %d track %d first %p second %p)",
533  tick.ticks(), int(o->type), track, cr1, cr2);
534  return false;
535  }
536  return true;
537  }

604  }
605  }

609  ticks = ft * o->fullMeasures;

623  Segment* s = m->getSegment(SegmentType::ChordRest, tick);

627  d1.setType(TDuration::DurationType::V_MEASURE);
628  rest->setTicks(m->ticks());
629  }

631  d1.setVal(ticks.ticks());
632  rest->setTicks(d1.fraction());
633  }

635  rest->setTrack(track);
636  rest->setVisible(!o->invisible);
637  s->add(rest);
638  if (tuplet)
639  tuplet->add(rest);

1131 // needPart -- determine if a staff needs its own part
1132 //---------------------------------------------------------
1133 
1134 // As Capella does not define parts (it only knows about staves,
1135 // MIDI instruments numbers, brackets and braces), the following
1136 // algorithm is used:
1137 // - every staff is a separate part
1138 // - unless:
1139 // - it is in a brace
1140 // - it is not the first staff in the brace
1141 // - it has the same MIDI instrument as the previous staff
1142 // Common cases:
1143 // - Keyboards: two or three staves with the same MIDI instrument and a brace
1144 // -> create one part
1145 // - SATB: two or four staves with the same MIDI instrument and a bracket
1146 // -> create two or four parts
1147 
1148 static bool needPart(const int prevInst, const int currInst, const int staffIdx, QList<CapBracket> const& bracketList)
1149  {
1150  foreach(CapBracket cb, bracketList) {
1151  if (prevInst == currInst && cb.from < staffIdx && staffIdx <= cb.to && cb.curly)
1152  return false;
1153  }
1154  return true;
1155  }
