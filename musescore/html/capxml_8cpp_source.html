55  e.readNext();
56  return f;

63 static bool qstring2timestep(QString& str, TIMESTEP& tstp)
64  {
65  if (str == "1/1") { tstp = TIMESTEP::D1; return true; }
66  else if (str == "1/2") { tstp = TIMESTEP::D2; return true; }
67  else if (str == "1/4") { tstp = TIMESTEP::D4; return true; }
68  else if (str == "1/8") { tstp = TIMESTEP::D8; return true; }
69  else if (str == "1/16") { tstp = TIMESTEP::D16; return true; }
70  else if (str == "1/32") { tstp = TIMESTEP::D32; return true; }
71  else if (str == "1/64") { tstp = TIMESTEP::D64; return true; }
72  else if (str == "1/128") { tstp = TIMESTEP::D128; return true; }
73  else if (str == "2/1") { tstp = TIMESTEP::D_BREVE; return true; }
74  return false;
75  }

104  {

106  noDuration = false;
107  postGrace = false;
108  bSmall = false;
109  invisible = false;
110  notBlack = false;
111  color = Qt::black;
112  t = TIMESTEP::D1;
113  horizontalShift = 0;
114  count = 0;
115  tripartite = 0;
116  isProlonging = 0;
117  QString base = e.attribute("base");
118  bool res = false;
119  // try to convert base to timestep ("first pattern")
120  res = qstring2timestep(base, t);
121  if (!res) {
122  // try multi-measure rest ("second pattern")
123  int i = base.toInt();
124  if (i > 0)

126  else

128  }
129  nDots = e.intAttribute("dots", 0);
130  noDuration = e.attribute("noDuration", "false") == "true";
131  while (e.readNextStartElement()) {
132  const QStringRef& tag(e.name());
133  if (tag == "tuplet") {
134  count = e.attribute("count").toInt();
135  tripartite = e.attribute("tripartite", "false") == "true";
136  isProlonging = e.attribute("prolong", "false") == "true";
137  e.readNext();
138  }
139  else
140  e.unknown();
141  }
142  qDebug("DurationObj ndots %d nodur %d postgr %d bsm %d inv %d notbl %d t %d hsh %d cnt %d trp %d ispro %d fullm %d",
143  nDots, noDuration, postGrace, bSmall, invisible, notBlack, int(t), horizontalShift, count, tripartite, isProlonging, fullm
144  );

226 // qstring2timesig -- convert string to timesig
227 // return true on success
228 //---------------------------------------------------------
229 
230 static void qstring2timesig(QString& time, uchar& numerator, int& log2Denom, bool& allaBreve)
231  {
232  bool res = true;
233  numerator = 4; log2Denom = 2; allaBreve = false; // set default
234  if (time == "allaBreve") { numerator = 2; log2Denom = 1; allaBreve = true; }
235  else if (time == "longAllaBreve") { numerator = 4; log2Denom = 1; allaBreve = true; }
236  else if (time == "C") { numerator = 4; log2Denom = 2; allaBreve = false; }
237  else if (time == "infinite") { qDebug("Meter infinite"); } // not supported by MuseScore ?
238  else {
239  QStringList splitTime = time.split("/");
240  if (splitTime.size() == 2) {
241  numerator = splitTime.at(0).toInt();
242  QString denom = splitTime.at(1);
243  if (denom == "1") log2Denom = 0;
244  else if (denom == "2") log2Denom = 1;
245  else if (denom == "4") log2Denom = 2;
246  else if (denom == "8") log2Denom = 3;
247  else if (denom == "16") log2Denom = 4;
248  else if (denom == "32") log2Denom = 5;
249  else if (denom == "64") log2Denom = 6;
250  else if (denom == "128") log2Denom = 7;
251  else res = false;
252  }
253  else res = false;
254  }
255  // TODO: recovery required if decoding the timesig failed ?
256  qDebug("Meter '%s' res %d %d/%d allaBreve %d", qPrintable(time), res, numerator, log2Denom, allaBreve);
257  }

353 //---------------------------------------------------------
354 // pitchStr2Char -- convert pitch string ("[A-G][0-9]") to signed char
355 // notes:
356 // - in .capx middle C is called C5 (which is C4 in MusicXML)
357 // - middle C is MIDI note number 60
358 // - in .cap, pitch contains the diatonic note number relative to clef and key
359 // - as .capx contains absolute notes, store the MIDI note number instead
360 //---------------------------------------------------------
361 
362 static signed char pitchStr2Char(QString& strPitch)
363  {
364  QRegExp pitchRegExp("[A-G][0-9]");
365 
366  if (!pitchRegExp.exactMatch(strPitch)) {
367  qDebug("pitchStr2Char: illegal pitch '%s'", qPrintable(strPitch));
368  return 0;
369  }
370 
371  QString steps("C.D.EF.G.A.B");
372  int istep = steps.indexOf(strPitch.left(1));
373  int octave = strPitch.right(1).toInt();
374  int pitch = istep + 12 * octave;
375 
376  if (pitch < 0)
377  pitch = -1;
378  if (pitch > 127)
379  pitch = -1;
380 
381  qDebug("pitchStr2Char: '%s' -> %d", qPrintable(strPitch), pitch);
382 
383  return static_cast<signed char>(pitch);
384  }
