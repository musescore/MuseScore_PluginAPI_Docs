180 void ChordView::drawBackground(QPainter* p, const QRectF& r)
181  {
182  if (chord == 0)
183  return;
184  QRectF r1(-1000000.0, 0.0, 1000000.0+CHORD_MAP_OFFSET, 1000000.0);
185  QRectF r2(ticks + CHORD_MAP_OFFSET, 0.0, 1000000.0, 1000000.0);
186  QRectF r3(-1000000.0, 127*keyHeight, 1000000.0+CHORD_MAP_OFFSET, keyHeight);
187  QRectF r4(ticks + CHORD_MAP_OFFSET, 127*keyHeight, 1000000.0, keyHeight);
188 
189  QColor bg(0x71, 0x8d, 0xbe);
190  QColor bg1 = bg.darker(150);
191  QColor bg2 = bg.lighter(150);
192  QColor bg3 = bg.darker(75);
193 
194  p->fillRect(r, bg);
195  p->fillRect(r.intersected(r1), bg1);
196  p->fillRect(r.intersected(r2), bg1);
197 
198  foreach (const Note* n, chord->notes()) {
199  p->fillRect(QRect(CHORD_MAP_OFFSET, (127 - n->pitch()) * keyHeight,
200  1000, keyHeight), bg2);
201  }
202 
203  p->fillRect(r.intersected(r3), bg3);
204  p->fillRect(r.intersected(r4), bg3);
205 
206  //---------------------------------------------------
207  // draw horizontal grid lines
208  //---------------------------------------------------
209 
210  qreal y1 = r.y();
211  qreal y2 = y1 + r.height();
212  qreal x1 = r.x();
213  qreal x2 = x1 + r.width();
214 
215  int key = floor(y1 / keyHeight);
216  qreal y = key * keyHeight;
217 
218  for (; key < 127; ++key, y += keyHeight) {
219  if (y < y1)
220  continue;
221  if (y > y2)
222  break;
223  p->setPen(QPen((key % 6) == 5 ? Qt::lightGray : Qt::gray));
224  p->drawLine(QLineF(x1, y, x2, y));
225  }
226 
227  //---------------------------------------------------
228  // draw raster
229  //---------------------------------------------------
230 
231  if (_evenGrid) {
232  for (int x = 0; x <= 1000; x += 50) {
233  if (x % 250)
234  p->setPen(Qt::lightGray);
235  else
236  p->setPen(Qt::black);
237  p->drawLine(pos2pix(x), y1, pos2pix(x), y2);
238  }
239  }
240  else {
241  double step1 = 1000.0 / 3;
242  double step2 = step1 / 4;
243  for (int i = 0; i < 3; ++i) {
244  p->setPen(Qt::lightGray);
245  for (int k = 1; k < 4; ++k) {
246  int x = lrint(i * step1 + k * step2);
247  p->drawLine(pos2pix(x), y1, pos2pix(x), y2);
248  }
249  p->setPen(Qt::black);
250  int x = lrint(i * step1);
251  p->drawLine(pos2pix(x), y1, pos2pix(x), y2);
252  }
253  }
254  }

260 void ChordView::setChord(Chord* c)
261  {
262  chord = c;
263  _dirty = false;
264  _pos = 0;
265  _locator = 0;
266 
267  scene()->blockSignals(true);
268  scene()->clear();
269  locatorLine = new QGraphicsLineItem(QLineF(0.0, 0.0, 0.0, keyHeight * 127.0 * 5));
270  QPen pen(Qt::red);
271  pen.setWidth(2);
272  locatorLine->setPen(pen);
273  locatorLine->setZValue(1000); // set stacking order
274  locatorLine->setFlag(QGraphicsItem::ItemIgnoresTransformations, true);
275  scene()->addItem(locatorLine);
276 
277  curEvent = 0;
278  _curNote = 0;
279  foreach(Note* note, c->notes()) {
280  if (note->selected() && _curNote == 0)
281  _curNote = note;
282  int n = note->playEvents().size();
283  for (int i = 0; i < n; ++i) {
284  NoteEvent* e = &note->playEvents()[i];
285  ChordItem* item = new ChordItem(this, note, e);
286  if (_curNote == note && curEvent == 0)
287  curEvent = item;
288  scene()->addItem(item);
289  }
290  }
291  lg = new GripItem(0, this);
292  rg = new GripItem(1, this);
293  lg->setZValue(100);
294  rg->setZValue(101);
295  lg->setVisible(false);
296  rg->setVisible(false);
297  scene()->addItem(lg);
298  scene()->addItem(rg);
299 
300  scene()->blockSignals(false);
301  setCurItem(curEvent);
302  // selectionChanged();
303 
304  scene()->setSceneRect(0.0, 0.0, double(ticks + CHORD_MAP_OFFSET * 2), keyHeight * 127);
305 
306  moveLocator();
307 
308  //
309  // move to something interesting
310  //
311  QList<QGraphicsItem*> items = scene()->selectedItems();
312  QRectF boundingRect;
313  foreach(QGraphicsItem* item, items) {
314  Note* note = static_cast<Note*>(item->data(0).value<void*>());
315  if (note)
316  boundingRect |= item->mapToScene(item->boundingRect()).boundingRect();
317  }
318  centerOn(boundingRect.center());
319  scene()->update();
320  }

341 void ChordView::wheelEvent(QWheelEvent* event)
342  {
343  int step = event->angleDelta().y() / 120;
344  double xmag = transform().m11();
345  double ymag = transform().m22();
346 
347  if (event->modifiers() == Qt::ControlModifier) {
348  if (step > 0) {
349  for (int i = 0; i < step; ++i) {
350  if (xmag > 10.0)
351  break;
352  scale(1.1, 1.0);
353  xmag *= 1.1;
354  }
355  }
356  else {
357  for (int i = 0; i < -step; ++i) {
358  if (xmag < 0.001)
359  break;
360  scale(.9, 1.0);
361  xmag *= .9;
362  }
363  }
364  emit magChanged(xmag, ymag);
365 
366  int tpix = 1000 * xmag;
367  magStep = -5;
368  if (tpix <= 4000)
369  magStep = -4;
370  if (tpix <= 2000)
371  magStep = -3;
372  if (tpix <= 1000)
373  magStep = -2;
374  if (tpix <= 500)
375  magStep = -1;
376  if (tpix <= 128)
377  magStep = 0;
378  if (tpix <= 64)
379  magStep = 1;
380  if (tpix <= 32)
381  magStep = 2;
382  if (tpix <= 16)
383  magStep = 3;
384  if (tpix <= 8)
385  magStep = 4;
386  if (tpix <= 4)
387  magStep = 5;
388  if (tpix <= 2)
389  magStep = 6;
390 
391  //
392  // if xpos <= 0, then the scene is centered
393  // there is no scroll bar anymore sending
394  // change signals, so we have to do it here:
395  //
396  double xpos = -(mapFromScene(QPointF()).x());
397  if (xpos <= 0)
398  emit xposChanged(xpos);
399  }
400  else if (event->modifiers() == Qt::ShiftModifier) {
401  QWheelEvent we(event->pos(), event->delta(), event->buttons(), 0, Qt::Horizontal);
402  QGraphicsView::wheelEvent(&we);
403  }
404  else if (event->modifiers() == 0) {
405  QGraphicsView::wheelEvent(event);
406  }
407  else if (event->modifiers() == (Qt::ShiftModifier | Qt::ControlModifier)) {
408  if (step > 0) {
409  for (int i = 0; i < step; ++i) {
410  if (ymag > 3.0)
411  break;
412  scale(1.0, 1.1);
413  ymag *= 1.1;
414  }
415  }
416  else {
417  for (int i = 0; i < -step; ++i) {
418  if (ymag < 0.4)
419  break;
420  scale(1.0, .9);
421  ymag *= .9;
422  }
423  }
424  emit magChanged(xmag, ymag);
425  }
426  }

450 void ChordView::mousePressEvent(QMouseEvent* event)
451  {
452  if (event->modifiers() & Qt::ControlModifier) {
453  QPointF p(mapToScene(event->pos()));
454  int pitch = y2pitch(int(p.y()));
455  int tick = int(p.x()) - CHORD_MAP_OFFSET;
456  int tcks = 1000 - tick;
457  foreach(const NoteEvent& e, _curNote->playEvents()) {
458  if (e.pitch() != pitch)
459  continue;
460  if (tick >= e.ontime() && tick < e.offtime()) {
461  return; // cannot place an event here
462  }
463  int nticks = e.ontime() - tick;
464  if (nticks > 0)
465  tcks = qMin(tcks, nticks);
466  }
467 
468  NoteEvent ne;
469  ne.setPitch(pitch);
470  ne.setOntime(tick);
471  ne.setLen(tcks);
472  _curNote->playEvents().append(ne);
473  NoteEvent* pne = &_curNote->playEvents()[_curNote->playEvents().size()-1];
474  ChordItem* item = new ChordItem(this, _curNote, pne);
475  scene()->addItem(item);
476  setCurItem(item);
477  _dirty = true;
478  }
479  QGraphicsView::mousePressEvent(event);
480  }
