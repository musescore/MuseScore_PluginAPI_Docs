112 void Clef::layout()
113  {
114  // determine current number of lines and line distance
115  int lines;
116  qreal lineDist;
117  Segment* clefSeg = segment();
118  int stepOffset;
119 
120  // check clef visibility and type compatibility
121  if (clefSeg && staff()) {
122  Fraction tick = clefSeg->tick();
123  const StaffType* st = staff()->staffType(tick);
124  bool show = st->genClef(); // check staff type allows clef display
125  StaffGroup staffGroup = st->group();
126 
127  // if not tab, use instrument->useDrumset to set staffGroup (to allow pitched to unpitched in same staff)
128  if ( staffGroup != StaffGroup::TAB)
129  staffGroup = staff()->part()->instrument(this->tick())->useDrumset() ? StaffGroup::PERCUSSION : StaffGroup::STANDARD;
130 
131  // check clef is compatible with staff type group:
132  if (ClefInfo::staffGroup(clefType()) != staffGroup) {
133  if (tick > Fraction(0,1) && !generated()) // if clef is not generated, hide it
134  show = false;
135  else // if generated, replace with initial clef type
136  // TODO : instead of initial staff clef (which is assumed to be compatible)
137  // use the last compatible clef previously found in staff
138  _clefTypes = staff()->clefType(Fraction(0,1));
139  }
140 
141  // if clef not to show or not compatible with staff group
142  if (!show) {
143  setbbox(QRectF());
144  symId = SymId::noSym;
145  qDebug("Clef::layout(): invisible clef at tick %d(%d) staff %d",
146  segment()->tick().ticks(), segment()->tick().ticks()/1920, staffIdx());
147  return;
148  }
149  lines = st->lines(); // init values from staff type
150  lineDist = st->lineDistance().val();
151  stepOffset = st->stepOffset();
152  }
153  else {
154  lines = 5;
155  lineDist = 1.0;
156  stepOffset = 0;
157  }
158 
159  qreal _spatium = spatium();
160  qreal yoff = 0.0;
161  if (clefType() != ClefType::INVALID && clefType() != ClefType::MAX) {
162  symId = ClefInfo::symId(clefType());
163  yoff = lineDist * (5 - ClefInfo::line(clefType()));
164  }
165  else
166  symId = SymId::noSym;
167 
168  switch (clefType()) {
169  case ClefType::C_19C: // 19th C clef is like a G clef
170  yoff = lineDist * 1.5;
171  break;
172  case ClefType::TAB: // TAB clef
173  // on tablature, position clef at half the number of spaces * line distance
174  yoff = lineDist * (lines - 1) * .5;
175  stepOffset = 0; // ignore stepOffset for TAB and pecussion clefs
176  break;
177  case ClefType::TAB4: // TAB clef 4 strings
178  // on tablature, position clef at half the number of spaces * line distance
179  yoff = lineDist * (lines - 1) * .5;
180  stepOffset = 0;
181  break;
182  case ClefType::TAB_SERIF: // TAB clef alternate style
183  // on tablature, position clef at half the number of spaces * line distance
184  yoff = lineDist * (lines - 1) * .5;
185  stepOffset = 0;
186  break;
187  case ClefType::TAB4_SERIF: // TAB clef alternate style
188  // on tablature, position clef at half the number of spaces * line distance
189  yoff = lineDist * (lines - 1) * .5;
190  stepOffset = 0;
191  break;
192  case ClefType::PERC: // percussion clefs
193  yoff = lineDist * (lines - 1) * 0.5;
194  stepOffset = 0;
195  break;
196  case ClefType::PERC2:
197  yoff = lineDist * (lines - 1) * 0.5;
198  stepOffset = 0;
199  break;
200  case ClefType::INVALID:
201  case ClefType::MAX:
202  qDebug("Clef::layout: invalid type");
203  return;
204  default:
205  break;
206  }
207  // clefs on palette or at start of system/measure are left aligned
208  // other clefs are right aligned
209  QRectF r(symBbox(symId));
210  qreal x = segment() && segment()->rtick().isNotZero() ? -r.right() : 0.0;
211  setPos(x, yoff * _spatium + (stepOffset * 0.5 * _spatium));
212 
213  setbbox(r);
214  }

242 Element* Clef::drop(EditData& data)
243  {
244  Element* e = data.dropElement;
245  Clef* c = 0;
246  if (e->isClef()) {
247  Clef* clef = toClef(e);
248  ClefType stype = clef->clefType();
249  if (clefType() != stype) {
250  score()->undoChangeClef(staff(), this, stype);
251  c = this;
252  }
253  }
254  else if (e->isAmbitus()) {
255  /*if (!generated())*/ {
256  Measure* meas = measure();
257  Segment* segm = meas->getSegment(SegmentType::Ambitus, meas->tick());
258  if (segm->element(track()))
259  score()->undoRemoveElement(segm->element(track()));
260  Ambitus* r = new Ambitus(score());
261  r->setParent(segm);
262  r->setTrack(track());
263  score()->undoAddElement(r);
264  }
265  }
266  delete e;
267  return c;
268  }

421 //---------------------------------------------------------
422 // otherClef
423 // try to locate the 'other clef' of a courtesy / main pair
424 //---------------------------------------------------------
425 
426 Clef* Clef::otherClef()
427  {
428  // if not in a clef-segment-measure hierarchy, do nothing
429  if (!parent() || !parent()->isSegment())
430  return nullptr;
431  Segment* segm = toSegment(parent());
432  if (!segm->parent() || !segm->parent()->isMeasure())
433  return 0;
434  Measure* meas = toMeasure(segm->parent());
435  Measure* otherMeas = nullptr;
436  Segment* otherSegm = nullptr;
437  Fraction segmTick = segm->tick();
438  SegmentType type = SegmentType::Clef;
439  if (segmTick == meas->tick() && segm->segmentType() == SegmentType::HeaderClef) // if clef segm is measure-initial
440  otherMeas = meas->prevMeasure(); // look for a previous measure
441  else if (segmTick == meas->tick() + meas->ticks()) { // if clef segm is measure-final
442  otherMeas = meas->nextMeasure(); // look for a next measure
443  type = SegmentType::HeaderClef;
444  }
445  if (!otherMeas)
446  return nullptr;
447  // look for a clef segment in the 'other' measure at the same tick of this clef segment
448  otherSegm = otherMeas->findSegment(type, segmTick);
449  if (!otherSegm)
450  return nullptr;
451  // if any 'other' segment found, look for a clef in the same track as this
452  return toClef(otherSegm->element(track()));
453  }
