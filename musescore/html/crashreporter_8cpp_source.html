73 void CrashReporter::uploadReport()
74  {
75  if (!_networkManager) {
76  _networkManager = new QNetworkAccessManager(this);
77  connect(_networkManager, &QNetworkAccessManager::finished, this, &CrashReporter::onUploadFinished);
78  }
79 
80  QHttpMultiPart* multiPart = new QHttpMultiPart(QHttpMultiPart::FormDataType);
81 
82  QHttpPart miniDumpPart;
83  miniDumpPart.setHeader(QNetworkRequest::ContentTypeHeader, QVariant("application/octet-stream"));
84  const QString filename = QFileInfo(_miniDump).fileName();
85  const QString contentDisposition(QString("form-data; name=\"upload_file_minidump\"; filename=\"%1\"").arg(filename));
86  miniDumpPart.setHeader(QNetworkRequest::ContentDispositionHeader, contentDisposition);
87  QFile* file = new QFile(_miniDump, multiPart);
88  file->open(QIODevice::ReadOnly);
89  miniDumpPart.setBodyDevice(file);
90  multiPart->append(miniDumpPart);
91 
92  QHttpPart releasePart;
93  releasePart.setHeader(QNetworkRequest::ContentDispositionHeader, QVariant("form-data; name=\"sentry[release]\""));
94  releasePart.setBody(version().toLatin1());
95  multiPart->append(releasePart);
96 
97  const QString comment(_ui->commentText->toPlainText());
98  const bool hasComment = !comment.isEmpty();
99 
100  QHttpPart hasCommentPart;
101  hasCommentPart.setHeader(QNetworkRequest::ContentDispositionHeader, QVariant("form-data; name=\"sentry[tags][hasComment]\""));
102  hasCommentPart.setBody(QString::number(hasComment).toLatin1());
103  multiPart->append(hasCommentPart);
104 
105  QHttpPart commentPart;
106  commentPart.setHeader(QNetworkRequest::ContentDispositionHeader, QVariant("form-data; name=\"comments\""));
107  commentPart.setBody(comment.toUtf8());
108  multiPart->append(commentPart);
109 
110  QNetworkRequest request(_uploadUrl);
111  // TODO: custom user-agent?
112 
113  QNetworkReply* reply = _networkManager->post(request, multiPart);
114  multiPart->setParent(reply);
115  }

121 void CrashReporter::onUploadFinished(QNetworkReply* reply)
122  {
