61 void Ladspa_CS_chorus1::runproc (unsigned long len, bool add)
62 {
63  unsigned long i, k, wi;
64  int j;
65  float *p0, *p1;
66  float t, x, y;
67  p0 = _port [INPUT];
68  p1 = _port [OUTPUT];
69 
70  wi = _wi;
71  do
72  {
73  if (_gi == 0)
74  {
75  _gi = 64;
76 
77  t = 402.12f * _port [FREQ1][0] / _fsam;
78  x = _x1 - t * _y1;
79  y = _y1 + t * _x1;
80  t = sqrtf (x * x + y * y);
81  _x1 = x / t;
82  _y1 = y / t;
83 
84  t = 402.12f * _port [FREQ2][0] / _fsam;
85  x = _x2 - t * _y2;
86  y = _y2 + t * _x2;
87  t = sqrtf (x * x + y * y);
88  _x2 = x / t;
89  _y2 = y / t;
90 
91  x = _port [TMOD1][0] * _x1 + _port [TMOD2][0] * _x2;
92  y = _port [TMOD1][0] * _y1 + _port [TMOD2][0] * _y2;
93 
94  _dr [0] = x;
95  _dr [1] = -0.500f * x + 0.866f * y;
96  _dr [2] = -0.500f * x - 0.866f * y;
97 
98  for (j = 0; j < 3; j++)
99  {
100  t = _port [DELAY][0] + _dr [j];
101  if (t < 0) t = 0;
102  if (t > 30) t = 30;
103  t *= _fsam / 1000.0f;
104  _dr [j] = (t - _ri [j]) / 64;
105  }
106  }
107 
108  k = (_gi < len) ? _gi : len;
109  _gi -= k;
110  len -= k;
111 
112  while (k--)
113  {
114  _line [++wi] = *p0++;
115  y = 0;
116  for (j = 0; j < 3; j++)
117  {
118  x = wi - _ri [j];
119  _ri [j] += _dr [j];
120  if (x < 0) x += _size;
121  i = (int)(floorf (x));
122  x -= i;
123  y += (1 - x) * _line [i] + x * _line [i + 1];
124  }
125  y *= 0.333f;
126  if (add) *p1++ += y * _gain;
127  else *p1++ = y;
128 
129  }
130  if (wi == _size) _line [wi = 0] = _line [_size];
131  }
132  while (len);
133 
134  _wi = wi;
135 }
