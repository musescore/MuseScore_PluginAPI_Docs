105 // rewind
106 /// Rewind cursor to a certain position.
107 /// \param mode Determines the position where to move
108 /// this cursor. See Cursor::RewindMode to see the list of
109 /// available rewind modes.
110 /// \note In MuseScore 2.X, this function took an integer
111 /// value (0, 1 or 2) as its parameter. For compatibility
112 /// reasons, the old values are still working, but it is
113 /// recommended to use RewindMode enumerators instead.
114 //---------------------------------------------------------
115 
116 void Cursor::rewind(RewindMode mode)
117  {
118  //
119  // rewind to start of score
120  //
121  if (mode == SCORE_START) {
122  Ms::Measure* m = _score->firstMeasure();
123  if (m) {
124  setSegment(m->first(_filter));
125  nextInTrack();
126  }
127  else {
128  setSegment(nullptr);
129  }
130  }
131  //
132  // rewind to start of selection
133  //
134  else if (mode == SELECTION_START) {
135  if (!_score->selection().isRange())
136  return;
137  setSegment(_score->selection().startSegment());
138  setTrack(_score->selection().staffStart() * VOICES);
139  nextInTrack();
140  }
141  //
142  // rewind to end of selection
143  //
144  else if (mode == SELECTION_END) {
145  if (!_score->selection().isRange())
146  return;
147  setSegment(_score->selection().endSegment());
148  setTrack((_score->selection().staffEnd() * VOICES) - 1); // be sure _track exists
149  }
150  }

154 /// Rewind cursor to a position defined by tick.
155 /// \param tick Determines the position where to move
156 /// this cursor.

158 /// \since MuseScore 3.5

160 
161 void Cursor::rewindToTick(int tick)
162  {
163  // integer ticks may contain numeric errors so it is
164  // better to search not precisely if possible
165  Ms::Fraction fTick = Ms::Fraction::fromTicks(tick + 1);
166  Ms::Segment* seg = _score->tick2leftSegment(fTick);
167  if (!(seg->segmentType() & _filter)) {
168  // we need another segment type, search by known tick
169  seg = _score->tick2segment(seg->tick(), /* first */ true, _filter);
170  }
171 
172  setSegment(seg);
173  nextInTrack();
174  }

232 /// Adds the given element to a score at this cursor's
233 /// position.
234 //---------------------------------------------------------
235 
236 void Cursor::add(Element* wrapped)
237  {
238  Ms::Element* s = wrapped ? wrapped->element() : nullptr;
239  if (!segment() || !s)
240  return;
241 
242  // Ensure that the object has the expected ownership
243  if (wrapped->ownership() == Ownership::SCORE) {
244  qWarning("Cursor::add: Cannot add this element. The element is already part of the score.");
245  return; // Don't allow operation.
246  }
247 
248  const int _track = track();
249  Ms::Segment* _segment = segment();
250 
251  wrapped->setOwnership(Ownership::SCORE);
252  s->setScore(_score);
253  s->setTrack(_track);
254  s->setParent(_segment);
255 
256  if (s->isChordRest())
257  s->score()->undoAddCR(toChordRest(s), _segment->measure(), _segment->tick());
258  else if (s->type() == ElementType::KEYSIG) {
259  Ms::Segment* ns = _segment->measure()->undoGetSegment(SegmentType::KeySig, _segment->tick());
260  s->setParent(ns);
261  _score->undoAddElement(s);
262  }
263  else if (s->type() == ElementType::TIMESIG) {
264  Ms::Measure* m = _segment->measure();
265  Fraction tick = m->tick();
266  _score->cmdAddTimeSig(m, _track, toTimeSig(s), false);
267  m = _score->tick2measure(tick);
268  _segment = m->first(_filter);
269  nextInTrack();
270  }
271  else {
272  switch (s->type()) {
273  // To be added at measure level
274  case ElementType::MEASURE_NUMBER:
275  case ElementType::MMREST_RANGE:
276  case ElementType::SPACER:
277  case ElementType::JUMP:
278  case ElementType::MARKER:
279  case ElementType::HBOX:
280  case ElementType::STAFFTYPE_CHANGE:
281  case ElementType::LAYOUT_BREAK: {
282  Ms::Measure* m = _segment->measure();
283  s->setParent(m);
284  _score->undoAddElement(s);
285  break;
286  }
287 
288  // To be added at chord level
289  case ElementType::NOTE:
290  case ElementType::ARPEGGIO:
291  case ElementType::TREMOLO:
292  case ElementType::CHORDLINE:
293  case ElementType::ARTICULATION: {
294  Ms::Element* curElement = currentElement();
295  if (curElement->isChord()) {
296  // call Chord::addInternal() (i.e. do the same as a call to Chord.add())
297  Chord::addInternal(toChord(curElement), s);
298  }
299  break;
300  break;
301  }
302 
303  // To be added at chord/rest level
304  case ElementType::LYRICS: {
305  Ms::Element* curElement = currentElement();
306  if (curElement->isChordRest()) {
307  s->setParent(curElement);
308  _score->undoAddElement(s);
309  }
310  break;
311  }
312 
313  // To be added to a note
314  case ElementType::SYMBOL:
315  case ElementType::FINGERING:
316  case ElementType::BEND:
317  case ElementType::NOTEHEAD: {
318  Ms::Element* curElement = currentElement();
319  if (curElement->isChord()) {
320  Ms::Chord* chord = toChord(curElement);
321  Ms::Note* note = nullptr;
322  if (chord->notes().size() > 0) {
323  // Get first note from chord to add element
324  note = chord->notes().front();
325  }
326  if (note) {
327  Note::addInternal(note, s);
328  }
329  }
330  break;
331  }
332 
333  // To be added to a segment (clef subtype)
334  case ElementType::CLEF:
335  case ElementType::AMBITUS: {
336  Ms::Element* parent = nullptr;
337  // Find backwards first measure containing a clef
338  for (Ms::Measure* m = _segment->measure(); m != 0; m = m->prevMeasure()) {
339  Ms::Segment* seg = m->findSegment(SegmentType::Clef | SegmentType::HeaderClef, m->tick());
340  if (seg != 0) {
341  parent = m->undoGetSegmentR(s->isAmbitus() ? SegmentType::Ambitus : seg->segmentType(), Fraction(0,1));
342  break;
343  }
344  }
345  if (parent && parent->isSegment()) {
346  if (s->isClef()) {
347  Ms::Clef* clef = toClef(s);
348  if (clef->clefType() == Ms::ClefType::INVALID) {
349  clef->setClefType(Ms::ClefType::G);
350  }
351  }
352  s->setParent(parent);
353  s->setTrack(_track);
354  _score->undoAddElement(s);
355  }
356  break;
357  }
358 
359  default: // All others will be added to the current segment
360  _score->undoAddElement(s);
361  break;
362  }
363  }
364  }

412 //---------------------------------------------------------
413 // addTuplet
414 /// \brief Adds a tuplet to the current cursor position.
415 /// \details This function provides a possibility to setup
416 /// the tuplet's ratio to any value (similarly to
417 /// Add > Tuplets > Other... dialog in MuseScore).
418 ///
419 /// Examples of most typical usage:
420 /// \code
421 /// // add a triplet of three eighth notes
422 /// cursor.addTuplet(fraction(3, 2), fraction(1, 4));
423 ///
424 /// // add a quintuplet in place of the current chord/rest
425 /// var cr = cursor.element;
426 /// if (cr)
427 /// cursor.addTuplet(fraction(5, 4), cr.duration);
428 /// \endcode
429 ///
430 /// \param ratio tuplet ratio. Numerator represents
431 /// actual number of notes in this tuplet, denominator is
432 /// a number of "normal" notes which correspond to the
433 /// same total duration. For example, a triplet has a
434 /// ratio of 3/2 as it has 3 notes fitting to the
435 /// duration which would normally be occupied by 2 notes
436 /// of the same nominal length.
437 /// \param duration total duration of the tuplet. To
438 /// create a tuplet with duration matching to duration of
439 /// existing chord or rest, use its
440 /// \ref DurationElement.duration "duration" value as
441 /// a parameter.
442 /// \since MuseScore 3.5
443 /// \see \ref DurationElement.tuplet
444 //---------------------------------------------------------
445 
446 void Cursor::addTuplet(FractionWrapper* ratio, FractionWrapper* duration)
447  {
448  if (!segment()) {
449  qWarning("Cursor::addTuplet: cursor location is undefined, use rewind() to define its location");
450  return;
451  }
452 
453  const Ms::Fraction fRatio = ratio->fraction();
454  const Ms::Fraction fDuration = duration->fraction();
455 
456  if (!fRatio.isValid() || fRatio.isZero() || fRatio.negative()
457  || !fDuration.isValid() || fDuration.isZero() || fDuration.negative()) {
458  qWarning("Cursor::addTuplet: invalid parameter values: %s, %s", qPrintable(fRatio.toString()), qPrintable(fDuration.toString()));
459  return;
460  }
461 
462  Ms::Measure* tupletMeasure = segment()->measure();
463  const Ms::Fraction tupletTick = segment()->tick();
464 
465  if (tupletTick + fDuration > tupletMeasure->endTick()) {
466  qWarning(
467  "Cursor::addTuplet: cannot add cross-measure tuplet (measure %d, rel.tick %s, duration %s)",
468  tupletMeasure->no() + 1, qPrintable(segment()->rtick().toString()), qPrintable(fDuration.toString()));
469  return;
470  }
471 
472  const Ms::Fraction baseLen = fDuration * Fraction(1, fRatio.denominator());
473  if (!TDuration::isValid(baseLen)) {
474  qWarning("Cursor::addTuplet: cannot create tuplet for ratio %s and duration %s", qPrintable(fRatio.toString()), qPrintable(fDuration.toString()));
475  return;
476  }
477 
478  _score->expandVoice(inputState().segment(), inputState().track());
479  Ms::ChordRest* cr = inputState().cr();
480  if (!cr) // shouldn't happen?
481  return;
482 
483  _score->changeCRlen(cr, fDuration);
484 
485  Ms::Tuplet* tuplet = new Ms::Tuplet(_score);
486  tuplet->setParent(tupletMeasure);
487  tuplet->setTrack(track());
488  tuplet->setTick(tupletTick);
489  tuplet->setRatio(fRatio);
490  tuplet->setTicks(fDuration);
491  tuplet->setBaseLen(baseLen);
492 
493  _score->cmdCreateTuplet(cr, tuplet);
494 
495  inputState().setSegment(tupletMeasure->tick2segment(tupletTick));
496  inputState().setDuration(baseLen);
497  }
