302 static void addChord(ElementItem* sei, Chord* chord)
303  {
304  if (chord->hook())
305  new ElementItem(sei, chord->hook());
306  if (chord->stem())
307  new ElementItem(sei, chord->stem());
308  if (chord->stemSlash())
309  new ElementItem(sei, chord->stemSlash());
310  if (chord->arpeggio())
311  new ElementItem(sei, chord->arpeggio());
312  if (chord->tremolo() && (chord->tremolo()->chord1() == chord))
313  new ElementItem(sei, chord->tremolo());
314 // if (chord->glissando())
315 // new ElementItem(sei, chord->glissando());
316 
317  for (Articulation* a : chord->articulations())
318  new ElementItem(sei, a);
319  for (LedgerLine* h = chord->ledgerLines(); h; h = h->next())
320  new ElementItem(sei, h);
321  for (Note* note : chord->notes()) {
322  ElementItem* ni = new ElementItem(sei, note);
323  if (note->accidental()) {
324  new ElementItem(ni, note->accidental());
325  }
326  for (Element* f : note->el()) {
327  if (f->type() == ElementType::SYMBOL || f->type() == ElementType::IMAGE) {
328  BSymbol* bs = static_cast<BSymbol*>(f);
329  addSymbol(ni, bs);
330  }
331  else
332  new ElementItem(ni, f);
333  }
334  for (NoteDot* dot : note->dots())
335  new ElementItem(ni, dot);
336 
337  if (note->tieFor()) {
338  Tie* tie = note->tieFor();
339  ElementItem* ti = new ElementItem(ni, tie);
340  for (Element* el1 : tie->spannerSegments())
341  new ElementItem(ti, el1);
342  }
343  for (Spanner* s : note->spannerFor()) {
344  ElementItem* si = new ElementItem(ni, s);
345  for (Element* ls : s->spannerSegments())
346  new ElementItem(si, ls);
347  }
348  }
349  for (Element* e : chord->el()) {
350  ElementItem* ei = new ElementItem(sei, e);
351  if (e->type() == ElementType::SLUR) {
352  Slur* gs = static_cast<Slur*>(e);
353  for (SpannerSegment* sp : gs->spannerSegments())
354  new ElementItem(ei, sp);
355  }
356  }
357  for (Chord* c : chord->graceNotes()) {
358  ElementItem* ssei = new ElementItem(sei, c);
359  addChord(ssei, c);
360  }
361 
362  if (chord->beam() && chord->beam()->elements().front() == chord)
363  new ElementItem(sei, chord->beam());
364  for (Lyrics* lyrics : chord->lyrics()) {
365  if (lyrics)
366  new ElementItem(sei, lyrics);
367  }
368  DurationElement* de = chord;
369  while (de->tuplet() && de->tuplet()->elements().front() == de) {
370  new ElementItem(sei, de->tuplet());
371  de = de->tuplet();
372  }
373  }

379 void Debugger::addMeasure(ElementItem* mi, Measure* measure)
380  {
381  int staves = cs->nstaves();
382  int tracks = staves * VOICES;
383  for (int staffIdx = 0; staffIdx < measure->score()->nstaves(); ++staffIdx) {
384  if (measure->vspacerUp(staffIdx))
385  new ElementItem(mi, measure->vspacerUp(staffIdx));
386  if (measure->vspacerDown(staffIdx))
387  new ElementItem(mi, measure->vspacerDown(staffIdx));
388  if (measure->noText(staffIdx))
389  new ElementItem(mi, measure->noText(staffIdx));
390  }
391  for (Segment* segment = measure->first(); segment; segment = segment->next()) {
392  ElementItem* segItem = new ElementItem(mi, segment);
393  for (int track = 0; track < tracks; ++track) {
394  Element* e = segment->element(track);
395  if (!e)
396  continue;
397  ElementItem* sei = new ElementItem(segItem, e);
398  if (e->isChord())
399  addChord(sei, toChord(e));
400  else if (e->isRest()) {
401  Rest* rest = toRest(e);
402  for (int i = 0; i < rest->dots(); ++i)
403  new ElementItem(sei, rest->dot(i));
404  ChordRest* cr = toChordRest(e);
405  if (cr->beam() && cr->beam()->elements().front() == cr)
406  new ElementItem(sei, cr->beam());
407  for (Lyrics* lyrics : cr->lyrics()) {
408  if (lyrics)
409  new ElementItem(sei, lyrics);
410  }
411  DurationElement* de = cr;
412  while (de->tuplet() && de->tuplet()->elements().front() == de) {
413  new ElementItem(sei, de->tuplet());
414  de = de->tuplet();
415  }
416  }
417  }
418 
419  foreach(Element* s, segment->annotations()) {
420  if (s->type() == ElementType::SYMBOL || s->type() == ElementType::IMAGE)
421  addBSymbol(segItem, static_cast<BSymbol*>(s));
422  else if (s->type() == ElementType::FRET_DIAGRAM) {
423  ElementItem* fdi = new ElementItem(segItem, s);
424  FretDiagram* fd = static_cast<FretDiagram*>(s);
425  if (fd->harmony())
426  new ElementItem(fdi, fd->harmony());
427  }
428  else
429  new ElementItem(segItem, s);
430  }
431  }
432  }

438 void Debugger::updateList(Score* s)
439  {
440  if (cs != s) {
441  backStack.clear();
442  forwardStack.clear();
443  back->setEnabled(false);
444  forward->setEnabled(false);
445  cs = s;
446  }
447  curElement = 0;
448  list->clear();
449  if (!isVisible())
450  return;
451 
452  if (s->masterScore()->movements()) {
453  QTreeWidgetItem* mi = new QTreeWidgetItem(list, int(ElementType::INVALID));
454  mi->setText(0, "Movements");
455  for (MasterScore* ms : *s->masterScore()->movements()) {
456  QTreeWidgetItem* i = new QTreeWidgetItem(mi, int(ElementType::INVALID));
457  i->setText(0, ms->metaTag("movementTitle"));
458  }
459  }
460 
461 
462  QTreeWidgetItem* li = new QTreeWidgetItem(list, int(ElementType::INVALID));
463  li->setText(0, "Global");
464  for (auto i : s->spanner()) {
465  ElementItem* it = new ElementItem(li, i.second);
466  if (i.second->type() == ElementType::TRILL) {
467  Trill* trill = static_cast<Trill*>(i.second);
468  if (trill->accidental())
469  new ElementItem(it, trill->accidental());
470  }
471  }
472 
473  foreach (Page* pg, cs->pages()) {
474  ElementItem* pi = new ElementItem(list, pg);
475 
476  for (System* system : pg->systems()) {
477  ElementItem* si = new ElementItem(pi, system);
478  for (Bracket* b : system->brackets())
479  new ElementItem(si, b);
480  if (system->systemDividerLeft())
481  new ElementItem(si, system->systemDividerLeft());
482  if (system->systemDividerRight())
483  new ElementItem(si, system->systemDividerRight());
484  for (SpannerSegment* ss : system->spannerSegments())
485  new ElementItem(si, ss);
486  for (SysStaff* ss : *system->staves()) {
487  for (InstrumentName* in : ss->instrumentNames)
488  new ElementItem(si, in);
489  }
490 
491  for (MeasureBase* mb : system->measures()) {
492  ElementItem* mi = new ElementItem(si, mb);
493  addMeasureBaseToList(mi, mb);
494 
495  if (mb->type() != ElementType::MEASURE)
496  continue;
497  Measure* measure = (Measure*) mb;
498  if (cs->styleB(Sid::concertPitch)) {
499  if (measure->mmRest()) {
500  ElementItem* mmi = new ElementItem(mi, measure->mmRest());
501  addMeasure(mmi, measure->mmRest());
502  }
503  }
504  else {
505  if (measure->isMMRest()) {
506  Measure* m1 = measure->mmRestFirst();
507  Measure* m2 = measure->mmRestLast();
508  for (;;) {
509  ElementItem* mmi = new ElementItem(mi, m1);
510  addMeasure(mmi, m1);
511  if (m1 == m2)
512  break;
513  m1 = m1->nextMeasure();
514  }
515  }
516  }
517  addMeasure(mi, measure);
518  }
519  }
520  }
521  }
