51 void Division::process()
52  {
53  memset (_buff, 0, NCHANN * PERIOD * sizeof (float));
54  for (int i = 0; i < _nrank; i++)
55  _ranks [i]->play (1);
56 
57  float g = _swel;
58  if (_trem) {
59  _s += _w * _c;
60  _c -= _w * _s;
61  float t = sqrtf (_c * _c + _s * _s);
62  _c /= t;
63  _s /= t;
64  if ((_trem == 2) && (fabsf (_s) < 0.05f)) {
65  _trem = 0;
66  _c = 1;
67  _s = 0;
68  }
69  g *= 1.0f + _m * _s;
70  }
71 
72  float t = 1.05f * _gain;
73  if (g > t)
74  g = t;
75  t = 0.95f * _gain;
76  if (g < t)
77  g = t;
78 
79  float d = (g - _gain) / PERIOD;
80  g = _gain;
81  float* p = _buff;
82  float* q = _asect->get_wptr ();
83 
84  for (int i = 0; i < PERIOD; i++) {
85  g += d;
86  q [0 * PERIOD * MIXLEN] += p [0 * PERIOD] * g;
87  q [1 * PERIOD * MIXLEN] += p [1 * PERIOD] * g;
88  q [2 * PERIOD * MIXLEN] += p [2 * PERIOD] * g;
89  q [3 * PERIOD * MIXLEN] += p [3 * PERIOD] * g;
90  p++;
91  q++;
92  }
93  _gain = g;
94  }
