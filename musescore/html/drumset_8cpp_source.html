27 void Drumset::save(XmlWriter& xml) const
28  {
29  for (int i = 0; i < 128; ++i) {
30  if (!isValid(i))
31  continue;
32  xml.stag(QString("Drum pitch=\"%1\"").arg(i));
33  const NoteHead::Group nh = noteHead(i);
34  //write custom as Normal notehead group + noteheads tag to keep compatibility with 2.X versions
35  const NoteHead::Group saveNHValue = (nh == NoteHead::Group::HEAD_CUSTOM) ? NoteHead::Group::HEAD_NORMAL : nh;
36  xml.tag("head", NoteHead::group2name(saveNHValue));
37  if (nh == NoteHead::Group::HEAD_CUSTOM) {
38  xml.stag("noteheads");
39  for (int j = 0; j < int(NoteHead::Type::HEAD_TYPES); j++) {
40  xml.tag(NoteHead::type2name(NoteHead::Type(j)), Sym::id2name(noteHeads(i, NoteHead::Type(j))));
41  }
42  xml.etag();
43  }
44  xml.tag("line", line(i));
45  xml.tag("voice", voice(i));
46  xml.tag("name", name(i));
47  xml.tag("stem", int(stemDirection(i)));
48  if (shortcut(i)) {
49  switch (shortcut(i)) {
50  case 'C':
51  case 'D':
52  case 'E':
53  case 'F':
54  case 'G':
55  case 'A':
56  case 'B':
57  {
58  char a[2];
59  a[0] = shortcut(i);
60  a[1] = 0;
61  xml.tag("shortcut", a);
62  }
63  break;
64  default:
65  qDebug("illegal drum shortcut");
66  break;
67  }
68  }
69  auto vs = variants(i);
70  if (!vs.isEmpty()) {
71  xml.stag("variants");
72  for (const auto &v : qAsConst(vs)) {
73  xml.stag(QString("variant pitch=\"%1\"").arg(v.pitch));
74  if (!v.articulationName.isEmpty())
75  xml.tag("articulation", v.articulationName);
76  if (v.tremolo != TremoloType::INVALID_TREMOLO)
77  xml.tag("tremolo", Tremolo::type2name(v.tremolo));
78  xml.etag();
79  }
80  xml.etag();
81  }
82  xml.etag();
83  }
84  }

86 bool Drumset::readProperties(XmlReader& e, int pitch)
87  {
88  if (pitch < 0 || pitch > DRUM_INSTRUMENTS - 1)
89  return false;
90 
91  const QStringRef& tag(e.name());
92  if (tag == "head")
93  _drum[pitch].notehead = NoteHead::name2group(e.readElementText());
94  else if (tag == "noteheads") {
95  _drum[pitch].notehead = NoteHead::Group::HEAD_CUSTOM;
96  while (e.readNextStartElement()) {
97  const QStringRef& nhTag(e.name());
98  int noteType = int(NoteHead::name2type(nhTag.toString()));
99  if (noteType > int(NoteHead::Type::HEAD_TYPES) - 1 || noteType < 0)
100  return false;
101 
102  _drum[pitch].noteheads[noteType] = Sym::name2id(e.readElementText());
103  }
104  }
105  else if (tag == "line")
106  _drum[pitch].line = e.readInt();
107  else if (tag == "voice")
108  _drum[pitch].voice = e.readInt();
109  else if (tag == "name")
110  _drum[pitch].name = e.readElementText();
111  else if (tag == "stem")
112  _drum[pitch].stemDirection = Direction(e.readInt());
113  else if (tag == "shortcut") {
114  bool isNum;
115  QString val(e.readElementText());
116  int i = val.toInt(&isNum);
117  _drum[pitch].shortcut = isNum ? i : toupper(val[0].toLatin1());
118  }
119  else if (tag == "variants") {
120  while(e.readNextStartElement()) {
121  const QStringRef& tagv(e.name());
122  if (tagv == "variant") {
123  DrumInstrumentVariant div;
124  div.pitch = e.attribute("pitch").toInt();
125  while (e.readNextStartElement()) {
126  const QStringRef& taga(e.name());
127  if (taga == "articulation") {
128  div.articulationName = e.readElementText();
129  }
130  else if (taga == "tremolo") {
131  div.tremolo = Tremolo::name2Type(e.readElementText());
132  }
133  }
134  _drum[pitch].addVariant(div);
135  }
136  }
137  }
138  else
139  return false;
140  return true;
141  }

147 void Drumset::load(XmlReader& e)
148  {
149  int pitch = e.intAttribute("pitch", -1);
150  if (pitch < 0 || pitch > 127) {
151  qDebug("load drumset: invalid pitch %d", pitch);
152  return;
153  }
154  while (e.readNextStartElement()) {
155  if (readProperties(e, pitch))
156  ;
157  else
158  e.unknown();
159  }
160  }

215 DrumInstrumentVariant Drumset::findVariant(int p, const QVector<Articulation*> articulations, Tremolo* tremolo) const
216  {
217  DrumInstrumentVariant div;
218  auto vs = variants(p);
219  for (const auto &v : qAsConst(vs)) {
220  bool matchTremolo = (!tremolo && v.tremolo == TremoloType::INVALID_TREMOLO) || (tremolo && v.tremolo == tremolo->tremoloType());
221  bool matchArticulation = v.articulationName.isEmpty() && articulations.isEmpty();
222  for (auto a : articulations) {
223  matchArticulation = a->articulationName() == v.articulationName;
224  if (!matchArticulation)
225  break;
226  }
227  if (matchArticulation && matchTremolo) {
228  div = v;
229  break;
230  }
231  }
232  return div;
233  }
