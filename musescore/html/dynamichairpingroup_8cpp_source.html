21 static std::pair<Hairpin*, Hairpin*> findAdjacentHairpins(Dynamic* d)
22  {
23  Score* score = d->score();
24  const Segment* dSeg = d->segment();
25  Hairpin* leftHairpin = nullptr;
26  Hairpin* rightHairpin = nullptr;
27 
28  const Fraction tick = dSeg->tick();
29  const int intTick = tick.ticks();
30 
31  const auto& nearSpanners = score->spannerMap().findOverlapping(intTick - 1, intTick + 1);
32  for (auto i : nearSpanners) {
33  Spanner* s = i.value;
34  if (s->track() == d->track() && s->isHairpin()) {
35  Hairpin* h = toHairpin(s);
36  if (h->tick() == tick)
37  rightHairpin = h;
38  else if (h->tick2() == tick)
39  leftHairpin = h;
40  }
41  }
42 
43  return { leftHairpin, rightHairpin };
44  }

46 std::unique_ptr<ElementGroup> HairpinWithDynamicsDragGroup::detectFor(HairpinSegment* hs, std::function<bool(const Element*)> isDragged)
47  {
48  if (!hs->isSingleType())

50

58  Dynamic* endDynamic = toDynamic(endSegment->findAnnotation(ElementType::DYNAMIC, track, track));

64  endDynamic = nullptr;

71 std::unique_ptr<ElementGroup> HairpinWithDynamicsDragGroup::detectFor(Dynamic* d, std::function<bool(const Element*)> isDragged)
72  {
73  Hairpin* leftHairpin = nullptr;
74  Hairpin* rightHairpin = nullptr;
75 
76  std::tie(leftHairpin, rightHairpin) = findAdjacentHairpins(d);
77 
78  // Dynamic will be governed bt HairpinWithDynamicsDragGroup if any of adjacent
79  // hairpins is dragged, disable separate drag logic for dynamic in this case.
80  if (isDragged(leftHairpin))
81  return std::unique_ptr<ElementGroup>(new DisabledElementGroup());
82  if (isDragged(rightHairpin))
83  return std::unique_ptr<ElementGroup>(new DisabledElementGroup());
84 
85  return nullptr;

97 QRectF HairpinWithDynamicsDragGroup::drag(EditData& ed)
98  {
99  QRectF r;
100 
101  if (startDynamic)
102  r |= static_cast<Element*>(startDynamic)->drag(ed);
103  r |= hairpinSegment->drag(ed);
104  if (endDynamic)
105  r |= static_cast<Element*>(endDynamic)->drag(ed);
106 
107  Hairpin* h = hairpinSegment->hairpin();
108 
109  const Fraction startTick = startDynamic ? startDynamic->segment()->tick() : h->tick();
110  const Fraction endTick = endDynamic ? endDynamic->segment()->tick() : h->tick2();
111 
112  if (endTick > startTick) {
113  if (h->tick() != startTick)
114  h->undoChangeProperty(Pid::SPANNER_TICK, startTick);
115  if (h->tick2() != endTick)
116  h->undoChangeProperty(Pid::SPANNER_TICKS, endTick - startTick);
117  }
118 
119  return r;
120  }

161 QRectF DynamicNearHairpinsDragGroup::drag(EditData& ed)
162  {
163  QRectF r(static_cast<Element*>(dynamic)->drag(ed));
164 
165  const Fraction tick = dynamic->segment()->tick();
166 
167  if (leftHairpin && leftHairpin->tick2() != tick && tick > leftHairpin->tick())
168  leftHairpin->undoChangeProperty(Pid::SPANNER_TICKS, tick - leftHairpin->tick());
169 
170  if (rightHairpin && rightHairpin->tick() != tick) {
171  const Fraction tick2 = rightHairpin->tick2();
172  if (tick < tick2) {
173  rightHairpin->undoChangeProperty(Pid::SPANNER_TICK, tick);
174  rightHairpin->undoChangeProperty(Pid::SPANNER_TICKS, tick2 - tick);
175  }
176  }
177 
178  if (leftHairpin || rightHairpin)
179  dynamic->triggerLayout();
180 
181  return r;
182  }
