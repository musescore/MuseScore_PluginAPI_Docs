24 // The following function is inspired by "A Primer on BÃ©zier Curve" sections 17 and 23 by Pomax:
25 // https://pomax.github.io/bezierinfo/
26 // However, the fuction is greatly specialized, simplified and optimized for use as an ease-in and
27 // ease-out transfer curve for bends, glissandi and portamenti in MuseScore. The function computes
28 // Y from X by performing a cubic root finding to compute t from X on the X component of the
29 // transfer curve and then a cubic Bezier evaluation to compute Y from t on the Y component of the
30 // transfer curve. The code could be simplified because the Bezier curve is constrained as a well
31 // defined transfer function. It is not suitable for general or arbitrary Bezier curve work. -YP
32 //-------------------------------------------------------------------------------------------------
33 qreal EaseInOut::tFromX(const qreal x) const {
34  const qreal pi = 3.14159265358979323846;
35  qreal c = -x;
36  qreal w1 = _easeIn - x;
37  qreal w2 = 1.0 - _easeOut - x;
38  qreal d = x + 3.0 * w1 - 3.0 * w2 + (1.0 - x);
39  qreal a = (-3.0 * x - 6.0 * w1 + 3 * w2) / d;
40  qreal b = (3.0 * x + 3.0 * w1) / d;
41  c = -x / d;
42  qreal a2 = a * a;
43  qreal p = (3.0 * b - a2) / 3.0;
44  qreal q = (2.0 * a2 * a - 9.0 * a * b + 27.0 * c) / 27.0;
45  qreal discr = (q * q) / 4.0 + (p * p * p) / 27.0;
46  qreal t = 0.0;
47  // Crazy idea to first test the least probable case with such an expensive test but...
48  if (qFuzzyCompare(discr, 0.0)) {
49  // Case that happens extremely rarely --> 2 roots.
50  qreal q2 = q / 2.0;
51  qreal u = q2 < 0.0 ? std::pow(-q2, 1.0 / 3.0) : -std::pow(q2, 1.0 / 3.0);
52  // Find the only root that is within the 0 to 1 interval.
53  t = 2.0 * u - a / 3.0;
54  if (0.0 > t || t > 1.0) {
55  t = -u - a / 3.0;
56  }
57  }
58  else if (discr < 0.0) {
59  // Case that happens about 75% of the time --> 3 roots.
60  qreal mp3 = -p / 3.0;
61  qreal r = std::sqrt(mp3 * mp3 * mp3);
62  qreal phi = std::acos(std::min(std::max(-1.0, -q / (2.0 * r)), 1.0));
63  qreal t1 = 2.0 * std::pow(r, 1.0 / 3.0);
64  // Find the only root that is within the 0 to 1 interval.
65  t = t1 * std::cos(phi / 3.0) - a / 3.0;
66  if (0.0 > t || t > 1.0) {
67  t = t1 * std::cos((phi + 2.0 * pi) / 3.0) - a / 3.0;
68  if (0.0 > t || t > 1.0) {
69  t = t1 * std::cos((phi + 4.0 * pi) / 3.0) - a / 3.0;
70  }
71  }
72  }
73  else if (discr > 0.0) {
74  // Case that happens about 25% of the time --> 1 root.
75  qreal q2 = q / 2.0;
76  qreal sd = std::sqrt(discr);
77  qreal u = std::pow(-q2 + sd, 1.0 / 3.0);
78  qreal v = std::pow(q2 + sd, 1.0 / 3.0);
79  t = u - v - a / 3.0;
80  }
81  return t;
82  }
