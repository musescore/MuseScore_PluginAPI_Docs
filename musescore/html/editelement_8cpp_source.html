39 void ScoreView::updateGrips()
40  {
41  if (!editData.element)
42  return;
43 
44  double dx = 1.5 / _matrix.m11();
45  double dy = 1.5 / _matrix.m22();
46 
47  for (const QRectF& r : editData.grip)
48  score()->addRefresh(r.adjusted(-dx, -dy, dx, dy));
49 
50  qreal w = 8.0 / _matrix.m11();
51  qreal h = 8.0 / _matrix.m22();
52  QRectF r(-w*.5, -h*.5, w, h);
53 
54  editData.grip.resize(editData.grips);
55 
56  if (editData.grips) {
57  for (QRectF& gr : editData.grip)
58  gr = r;
59  editData.element->updateGrips(editData);
60 
61  // updateGrips returns grips in page coordinates,
62  // transform to view coordinates:
63  Element* const page = editData.element->findAncestor(ElementType::PAGE);
64  const QPointF pageOffset((page ? page : editData.element)->pos());
65 
66  for (QRectF& grip : editData.grip) {
67  grip.translate(pageOffset);
68  score()->addRefresh(grip.adjusted(-dx, -dy, dx, dy));
69  }
70 
71  const Grip anchorLinesGrip = editData.curGrip == Grip::NO_GRIP ? editData.element->defaultGrip() : editData.curGrip;
72  QVector<QLineF> anchorLines = editData.element->gripAnchorLines(anchorLinesGrip);
73 
74  if (!anchorLines.isEmpty()) {
75  setDropAnchorLines(anchorLines);
76  }
77  else
78  setDropTarget(0); // this also resets dropAnchor
79  }
80  score()->addRefresh(editData.element->canvasBoundingRect());
81  }

87 void ScoreView::startEditMode(Element* e)
88  {
89  if (score()->selection().elements().size() != 1)
90  score()->select(e);
91  if (!e || !e->isEditable()) {
92  qDebug("The element cannot be edited");
93  return;
94  }
95  if (textEditMode()) {
96  // leave text edit mode before starting editing another element
97  changeState(ViewState::NORMAL);
98  }
99  if (score()->undoStack()->active())
100  score()->endCmd();
101  // Restart edit mode to reinit edit values
102  if (editMode())
103  changeState(ViewState::NORMAL);
104  editData.element = e;
105  changeState(ViewState::EDIT);
106  }

112 void ScoreView::startEdit(Element* element, Grip startGrip)
113  {
114  if (!element || !element->isEditable()) {
115  qDebug("The element cannot be edited");
116  return;
117  }
118 
119  const bool forceStartEdit = (
120  (state == ViewState::EDIT || state == ViewState::DRAG_EDIT)
121  && element != editData.element
122  );
123  editData.element = element;
124  if (forceStartEdit) // call startEdit() forcibly to reinitialize edit mode.
125  startEdit();
126  else if (state != ViewState::DRAG_EDIT)
127  changeState(ViewState::EDIT);
128 
129  if (startGrip != Grip::NO_GRIP)
130  editData.curGrip = startGrip;
131  }

163 void ScoreView::endEdit()
164  {
165  setAttribute(Qt::WA_InputMethodEnabled, false);
166  setDropTarget(0);
167  if (!editData.element)
168  return;
169  _score->addRefresh(editData.element->canvasBoundingRect());
170  for (int i = 0; i < editData.grips; ++i)
171  score()->addRefresh(editData.grip[i]);
172  editData.element->endEdit(editData);
173  //! NOTE After endEdit, the element may be null
174  if (editData.element) {
175  _score->addRefresh(editData.element->canvasBoundingRect());
176  ElementType tp = editData.element->type();
177  if (tp == ElementType::LYRICS)
178  lyricsEndEdit();
179  }
180  editData.clearData();
181  mscore->updateInspector();
182  }
