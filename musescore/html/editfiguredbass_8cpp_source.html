26 // derived from harmonyTab() (for Harmony)
27 // manages [Space] / [Shift][Space] keys, moving editing to FB of next/prev ChordRest
28 // and [Tab] / [Shift][Tab] keys, moving to FB of next/prev measure

31 void ScoreView::figuredBassTab(bool bMeas, bool bBack)
32  {
33  FiguredBass* fb = (FiguredBass*)editData.element;
34  Segment* nextSegm;
35  Segment* segm = fb->segment();
36  int track = fb->track();
37 
38  if (!segm) {
39  qDebug("figuredBassTab: no segment");
40  return;
41  }
42 
43  // if moving to next/prev measure
44 
45  if (bMeas) {
46  Measure* meas = segm->measure();
47  if (meas) {
48  if (bBack)
49  meas = meas->prevMeasure();
50  else
51  meas = meas->nextMeasure();
52  }
53  if (!meas) {
54  qDebug("figuredBassTab: no prev/next measure");
55  return;
56  }
57  // find initial ChordRest segment
58  nextSegm = meas->findSegment(SegmentType::ChordRest, meas->tick());
59  if (!nextSegm) {
60  qDebug("figuredBassTab: no ChordRest segment at measure");
61  return;
62  }
63  }
64 
65  // if moving to next/prev chord segment
66 
67  else {
68  // search next chord segment in same staff
69  nextSegm = bBack ? segm->prev1(SegmentType::ChordRest) : segm->next1(SegmentType::ChordRest);
70  int minTrack = (track / VOICES ) * VOICES;
71  int maxTrack = minTrack + (VOICES-1);
72 
73  while (nextSegm) { // look for a ChordRest in the compatible track range
74  if(nextSegm->hasAnnotationOrElement(ElementType::FIGURED_BASS, minTrack, maxTrack))
75  break;
76  nextSegm = bBack ? nextSegm->prev1(SegmentType::ChordRest) : nextSegm->next1(SegmentType::ChordRest);
77  }
78 
79  if (!nextSegm) {
80  qDebug("figuredBassTab: no prev/next segment");
81  return;
82  }
83  }
84 
85  changeState(ViewState::NORMAL);
86 
87  bool bNew;
88  // add a (new) FB element, using chord duration as default duration
89  FiguredBass * fbNew = FiguredBass::addFiguredBassToSegment(nextSegm, track, Fraction(0,1), &bNew);
90  if (bNew) {
91  _score->startCmd();
92  _score->undoAddElement(fbNew);
93  _score->endCmd();
94  }
95  _score->select(fbNew, SelectType::SINGLE, 0);
96  startEdit(fbNew, Grip::NO_GRIP);
97 
98  mscore->changeState(mscoreState());
99  adjustCanvasPosition(fbNew, false);
100  fbNew->cursor(editData)->moveCursorToEnd();
101 // _score->update(); // used by lyricsTab() but not by harmonyTab(): needed or not?
102  }

109 void ScoreView::figuredBassTicksTab(const Fraction& ticks)
110  {
111  FiguredBass* fb = toFiguredBass(editData.element);
112  int track = fb->track();
113  Segment* segm = fb->segment();
114  if (!segm) {
115  qDebug("figuredBassTicksTab: no segment");
116  return;
117  }
118  Measure* measure = segm->measure();
119 
120  Fraction nextSegTick = segm->tick() + ticks;
121 
122  // find the measure containing the target tick
123  while (nextSegTick >= measure->tick() + measure->ticks()) {
124  measure = measure->nextMeasure();
125  if (!measure) {
126  qDebug("figuredBassTicksTab: no next measure");
127  return;
128  }
129  }
130 
131  changeState(ViewState::NORMAL);
132 
133  // look for a segment at this tick; if none, create one
134  Segment * nextSegm = segm;
135  while (nextSegm && nextSegm->tick() < nextSegTick)
136  nextSegm = nextSegm->next1(SegmentType::ChordRest);
137  if (!nextSegm || nextSegm->tick() > nextSegTick) { // no ChordRest segm at this tick
138  nextSegm = new Segment(measure, SegmentType::ChordRest, nextSegTick - measure->tick());
139  if (!nextSegm) {
140  qDebug("figuredBassTicksTab: no next segment");
141  return;
142  }
143  _score->startCmd();
144  _score->undoAddElement(nextSegm);
145  _score->endCmd();
146  }
147 
148  bool bNew;
149  FiguredBass * fbNew = FiguredBass::addFiguredBassToSegment(nextSegm, track, ticks, &bNew);
150  if (bNew) {
151  _score->startCmd();
152  _score->undoAddElement(fbNew);
153  _score->endCmd();
154  }
155  _score->select(fbNew, SelectType::SINGLE, 0);
156  startEdit(fbNew, Grip::NO_GRIP);
157  mscore->changeState(mscoreState());
158  adjustCanvasPosition(fbNew, false);
159 //TODO-edit fb->moveCursorToEnd();
160  }
