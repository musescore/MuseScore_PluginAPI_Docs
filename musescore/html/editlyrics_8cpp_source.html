29 bool ScoreView::editKeyLyrics()
30  {
31  Q_ASSERT(editData.element->isLyrics());
32 
33  const bool textEditing = true;
34  switch (editData.key) {
35  case Qt::Key_Space:
36  if (!editData.control(textEditing)) {
37  if (editData.s == "_")
38  lyricsUnderscore();
39  else // TODO: shift+tab events are filtered by qt
40  lyricsTab(editData.modifiers & Qt::ShiftModifier, true, false);
41  }
42  else
43  return false;
44  break;
45 
46  case Qt::Key_Left:
47  case Qt::Key_Right:
48  if (!editData.control(textEditing) && editData.element->edit(editData))
49  mscore->textTools()->updateTools(editData);
50  else {
51  bool kl = editData.key == Qt::Key_Left;
52  lyricsTab(kl, kl, true); // go to previous/next lyrics
53  }
54  if (editData.modifiers & Qt::ShiftModifier)
55  _score->update(); // draw updated selection
56  break;
57 
58  case Qt::Key_Up:
59  case Qt::Key_Down:
60  lyricsUpDown(editData.key == Qt::Key_Up, true);
61  break;
62 
63  case Qt::Key_Return:
64  lyricsReturn();
65  break;
66 
67  case Qt::Key_Minus:
68  if (editData.control(textEditing)) {
69  if ((editData.modifiers & Qt::AltModifier) && editData.element->edit(editData)) {
70  _score->update();
71  break;
72  }
73  // change into normal minus
74  editData.modifiers &= ~CONTROL_MODIFIER;
75  return false;
76  }
77  else
78  lyricsMinus();
79  break;
80 
81  case Qt::Key_Underscore:
82  if (editData.control(textEditing)) {
83  // change into normal underscore
84  editData.modifiers = {}; // &= ~CONTROL_MODIFIER;
85  return false;
86  }
87  else
88  lyricsUnderscore();
89  break;
90 
91  default:
92  return false;
93  }
94  return true;
95  }

101 void ScoreView::lyricsUpDown(bool up, bool end)
102  {
103  Lyrics* lyrics = toLyrics(editData.element);
104  int track = lyrics->track();
105  ChordRest* cr = lyrics->chordRest();
106  int verse = lyrics->no();
107  Placement placement = lyrics->placement();
108  PropertyFlags pFlags = lyrics->propertyFlags(Pid::PLACEMENT);
109 
110  if (up) {
111  if (verse == 0)
112  return;
113  --verse;
114  }
115  else {
116  ++verse;
117  if (verse > cr->lastVerse(placement))
118  return;
119  }
120 
121  changeState(ViewState::NORMAL);
122  lyrics = cr->lyrics(verse, placement);
123  if (!lyrics) {
124  lyrics = new Lyrics(_score);
125  lyrics->setTrack(track);
126  lyrics->setParent(cr);
127  lyrics->setNo(verse);
128  lyrics->setPlacement(placement);
129  lyrics->setPropertyFlags(Pid::PLACEMENT, pFlags);
130  _score->startCmd();
131  _score->undoAddElement(lyrics);
132  _score->endCmd();
133  }
134 
135  _score->select(lyrics, SelectType::SINGLE, 0);
136  startEdit(lyrics, Grip::NO_GRIP);
137  mscore->changeState(mscoreState());
138  adjustCanvasPosition(lyrics, false);
139 
140  lyrics = toLyrics(editData.element);
141  TextCursor* cursor = lyrics->cursor(editData);
142  if (end) {
143  cursor->movePosition(QTextCursor::Start, QTextCursor::MoveAnchor);
144  cursor->movePosition(QTextCursor::End, QTextCursor::KeepAnchor);
145  }
146  else {
147  cursor->movePosition(QTextCursor::End, QTextCursor::MoveAnchor);
148  cursor->movePosition(QTextCursor::Start, QTextCursor::KeepAnchor);
149  }
150 
151  _score->setLayoutAll();
152  _score->update();
153  }

159 void ScoreView::lyricsTab(bool back, bool end, bool moveOnly)
160  {
161  Lyrics* lyrics = toLyrics(editData.element);
162  int track = lyrics->track();
163  Segment* segment = lyrics->segment();
164  int verse = lyrics->no();
165  Placement placement = lyrics->placement();
166  PropertyFlags pFlags = lyrics->propertyFlags(Pid::PLACEMENT);
167 
168  Segment* nextSegment = segment;
169  if (back) {
170  // search prev chord
171  while ((nextSegment = nextSegment->prev1(SegmentType::ChordRest))) {
172  Element* el = nextSegment->element(track);
173  if (el && el->isChord())
174  break;
175  }
176  }
177  else {
178  // search next chord
179  while ((nextSegment = nextSegment->next1(SegmentType::ChordRest))) {
180  Element* el = nextSegment->element(track);
181  if (el && el->isChord())
182  break;
183  }
184  }
185  if (nextSegment == 0)
186  return;
187 
188  changeState(ViewState::NORMAL);
189 
190  // look for the lyrics we are moving from; may be the current lyrics or a previous one
191  // if we are skipping several chords with spaces
192  Lyrics* fromLyrics = 0;
193  if (!back) {
194  while (segment) {
195  ChordRest* cr = toChordRest(segment->element(track));
196  if (cr) {
197  fromLyrics = cr->lyrics(verse, placement);
198  if (fromLyrics)
199  break;
200  }
201  segment = segment->prev1(SegmentType::ChordRest);
202  }
203  }
204 
205  ChordRest* cr = toChordRest(nextSegment->element(track));
206  if (!cr) {
207  qDebug("no next lyrics list: %s", nextSegment->element(track)->name());
208  return;
209  }
210  Lyrics* _toLyrics = cr->lyrics(verse, placement);
211 
212  bool newLyrics = false;
213  if (!_toLyrics) {
214  _toLyrics = new Lyrics(_score);
215  _toLyrics->setTrack(track);
216  cr = toChordRest(nextSegment->element(track));
217  _toLyrics->setParent(cr);
218  _toLyrics->setNo(verse);
219  _toLyrics->setPlacement(placement);
220  _toLyrics->setPropertyFlags(Pid::PLACEMENT, pFlags);
221  _toLyrics->setSyllabic(Lyrics::Syllabic::SINGLE);
222  newLyrics = true;
223  }
224 
225  _score->startCmd();
226  if (fromLyrics && !moveOnly) {
227  switch (_toLyrics->syllabic()) {
228  // as we arrived at toLyrics by a [Space], it can be the beginning
229  // of a multi-syllable, but cannot have syllabic dashes before
230  case Lyrics::Syllabic::SINGLE:
231  case Lyrics::Syllabic::BEGIN:
232  break;
233  case Lyrics::Syllabic::END:
234  _toLyrics->undoChangeProperty(Pid::SYLLABIC, int(Lyrics::Syllabic::SINGLE));
235  break;
236  case Lyrics::Syllabic::MIDDLE:
237  _toLyrics->undoChangeProperty(Pid::SYLLABIC, int(Lyrics::Syllabic::BEGIN));
238  break;
239  }
240  // as we moved away from fromLyrics by a [Space], it can be
241  // the end of a multi-syllable, but cannot have syllabic dashes after
242  switch (fromLyrics->syllabic()) {
243  case Lyrics::Syllabic::SINGLE:
244  case Lyrics::Syllabic::END:
245  break;
246  case Lyrics::Syllabic::BEGIN:
247  fromLyrics->undoChangeProperty(Pid::SYLLABIC, int(Lyrics::Syllabic::SINGLE));
248  break;
249  case Lyrics::Syllabic::MIDDLE:
250  fromLyrics->undoChangeProperty(Pid::SYLLABIC, int(Lyrics::Syllabic::END));
251  break;
252  }
253  // for the same reason, it cannot have a melisma
254  fromLyrics->undoChangeProperty(Pid::LYRIC_TICKS, 0);
255  }
256 
257  if (newLyrics)
258  _score->undoAddElement(_toLyrics);
259  _score->endCmd();
260 
261  _score->select(_toLyrics, SelectType::SINGLE, 0);
262  startEdit(_toLyrics, Grip::NO_GRIP);
263 
264  adjustCanvasPosition(_toLyrics, false);
265 
266  TextCursor* cursor = toLyrics(editData.element)->cursor(editData);
267  if (end) {
268  cursor->movePosition(QTextCursor::Start, QTextCursor::MoveAnchor);
269  cursor->movePosition(QTextCursor::End, QTextCursor::KeepAnchor);
270  }
271  else {
272  cursor->movePosition(QTextCursor::End, QTextCursor::MoveAnchor);
273  cursor->movePosition(QTextCursor::Start, QTextCursor::KeepAnchor);
274  }
275  _score->setLayoutAll();
276  }

329  toLyrics->setSyllabic(Lyrics::Syllabic::END);

336  toLyrics->undoChangeProperty(Pid::SYLLABIC, int(Lyrics::Syllabic::END));
337  }
338

341  // it can have syll. dashes before and after but cannot be isolated or terminal
342  switch(fromLyrics->syllabic()) {
343  case Lyrics::Syllabic::BEGIN:
344  case Lyrics::Syllabic::MIDDLE:
345  break;
346  case Lyrics::Syllabic::SINGLE:
347  fromLyrics->undoChangeProperty(Pid::SYLLABIC, int(Lyrics::Syllabic::BEGIN));
348  break;
349  case Lyrics::Syllabic::END:

352  }

356 
357  if (newLyrics)
358  _score->undoAddElement(toLyrics);
359  _score->endCmd();
360 
361  _score->select(toLyrics, SelectType::SINGLE, 0);
362  startEdit(toLyrics, Grip::NO_GRIP);
363 
364  adjustCanvasPosition(toLyrics, false);
365  TextCursor* cursor = Ms::toLyrics(editData.element)->cursor(editData);
366  Ms::toLyrics(editData.element)->selectAll(cursor);
367  _score->setLayoutAll();

374 void ScoreView::lyricsUnderscore()
375  {
376  Lyrics* lyrics = toLyrics(editData.element);
377  int track = lyrics->track();
378  Segment* segment = lyrics->segment();
379  int verse = lyrics->no();
380  Placement placement = lyrics->placement();
381  PropertyFlags pFlags = lyrics->propertyFlags(Pid::PLACEMENT);
382  Fraction endTick = segment->tick(); // a previous melisma cannot extend beyond this point
383 
384  changeState(ViewState::NORMAL);
385 
386  // search next chord
387  Segment* nextSegment = segment;
388  while ((nextSegment = nextSegment->next1(SegmentType::ChordRest))) {
389  Element* el = nextSegment->element(track);
390  if (el && el->isChord())
391  break;
392  }
393 
394  // look for the lyrics we are moving from; may be the current lyrics or a previous one
395  // we are extending with several underscores
396  Lyrics* fromLyrics = 0;
397  while (segment) {
398  ChordRest* cr = toChordRest(segment->element(track));
399  if (cr) {
400  fromLyrics = cr->lyrics(verse, placement);
401  if (fromLyrics)
402  break;
403  }
404  segment = segment->prev1(SegmentType::ChordRest);
405  // if the segment has a rest in this track, stop going back
406  Element* e = segment ? segment->element(track) : 0;
407  if (e && !e->isChord())
408  break;
409  }
410 
411  // one-chord melisma?
412  // if still at melisma initial chord and there is a valid next chord (if not,
413  // there will be no melisma anyway), set a temporary melisma duration
414  if (fromLyrics == lyrics && nextSegment) {
415  _score->startCmd();
416  lyrics->undoChangeProperty(Pid::LYRIC_TICKS, Fraction::fromTicks(Lyrics::TEMP_MELISMA_TICKS));
417  _score->setLayoutAll();
418  _score->endCmd();
419  }
420 
421  if (nextSegment == 0) {
422  _score->startCmd();
423  if (fromLyrics) {
424  switch(fromLyrics->syllabic()) {
425  case Lyrics::Syllabic::SINGLE:
426  case Lyrics::Syllabic::END:
427  break;
428  default:
429  fromLyrics->undoChangeProperty(Pid::SYLLABIC, int(Lyrics::Syllabic::END));
430  break;
431  }
432  if (fromLyrics->segment()->tick() < endTick)
433  fromLyrics->undoChangeProperty(Pid::LYRIC_TICKS, endTick - fromLyrics->segment()->tick());
434  }
435  // leave edit mode, select something (just for user feedback) and update to show extended melisam
436  mscore->changeState(STATE_NORMAL);
437  if (fromLyrics)
438  _score->select(fromLyrics, SelectType::SINGLE, 0);
439  _score->setLayoutAll();
440  _score->endCmd();
441  return;
442  }
443 
444  // if a place for a new lyrics has been found, create a lyrics there
445 
446  _score->startCmd();
447  ChordRest* cr = toChordRest(nextSegment->element(track));
448  Lyrics* toLyrics = cr->lyrics(verse, placement);
449  bool newLyrics = (toLyrics == 0);
450  if (!toLyrics) {
451  toLyrics = new Lyrics(_score);
452  toLyrics->setTrack(track);
453  toLyrics->setParent(nextSegment->element(track));
454  toLyrics->setNo(verse);
455  toLyrics->setPlacement(placement);
456  toLyrics->setPropertyFlags(Pid::PLACEMENT, pFlags);
457  toLyrics->setSyllabic(Lyrics::Syllabic::SINGLE);
458  }
459  // as we arrived at toLyrics by an underscore, it cannot have syllabic dashes before
460  else if (toLyrics->syllabic() == Lyrics::Syllabic::MIDDLE)
461  toLyrics->undoChangeProperty(Pid::SYLLABIC, int(Lyrics::Syllabic::BEGIN));
462  else if (toLyrics->syllabic() == Lyrics::Syllabic::END)
463  toLyrics->undoChangeProperty(Pid::SYLLABIC, int(Lyrics::Syllabic::SINGLE));
464 
465  if (fromLyrics) {
466  // as we moved away from fromLyrics by an underscore,
467  // it can be isolated or terminal but cannot have dashes after
468  switch(fromLyrics->syllabic()) {
469  case Lyrics::Syllabic::SINGLE:
470  case Lyrics::Syllabic::END:
471  break;
472  default:
473  fromLyrics->undoChangeProperty(Pid::SYLLABIC, int(Lyrics::Syllabic::END));
474  break;
475  }
476  // for the same reason, if it has a melisma, this cannot extend beyond toLyrics
477  if (fromLyrics->segment()->tick() < endTick)
478  fromLyrics->undoChangeProperty(Pid::LYRIC_TICKS, endTick - fromLyrics->segment()->tick());
479  }
480  if (newLyrics)
481  _score->undoAddElement(toLyrics);
482  _score->endCmd();
483 
484  _score->select(toLyrics, SelectType::SINGLE, 0);
485  startEdit(toLyrics, Grip::NO_GRIP);
486 
487  adjustCanvasPosition(toLyrics, false);

489  Ms::toLyrics(editData.element)->selectAll(cursor);
490  }
