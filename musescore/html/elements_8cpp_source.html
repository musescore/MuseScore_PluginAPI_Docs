287 // wrap
288 /// \cond PLUGIN_API \private \endcond
289 /// Wraps Ms::Element choosing the correct wrapper type
290 /// at runtime based on the actual element type.
291 //---------------------------------------------------------
292 
293 Element* wrap(Ms::Element* e, Ownership own)
294  {
295  if (!e)
296  return nullptr;
297 
298  using Ms::ElementType;
299  switch(e->type()) {
300  case ElementType::NOTE:
301  return wrap<Note>(toNote(e), own);
302  case ElementType::CHORD:
303  return wrap<Chord>(toChord(e), own);
304  case ElementType::TUPLET:
305  return wrap<Tuplet>(toTuplet(e), own);
306  case ElementType::SEGMENT:
307  return wrap<Segment>(toSegment(e), own);
308  case ElementType::MEASURE:
309  return wrap<Measure>(toMeasure(e), own);
310  case ElementType::PAGE:
311  return wrap<Page>(toPage(e), own);
312  default:
313  if (e->isDurationElement()) {
314  if (e->isChordRest())
315  return wrap<ChordRest>(toChordRest(e), own);
316  return wrap<DurationElement>(toDurationElement(e), own);
317  }
318  break;
319  }
320  return wrap<Element>(e, own);
321  }
322 }
