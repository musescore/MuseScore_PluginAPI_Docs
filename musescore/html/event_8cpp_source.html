29 void MidiCoreEvent::write(XmlWriter& xml) const
30  {
31  switch(_type) {
32  case ME_NOTEON:
33  xml.tagE(QString("note-on channel=\"%1\" pitch=\"%2\" velo=\"%3\"")
34  .arg(_channel).arg(_a).arg(_b));
35  break;
36 
37  case ME_NOTEOFF:
38  xml.tagE(QString("note-off channel=\"%1\" pitch=\"%2\" velo=\"%3\"")
39  .arg(_channel).arg(_a).arg(_b));
40  break;
41 
42  case ME_CONTROLLER:
43  if (_a == CTRL_PROGRAM) {
44  if (_channel == 0) {
45  xml.tagE(QString("program value=\"%1\"").arg(_b));
46  }
47  else {
48  xml.tagE(QString("program channel=\"%1\" value=\"%2\"")
49  .arg(channel()).arg(_b));
50  }
51  }
52  else {
53  if (channel() == 0) {
54  xml.tagE(QString("controller ctrl=\"%1\" value=\"%2\"")
55  .arg(_a).arg(_b));
56  }
57  else {
58  xml.tagE(QString("controller channel=\"%1\" ctrl=\"%2\" value=\"%3\"")
59  .arg(channel()).arg(_a).arg(_b));
60  }
61  }
62  break;
63  default:
64  qDebug("MidiCoreEvent::write: unknown type");
65  break;
66  }

175 bool NPlayEvent::isMuted() const
176  {
177  const Note* n = note();
178  if (n) {
179  MasterScore* cs = n->masterScore();
180  Staff* staff = n->staff();
181  Instrument* instr = staff->part()->instrument(n->tick());
182  const Channel* a = instr->playbackChannel(n->subchannel(), cs);
183  return a->mute() || a->soloMute() || !staff->playbackVoice(n->voice());
184  }
185 
186  const Harmony* h = harmony();
187  if (h) {
188  const Channel* hCh = h->part()->harmonyChannel();
189  if (hCh) { //if there is a harmony channel
190  const Channel* pCh = h->masterScore()->playbackChannel(hCh);
191  return pCh->mute() || pCh->soloMute();
192  }
193  }
194 
195  return false;
196  }

243 void Event::write(XmlWriter& xml) const
244  {
245  switch(_type) {
246  case ME_NOTE:
247  xml.tagE(QString("note tick=\"%1\" channel=\"%2\" len=\"%3\" pitch=\"%4\" velo=\"%5\"")
248  .arg(_ontime).arg(_channel).arg(_duration).arg(_a).arg(_b));
249  break;
250 
251  case ME_NOTEON:
252  xml.tagE(QString("note-on tick=\"%1\" channel=\"%2\" pitch=\"%3\" velo=\"%4\"")
253  .arg(_ontime).arg(_channel).arg(_a).arg(_b));
254  break;
255 
256  case ME_NOTEOFF:
257  xml.tagE(QString("note-off tick=\"%1\" channel=\"%2\" pitch=\"%3\" velo=\"%4\"")
258  .arg(_ontime).arg(_channel).arg(_a).arg(_b));
259  break;
260 
261  case ME_CONTROLLER:
262  if (_a == CTRL_PROGRAM) {
263  if ((_ontime == -1) && (_channel == 0)) {
264  xml.tagE(QString("program value=\"%1\"").arg(_b));
265  }
266  else {
267  xml.tagE(QString("program tick=\"%1\" channel=\"%2\" value=\"%3\"")
268  .arg(ontime()).arg(channel()).arg(_b));
269  }
270  }
271  else {
272  if ((ontime() == -1) && (channel() == 0)) {
273  xml.tagE(QString("controller ctrl=\"%1\" value=\"%2\"")
274  .arg(_a).arg(_b));
275  }
276  else {
277  xml.tagE(QString("controller tick=\"%1\" channel=\"%2\" ctrl=\"%3\" value=\"%4\"")
278  .arg(ontime()).arg(channel()).arg(_a).arg(_b));
279  }
280  }
281  break;
282 
283  case ME_SYSEX:
284  xml.stag(QString("sysex tick=\"%1\" len=\"%2\"").arg(ontime()).arg(_len));
285  xml.dump(_len, _edata);
286  xml.etag();
287  break;
288 
289  case ME_META:
290  switch(metaType()) {
291  case META_TRACK_NAME:
292  xml.tag(QString("TrackName tick=\"%1\"").arg(ontime()), QString((char*)(edata())));
293  break;
294 
295  case META_LYRIC:
296  xml.tag(QString("Lyric tick=\"%1\"").arg(ontime()), QString((char*)(edata())));
297  break;
298 
299  case META_KEY_SIGNATURE:
300  {
301  const char* keyTable[] = {
302  "Ces", "Ges", "Des", "As", "Es", "Bes", "F",
303  "C",
304  "G", "D", "A", "E", "B", "Fis", "Cis"
305  };
306  int key = (char)(_edata[0]) + 7;
307  if (key < 0 || key > 14) {
308  qDebug("bad key signature %d", key);
309  key = 0;
310  }
311  QString sex(_edata[1] ? "Minor" : "Major");
312  QString keyName(keyTable[key]);
313  xml.tag(QString("Key tick=\"%1\" key=\"%2\" sex=\"%3\"").arg(ontime()).arg(_edata[0]).arg(_edata[1]),
314  QString("%1 %2").arg(keyName, sex));
315  }
316  break;
317 
318  case META_TIME_SIGNATURE:
319  xml.tagE(QString("TimeSig tick=\"%1\" num=\"%2\" denom=\"%3\" metro=\"%4\" quarter=\"%5\"")
320  .arg(ontime())
321  .arg(int(_edata[0]))
322  .arg(int(_edata[1]))
323  .arg(int(_edata[2]))
324  .arg(int(_edata[3])));
325  break;
326 
327  case META_TEMPO:
328  {
329  unsigned tempo = _edata[2] + (_edata[1] << 8) + (_edata[0] << 16);
330  xml.tagE(QString("Tempo tick=\"%1\" value=\"%2\"").arg(ontime()).arg(tempo));
331  }
332  break;
333 
334  default:
335  xml.stag(QString("Meta tick=\"%1\" type=\"%2\" len=\"%3\" name=\"%4\"")
336  .arg(ontime()).arg(metaType()).arg(_len).arg(midiMetaName(metaType())));
337  xml.dump(_len, _edata);
338  xml.etag();
339  break;
340  }
341  break;
342  }
343  }

411 void EventMap::fixupMIDI()
412  {
413  /* track info for each of the 128 possible MIDI notes */
414  struct channelInfo {
415  /* which event the first ME_NOTEON came from */
416  NPlayEvent *event[128];
417  /* how often is the note on right now? */
418  unsigned short nowPlaying[128];
419  };
420 
421  /* track info for each channel (on the heap, 0-initialised) */
422  struct channelInfo *info = (struct channelInfo *)calloc(_highestChannel + 1, sizeof(struct channelInfo));
423 
424  auto it = begin();
425  while (it != end()) {
426  NPlayEvent& event = it->second;
427  /* ME_NOTEOFF is never emitted, no need to check for it */
428  if (event.type() == ME_NOTEON && !event.isMuted()) {
429  unsigned short np = info[event.channel()].nowPlaying[event.pitch()];
430  if (event.velo() == 0) {
431  /* already off (should not happen) or still playing? */
432  if (np == 0 || --np > 0)
433  event.setDiscard(1);
434  else {
435  /* hoist NOTEOFF to same track as NOTEON */
436  event.setOriginatingStaff(info[event.channel()].event[event.pitch()]->getOriginatingStaff());
437  }
438  }
439  else {
440  if (++np > 1)
441  /* restrike, possibly on different track */
442  event.setDiscard(info[event.channel()].event[event.pitch()]->getOriginatingStaff() + 1);
443  info[event.channel()].event[event.pitch()] = &event;
444  }
445  info[event.channel()].nowPlaying[event.pitch()] = np;
446  }
447 
448  ++it;
449  }
450 
451  free((void *)info);
452  }
