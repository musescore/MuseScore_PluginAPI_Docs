41 bool ScoreView::event(QEvent* event)
42  {
43  switch (event->type()) {
44  case QEvent::KeyPress: {
45  QKeyEvent* ke = static_cast<QKeyEvent*>(event);
46  const int key = ke->key();
47  if (key != Qt::Key_Tab && key != Qt::Key_Backtab)
48  break;
49 
50  if (textEditMode()) {
51  // block Tab/Backtab in text editing mode
52  return true;
53  }
54 
55  if (hasEditGrips() || editMode()) {
56  if (ke->key() == Qt::Key_Tab)
57  editData.element->nextGrip(editData);
58  else
59  editData.element->prevGrip(editData);
60  updateGrips();
61  _score->update();
62  return true;
63  }
64  }
65  break;
66  case QEvent::ShortcutOverride: {
67  QKeyEvent* ke = static_cast<QKeyEvent*>(event);
68  switch (ke->key()) {
69  case Qt::Key_Left:
70  case Qt::Key_Right:
71  case Qt::Key_Up:
72  case Qt::Key_Down: {
73  if (!hasEditGrips())
74  break;
75  const auto m = ke->modifiers();
76  // KeypadModifier is necessary on MacOS as arrow keys seem to always
77  // trigger that modifier there. However it would probably be appropriate
78  // to allow it on other systems too.
79  const auto allowedModifiers = (editData.curGrip == Grip::NO_GRIP)
80  ? (Qt::KeypadModifier | Qt::ShiftModifier)
81  : (Qt::KeypadModifier | Qt::ShiftModifier | Qt::ControlModifier);
82  if ((m & ~allowedModifiers) == 0) {
83  ke->accept();
84  return true;
85  }
86  }
87  break;
88  default:
89  break;
90  }
91  }
92  break;
93  case QEvent::Gesture:
94  return gestureEvent(static_cast<QGestureEvent*>(event));
95  case QEvent::MouseButtonPress:
96  if (qApp->focusWidget() != this) {
97  QMouseEvent* me = static_cast<QMouseEvent*>(event);
98  if (me->button() == Qt::LeftButton)
99  this->setFocus();
100  }
101  break;
102  default:
103  break;
104  }
105 
106  return QWidget::event(event);
107  }

114 bool ScoreView::gestureEvent(QGestureEvent *event)
115  {
116  if (QGesture *gesture = event->gesture(Qt::PinchGesture)) {
117  // Zoom in/out when receiving a pinch gesture
118  QPinchGesture *pinch = static_cast<QPinchGesture *>(gesture);
119 
120  static qreal startLogicalZoomLevel = 1.0;
121  if (pinch->state() == Qt::GestureStarted) {
122  startLogicalZoomLevel = logicalZoomLevel();
123  }
124  if (pinch->changeFlags() & QPinchGesture::ScaleFactorChanged) {
125  // On Windows, totalScaleFactor() contains the net zoom.
126  // On OS X, totalScaleFactor() is 1, and scaleFactor() contains the net zoom.
127  qreal value = pinch->totalScaleFactor();
128  if (value == 1) {
129  value = pinch->scaleFactor();
130  }
131  setLogicalZoom(ZoomIndex::ZOOM_FREE, startLogicalZoomLevel * value, pinch->centerPoint());
132  }
133  }
134  return true;
135  }

141 void ScoreView::wheelEvent(QWheelEvent* event)
142  {
143 #define PIXELSSTEPSFACTOR 5
144 
145  QPoint pixelsScrolled = event->pixelDelta();
146  QPoint stepsScrolled = event->angleDelta();
147 
148  int dx = 0, dy = 0, n = 0;
149  qreal nReal = 0.0;
150 
151  if (!pixelsScrolled.isNull()) {
152  dx = pixelsScrolled.x();
153  dy = pixelsScrolled.y();
154  nReal = static_cast<qreal>(dy) / PIXELSSTEPSFACTOR;
155  }
156  else if (!stepsScrolled.isNull()) {
157  dx = static_cast<qreal>(stepsScrolled.x()) * qMax(2, width() / 10) / 120;
158  dy = static_cast<qreal>(stepsScrolled.y()) * qMax(2, height() / 10) / 120;
159  nReal = static_cast<qreal>(stepsScrolled.y()) / 120;
160  }
161 
162  n = static_cast<int>(nReal);
163 
164  //this functionality seems currently blocked by the context menu
165  if (event->buttons() & Qt::RightButton) {
166  bool up = n > 0;
167  if (!up)
168  n = -n;
169  score()->startCmd();
170  for (int i = 0; i < n; ++i)
171  score()->upDown(up, UpDownMode::CHROMATIC);
172  score()->endCmd();
173  return;
174  }
175 
176  if (event->modifiers() & Qt::ControlModifier) { // Windows touch pad pinches also execute this
177  QApplication::sendPostedEvents(this, 0);
178  zoomBySteps(nReal, true, event->posF());
179  return;
180  }
181 
182  //make shift+scroll go horizontally
183  if (event->modifiers() & Qt::ShiftModifier && dx == 0) {
184  dx = dy;
185  dy = 0;
186  }
187 
188  if (dx == 0 && dy == 0)
189  return;
190 
191  constraintCanvas(&dx, &dy);
192 
193  _matrix.setMatrix(_matrix.m11(), _matrix.m12(), _matrix.m13(), _matrix.m21(),
194  _matrix.m22(), _matrix.m23(), _matrix.dx()+dx, _matrix.dy()+dy, _matrix.m33());
195  imatrix = _matrix.inverted();
196 
197  scroll(dx, dy, QRect(0, 0, width(), height()));
198  emit viewRectChanged();
199  emit offsetChanged(_matrix.dx(), _matrix.dy());
200  }

274 bool ScoreView::startTextEditingOnMouseRelease(QMouseEvent* mouseEvent)
275  {
276  if (!editData.element || mouseEvent->button() != Qt::LeftButton)
277  return false;
278 
279  if (!(editData.element->isEditable() && editData.element->isTextBase()))
280  return false;
281 
282  TextBase* textBase = toTextBase(editData.element);
283 
284  if (!editData.element->canvasBoundingRect().contains(toLogical(mouseEvent->pos()))) {
285  // mouse up OUTSIDE textbase - clear any priming
286  textBase->setPrimed(false);
287  return false;
288  }
289 
290  if (!textBase->isPrimed()) {
291  // mouse up INSIDE textBase - set priming
292  textBase->setPrimed(true);
293  return false;
294  }
295 
296  // mouse up INSIDE primed textBase - start editing
297  startEditMode(textBase);
298  setCursor(QCursor(Qt::IBeamCursor));
299  textBase->setPrimed(false);
300  return true;

355 // mousePressEventNormal
356 // handle mouse press event for NORMAL mode
357 //---------------------------------------------------------
358 
359 void ScoreView::mousePressEventNormal(QMouseEvent* ev)
360  {
361  _score->masterScore()->cmdState().reset(); // DEBUG: should not be necessary
362  modifySelection = false;
363  elementToSelect = nullptr;
364 
365  Qt::KeyboardModifiers keyState = ev->modifiers();
366  SelectType st = SelectType::SINGLE;
367  if (keyState == Qt::NoModifier)
368  st = SelectType::SINGLE;
369  else if (keyState & Qt::ShiftModifier)
370  st = SelectType::RANGE;
371  else if (keyState & Qt::ControlModifier)
372  st = SelectType::ADD;
373 
374  Element* e = editData.element;
375  if (e) {
376  if (keyState == (Qt::ShiftModifier | Qt::ControlModifier)) {
377  cloneElement(e);
378  return;
379  }
380  if (e->isKeySig() && (keyState != Qt::ControlModifier) && st == SelectType::SINGLE) {
381  // special case: select for all staves
382  Segment* s = toKeySig(e)->segment();
383  bool first = true;
384  for (int staffIdx = 0; staffIdx < _score->nstaves(); ++staffIdx) {
385  Element* ee = s->element(staffIdx * VOICES);
386  if (ee) {
387  ee->score()->select(ee, first ? SelectType::SINGLE : SelectType::ADD);
388  first = false;
389  }
390  }
391  }
392  else if (e->isTimeSig() && !toTimeSig(e)->isLocal() && (keyState != Qt::ControlModifier) && st == SelectType::SINGLE) {
393  // special case: select for all staves except when TimeSig is local.
394  Segment* s = toTimeSig(e)->segment();
395  bool first = true;
396  for (int staffIdx = 0; staffIdx < _score->nstaves(); ++staffIdx) {
397  Element* ee = s->element(staffIdx * VOICES);
398  if (ee) {
399  ee->score()->select(ee, first ? SelectType::SINGLE : SelectType::ADD);
400  first = false;
401  }
402  }
403  }
404  else {
405  if (st == SelectType::ADD) {
406  // convert range to list
407  if (e->score()->selection().isRange()) {
408  e->score()->selection().setState(SelState::LIST);
409  e->score()->setUpdateAll(); // needed to clear selection rectangle
410  }
411  // e is the top element in stacking order,
412  // but we want to replace it with "first non-measure element after a selected element"
413  // (if such an element exists)
414  QList<Element*> ll = elementsNear(editData.startMove);
415  bool found = false;
416  for (Element* ee : qAsConst(ll)) {
417  if (found) {
418  e = ee;
419  if (!e->isMeasure())
420  break;
421  }
422  else if (ee->selected()) {
423  found = true;
424  ee->score()->deselect(ee);
425  e = nullptr;
426  }
427  }
428  }
429  if (e) {
430  if (!e->selected())
431  e->score()->select(e, st, -1);
432  else if (st != SelectType::ADD) {
433  modifySelection = true;
434  elementToSelect = e;
435  }
436  }
437  }
438  if (e) {
439  if (e->isNote() || e->isHarmony()) {
440  e->score()->updateCapo();
441  mscore->play(e);
442  }
443  _score = e->score();
444  _score->setUpdateAll();
445  }
446  }
447  else {
448  // special case: check if measure is selected
449  int staffIdx;
450  Measure* m = _score->pos2measure(editData.startMove, &staffIdx, 0, 0, 0);
451  if (m) {
452  QRectF r = m->staffLines(staffIdx)->canvasBoundingRect();
453  if (_score->staff(staffIdx)->lines(m->tick()) == 1) {
454  r.setHeight(m->spatium() * 2);
455  r.translate(0.0, -m->spatium());
456  }
457  if (r.contains(editData.startMove)) {
458  //TourHandler::startTour("select-tour");
459  _score->select(m, st, staffIdx);
460  _score->setUpdateAll();
461  }
462  else if (st == SelectType::SINGLE)
463  modifySelection = true;
464  }
465  else if (st == SelectType::SINGLE)
466  modifySelection = true;
467  }
468  _score->update();
469  mscore->endCmd();
470  }

476 void ScoreView::mousePressEvent(QMouseEvent* ev)
477  {
478 
479  if (tripleClickPending) {
480  if (textEditMode()) {
481  TextBase* textBase = toTextBase(editData.element);
482  textBase->multiClickSelect(editData, MultiClick::Triple);
483  mscore->textTools()->updateTools(editData);
484  textBase->endHexState(editData);
485  update();
486  return;
487  }
488  }
489 
490  editData.startMovePixel = ev->pos();
491  editData.startMove = toLogical(ev->pos());
492  editData.lastPos = editData.startMove;
493  editData.pos = editData.startMove;
494  editData.buttons = ev->buttons();
495  editData.modifiers = qApp->keyboardModifiers();
496  editData.key = 0;
497 
498  bool gripFound = false;
499  if (hasEditGrips() && ev->button() == Qt::LeftButton) {
500  switch (state) {
501  case ViewState::NORMAL:
502  case ViewState::EDIT:
503  case ViewState::FOTO: {
504  const qreal a = editData.grip[0].width() * 0.5;
505  for (int i = 0; i < editData.grips; ++i) {
506  if (editData.grip[i].adjusted(-a, -a, a, a).contains(editData.startMove)) {
507  editData.curGrip = Grip(i);
508  updateGrips();
509  score()->update();
510  gripFound = true;
511  break;
512  }
513  }
514 
515  if (!gripFound)
516  editData.curGrip = Grip::NO_GRIP;
517  }
518  break;
519  default:
520  break;
521  }
522  }
523 
524  switch (state) {
525  case ViewState::NORMAL: {
526  if (gripFound)
527  break;
528  if (ev->button() == Qt::RightButton) // context menu?
529  break;
530 
531  if (editData.element
532  && editData.element->isEditable()
533  && editData.element->isTextBase()
534  && !editData.element->canvasBoundingRect().contains(toLogical(ev->pos()))) {
535  // mouse down OUTSIDE of textBase - clear priming -
536  toTextBase(editData.element)->setPrimed(false);
537  }
538 
539  setEditElement(elementNear(editData.startMove));
540  mousePressEventNormal(ev);
541  }
542  break;
543 
544  case ViewState::FOTO: {
545  if (ev->buttons() & Qt::RightButton)
546  break;
547  setEditElement(_foto);
548 
549  if (gripFound)
550  changeState(ViewState::FOTO_DRAG_EDIT);
551  else if (_foto->canvasBoundingRect().contains(editData.startMove))
552  changeState(ViewState::FOTO_DRAG_OBJECT);
553  else if (ev->modifiers() & Qt::ShiftModifier)
554  changeState(ViewState::FOTO_LASSO);
555  else
556  changeState(ViewState::FOTO_DRAG);
557  }
558  break;
559 
560  case ViewState::NOTE_ENTRY: {
561  _score->startCmd();
562  bool restMode = _score->inputState().rest();
563  if (ev->button() == Qt::RightButton)
564  _score->inputState().setRest(!restMode);
565  _score->putNote(editData.startMove, ev->modifiers() & Qt::ShiftModifier, ev->modifiers() & Qt::ControlModifier);
566  if (ev->button() == Qt::RightButton)
567  _score->inputState().setRest(restMode);
568  _score->endCmd();
569  if (_score->inputState().cr())
570  adjustCanvasPosition(_score->inputState().cr(), false);
571  shadowNote->setVisible(false);
572  }
573  break;
574 
575  case ViewState::EDIT: {
576  if (gripFound)
577  break;
578  if (!editData.element->canvasBoundingRect().contains(editData.startMove)) {
579  changeState(ViewState::NORMAL);
580  // changeState may trigger layout and destroy some elements
581  // so we should search elementNear after changeState.
582  setEditElement(elementNear(editData.startMove));
583  mousePressEventNormal(ev);
584  }
585  else {
586  editData.element->mousePress(editData);
587  score()->update();
588  if (editData.element->isTextBase() && mscore->textTools())
589  mscore->textTools()->updateTools(editData);
590  }
591  }
592  break;
593 
594  case ViewState::PLAY: {
595  Element* e = elementNear(editData.startMove);
596  if (seq && e && (e->isNote() || e->isRest())) {
597  if (e->isNote())
598  e = e->parent();
599  ChordRest* cr = toChordRest(e);
600  seq->seek(seq->score()->repeatList().tick2utick(cr->tick().ticks()));
601  }
602  }
603  break;
604 
605  default:
606  qDebug("mousePressEvent in state %d", int(state));
607  break;
608  }
609  }

635 void ScoreView::mouseMoveEvent(QMouseEvent* me)
636  {
637  adjustCursorForTextEditing(me);
638 
639  if (state != ViewState::NOTE_ENTRY && editData.buttons == Qt::NoButton)
640  return;
641 
642  // start some drag operations after a minimum of movement:
643  bool drag = (me->pos() - editData.startMovePixel).manhattanLength() > 4;
644 
645  switch (state) {
646  case ViewState::NORMAL:
647  if (!drag)
648  return;
649  if (!editData.element && (me->modifiers() & Qt::ShiftModifier)) {
650  changeState(ViewState::LASSO);
651  break;
652  }
653  if (editData.element) {
654  if (editData.element->normalModeEditBehavior() == Element::EditBehavior::Edit && editData.curGrip != Grip::NO_GRIP) {
655  score()->startCmd();
656  editData.element->startEditDrag(editData);
657  changeState(ViewState::DRAG_EDIT);
658  break;
659  }
660  if (editData.element->isMovable()) {
661  if (editData.element->normalModeEditBehavior() == Element::EditBehavior::Edit)
662  endEdit();
663  changeState(ViewState::DRAG_OBJECT);
664  break;
665  }
666  }
667  changeState(ViewState::DRAG);
668  modifySelection = false;
669  break;
670 
671  case ViewState::NOTE_ENTRY: {
672  QPointF p = toLogical(me->pos());
673  QRectF r(shadowNote->canvasBoundingRect());
674  setShadowNote(p);
675  r |= shadowNote->canvasBoundingRect();
676  update(toPhysical(r).adjusted(-2, -2, 2, 2));
677  }
678  break;
679 
680  case ViewState::DRAG:
681  case ViewState::FOTO_DRAG:
682  dragScoreView(me);
683  break;
684 
685  case ViewState::DRAG_OBJECT:
686  case ViewState::FOTO_DRAG_OBJECT:
687  doDragElement(me);
688  break;
689 
690  case ViewState::DRAG_EDIT:
691  case ViewState::FOTO_DRAG_EDIT:
692  doDragEdit(me);
693  break;
694 
695  case ViewState::LASSO:
696  doDragLasso(me);
697  break;
698 
699  case ViewState::EDIT:
700  if (!drag)
701  return;
702  score()->startCmd();
703  editData.element->startEditDrag(editData);
704  changeState(ViewState::DRAG_EDIT);
705  break;
706 
707  case ViewState::PLAY:
708  if (drag)
709  dragScoreView(me);
710  break;
711 
712  case ViewState::FOTO_LASSO:
713  doDragFoto(me);
714  break;
715 
716  default:
717  break;
718  }
719  update();
720  }

898 
899 bool ScoreView::handleArrowKeyPress(const QKeyEvent* ev)
900  {
901  QPointF delta;
902  qreal _spatium = editData.element->spatium();
903 
904  qreal xval, yval;
905  if (editData.element->isBeam()) {
906  xval = 0.25 * _spatium;
907  if (editData.modifiers & Qt::ControlModifier)
908  xval = _spatium;
909  else if (editData.modifiers & Qt::AltModifier)
910  xval = 4 * _spatium;
911  }
912  else {
913  xval = MScore::nudgeStep * _spatium;
914  if (editData.modifiers & Qt::ControlModifier)
915  xval = MScore::nudgeStep10 * _spatium;
916  else if (editData.modifiers & Qt::AltModifier)
917  xval = MScore::nudgeStep50 * _spatium;
918  }
919  yval = xval;
920 
921  if (mscore->vRaster()) {
922  qreal vRaster = _spatium / MScore::vRaster();
923  if (yval < vRaster)
924  yval = vRaster;
925  }
926  if (mscore->hRaster()) {
927  qreal hRaster = _spatium / MScore::hRaster();
928  if (xval < hRaster)
929  xval = hRaster;
930  }
931  // TODO: if raster, then xval/yval should be multiple of raster
932 
933  switch (ev->key()) {
934  case Qt::Key_Left:
935  delta = QPointF(-xval, 0);
936  break;
937  case Qt::Key_Right:
938  delta = QPointF(xval, 0);
939  break;
940  case Qt::Key_Up:
941  delta = QPointF(0, -yval);
942  break;
943  case Qt::Key_Down:
944  delta = QPointF(0, yval);
945  break;
946  default:
947  return false;
948  }
949  editData.delta = delta;
950  editData.evtDelta = editData.moveDelta = delta;
951  editData.hRaster = mscore->hRaster();
952  editData.vRaster = mscore->vRaster();
953  if (editData.curGrip != Grip::NO_GRIP && int(editData.curGrip) < editData.grips)
954  editData.pos = editData.grip[int(editData.curGrip)].center() + delta;
955  editData.element->startEditDrag(editData);
956  editData.element->editDrag(editData);
957  editData.element->endEditDrag(editData);
958  return true;
959  }
