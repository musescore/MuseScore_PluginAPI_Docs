247 bool ExportMidi::write(QIODevice* device, bool midiExpandRepeats, bool exportRPNs, const SynthesizerState& synthState)
248  {
249  mf.setDivision(MScore::division);
250  mf.setFormat(1);
251  QList<MidiTrack>& tracks = mf.tracks();
252 
253  for (int i = 0; i < cs->nstaves(); ++i)
254  tracks.append(MidiTrack());
255 
256  EventMap events;
257  cs->renderMidi(&events, false, midiExpandRepeats, synthState);
258 
259  pauseMap.calculate(cs);
260  writeHeader();
261 
262  int staffIdx = 0;
263  for (auto &track: tracks) {
264  Staff* staff = cs->staff(staffIdx);
265  Part* part = staff->part();
266 
267  track.setOutPort(part->midiPort());
268  track.setOutChannel(part->midiChannel());
269 
270  // Pass through the all instruments in the part
271  const InstrumentList* il = part->instruments();
272  for(auto j = il->begin(); j!= il->end(); j++) {
273  // Pass through the all channels of the instrument
274  // "normal", "pizzicato", "tremolo" for Strings,
275  // "normal", "mute" for Trumpet
276  for (const Channel* instrChan : j->second->channel()) {
277  const Channel* ch = part->masterScore()->playbackChannel(instrChan);
278  char port = part->masterScore()->midiPort(ch->channel());
279  char channel = part->masterScore()->midiChannel(ch->channel());
280 
281  if (staff->isTop()) {
282  track.insert(0, MidiEvent(ME_CONTROLLER, channel, CTRL_RESET_ALL_CTRL, 0));
283  // We need this to get the correct pitch of bends
284  // Hidden under preferences because some software
285  // crashes when receiving RPNs: https://musescore.org/en/node/37431
286  if (channel != 9 && exportRPNs) {
287  // set pitch bend sensitivity to 12 semitones:
288  track.insert(0, MidiEvent(ME_CONTROLLER, channel, CTRL_LRPN, 0));
289  track.insert(0, MidiEvent(ME_CONTROLLER, channel, CTRL_HRPN, 0));
290  track.insert(0, MidiEvent(ME_CONTROLLER, channel, CTRL_HDATA, 12));
291 
292  // reset fine tuning
293  /*track.insert(0, MidiEvent(ME_CONTROLLER, channel, CTRL_LRPN, 1));
294  track.insert(0, MidiEvent(ME_CONTROLLER, channel, CTRL_HRPN, 0));
295  track.insert(0, MidiEvent(ME_CONTROLLER, channel, CTRL_HDATA, 64));*/
296 
297  // deactivate rpn
298  track.insert(0, MidiEvent(ME_CONTROLLER, channel, CTRL_LRPN, 127));
299  track.insert(0, MidiEvent(ME_CONTROLLER, channel, CTRL_HRPN, 127));
300  }
301 
302  if (ch->program() != -1)
303  track.insert(0, MidiEvent(ME_CONTROLLER, channel, CTRL_PROGRAM, ch->program()));
304  track.insert(0, MidiEvent(ME_CONTROLLER, channel, CTRL_VOLUME, ch->volume()));
305  track.insert(0, MidiEvent(ME_CONTROLLER, channel, CTRL_PANPOT, ch->pan()));
306  track.insert(0, MidiEvent(ME_CONTROLLER, channel, CTRL_REVERB_SEND, ch->reverb()));
307  track.insert(0, MidiEvent(ME_CONTROLLER, channel, CTRL_CHORUS_SEND, ch->chorus()));
308  }
309 
310  // Export port to MIDI META event
311  if (track.outPort() >= 0 && track.outPort() <= 127) {
312  MidiEvent ev;
313  ev.setType(ME_META);
314  ev.setMetaType(META_PORT_CHANGE);
315  ev.setLen(1);
316  unsigned char* data = new unsigned char[1];
317  data[0] = int(track.outPort());
318  ev.setEData(data);
319  track.insert(0, ev);
320  }
321 
322  for (auto i = events.begin(); i != events.end(); ++i) {
323  const NPlayEvent& event = i->second;
324 
325  if (event.isMuted())
326  continue;
327  if (event.discard() == staffIdx + 1 && event.velo() > 0)
328  // turn note off so we can restrike it in another track
329  track.insert(pauseMap.addPauseTicks(i->first), MidiEvent(ME_NOTEON, channel,
330  event.pitch(), 0));
331 
332  if (event.getOriginatingStaff() != staffIdx)
333  continue;
334 
335  if (event.discard() && event.velo() == 0)
336  // ignore noteoff but restrike noteon
337  continue;
338 
339  if (!exportRPNs && event.type() == ME_CONTROLLER && event.portamento())
340  // ignore portamento control events if exportRPN isn't switched on
341  continue;
342 
343  char eventPort = cs->masterScore()->midiPort(event.channel());
344  char eventChannel = cs->masterScore()->midiChannel(event.channel());
345  if (port != eventPort || channel != eventChannel)
346  continue;
347 
348  if (event.type() == ME_NOTEON) {
349  // use the note values instead of the event values if portamento is suppressed
350  if (!exportRPNs && event.portamento())
351  track.insert(pauseMap.addPauseTicks(i->first), MidiEvent(ME_NOTEON, channel,
352  event.note()->pitch(), event.velo()));
353  else
354  track.insert(pauseMap.addPauseTicks(i->first), MidiEvent(ME_NOTEON, channel,
355  event.pitch(), event.velo()));
356  }
357  else if (event.type() == ME_CONTROLLER) {
358  track.insert(pauseMap.addPauseTicks(i->first), MidiEvent(ME_CONTROLLER, channel,
359  event.controller(), event.value()));
360  }
361  else if(event.type() == ME_PITCHBEND) {
362  track.insert(pauseMap.addPauseTicks(i->first), MidiEvent(ME_PITCHBEND, channel,
363  event.dataA(), event.dataB()));
364  }
365  else {
366  qDebug("writeMidi: unknown midi event 0x%02x", event.type());
367  }
368  }
369  }
370  }
371  ++staffIdx;
372  }
373  return !mf.write(device);
374  }

396 //---------------------------------------------------------
397 // PauseMap::calculate
398 // MIDI files cannot contain pauses so insert extra ticks and tempo changes instead.
399 // The PauseMap and new TempoMap are fully unwound to account for pauses on repeats.
400 //---------------------------------------------------------
401 
402 void ExportMidi::PauseMap::calculate(const Score* s)
403  {
404  Q_ASSERT(s);
405  TimeSigMap* sigmap = s->sigmap();
406  TempoMap* tempomap = s->tempomap();
407 
408  this->insert(std::pair<const int, int> (0, 0)); // can't start with a pause
409 
410  tempomapWithPauses = new TempoMap();
411  tempomapWithPauses->setRelTempo(tempomap->relTempo());
412 
413  for (const RepeatSegment* rs : s->repeatList()) {
414  int startTick = rs->tick;
415  int endTick = startTick + rs->len();
416  int tickOffset = rs->utick - rs->tick;
417 
418  auto se = tempomap->lower_bound(startTick);
419  auto ee = tempomap->lower_bound(endTick+1); // +1 to include first tick of next RepeatSegment
420 
421  for (auto it = se; it != ee; ++it) {
422  int tick = it->first;
423  int utick = tick + tickOffset;
424 
425  if (it->second.pause == 0.0) {
426  // We have a regular tempo change. Don't include tempo change from first tick of next RepeatSegment (it will be included later).
427  if (tick != endTick)
428  tempomapWithPauses->insert(std::pair<const int, TEvent> (this->addPauseTicks(utick), it->second));
429  }
430  else {
431  // We have a pause event. Don't include pauses from first tick of current RepeatSegment (it was included in the previous one).
432  if (tick != startTick) {
433  Fraction timeSig(sigmap->timesig(tick).timesig());
434  qreal quarterNotesPerMeasure = (4.0 * timeSig.numerator()) / timeSig.denominator();
435  int ticksPerMeasure = quarterNotesPerMeasure * MScore::division; // store a full measure of ticks to keep barlines in same places
436  tempomapWithPauses->setTempo(this->addPauseTicks(utick), quarterNotesPerMeasure / it->second.pause); // new tempo for pause
437  this->insert(std::pair<const int, int> (utick, ticksPerMeasure + this->offsetAtUTick(utick))); // store running total of extra ticks
438  tempomapWithPauses->setTempo(this->addPauseTicks(utick), it->second.tempo); // restore previous tempo
439  }
440  }
441  }
442  }
443  }
