73 bool Fermata::readProperties(XmlReader& e)
74  {
75  const QStringRef& tag(e.name());
76 
77  if (tag == "subtype") {
78  QString s = e.readElementText();
79  SymId id = Sym::name2id(s);
80  setSymId(id);
81  }
82  else if (tag == "play")
83  setPlay(e.readBool());
84  else if (tag == "timeStretch")
85  _timeStretch = e.readDouble();
86  else if (tag == "offset") {
87  if (score()->mscVersion() > 114)
88  Element::readProperties(e);
89  else
90  e.skipCurrentElement(); // ignore manual layout in older scores
91  }
92  else if (Element::readProperties(e))
93  ;
94  else
95  return false;
96  return true;
97  }

103 void Fermata::write(XmlWriter& xml) const
104  {

116  Element::writeProperties(xml);

213 void Fermata::layout()
214  {
215  Segment* s = segment();
216  setPos(QPointF());
217  if (!s) { // for use in palette
218  setOffset(0.0, 0.0);
219  QRectF b(symBbox(_symId));
220  setbbox(b.translated(-0.5 * b.width(), 0.0));
221  return;
222  }
223 
224  if (isStyled(Pid::OFFSET))
225  setOffset(propertyDefault(Pid::OFFSET).toPointF());
226  Element* e = s->element(track());
227  if (e) {
228  if (e->isChord())
229  rxpos() += score()->noteHeadWidth() * staff()->mag(Fraction(0, 1)) * .5;
230  else
231  rxpos() += e->x() + e->width() * staff()->mag(Fraction(0, 1)) * .5;
232  }
233 
234  QString name = Sym::id2name(_symId);
235  if (placeAbove()) {
236  if (name.endsWith("Below"))
237  _symId = Sym::name2id(name.left(name.size() - 5) + "Above");
238  }
239  else {
240  rypos() += staff()->height();
241  if (name.endsWith("Above"))
242  _symId = Sym::name2id(name.left(name.size() - 5) + "Below");
243  }
244  QRectF b(symBbox(_symId));
245  setbbox(b.translated(-0.5 * b.width(), 0.0));
246  autoplaceSegmentElement();
247  }

282 bool Fermata::setProperty(Pid propertyId, const QVariant& v)
283  {
284  switch (propertyId) {
285  case Pid::SYMBOL:
286  setSymId(v.value<SymId>());
287  break;
288  case Pid::PLACEMENT: {
289  Placement p = Placement(v.toInt());
290  if (p != placement()) {
291  QString s = Sym::id2name(_symId);
292  bool up = placeAbove();
293  if (s.endsWith(up ? "Above" : "Below")) {
294  QString s2 = s.left(s.size() - 5) + (up ? "Below" : "Above");
295  _symId = Sym::name2id(s2);
296  }
297  setPlacement(p);
298  }
299  }
300  break;
301  case Pid::PLAY:
302  setPlay(v.toBool());
303  break;
304  case Pid::TIME_STRETCH:
305  setTimeStretch(v.toDouble());
306  score()->fixTicks();
307  break;
308  default:
309  return Element::setProperty(propertyId, v);
310  }
311  triggerLayout();
312  return true;
313  }
