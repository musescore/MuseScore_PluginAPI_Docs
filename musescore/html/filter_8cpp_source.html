27 ZFilter::ZFilter()
28  {
29  constexpr double ff = 1.0 / 32768.0;
30  for (int i = 0; i < INTERP_MAX; i++) {
31  double x = (double) i / (double) INTERP_MAX;
32  interpCoeff[i][0] = (x * (-0.5 + x * (1 - 0.5 * x))) * ff;
33  interpCoeff[i][1] = (1.0 + x * x * (1.5 * x - 2.5)) * ff;
34  interpCoeff[i][2] = (x * (0.5 + x * (2.0 - 1.5 * x))) * ff;
35  interpCoeff[i][3] = (0.5 * x * x * (x - 1.0)) * ff;
36  }
37  }

43 void ZFilter::initialize(const Zerberus* _zerberus, const Zone* z, int velocity)
44  {
45  this->zerberus = _zerberus;
46  this->sampleZone = z;
47 
48  resonanceF = _zerberus->ct2hz(13500.0);
49  if (z->isCutoffDefined) {
50  //calculate current cutoff value
51  float cutoffHz = z->cutoff;
52  //Formula for converting the interval frequency ratio f2 / f1 to cents (c or ¢).
53  //¢ or c = 1200 × log2 (f2 / f1)
54  cutoffHz *= pow(2.0, velocity / 127.0f * z->fil_veltrack / 1200.0);
55  resonanceF = cutoffHz;
56  }
57 
58  last_resonanceF = -1.0;
59  float q_db = 0; //no resonance by default
60  q_lin = pow(10.0f, q_db / 20.0f);
61  gain = 1.0 / sqrt(q_lin);
62 
63  firstRun = true;
64  }

70 void ZFilter::update()
71  {
72  float adaptedFrequency = resonanceF;
73  const float sr = zerberus->sampleRate();
74  if (adaptedFrequency > 0.45f * sr)
75  adaptedFrequency = 0.45f * sr;
76  else if (adaptedFrequency < 5.f)
77  adaptedFrequency = 5.f;
78 
79  const bool freqWasUpdated = fabs(adaptedFrequency - last_resonanceF) > 0.01f;
80  if (!freqWasUpdated)
81  return;
82 
83  last_resonanceF = adaptedFrequency;
84 
85  // The filter coefficients have to be recalculated (filter
86  // parameters have changed). Recalculation for various reasons is
87  // forced by setting last_fres to -1. The flag filter_startup
88  // indicates, that the DSP loop runs for the first time, in this
89  // case, the filter is set directly, instead of smoothly fading
90  // between old and new settings.
91  //
92  // Those equations from Robert Bristow-Johnson's `Cookbook
93  // formulae for audio EQ biquad filter coefficients', obtained
94  // from Harmony-central.com / Computer / Programming. They are
95  // the result of the bilinear transform on an analogue filter
96  // prototype. To quote, `BLT frequency warping has been taken
97  // into account for both significant frequency relocation and for
98  // bandwidth readjustment'.
99 
100  const float omega = (float) 2.0f * M_PI * (adaptedFrequency / sr);
101  const float sin_coeff = sin(omega);
102  const float cos_coeff = cos(omega);
103  const float alpha_coeff = sin_coeff / (2.0f * q_lin);
104  const float a0_inv = 1.0f / (1.0f + alpha_coeff);
105  const float linCosCoeff = 2.f - cos_coeff;
106 
107  float a1_temp = 0.f;
108  float a2_temp = 0.f;
109  float b1_temp = 0.f;
110  float b0_temp = 0.f;
111  float b2_temp = 0.f;
112  switch (sampleZone->fil_type) {
113  case FilterType::lpf_2p: {
114  a1_temp = 2.0f * cos_coeff * a0_inv;
115  a2_temp = (alpha_coeff - 1.f) * a0_inv;
116  b1_temp = (1.0f - cos_coeff) * a0_inv;
117  b0_temp = b2_temp = b1_temp * 0.5f;
118  break;
119  }
120  case FilterType::hpf_2p: {
121  a1_temp = 2.0f * cos_coeff * a0_inv;
122  a2_temp = (alpha_coeff - 1.f) * a0_inv;
123  b1_temp = -(1.0f + cos_coeff) * a0_inv;
124  b0_temp = b2_temp = -b1_temp * 0.5f;
125  break;
126  }
127  case FilterType::bpf_2p: {
128  a1_temp = 2.0f * cos_coeff * a0_inv;
129  a2_temp = (alpha_coeff - 1.f) * a0_inv;
130  b0_temp = alpha_coeff * a0_inv;
131  b1_temp = 0.f;
132  b2_temp = -b0_temp;
133  break;
134  }
135  case FilterType::brf_2p: {
136  a1_temp = 2.0f * cos_coeff * a0_inv;
137  a2_temp = (alpha_coeff - 1.f) * a0_inv;
138  b1_temp = a0_inv * (-2.f * cos_coeff);
139  b0_temp = b2_temp = a0_inv;
140  break;
141  }
142  case FilterType::lpf_1p: {
143  a1_temp = -(linCosCoeff - sqrt(linCosCoeff * linCosCoeff - 1));
144  b0_temp = 1 + a1_temp;
145  break;
146  }
147  case FilterType::hpf_1p: {
148  a1_temp = -(linCosCoeff - sqrt(linCosCoeff * linCosCoeff - 1));
149  b0_temp = -a1_temp;
150  b1_temp = a1_temp;
151  break;
152  }
153  default:
154  qWarning() << "fil_type is not implemented: " << (int)sampleZone->fil_type;
155  }
156 
157  if (firstRun) {
158  /* The filter is calculated, because the voice was started up.
159  * In this case set the filter coefficients without delay.
160  */
161  a1 = a1_temp;
162  a2 = a2_temp;
163  b0 = b0_temp;
164  b2 = b2_temp;
165  b1 = b1_temp;
166  filter_coeff_incr_count = 0;
167  firstRun = false;
168  }
169  else {
170  /* The filter frequency is changed. Calculate an increment
171  * factor, so that the new setting is reached after some time.
172  */
173 
174  static const int FILTER_TRANSITION_SAMPLES = 64;
175 
176  a1_incr = (a1_temp - a1) / FILTER_TRANSITION_SAMPLES;
177  a2_incr = (a2_temp - a2) / FILTER_TRANSITION_SAMPLES;
178  b0_incr = (b0_temp - b0) / FILTER_TRANSITION_SAMPLES;
179  b1_incr = (b1_temp - b1) / FILTER_TRANSITION_SAMPLES;
180  b2_incr = (b2_temp - b2) / FILTER_TRANSITION_SAMPLES;
181  /* Have to add the increments filter_coeff_incr_count times. */
182  filter_coeff_incr_count = FILTER_TRANSITION_SAMPLES;
183  }
184  }

190 float ZFilter::apply(float inputValue, bool leftChannel)
191  {
192  float value = 0.f;
193  const auto applyBqEquation = [this, &inputValue](float& histX1, float& histX2, float& histY1, float& histY2) {
194  //apply filter
195  /*
196  float y = d.b0 * x + d.b1 * d.x1 + d.b2 * d.x2 +
197  d.a1 * d.y1 + d.a2 * d.y2;
198  d.x2 = d.x1;
199  d.x1 = x;
200  d.y2 = d.y1;
201  d.y1 = y;
202  return y;
203  */
204  float value = b0 * inputValue + b1 * histX1 + b2 * histX2 + a1 * histY1 + a2 * histY2;
205  histX2 = histX1;
206  histX1 = inputValue;
207  histY2 = histY1;
208  histY1 = value;
209  return value;
210  };
211  const auto applyHPF1PEquation = [this, &inputValue](float& histX1, float& histY1) {
212  float value = b0 * inputValue + b1 * histX1 - a1 * histY1;
213  histX1 = inputValue;
214  histY1 = value;
215  return value;
216  };
217  const auto applyLPF1PEquation = [this, &inputValue](float& histY1) {
218  float value = b0 * inputValue - a1 * histY1;
219  histY1 = value;
220  return value;
221  };
222  if (leftChannel) {
223  switch (sampleZone->fil_type) {
224  case FilterType::hpf_2p:
225  case FilterType::lpf_2p:
226  case FilterType::bpf_2p:
227  case FilterType::brf_2p: {
228  value = applyBqEquation(monoL.histX1, monoL.histX2, monoL.histY1, monoL.histY2);
229  break;
230  }
231  case FilterType::hpf_1p: {
232  value = applyHPF1PEquation(monoL.histX1, monoL.histY1);
233  break;
234  }
235  case FilterType::lpf_1p: {
236  value = applyLPF1PEquation(monoL.histY1);
237  break;
238  }
239  default:
240  qWarning() << "this equation is not implemented" << (int)sampleZone->fil_type;
241  }
242  }
243  else { //right channel in stereo samples
244  switch (sampleZone->fil_type) {
245  case FilterType::hpf_2p:
246  case FilterType::lpf_2p:
247  case FilterType::bpf_2p:
248  case FilterType::brf_2p: {
249  value = applyBqEquation(monoR.histX1, monoR.histX2, monoR.histY1, monoR.histY2);
250  break;
251  }
252  case FilterType::hpf_1p: {
253  value = applyHPF1PEquation(monoR.histX1, monoR.histY1);
254  break;
255  }
256  case FilterType::lpf_1p: {
257  value = applyLPF1PEquation(monoR.histY1);
258  break;
259  }
260  default:
261  qWarning() << "this equation is not implemented" << (int)sampleZone->fil_type;
262  }
263  }
264 
265  if (filter_coeff_incr_count) {
266  --filter_coeff_incr_count;
267  a1 += a1_incr;
268  a2 += a2_incr;
269  b0 += b0_incr;
270  b1 += b1_incr;
271  b2 += b2_incr;
272  }
273 
274  return value;
275  }
