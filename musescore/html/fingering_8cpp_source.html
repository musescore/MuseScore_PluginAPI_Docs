73 Placement Fingering::calculatePlacement() const
74  {
75  Note* n = note();
76  if (!n)
77  return Placement::ABOVE;
78  Chord* chord = n->chord();
79  Staff* staff = chord->staff();
80  Part* part = staff->part();
81  int nstaves = part->nstaves();
82  bool voices = chord->measure()->hasVoices(staff->idx(), chord->tick(), chord->actualTicks());
83  bool below = voices ? !chord->up() : (nstaves > 1) && (staff->rstaff() == nstaves - 1);
84  return below ? Placement::BELOW : Placement::ABOVE;
85  }

91 void Fingering::layout()
92  {
93  if (parent()) {
94  Fraction tick = parent()->tick();
95  const Staff* st = staff();
96  if (st && st->isTabStaff(tick) && !st->staffType(tick)->showTabFingering()) {
97  setbbox(QRectF());
98  return;
99  }
100  }
101 
102  TextBase::layout();
103  rypos() = 0.0; // handle placement below
104 
105  if (autoplace() && note()) {
106  Note* n = note();
107  Chord* chord = n->chord();
108  bool voices = chord->measure()->hasVoices(chord->staffIdx(), chord->tick(), chord->actualTicks());
109  bool tight = voices && chord->notes().size() == 1 && !chord->beam() && tid() != Tid::STRING_NUMBER;
110 
111  qreal headWidth = n->bboxRightPos();
112 
113  // update offset after drag
114  qreal rebase = 0.0;
115  if (offsetChanged() != OffsetChange::NONE && !tight)
116  rebase = rebaseOffset();
117 
118  // temporarily exclude self from chord shape
119  setAutoplace(false);
120 
121  if (layoutType() == ElementType::CHORD) {
122  bool above = placeAbove();
123  Stem* stem = chord->stem();
124  Segment* s = chord->segment();
125  Measure* m = s->measure();
126  qreal sp = spatium();
127  qreal md = minDistance().val() * sp;
128  SysStaff* ss = m->system()->staff(chord->vStaffIdx());
129  Staff* vStaff = chord->staff(); // TODO: use current height at tick
130 
131  if (n->mirror())
132  rxpos() -= n->ipos().x();
133  rxpos() += headWidth * .5;
134  if (above) {
135  if (tight) {
136  if (chord->stem())
137  rxpos() -= 0.8 * sp;
138  rypos() -= 1.5 * sp;
139  }
140  else {
141  QRectF r = bbox().translated(m->pos() + s->pos() + chord->pos() + n->pos() + pos());
142  SkylineLine sk(false);
143  sk.add(r.x(), r.bottom(), r.width());
144  qreal d = sk.minDistance(ss->skyline().north());
145  qreal yd = 0.0;
146  if (d > 0.0 && isStyled(Pid::MIN_DISTANCE))
147  yd -= d + height() * .25;
148  // force extra space above staff & chord (but not other fingerings)
149  qreal top;
150  if (chord->up() && chord->beam() && stem) {
151  top = stem->y() + stem->bbox().top();
152  }
153  else {
154  Note* un = chord->upNote();
155  top = qMin(0.0, un->y() + un->bbox().top());
156  }
157  top -= md;
158  qreal diff = (bbox().bottom() + ipos().y() + yd + n->y()) - top;
159  if (diff > 0.0)
160  yd -= diff;
161  if (offsetChanged() != OffsetChange::NONE) {
162  // user moved element within the skyline
163  // we may need to adjust minDistance, yd, and/or offset
164  bool inStaff = above ? r.bottom() + rebase > 0.0 : r.top() + rebase < staff()->height();
165  rebaseMinDistance(md, yd, sp, rebase, above, inStaff);
166  }
167  rypos() += yd;
168  }
169  }
170  else {
171  if (tight) {
172  if (chord->stem())
173  rxpos() += 0.8 * sp;
174  rypos() += 1.5 * sp;
175  }
176  else {
177  QRectF r = bbox().translated(m->pos() + s->pos() + chord->pos() + n->pos() + pos());
178  SkylineLine sk(true);
179  sk.add(r.x(), r.top(), r.width());
180  qreal d = ss->skyline().south().minDistance(sk);
181  qreal yd = 0.0;
182  if (d > 0.0 && isStyled(Pid::MIN_DISTANCE))
183  yd += d + height() * .25;
184  // force extra space below staff & chord (but not other fingerings)
185  qreal bottom;
186  if (!chord->up() && chord->beam() && stem) {
187  bottom = stem->y() + stem->bbox().bottom();
188  }
189  else {
190  Note* dn = chord->downNote();
191  bottom = qMax(vStaff->height(), dn->y() + dn->bbox().bottom());
192  }
193  bottom += md;
194  qreal diff = bottom - (bbox().top() + ipos().y() + yd + n->y());
195  if (diff > 0.0)
196  yd += diff;
197  if (offsetChanged() != OffsetChange::NONE) {
198  // user moved element within the skyline
199  // we may need to adjust minDistance, yd, and/or offset
200  bool inStaff = above ? r.bottom() + rebase > 0.0 : r.top() + rebase < staff()->height();
201  rebaseMinDistance(md, yd, sp, rebase, above, inStaff);
202  }
203  rypos() += yd;
204  }
205  }
206  }
207  else if (tid() == Tid::LH_GUITAR_FINGERING) {
208  // place to left of note
209  qreal left = n->shape().left();
210  if (left - n->x() > 0.0)
211  rxpos() -= left;
212  else
213  rxpos() -= n->x();
214  }
215  // for other fingering styles, do not autoplace
216 
217  // restore autoplace
218  setAutoplace(true);
219  }
220  else if (offsetChanged() != OffsetChange::NONE) {
221  // rebase horizontally too, as autoplace may have adjusted it
222  rebaseOffset(false);
223  }
224  setOffsetChanged(false);
225  }
