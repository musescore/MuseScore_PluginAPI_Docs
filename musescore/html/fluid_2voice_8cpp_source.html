63  pos += dur/4;
64  pos %= dur;

281 bool Voice::generateDataForDSPChain(unsigned framesBufCount)
282  {
283  /* Range checking for sample- and loop-related parameters
284  * Initial phase is calculated here*/
285  check_sample_sanity();
286 
287  /******************* vol env **********************/
288 
289  fluid_env_data_t* env_data = &volenv_data[volenv_section];
290  Sample2AmpInc.clear();
291  std::map<int, int> sample2VolEnvSection;
292  std::set<int> volumeChanges;
293 
294  if (volenv_section >= FLUID_VOICE_ENVFINISHED) {
295  off();
296  return false;
297  }
298 
299  // determine points where volume envelope changes
300  unsigned curVolEnvCount = volenv_count;
301  unsigned restN = framesBufCount;
302  while (curVolEnvCount + restN >= env_data->count) {
303  restN -= env_data->count - curVolEnvCount;
304 
305  sample2VolEnvSection.insert(std::pair<int, int>(framesBufCount - restN, volenv_section));
306  volumeChanges.insert(framesBufCount-restN);
307 
308  curVolEnvCount = 0;
309  volenv_section++;
310 
311  env_data = &volenv_data[volenv_section];
312  }
313 
314  sample2VolEnvSection.insert(std::pair<int, int>(framesBufCount, volenv_section));
315  volumeChanges.insert(framesBufCount);
316 
317  fluid_check_fpe ("voice_write vol env");
318 
319  /******************* mod env **********************/
320 
321  /* Skip to decay phase if delay and attack envelope sections each are
322  * less than 100 samples long. This avoids popping noises due to the
323  * mod envelope being out-of-sync with the sample-based volume envelope. */
324  if (modenv_section < 2 && modenv_data[FLUID_VOICE_ENVDELAY].count < 100 && modenv_data[FLUID_VOICE_ENVATTACK].count < 100) {
325  modenv_section = 2;
326  modenv_val = 1.0f;
327  }
328 
329  env_data = &modenv_data[modenv_section];
330 
331  /* skip to the next section of the envelope if necessary */
332  while (modenv_count >= env_data->count) {
333  env_data = &modenv_data[++modenv_section];
334  modenv_count = 0;
335  }
336 
337  /* calculate the envelope value and check for valid range */
338  float x = env_data->coeff * modenv_val + env_data->incr * framesBufCount;
339 
340  if (x < env_data->min) {
341  x = env_data->min;
342  modenv_section++;
343  modenv_count = 0;
344  }
345  else if (x > env_data->max) {
346  x = env_data->max;
347  modenv_section++;
348  modenv_count = 0;
349  }
350 
351  modenv_val = x;
352  modenv_count += framesBufCount;
353  fluid_check_fpe ("voice_write mod env");
354 
355  /******************* mod lfo **********************/
356  // calculate all points where we need to consider
357  // the mod lfo (where it changes its slope)
358 
359  int modLfoStart = -1;
360 
361  if (fabs(modlfo_to_vol) > 0) {
362  if (ticks >= modlfo_delay)
363  modLfoStart = 0;
364  else if (framesBufCount >= modlfo_delay)
365  modLfoStart = modlfo_delay;
366 
367  if (modLfoStart >= 0) {
368  if (modLfoStart > 0)
369  volumeChanges.insert(modLfoStart);
370 
371  unsigned int modLfoNextTurn = samplesToNextTurningPoint(modlfo_dur, modlfo_pos);
372 
373  while (modLfoNextTurn+modLfoStart < framesBufCount) {
374  volumeChanges.insert(modLfoNextTurn+modLfoStart);
375  modLfoNextTurn++;
376  modLfoNextTurn += samplesToNextTurningPoint(modlfo_dur, modLfoNextTurn);
377  }
378  }
379  }
380 
381  fluid_check_fpe ("voice_write mod LFO");
382 
383  /******************* vib lfo **********************/
384 
385  if (ticks >= viblfo_delay) {
386  viblfo_val += viblfo_incr * framesBufCount;
387 
388  if (viblfo_val > (float) 1.0) {
389  viblfo_incr = -viblfo_incr;
390  viblfo_val = (float) 2.0 - viblfo_val;
391  }
392  else if (viblfo_val < -1.0) {
393  viblfo_incr = -viblfo_incr;
394  viblfo_val = (float) -2.0 - viblfo_val;
395  }
396  }
397 
398  fluid_check_fpe ("voice_write Vib LFO");
399 
400  /******************* amplitude **********************/
401 
402  if (volenv_section == FLUID_VOICE_ENVDELAY) {
403  return false; /* The volume amplitude is in hold phase. No sound is produced. */
404  }
405 
406  qreal oldTargetAmp = amp;
407  int lastPos = 0;
408  auto oldVolEnvSection = sample2VolEnvSection.begin();
409  auto curVolEnvSection = oldVolEnvSection;
410 
411  for (auto curPos : volumeChanges)
412  {
413  if (modLfoStart >= 0 && curPos >= modLfoStart)
414  modlfo_val = triangle(modlfo_dur, modlfo_pos+curPos-modLfoStart);
415  else
416  modlfo_val = 0;
417 
418  // never calculate anything for the very first sample
419  // everything should have been calculated in the last
420  // cycle - it would also cause a divion by zero later
421  if (curPos == 0) {
422  curPos = 1;
423 
424  // if we should calculate for position 1 already make sure we don't do it twice
425  // could lead to curPos==lastPos which causes devision by zero
426  if (volumeChanges.find(1) != volumeChanges.end())
427  volumeChanges.erase(volumeChanges.find(1));
428  }
429 
430  // just go to the next volume section if we're below last volume point
431  if (curPos >= curVolEnvSection->first && (unsigned int) curVolEnvSection->first < framesBufCount)
432  curVolEnvSection++;
433 
434  volenv_count += curPos-lastPos;
435  calcVolEnv(curPos-lastPos, &volenv_data[oldVolEnvSection->second]);
436 
437  volenv_section = oldVolEnvSection->second;
438 
439  qreal target_amp {0.0}; /* target amplitude */
440  if (volenv_section <= FLUID_VOICE_ENVATTACK) {
441  /* the envelope is in the attack section: ramp linearly to max value.
442  * A positive modlfo_to_vol should increase volume (negative attenuation).
443  */
444  target_amp = fluid_atten2amp (attenuation)
445  * fluid_cb2amp (modlfo_val * -modlfo_to_vol)
446  * volenv_val;
447  }
448  else {
449  //float amplitude_that_reaches_noise_floor;
450  //float amp_max;
451 
452  target_amp = fluid_atten2amp (attenuation)
453  * fluid_cb2amp (960.0f * (1.0f - volenv_val)
454  + modlfo_val * -modlfo_to_vol);
455 
456  /* A voice can be turned off, when an estimate for the volume
457  * (upper bound) falls below that volume, that will drop the
458  * sample below the noise floor.
459  */
460 
461  /* If the loop amplitude is known, we can use it if the voice loop is within
462  * the sample loop
463  */
464 
465  float amplitude_that_reaches_noise_floor;
466  /* Is the playing pointer already in the loop? */
467  if (has_looped)
468  amplitude_that_reaches_noise_floor = amplitude_that_reaches_noise_floor_loop;
469  else
470  amplitude_that_reaches_noise_floor = amplitude_that_reaches_noise_floor_nonloop;
471 
472  /* voice->attenuation_min is a lower boundary for the attenuation
473  * now and in the future (possibly 0 in the worst case). Now the
474  * amplitude of sample and volenv cannot exceed amp_max (since
475  * volenv_val can only drop):
476  */
477 
478  float amp_max = fluid_atten2amp (min_attenuation_cB) * volenv_val;
479 
480  /* And if amp_max is already smaller than the known amplitude,
481  * which will attenuate the sample below the noise floor, then we
482  * can safely turn off the voice. Duh. */
483  if (amp_max < amplitude_that_reaches_noise_floor) {
484  positionToTurnOff = curPos;
485  }
486 
487  }
488 
489  if (curVolEnvSection->second != oldVolEnvSection->second) {
490  if (oldVolEnvSection->second == FLUID_VOICE_ENVDECAY) {
491  env_data = &volenv_data[oldVolEnvSection->second];
492  volenv_val = env_data->min * env_data->coeff;
493  }
494  volenv_count = 0;
495  oldVolEnvSection = curVolEnvSection;
496  }
497  /* Volume increment to go from voice->amp to target_amp in FLUID_BUFSIZE steps */
498  amp_incr = (target_amp - oldTargetAmp) / (curPos - lastPos);
499  lastPos = curPos;
500  Sample2AmpInc.insert(std::pair<int, qreal>(curPos, amp_incr));
501 
502  // if voice is turned off after this no need to calculate any more values
503  if (positionToTurnOff > 0)
504  break;
505 
506  oldTargetAmp = target_amp;
507  }
508 
509  if (modLfoStart >= 0) {
510  modlfo_pos += framesBufCount - modLfoStart;
511  modlfo_val = triangle(modlfo_dur, modlfo_pos-modLfoStart);
512  }
513 
514  fluid_check_fpe ("voice_write amplitude calculation");
515 
516  /* Calculate the number of samples, that the DSP loop advances
517  * through the original waveform with each step in the output
518  * buffer. It is the ratio between the frequencies of original
519  * waveform and output waveform.*/
520 
521  {
522  float cent = pitch + pitchoffset + modlfo_val * modlfo_to_pitch
523  + viblfo_val * viblfo_to_pitch
524  + modenv_val * modenv_to_pitch;
525  phase_incr = _fluid->ct2hz_real(cent) / root_pitch_hz;
526  }
527 
528  if (pitchoffset < 0.0f) {
529  float diff = pitchinc * framesBufCount;
530  pitchoffset -= diff;
531  if ( pitchoffset > 0.0f)
532  pitchoffset = 0.0f;
533  }
534  else if (pitchoffset > 0.0f) {
535  float diff = pitchinc * framesBufCount;
536  pitchoffset -= diff;
537  if (pitchoffset < 0.0f)
538  pitchoffset = 0.0f;
539  }
540 
541  /* if phase_incr is not advancing, set it to the minimum fraction value (prevent stuckage) */
542  if (phase_incr == 0)
543  phase_incr = 1;
544 
545  /*************** resonant filter ******************/
546 
547  /* calculate the frequency of the resonant filter in Hz */
548  float _fres = _fluid->ct2hz_real(fres
549  + modlfo_val * modlfo_to_fc
550  + modenv_val * modenv_to_fc);
551 
552  /* FIXME - Still potential for a click during turn on, can we interpolate
553  between 20khz cutoff and 0 Q? */
554 
555  /* I removed the optimization of turning the filter off when the
556  * resonance frequence is above the maximum frequency. Instead, the
557  * filter frequency is set to a maximum of 0.45 times the sampling
558  * rate. For a 44100 kHz sampling rate, this amounts to 19845
559  * Hz. The reason is that there were problems with anti-aliasing when the
560  * synthesizer was run at lower sampling rates. Thanks to Stephan
561  * Tassart for pointing me to this bug. By turning the filter on and
562  * clipping the maximum filter frequency at 0.45*srate, the filter
563  * is used as an anti-aliasing filter. */
564 
565  if (_fres > 0.45f * _fluid->sample_rate)
566  _fres = 0.45f * _fluid->sample_rate;
567  else if (_fres < 5)
568  _fres = 5;
569 
570  /* if filter enabled and there is a significant frequency change.. */
571  if ((qAbs(_fres - last_fres) > 0.01)) {
572  /* The filter coefficients have to be recalculated (filter
573  * parameters have changed). Recalculation for various reasons is
574  * forced by setting last_fres to -1. The flag filter_startup
575  * indicates, that the DSP loop runs for the first time, in this
576  * case, the filter is set directly, instead of smoothly fading
577  * between old and new settings.
578  *
579  * Those equations from Robert Bristow-Johnson's `Cookbook
580  * formulae for audio EQ biquad filter coefficients', obtained
581  * from Harmony-central.com / Computer / Programming. They are
582  * the result of the bilinear transform on an analogue filter
583  * prototype. To quote, `BLT frequency warping has been taken
584  * into account for both significant frequency relocation and for
585  * bandwidth readjustment'. */
586 
587  float omega = (float) (2.0 * M_PI * (_fres / ((float) _fluid->sample_rate)));
588  float sin_coeff = (float) sin(omega);
589  float cos_coeff = (float) cos(omega);
590  float alpha_coeff = sin_coeff / (2.0f * q_lin);
591  float a0_inv = 1.0f / (1.0f + alpha_coeff);
592 
593  /* Calculate the filter coefficients. All coefficients are
594  * normalized by a0. Think of `a1' as `a1/a0'.
595  *
596  * Here a couple of multiplications are saved by reusing common expressions.
597  * The original equations should be:
598  * b0=(1.-cos_coeff)*a0_inv*0.5*voice->filter_gain;
599  * b1=(1.-cos_coeff)*a0_inv*voice->filter_gain;
600  * b2=(1.-cos_coeff)*a0_inv*0.5*voice->filter_gain; */
601 
602  float a1_temp = -2.0f * cos_coeff * a0_inv;
603  float a2_temp = (1.0f - alpha_coeff) * a0_inv;
604  float b1_temp = (1.0f - cos_coeff) * a0_inv * filter_gain;
605  /* both b0 -and- b2 */
606  float b02_temp = b1_temp * 0.5f;
607 
608  if (filter_startup) {
609  /* The filter is calculated, because the voice was started up.
610  * In this case set the filter coefficients without delay.
611  */
612  a1 = a1_temp;
613  a2 = a2_temp;
614  b02 = b02_temp;
615  b1 = b1_temp;
616  filter_coeff_incr_count = 0;
617  filter_startup = 0;
618 //printf("Setting initial filter coefficients.\n");
619  }
620  else {
621  /* The filter frequency is changed. Calculate an increment
622  * factor, so that the new setting is reached after one buffer
623  * length. x_incr is added to the current value FLUID_BUFSIZE
624  * times. The length is arbitrarily chosen. Longer than one
625  * buffer will sacrifice some performance, though. Note: If
626  * the filter is still too 'grainy', then increase this number

683  / put all cached frames to buffer
684  / (buffer* is a buffer after putting left cache data)
685  / generate DSP data and interpolation for buffer* size data or required buffer size if buffer* is too small

1372 /**
1373  * fluid_voice_modulate
1374  *
1375  * In this implementation, I want to make sure that all controllers
1376  * are event based: the parameter values of the DSP algorithm should
1377  * only be updates when a controller event arrived and not at every
1378  * iteration of the audio cycle (which would probably be feasible if
1379  * the synth was made in silicon).
1380  *
1381  * The update is done in three steps:
1382  *
1383  * - first, we look for all the modulators that have the changed
1384  * controller as a source. This will yield a list of generators that
1385  * will be changed because of the controller event.
1386  *
1387  * - For every changed generator, calculate its new value. This is the
1388  * sum of its original value plus the values of al the attached
1389  * modulators.
1390  *
1391  * - For every changed generator, convert its value to the correct
1392  * unit of the corresponding DSP parameter
1393  * */
1394 
1395 void Voice::modulate(bool _cc, int _ctrl)
1396  {
1397  // For our purposes in MuseScore, we don't need to be able to control all
1398  // modulators in the release phase. In fact, we don't want these modulators to
1399  // have an effect, since it can lead to 'volume bumps' for single note
1400  // dynamics, when there is a sudden change from quiet to loud sounds.
1401  // So, if the voice is in release phase, don't apply these modulators.
1402  if (volenv_section == FLUID_VOICE_ENVRELEASE && _cc &&
1403  (_ctrl == BREATH_MSB || _ctrl == FOOT_MSB || _ctrl == EXPRESSION_MSB))
1404  return;
1405 
1406  for (int i = 0; i < mod_count; i++) {
1407  Mod* m = &mod[i];
1408 
1409  /* step 1: find all the modulators that have the changed controller
1410  * as input source.
1411  */
1412  if (m->has_source(_cc, _ctrl)) {
1413  int g = m->get_dest();
1414  float modval = 0.0;
1415 
1416  /* step 2: for every changed modulator, calculate the modulation
1417  * value of its associated generator
1418  */
1419  for (int k = 0; k < mod_count; k++) {
1420  if (fluid_mod_has_dest(&mod[k], g)) {
1421  modval += mod[k].get_value(channel, this);
1422  }
1423  }
1424  gen[g].set_mod(modval);
1425 
1426  /* step 3: now that we have the new value of the generator,
1427  * recalculate the parameter values that are derived from the
1428  * generator
1429  */
1430  update_param(g);
1431  }
1432  }
1433  }
