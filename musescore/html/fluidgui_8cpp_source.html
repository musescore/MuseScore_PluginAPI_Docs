308 void FluidGui::loadSf()
309  {
310  if (_sfToLoad.empty())
311  return;
312 
313  struct SfNamePath item = _sfToLoad.front();
314  QString sfName = item.name;
315  QString sfPath = item.path;
316  _sfToLoad.pop_front();
317 
318  QStringList sl;
319  for (int i = 0; i < soundFonts->count(); ++i) {
320  QListWidgetItem* widgetItem = soundFonts->item(i);
321  sl.append(widgetItem->text());
322  }
323 
324  if (sl.contains(sfPath)) {
325  QMessageBox::warning(this,
326  tr("MuseScore"),
327  tr("SoundFont %1 already loaded").arg(sfPath));
328  }
329  else {
330  _loadedSfName = sfName;
331  _loadedSfPath = sfPath;
332  QFuture<bool> future = QtConcurrent::run(fluid(), &FluidS::Fluid::addSoundFont, sfPath);
333  _futureWatcher.setFuture(future);
334  _progressTimer->start(1000);
335  _progressDialog->exec();
336  }
337  }

362  }
363  loadSf();
364  }
