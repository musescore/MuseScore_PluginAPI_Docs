144 void Freeverb::process(int n, float* in, float* out)
145  {
146  if (parameterChanged) {
147  roomsize = newRoomsize;
148  damp = newDamp;
149  width = newWidth;
150  sendLevel = newSendLevel;
151  wet = newWet;
152  update();
153  parameterChanged = false;
154  }
155  float* lp = in;
156  float* rp = in+1;
157  float* lo = out;
158  float* ro = out+1;
159 
160  float dry = 1.0 - wet;
161 
162  for (int k = 0; k < n; k++) {
163  float outL = 0.0;
164  float outR = 0.0;
165 
166  float input = ((*lp + *rp) * sendLevel) + DC_OFFSET;
167 
168  for (int i = 0; i < numcombs; i++) { // Accumulate comb filters in parallel
169  outL += combL[i].process(input);
170  outR += combR[i].process(input);
171  }
172  for (int i = 0; i < numallpasses; i++) { // Feed through allpasses in series
173  outL = allpassL[i].process(outL);
174  outR = allpassR[i].process(outR);
175  }
176 
177  // Remove the DC offset
178  outL -= DC_OFFSET;
179  outR -= DC_OFFSET;
180 
181  *lo = *lp * dry + (outL * wet1 + outR * wet2) * wet;
182  *ro = *rp * dry + (outR * wet1 + outL * wet2) * wet;
183  lp += 2;
184  rp += 2;
185  lo += 2;
186  ro += 2;
187  }
188  }
