400 static void parseLILC(char* buffer)
401  {
402  QString s(buffer);
403  QStringList sl = s.split("\n");
404  QRegExp ra("\\(attachment\\s\\.\\s\\(([0-9\\+\\-\\.]{1,})\\s\\.\\s([0-9\\+\\-\\.]{1,})\\)\\)\\)\\)");
405  QRegExp rb("\\(\\(bbox\\s\\.\\s\\(([0-9\\+\\-\\.]{1,})\\s([0-9\\+\\-\\.]{1,})\\s([0-9\\+\\-\\.]{1,})\\s([0-9\\+\\-\\.]{1,})\\)\\)");
406  int n = sl.size();
407  for (int i = 0; (i+4) < n; i += 5) {
408  Glyph g;
409  QString s = sl[i];
410  int nn = s.size();
411  s = s.mid(1, nn - 3);
412  g.name = s;
413  int idx = 0;
414  if (namemap.contains(s))
415  idx = namemap[s];
416  else
417  qDebug("genft: <%s> not in map", qPrintable(s));
418  int code = 0;
419  if (codemap.contains(idx))
420  code = codemap[idx];
421  else
422  qDebug("codemap has no index %d", idx);
423  g.code = code;
424 
425  s = sl[i+4];
426  int val = ra.indexIn(s);
427  if (val == -1 || ra.captureCount() != 2) {
428  qDebug("bad reg expr a");
429  exit(-5);
430  }
431  g.attach.rx() = ra.cap(1).toDouble();
432  g.attach.ry() = -ra.cap(2).toDouble();
433 
434  s = sl[i+1];
435  val = rb.indexIn(s);
436  if (val == -1 || rb.captureCount() != 4) {
437  qDebug("bad reg expr b");
438  exit(-5);
439  }
440  double a = rb.cap(1).toDouble();
441  double b = rb.cap(2).toDouble();
442  double c = rb.cap(3).toDouble();
443  double d = rb.cap(4).toDouble();
444  g.bbox = QRectF(a, -d, c - a, d - b);
445  glyphs.append(g);
446  }
447  }

454 static void genJson()
455  {
456  QJsonObject o;
457 
458  QMapIterator<QString, int> i(namemap);
459  while (i.hasNext()) {
460  i.next();
461 
462  if (!nmap.contains(i.key())) {
463  fprintf(stderr, "not found: <%s>\n", qPrintable(i.key()));
464  continue;
465  }
466 
467  int code = codemap[i.value()];
468 
469  QString smufl = nmap.value(i.key());
470 
471  QJsonObject jg;
472  QString s = QString("U+%1").arg(code, 0, 16);
473  jg.insert("codepoint", s);
474 
475  QJsonValue val(s);
476  o.insert(smufl, jg);
477  }
478  QJsonDocument d(o);
479  QByteArray ba = d.toJson();
480  QFile f("glyphnames.json");
481  f.open(QIODevice::WriteOnly);
482  f.write(ba);
483  f.close();
484  }
