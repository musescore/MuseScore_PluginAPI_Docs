79 void GlissandoSegment::draw(QPainter* painter) const
80  {
81  painter->save();
82  qreal _spatium = spatium();
83 
84  QPen pen(curColor(visible(), glissando()->lineColor()));
85  pen.setWidthF(glissando()->lineWidth());
86  pen.setCapStyle(Qt::RoundCap);
87  painter->setPen(pen);
88 
89  // rotate painter so that the line become horizontal
90  qreal w = pos2().x();
91  qreal h = pos2().y();
92  qreal l = sqrt(w * w + h * h);
93  qreal wi = asin(-h / l) * 180.0 / M_PI;
94  qreal scale = painter->worldTransform().m11();
95  painter->rotate(-wi);
96 
97  if (glissando()->glissandoType() == GlissandoType::STRAIGHT) {
98  painter->drawLine(QLineF(0.0, 0.0, l, 0.0));
99  }
100  else if (glissando()->glissandoType() == GlissandoType::WAVY) {
101  QRectF b = symBbox(SymId::wiggleTrill);
102  qreal a = symAdvance(SymId::wiggleTrill);
103  int n = static_cast<int>(l / a); // always round down (truncate) to avoid overlap
104  qreal x = (l - n*a) * 0.5; // centre line in available space
105  std::vector<SymId> ids;
106  for (int i = 0; i < n; ++i)
107  ids.push_back(SymId::wiggleTrill);
108  // this is very ugly but fix #68846 for now
109  bool tmp = MScore::pdfPrinting;
110  MScore::pdfPrinting = true;
111  score()->scoreFont()->draw(ids, painter, magS(), QPointF(x, -(b.y() + b.height()*0.5) ), scale);
112  MScore::pdfPrinting = tmp;
113  }
114 
115  if (glissando()->showText()) {
116  QFont f(glissando()->fontFace());
117  f.setPointSizeF(glissando()->fontSize() * MScore::pixelRatio * _spatium / SPATIUM20);
118  f.setBold(glissando()->fontStyle() & FontStyle::Bold);
119  f.setItalic(glissando()->fontStyle() & FontStyle::Italic);
120  f.setUnderline(glissando()->fontStyle() & FontStyle::Underline);
121  QFontMetricsF fm(f, painter->device()); // use the QPaintDevice, otherwise calculations will be done in screen metrics
122  QRectF r = fm.boundingRect(glissando()->text());
123 
124  // if text longer than available space, skip it
125  if (r.width() < l) {
126  qreal yOffset = r.height() + r.y(); // find text descender height
127  // raise text slightly above line and slightly more with WAVY than with STRAIGHT
128  yOffset += _spatium * (glissando()->glissandoType() == GlissandoType::WAVY ? 0.4 : 0.1);
129  painter->setFont(f);
130  qreal x = (l - r.width()) * 0.5;
131  painter->drawText(QPointF(x, -yOffset), glissando()->text());
132  }
133  }
134  painter->restore();
135  }

227 void Glissando::layout()
228  {
229  qreal _spatium = spatium();
230 
231  if (score() == gscore || !startElement() || !endElement()) { // for use in palettes or while dragging
232  if (spannerSegments().empty())
233  add(createLineSegment());
234  LineSegment* s = frontSegment();
235  s->setPos(QPointF());
236  s->setPos2(QPointF(_spatium * GLISS_PALETTE_WIDTH, -_spatium * GLISS_PALETTE_HEIGHT));
237  s->layout();
238  return;
239  }
240  SLine::layout();
241  if (spannerSegments().empty()) {
242  qDebug("no segments");
243  return;
244  }
245  setPos(0.0, 0.0);
246 
247  Note* anchor1 = toNote(startElement());
248  Note* anchor2 = toNote(endElement());
249  Chord* cr1 = anchor1->chord();
250  Chord* cr2 = anchor2->chord();
251  GlissandoSegment* segm1 = toGlissandoSegment(frontSegment());
252  GlissandoSegment* segm2 = toGlissandoSegment(backSegment());
253 
254  // Note: line segments are defined by
255  // initial point: ipos() (relative to system origin)
256  // ending point: pos2() (relative to initial point)
257 
258  // LINE ENDING POINTS TO NOTEHEAD CENTRES
259 
260  // assume gliss. line goes from centre of initial note centre to centre of ending note:
261  // move first segment origin and last segment ending point from notehead origin to notehead centre
262  // For TAB: begin at the right-edge of initial note rather than centre
263  QPointF offs1 = (cr1->staff()->isTabStaff(cr1->tick())) ? QPointF(anchor1->bbox().right(), 0.0) : QPointF(anchor1->headWidth() * 0.5, 0.0);
264  QPointF offs2 = QPointF(anchor2->headWidth() * 0.5, 0.0);
265 
266  // AVOID HORIZONTAL LINES
267 
268  int upDown = (0 < (anchor2->pitch() - anchor1->pitch())) - ((anchor2->pitch() - anchor1->pitch()) < 0);
269  // on TAB's, glissando are by necessity on the same string, this gives an horizontal glissando line;
270  // make bottom end point lower and top ending point higher
271  if (cr1->staff()->isTabStaff(cr1->tick())) {
272  qreal yOff = cr1->staff()->lineDistance(cr1->tick()) * 0.4 * _spatium;
273  offs1.ry() += yOff * upDown;
274  offs2.ry() -= yOff * upDown;
275  }
276  // if not TAB, angle glissando between notes on the same line
277  else {
278  if (anchor1->line() == anchor2->line()) {
279  offs1.ry() += _spatium * 0.25 * upDown;
280  offs2.ry() -= _spatium * 0.25 * upDown;
281  }
282  }
283 
284  // move initial point of first segment and adjust its length accordingly
285  segm1->setPos (segm1->ipos() + offs1);
286  segm1->setPos2(segm1->ipos2() - offs1);
287  // adjust ending point of last segment
288  segm2->setPos2(segm2->ipos2() + offs2);
289 
290  // FINAL SYSTEM-INITIAL NOTE
291  // if the last gliss. segment attaches to a system-initial note, some extra width has to be added
292  if (cr2->segment()->measure()->isFirstInSystem() && cr2->rtick().isZero()
293  // but ignore graces after, as they are not the first note of the system,
294  // even if their segment is the first segment of the system
295  && !(cr2->noteType() == NoteType::GRACE8_AFTER
296  || cr2->noteType() == NoteType::GRACE16_AFTER || cr2->noteType() == NoteType::GRACE32_AFTER)
297  // also ignore if cr1 is a child of cr2, which means cr1 is a grace-before of cr2
298  && !(cr1->parent() == cr2)) {
299  // in theory we should be reserving space for the gliss prior to the first note of a system
300  // but in practice we are not (and would be difficult to get right in current layout algorithms)
301  // so, a compromise is to at least use the available space to the left -
302  // the default layout for lines left a margin after the header
303  segm2->rxpos() -= _spatium;
304  segm2->rxpos2()+= _spatium;
305  }
306 
307  // INTERPOLATION OF INTERMEDIATE POINTS
308  // This probably belongs to SLine class itself; currently it does not seem
309  // to be needed for anything else than Glissando, though
310 
311  // get total x-width and total y-height of all segments
312  qreal xTot = 0.0;
313  for (SpannerSegment* segm : spannerSegments())
314  xTot += segm->ipos2().x();
315  qreal y0 = segm1->ipos().y();
316  qreal yTot = segm2->ipos().y() + segm2->ipos2().y() - y0;
317  qreal ratio = yTot / xTot;
318  // interpolate y-coord of intermediate points across total width and height
319  qreal xCurr = 0.0;
320  qreal yCurr;
321  for (unsigned i = 0; i + 1 < spannerSegments().size(); i++) {
322  SpannerSegment* segm = segmentAt(i);
323  xCurr += segm->ipos2().x();
324  yCurr = y0 + ratio * xCurr;
325  segm->rypos2() = yCurr - segm->ipos().y(); // position segm. end point at yCurr
326  // next segment shall start where this segment stopped
327  segm = segmentAt(i+1);
328  segm->rypos2() += segm->ipos().y() - yCurr; // adjust next segm. vertical length
329  segm->rypos() = yCurr; // position next segm. start point at yCurr
330  }
331 
332  // STAY CLEAR OF NOTE APPENDAGES
333 
334  // initial note dots / ledger line / notehead
335  offs1 *= -1.0; // discount changes already applied
336  int dots = anchor1->dots().size();
337  LedgerLine * ledLin = cr1->ledgerLines();
338 
339  // If TAB: completely zero first offset since it was already applied as right edge of first note
340  if (cr1->staff()->isTabStaff(cr1->tick()))
341  offs1.rx() = 0.0;
342  // if dots, start at right of last dot
343  // if no dots, from right of ledger line, if any; from right of notehead, if no ledger line
344  else offs1.rx() += (dots && anchor1->dot(dots-1) ? anchor1->dot(dots-1)->pos().x() + anchor1->dot(dots-1)->width()
345  : (ledLin ? ledLin->pos().x() + ledLin->width() : anchor1->headWidth()) );
346 
347  // final note arpeggio / accidental / ledger line / accidental / arpeggio (i.e. from outermost to innermost)
348  offs2 *= -1.0; // discount changes already applied
349  if (Arpeggio* ap = cr2->arpeggio())
350  offs2.rx() += ap->pos().x() + ap->offset().x();
351  else if (Accidental* ac = anchor2->accidental())
352  offs2.rx() += ac->pos().x() + ac->offset().x();
353  else if ( (ledLin = cr2->ledgerLines()) != nullptr)
354  offs2.rx() += ledLin->pos().x();
355 
356  // add another a quarter spatium of 'air'
357  offs1.rx() += _spatium * 0.25;
358  offs2.rx() -= _spatium * 0.25;
359 
360  // apply offsets: shorten first segment by x1 (and proportionally y) and adjust its length accordingly
361  offs1.ry() = segm1->ipos2().y() * offs1.x() / segm1->ipos2().x();
362  segm1->setPos(segm1->ipos() + offs1);
363  segm1->setPos2(segm1->ipos2() - offs1);
364  // adjust last segment length by x2 (and proportionally y)
365  offs2.ry() = segm2->ipos2().y() * offs2.x() / segm2->ipos2().x();
366  segm2->setPos2(segm2->ipos2() + offs2);
367 
368  for (SpannerSegment* segm : spannerSegments())
369  segm->layout();
370 
371  // compute glissando bbox as the bbox of the last segment, relative to the end anchor note
372  QPointF anchor2PagePos = anchor2->pagePos();
373  QPointF system2PagePos;
374  IF_ASSERT_FAILED(cr2->segment()->system()) {
375  system2PagePos = segm2->pos();
376  }
377  else {
378  system2PagePos = cr2->segment()->system()->pagePos();
379  }
380 
381  QPointF anchor2SystPos = anchor2PagePos - system2PagePos;
382  QRectF r = QRectF(anchor2SystPos - segm2->pos(), anchor2SystPos - segm2->pos() - segm2->pos2()).normalized();
383  qreal lw = lineWidth() * .5;
384  setbbox(r.adjusted(-lw, -lw, lw, lw));
385  }

443 //---------------------------------------------------------
444 // STATIC FUNCTIONS: guessInitialNote
445 //
446 // Used while reading old scores (either 1.x or transitional 2.0) to determine (guess!)
447 // the glissando initial note from its final chord. Returns the top note of previous chord
448 // of the same instrument, preferring the chord in the same track as chord, if it exists.
449 //
450 // CANNOT be called if the final chord and/or its segment do not exist yet in the score
451 //
452 // Parameter: chord: the chord this glissando ends into
453 // Returns: the top note in a suitable previous chord or nullptr if none found.
454 //---------------------------------------------------------
455 
456 Note* Glissando::guessInitialNote(Chord* chord)
457  {
458  switch (chord->noteType()) {
459 // case NoteType::INVALID:
460 // return 0;
461  // for grace notes before, previous chord is previous chord of parent chord
462  case NoteType::ACCIACCATURA:
463  case NoteType::APPOGGIATURA:
464  case NoteType::GRACE4:
465  case NoteType::GRACE16:
466  case NoteType::GRACE32:
467  // move unto parent chord and proceed to standard case
468  if (chord->parent() && chord->parent()->isChord())
469  chord = toChord(chord->parent());
470  else
471  return 0;
472  break;
473  // for grace notes after, return top note of parent chord
474  case NoteType::GRACE8_AFTER:
475  case NoteType::GRACE16_AFTER:
476  case NoteType::GRACE32_AFTER:
477  if (chord->parent() && chord->parent()->isChord())
478  return toChord(chord->parent())->upNote();
479  else // no parent or parent is not a chord?
480  return nullptr;
481  case NoteType::NORMAL:
482  {
483  // if chord has grace notes before, the last one is the previous note
484  QVector<Chord*> graces = chord->graceNotesBefore();
485  if (graces.size() > 0)
486  return graces.last()->upNote();
487  }
488  break; // else process to standard case
489  default:
490  break;
491  }
492 
493  // standard case (NORMAL or grace before chord)
494 
495  // if parent not a segment, can't locate a target note
496  if (!chord->parent()->isSegment())
497  return 0;
498 
499  int chordTrack = chord->track();
500  Segment* segm = chord->segment();
501  Part* part = chord->part();
502  if (segm != nullptr)
503  segm = segm->prev1();
504  while (segm) {
505  // if previous segment is a ChordRest segment
506  if (segm->segmentType() == SegmentType::ChordRest) {
507  Chord* target = nullptr;
508  // look for a Chord in the same track
509  if (segm->element(chordTrack) && segm->element(chordTrack)->isChord())
510  target = toChord(segm->element(chordTrack));
511  else { // if no same track, look for other chords in the same instrument
512  for (Element* currChord : segm->elist()) {
513  if (currChord && currChord->isChord() && toChord(currChord)->part() == part) {
514  target = toChord(currChord);
515  break;
516  }
517  }
518  }
519  // if we found a target previous chord
520  if (target) {
521  // if chord has grace notes after, the last one is the previous note
522  QVector<Chord*>graces = target->graceNotesAfter();
523  if (graces.size() > 0)
524  return graces.last()->upNote();
525  return target->upNote(); // if no grace after, return top note
526  }
527  }
528  segm = segm->prev1();
529  }
530  qDebug("no first note for glissando found");
531  return 0;
532  }

534 //---------------------------------------------------------
535 // STATIC FUNCTIONS: guessFinalNote
536 //
537 // Used while dropping a glissando on a note to determine (guess!) the glissando final
538 // note from its initial chord.
539 // Returns the top note of next chord of the same instrument,
540 // preferring the chord in the same track as chord, if it exists.
541 //
542 // Parameter: chord: the chord this glissando start from
543 // Returns: the top note in a suitable following chord or nullptr if none found
544 //---------------------------------------------------------
545 
546 Note* Glissando::guessFinalNote(Chord* chord)
547  {
548  switch (chord->noteType()) {
549 // case NoteType::INVALID:
550 // return nullptr;
551  // for grace notes before, return top note of parent chord
552  // TODO : if the grace-before is not the LAST ONE, this still returns the main note
553  // which is probably not correct; however a glissando between two grace notes
554  // probably makes little sense.
555  case NoteType::ACCIACCATURA:
556  case NoteType::APPOGGIATURA:
557  case NoteType::GRACE4:
558  case NoteType::GRACE16:
559  case NoteType::GRACE32:
560  if (chord->parent() && chord->parent()->isChord())
561  return toChord(chord->parent())->upNote();
562  else // no parent or parent is not a chord?
563  return nullptr;
564  // for grace notes after, next chord is next chord of parent chord
565  // TODO : same note as case above!
566  case NoteType::GRACE8_AFTER:
567  case NoteType::GRACE16_AFTER:
568  case NoteType::GRACE32_AFTER:
569  // move unto parent chord and proceed to standard case
570  if (chord->parent() && chord->parent()->isChord())
571  chord = toChord(chord->parent());
572  else
573  return 0;
574  break;
575  case NoteType::NORMAL:
576  {
577  // if chord has grace notes after, the first one is the next note
578  QVector<Chord*>graces = chord->graceNotesAfter();
579  if (graces.size() > 0)
580  return graces.first()->upNote();
581  }
582  break;
583  default:
584  break;
585  }
586 
587  // standard case (NORMAL or grace after chord)
588 
589  // if parent not a segment, can't locate a target note
590  if (!chord->parent()->isSegment())
591  return 0;
592 
593  // look for first ChordRest segment after initial note is elapsed
594  Segment* segm = chord->score()->tick2rightSegment(chord->tick() + chord->actualTicks());
595  int chordTrack = chord->track();
596  Part* part = chord->part();
597  while (segm) {
598  // if next segment is a ChordRest segment
599  if (segm->segmentType() == SegmentType::ChordRest) {
600  Chord* target = nullptr;
601 
602  // look for a Chord in the same track
603  if (segm->element(chordTrack) && segm->element(chordTrack)->isChord())
604  target = toChord(segm->element(chordTrack));
605  else { // if no same track, look for other chords in the same instrument
606  for (Element* currChord : segm->elist()) {
607  if (currChord && currChord->isChord() && toChord(currChord)->part() == part) {
608  target = toChord(currChord);
609  break;
610  }
611  }
612  }
613 
614  // if we found a target next chord
615  if (target) {
616  // if chord has grace notes before, the first one is the next note
617  QVector<Chord*>graces = target->graceNotesBefore();
618  if (graces.size() > 0)
619  return graces.first()->upNote();
620  return target->upNote(); // if no grace before, return top note
621  }
622  }
623  segm = segm->next1();
624  }
625  qDebug("no second note for glissando found");
626  return 0;
627  }
