75 Beam::Mode Groups::endBeam(ChordRest* cr, ChordRest* prev)
76  {
77  if (cr->isGrace() || cr->beamMode() != Beam::Mode::AUTO)
78  return cr->beamMode();
79  Q_ASSERT(cr->staff());
80 
81  TDuration d = cr->durationType();
82  const Groups& g = cr->staff()->group(cr->tick());
83  Fraction stretch = cr->staff()->timeStretch(cr->tick());
84  Fraction tick = cr->rtick() * stretch;
85 
86  Beam::Mode val = g.beamMode(tick.ticks(), d.type());
87 
88  // context-dependent checks
89  if (val == Beam::Mode::AUTO && tick.isNotZero()) {
90  // if current or previous cr is in tuplet (but not both in same tuplet):
91  // consider it as if this were next shorter duration
92  if (prev && (cr->tuplet() != prev->tuplet()) && (d == prev->durationType())) {
93  if (d >= TDuration::DurationType::V_EIGHTH)
94  val = g.beamMode(tick.ticks(), TDuration::DurationType::V_16TH);
95  else if (d == TDuration::DurationType::V_16TH)
96  val = g.beamMode(tick.ticks(), TDuration::DurationType::V_32ND);
97  else
98  val = g.beamMode(tick.ticks(), TDuration::DurationType::V_64TH);
99  }
100  // if there is a hole between previous and current cr, break beam
101  // exclude tuplets from this check; tick calculations can be unreliable
102  // and they seem to be handled well anyhow
103  if (cr->voice() && prev && !prev->tuplet() && prev->tick() + prev->actualTicks() < cr->tick())
104  val = Beam::Mode::BEGIN;
105  }
106 
107  return val;
108  }

115 Beam::Mode Groups::beamMode(int tick, TDuration::DurationType d) const
116  {
117  int shift;
118  switch (d) {
119  case TDuration::DurationType::V_EIGHTH: shift = 0; break;
120  case TDuration::DurationType::V_16TH: shift = 4; break;
121  case TDuration::DurationType::V_32ND: shift = 8; break;
122  default:
123  return Beam::Mode::AUTO;
124  }
125  const int dm = MScore::division / 8;
126  for (const GroupNode& e : *this) {
127  if (e.pos * dm < tick)
128  continue;
129  if (e.pos * dm > tick)
130  break;
131 
132  int action = (e.action >> shift) & 0xf;
133  switch (action) {
134  case 0: return Beam::Mode::AUTO;
135  case 1: return Beam::Mode::BEGIN;
136  case 2: return Beam::Mode::BEGIN32;
137  case 3: return Beam::Mode::BEGIN64;
138  default:
139  qDebug(" Groups::beamMode: bad action %d", action);
140  return Beam::Mode::AUTO;
141  }
142  }
143  return Beam::Mode::AUTO;
144  }

150 const Groups& Groups::endings(const Fraction& f)
151  {
152  for (const NoteGroup& g : noteGroups) {
153  if (g.timeSig.identical(f)) {
154  return g.endings;
155  }
156  }
157  NoteGroup g;
158  g.timeSig = f;
159  noteGroups.push_back(g);
160 
161  int pos = 0;
162  switch (f.denominator()) {
163  case 2: pos = 16; break;
164  case 4: pos = 8; break;
165  case 8: pos = 4; break;
166  case 16: pos = 2; break;
167  case 32: pos = 1; break;
168  break;
169  }
170  for (int i = 1; i < f.numerator(); ++i) {
171  GroupNode n;
172  n.pos = pos * i;
173  n.action = 0x111;
174  g.endings.push_back(n);
175  }
176  return noteGroups.back().endings;
177  }

216 void Groups::addStop(int pos, TDuration::DurationType d, Beam::Mode bm)
217  {
218  int shift;
219  switch (d) {
220  case TDuration::DurationType::V_EIGHTH: shift = 0; break;
221  case TDuration::DurationType::V_16TH: shift = 4; break;
222  case TDuration::DurationType::V_32ND: shift = 8; break;
223  default:
224  return;
225  }
226  int action;
227  if (bm == Beam::Mode::BEGIN)
228  action = 1;
229  else if (bm == Beam::Mode::BEGIN32)
230  action = 2;
231  else if (bm == Beam::Mode::BEGIN64)
232  action = 3;
233  else
234  return;
235 
236  pos /= 60;
237  action <<= shift;
238 
239  auto i = begin();
240  for (; i != end(); ++i) {
241  if (i->pos == pos) {
242  i->action = (i->action & ~(0xf << shift)) | action;
243  return;
244  }
245  if (i->pos > pos)
246  break;
247  }
248  insert(i, GroupNode({ pos, action }));
249  }
