50 /// root note and bass note are notated as "tonal pitch class":
51 /// <table>
52 /// <tr><td>&nbsp;</td><td>bb</td><td> b</td><td> -</td><td> #</td><td>##</td></tr>
53 /// <tr><td>C</td> <td> 0</td><td> 7</td><td>14</td><td>21</td><td>28</td></tr>
54 /// <tr><td>D</td> <td> 2</td><td> 9</td><td>16</td><td>23</td><td>30</td></tr>
55 /// <tr><td>E</td> <td> 4</td><td>11</td><td>18</td><td>25</td><td>32</td></tr>
56 /// <tr><td>F</td> <td>-1</td><td> 6</td><td>13</td><td>20</td><td>27</td></tr>
57 /// <tr><td>G</td> <td> 1</td><td> 8</td><td>15</td><td>22</td><td>29</td></tr>
58 /// <tr><td>A</td> <td> 3</td><td>10</td><td>17</td><td>24</td><td>31</td></tr>
59 /// <tr><td>B</td> <td> 5</td><td>12</td><td>19</td><td>26</td><td>33</td></tr></table>
60 //
