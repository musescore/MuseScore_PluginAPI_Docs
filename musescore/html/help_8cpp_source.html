68 void HelpQuery::textChanged(const QString& ss)
69  {
70  QString s = ss.toLower();
71  QWidget* menu = static_cast<QWidget*>(parent());
72  if (s.isEmpty()) {
73  if (!emptyState) { // restore old menu entries
74  QList<QAction*> al = menu->actions();
75  for (QAction* a : qAsConst(al)) {
76  if (a != this)
77  menu->removeAction(a);
78  }
79  for (QAction* a : qAsConst(actions)) {
80  if (a != this)
81  menu->addAction(a);
82  }
83  }
84  emptyState = true;
85  return;
86  }
87  if (emptyState)
88  actions = menu->actions();
89  for (QAction* a : menu->actions()) {
90  if (a != this)
91  menu->removeAction(a);
92  }
93  emptyState = false;
94  if (!mscore->helpEngine())
95  return;
96  QMap<QString,QUrl>list = mscore->helpEngine()->linksForIdentifier(s);
97 // QMap<QString,QUrl>list = mscore->helpEngine()->indexModel()->linksForKeyword(s);
98  int k = 0;
99  for (auto i = list.begin(); i != list.end(); ++i) {
100  QAction* action = new QAction(i.key(), this);
101  action->setData(i.value());
102 //printf("add action <%s> <%s>\n", qPrintable(i.key()), qPrintable(i.value().toString()));
103  menu->addAction(action);
104  connect(action, SIGNAL(triggered()), mapper, SLOT(map()));
105  mapper->setMapping(action, action);
106  if (++k > 10)
107  break;
108  }
109  }
