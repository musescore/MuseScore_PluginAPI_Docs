177 void genIcons()
178  {
179  for (int i = 0; i < int(Icons::voice1_ICON); ++i) {
180  QIcon* icon = new QIcon(new MIconEngine);
181  icon->addFile(iconPath + iconNames[i]);
182  icons[i] = icon;
183  if (icon->isNull() || icon->pixmap(12).isNull()) {
184  qDebug("cannot load Icon <%s>", qPrintable(iconPath + iconNames[i]));
185  }
186  }
187 
188  static const char* vtext[VOICES] = { "1","2","3","4" };
189  int iw = preferences.getInt(PREF_UI_THEME_ICONHEIGHT) * 2 / 3; // 16;
190  int ih = preferences.getInt(PREF_UI_THEME_ICONHEIGHT); // 24;
191  for (int i = 0; i < VOICES; ++i) {
192  icons[int(Icons::voice1_ICON) + i] = new QIcon;
193  QPixmap image(iw, ih);
194  QColor c(MScore::selectColor[i].lighter(180));
195  image.fill(c);
196  QPainter painter(&image);
197  painter.setFont(QFont("FreeSans", 8));
198  painter.setRenderHint(QPainter::Antialiasing);
199  painter.setRenderHint(QPainter::TextAntialiasing);
200  painter.setPen(QPen(Qt::black));
201  painter.drawText(QRect(0, 0, iw, ih), Qt::AlignCenter, vtext[i]);
202  painter.end();
203  icons[int(Icons::voice1_ICON) +i]->addPixmap(image);
204 
205  painter.begin(&image);
206  c = QColor(MScore::selectColor[i].lighter(140));
207  painter.fillRect(0, 0, iw, ih, c);
208  painter.setPen(QPen(Qt::black));
209  painter.drawText(QRect(0, 0, iw, ih), Qt::AlignCenter, vtext[i]);
210  painter.end();
211  icons[int(Icons::voice1_ICON) + i]->addPixmap(image, QIcon::Normal, QIcon::On);
212  }
213  }
214 }
