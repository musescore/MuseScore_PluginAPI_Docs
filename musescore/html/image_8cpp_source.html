114 void Image::draw(QPainter* painter) const
115  {
116  bool emptyImage = false;
117  if (imageType == ImageType::SVG) {
118  if (!svgDoc)
119  emptyImage = true;
120  else
121  svgDoc->render(painter, bbox());
122  }
123  else if (imageType == ImageType::RASTER) {
124  if (rasterDoc == nullptr)
125  emptyImage = true;
126  else {
127  painter->save();
128  QSizeF s;
129  if (_sizeIsSpatium)
130  s = _size * spatium();
131  else
132  s = _size * DPMM;
133  if (score() && score()->printing() && !MScore::svgPrinting) {
134  // use original image size for printing, but not for svg for reasonable file size.
135  painter->scale(s.width() / rasterDoc->width(), s.height() / rasterDoc->height());
136  painter->drawPixmap(QPointF(0, 0), QPixmap::fromImage(*rasterDoc));
137  }
138  else {
139  QTransform t = painter->transform();
140  QSize ss = QSizeF(s.width() * t.m11(), s.height() * t.m22()).toSize();
141  t.setMatrix(1.0, t.m12(), t.m13(), t.m21(), 1.0, t.m23(), t.m31(), t.m32(), t.m33());
142  painter->setWorldTransform(t);
143  if ((buffer.size() != ss || _dirty) && rasterDoc && !rasterDoc->isNull()) {
144  buffer = QPixmap::fromImage(rasterDoc->scaled(ss, Qt::IgnoreAspectRatio, Qt::SmoothTransformation));
145  _dirty = false;
146  }
147  if (buffer.isNull())
148  emptyImage = true;
149  else
150  painter->drawPixmap(QPointF(0.0, 0.0), buffer);
151  }
152  painter->restore();
153  }
154  }
155  if (emptyImage) {
156  painter->setBrush(Qt::NoBrush);
157  painter->setPen(Qt::black);
158  painter->drawRect(bbox());
159  painter->drawLine(0.0, 0.0, bbox().width(), bbox().height());
160  painter->drawLine(bbox().width(), 0.0, 0.0, bbox().height());
161  }
162  if (selected() && !(score() && score()->printing())) {
163  painter->setBrush(Qt::NoBrush);
164  painter->setPen(MScore::selectColor[0]);
165  painter->drawRect(bbox());
166  }
167  }

173 void Image::write(XmlWriter& xml) const
174  {
175  // attempt to convert the _linkPath to a path relative to the score
176  //
177  // TODO : on Save As, score()->fileInfo() still contains the old path and fname
178  // if the Save As path is different, image relative path will be wrong!
179  //
180  QString relativeFilePath= QString();
181  if (!_linkPath.isEmpty() && _linkIsValid) {
182  QFileInfo fi(_linkPath);
183  // score()->fileInfo()->canonicalPath() would be better
184  // but we are saving under a temp file name and the 'final' file
185  // might not exist yet, so canonicalFilePath() may return only "/"
186  // OTOH, the score 'final' file name is practically always canonical, at this point
187  QString scorePath = score()->masterScore()->fileInfo()->absolutePath();
188  QString imgFPath = fi.canonicalFilePath();
189  // if imgFPath is in (or below) the directory of scorePath
190  if (imgFPath.startsWith(scorePath, Qt::CaseSensitive)) {
191  // relative img path is the part exceeding scorePath
192  imgFPath.remove(0, scorePath.size());
193  if(imgFPath.startsWith('/'))
194  imgFPath.remove(0, 1);
195  relativeFilePath = imgFPath;
196  }
197  // try 1 level up
198  else {
199  // reduce scorePath by one path level
200  fi.setFile(scorePath);
201  scorePath = fi.path();
202  // if imgFPath is in (or below) the directory up the score directory
203  if (imgFPath.startsWith(scorePath, Qt::CaseSensitive)) {
204  // relative img path is the part exceeding new scorePath plus "../"
205  imgFPath.remove(0, scorePath.size());
206  if (!imgFPath.startsWith('/'))
207  imgFPath.prepend('/');
208  imgFPath.prepend("..");
209  relativeFilePath = imgFPath;
210  }
211  }
212  }
213  // if no match, use full _linkPath
214  if (relativeFilePath.isEmpty())
215  relativeFilePath = _linkPath;
216 
217  xml.stag(this);
218  BSymbol::writeProperties(xml);
219  // keep old "path" tag, for backward compatibility and because it is used elsewhere
220  // (for instance by Box:read(), Measure:read(), Note:read(), ...)
221  xml.tag("path", _storeItem ? _storeItem->hashName() : relativeFilePath);
222  xml.tag("linkPath", relativeFilePath);
223 
224  writeProperty(xml, Pid::AUTOSCALE);
225  writeProperty(xml, Pid::SIZE);
226  writeProperty(xml, Pid::LOCK_ASPECT_RATIO);
227  writeProperty(xml, Pid::SIZE_IS_SPATIUM);
228 
229  xml.etag();
230  }

236 void Image::read(XmlReader& e)
237  {
238  if (score()->mscVersion() <= 114)
239  _sizeIsSpatium = false;
240 
241  while (e.readNextStartElement()) {
242  const QStringRef& tag(e.name());
243  if (tag == "autoScale")
244  readProperty(e, Pid::AUTOSCALE);
245  else if (tag == "size")
246  readProperty(e, Pid::SIZE);
247  else if (tag == "lockAspectRatio")
248  readProperty(e, Pid::LOCK_ASPECT_RATIO);
249  else if (tag == "sizeIsSpatium")
250  // setting this using the property Pid::SIZE_IS_SPATIUM breaks, because the
251  // property setter attempts to maintain a constant size. If we're reading, we
252  // don't want to do that, because the stored size will be in:
253  // mm if size isn't spatium
254  // sp if size is spatium
255  _sizeIsSpatium = e.readBool();
256  else if (tag == "path")
257  _storePath = e.readElementText();
258  else if (tag == "linkPath")
259  _linkPath = e.readElementText();
260  else if (tag == "subtype") // obsolete
261  e.skipCurrentElement();
262  else if (!BSymbol::readProperties(e))
263  e.unknown();
264  }
265 
266  // once all paths are read, load img or retrieve it from store
267  // loading from file is tried first to update the stored image, if necessary
268 
269  qDebug("linkPath <%s>", qPrintable(_linkPath));
270  qDebug("storePath <%s>", qPrintable(_storePath));
271 
272  QString path;
273  bool loaded = false;
274  // if a store path is given, attempt to get the image from the store
275  if (!_storePath.isEmpty()) {
276  _storeItem = imageStore.getImage(_storePath);
277  if (_storeItem) {
278  _storeItem->reference(this);
279  loaded = true;
280  }
281  // if no image in store, attempt to load from path (for backward compatibility)
282  else
283  loaded = load(_storePath);
284  path = _storePath;
285  }
286  // if no success from store path, attempt loading from link path (for .mscx files)
287  if (!loaded) {
288  _linkIsValid = load(_linkPath);
289  path = _linkPath;
290  }
291 
292  if (path.endsWith(".svg"))
293  setImageType(ImageType::SVG);
294  else
295  setImageType(ImageType::RASTER);
296  }

304 bool Image::load(const QString& ss)
305  {
306  qDebug("Image::load <%s>", qPrintable(ss));
307  QString path(ss);
308  // if file path is relative, prepend score path
309  QFileInfo fi(path);
310  if (fi.isRelative()) {
311  path.prepend(score()->masterScore()->fileInfo()->absolutePath() + "/");
312  fi.setFile(path);
313  }
314 
315  _linkIsValid = false; // assume link fname is invalid
316  QFile f(path);
317  if (!f.open(QIODevice::ReadOnly)) {
318  qDebug("Image::load<%s> failed", qPrintable(path));
319  return false;
320  }
321  QByteArray ba = f.readAll();
322  f.close();
323 
324  _linkIsValid = true;
325  _linkPath = fi.canonicalFilePath();
326  _storeItem = imageStore.add(_linkPath, ba);
327  _storeItem->reference(this);
328  if (path.endsWith(".svg"))
329  setImageType(ImageType::SVG);
330  else
331  setImageType(ImageType::RASTER);
332  return true;
333  }

372 void Image::editDrag(EditData& ed)
373  {
374  qreal ratio = _size.width() / _size.height();
375  qreal dx = ed.delta.x();
376  qreal dy = ed.delta.y();
377  if (_sizeIsSpatium) {
378  qreal _spatium = spatium();
379  dx /= _spatium;
380  dy /= _spatium;
381  }
382  else {
383  dx /= DPMM;
384  dy /= DPMM;
385  }
386  if (ed.curGrip == Grip::START) {
387  _size.setWidth(_size.width() + dx);
388  if (_lockAspectRatio)
389  _size.setHeight(_size.width() / ratio);
390  }
391  else {
392  _size.setHeight(_size.height() + dy);
393  if (_lockAspectRatio)
394  _size.setWidth(_size.height() * ratio);
395  }
396  layout();
397  }

434 void Image::layout()
435  {
436  setPos(0.0, 0.0);
437  if (imageType == ImageType::SVG && !svgDoc) {
438  if (_storeItem)
439  svgDoc = new QSvgRenderer(_storeItem->buffer());
440  }
441  else if (imageType == ImageType::RASTER && !rasterDoc) {
442  if (_storeItem) {
443  rasterDoc = new QImage;
444  rasterDoc->loadFromData(_storeItem->buffer());
445  if (!rasterDoc->isNull())
446  _dirty = true;
447  }
448  }
449  if (_size.isNull())
450  _size = pixel2size(imageSize());
451 
452  // if autoscale && inside a box, scale to box relevant size
453  if (autoScale() && parent() && ((parent()->isHBox() || parent()->isVBox()))) {
454  if (_lockAspectRatio) {
455  qreal f = _sizeIsSpatium ? spatium() : DPMM;
456  QSizeF size(imageSize());
457  qreal ratio = size.width() / size.height();
458  qreal w = parent()->width();
459  qreal h = parent()->height();
460  if ((w / h) < ratio) {
461  _size.setWidth(w / f);
462  _size.setHeight((w / ratio) / f);
463  }
464  else {
465  _size.setHeight(h / f);
466  _size.setWidth(h * ratio / f);
467  }
468  }
469  else
470  _size = pixel2size(parent()->bbox().size());
471  }
472 
473  // in any case, adjust position relative to parent
474  setbbox(QRectF(QPointF(), size2pixel(_size)));
475  }
