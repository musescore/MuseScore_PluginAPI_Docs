72  Convert MusicXML \a step / \a alter / \a octave to midi pitch,
73  set pitch and tpc.
74  */
75 
76 static void xmlSetPitch(Ms::Note* n, char step, int alter, int octave)
77  {
78  int istep = step - 'A';
79  // a b c d e f g
80  static int table[7] = { 9, 11, 0, 2, 4, 5, 7 };
81  if (istep < 0 || istep > 6) {
82  qDebug("xmlSetPitch: illegal pitch %d, <%c>", istep, step);
83  return;
84  }
85  int pitch = table[istep] + alter + (octave+1) * 12;
86 
87  if (pitch < 0)
88  pitch = 0;
89  if (pitch > 127)
90  pitch = 127;
91 
92  n->setPitch(pitch);
93 
94  // a b c d e f g
95  static int table1[7] = { 5, 6, 0, 1, 2, 3, 4 };
96  int tpc = step2tpc(table1[istep], Ms::AccidentalVal(alter));
97  n->setTpc(tpc);
98  }

207 void MsScWriter::beginMeasure(const Bww::MeasureBeginFlags mbf)
208  {
209  qDebug() << "MsScWriter::beginMeasure()";

211 
212  // create a new measure
213  currentMeasure = new Ms::Measure(score);
214  currentMeasure->setTick(tick);
215  currentMeasure->setTimesig(Ms::Fraction(beats, beat));
216  currentMeasure->setNo(measureNumber);
217  score->measures()->add(currentMeasure);
218 
219  if (mbf.repeatBegin)
220  currentMeasure->setRepeatStart(true);
221 
222  if (mbf.irregular)
223  currentMeasure->setIrregular(true);
224 
225  if (mbf.endingFirst || mbf.endingSecond) {
226  Ms::Volta* volta = new Ms::Volta(score);
227  volta->setTrack(0);
228  volta->endings().clear();
229  if (mbf.endingFirst) {
230  volta->setText("1");
231  volta->endings().append(1);
232  ending = 1;
233  }
234  else {
235  volta->setText("2");
236  volta->endings().append(2);
237  ending = 2;
238  }
239  volta->setTick(currentMeasure->tick());
240  score->addElement(volta);
241  lastVolta = volta;
242  }
243 
244  // set clef, key and time signature in the first measure
245  if (measureNumber == 1) {
246  // clef
247  Ms::Clef* clef = new Ms::Clef(score);
248  clef->setClefType(Ms::ClefType::G);
249  clef->setTrack(0);
250  Ms::Segment* s = currentMeasure->getSegment(Ms::SegmentType::Clef, tick);
251  s->add(clef);
252  // keysig
253  Ms::KeySigEvent key;
254  key.setKey(Ms::Key::D);
255  Ms::KeySig* keysig = new Ms::KeySig(score);
256  keysig->setKeySigEvent(key);
257  keysig->setTrack(0);
258  s = currentMeasure->getSegment(Ms::SegmentType::KeySig, tick);
259  s->add(keysig);
260  // timesig
261  Ms::TimeSig* timesig = new Ms::TimeSig(score);
262  timesig->setSig(Ms::Fraction(beats, beat));
263  timesig->setTrack(0);
264  s = currentMeasure->getSegment(Ms::SegmentType::TimeSig, tick);
265  s->add(timesig);
266  qDebug("tempo %d", tempo);
267  }

319 void MsScWriter::note(const QString pitch, const QVector<Bww::BeamType> beamList,
320  const QString type, const int dots,
321  bool tieStart, bool /*TODO tieStop */,
322  StartStop triplet,
323  bool grace)
324  {
325  qDebug() << "MsScWriter::note()"
326  << "type:" << type
327  << "dots:" << dots
328  << "grace" << grace
329  ;
330 
331  if (!stepAlterOctMap.contains(pitch)
332  || !typeMap.contains(type)) {
333  // TODO: error message
334  return;
335  }
336  StepAlterOct sao = stepAlterOctMap.value(pitch);
337 
338  int ticks = 4 * Ms::MScore::division / type.toInt();
339  if (dots) ticks = 3 * ticks / 2;
340  qDebug() << "ticks:" << ticks;
341  Ms::TDuration durationType(Ms::TDuration::DurationType::V_INVALID);
342  durationType.setVal(ticks);
343  qDebug() << "duration:" << durationType.name();
344  if (triplet != ST_NONE) ticks = 2 * ticks / 3;
345 
346  Ms::Beam::Mode bm = (beamList.at(0) == Bww::BM_BEGIN) ? Ms::Beam::Mode::BEGIN : Ms::Beam::Mode::AUTO;
347  Ms::Direction sd = Ms::Direction::AUTO;
348 
349  // create chord
350  Ms::Chord* cr = new Ms::Chord(score);
351  //ws cr->setTick(tick);
352  cr->setBeamMode(bm);
353  cr->setTrack(0);
354  if (grace) {
355  cr->setNoteType(Ms::NoteType::GRACE32);
356  cr->setDurationType(Ms::TDuration::DurationType::V_32ND);
357  sd = Ms::Direction::UP;
358  }
359  else {
360  if (durationType.type() == Ms::TDuration::DurationType::V_INVALID)
361  durationType.setType(Ms::TDuration::DurationType::V_QUARTER);
362  cr->setDurationType(durationType);
363  sd = Ms::Direction::DOWN;
364  }
365  cr->setTicks(durationType.fraction());
366  cr->setDots(dots);
367  cr->setStemDirection(sd);
368  // add note to chord
369  Ms::Note* note = new Ms::Note(score);
370  note->setTrack(0);
371  xmlSetPitch(note, sao.s.toLatin1(), sao.a, sao.o);
372  if (tieStart) {
373  Ms::Tie* tie = new Ms::Tie(score);
374  note->setTieFor(tie);
375  tie->setStartNote(note);
376  tie->setTrack(0);
377  }
378  cr->add(note);
379  // add chord to measure
380  if (!grace) {
381  Ms::Segment* s = currentMeasure->getSegment(Ms::SegmentType::ChordRest, tick);
382  s->add(cr);
383  if (!currentGraceNotes.isEmpty()) {
384  for (int i = currentGraceNotes.size() - 1; i >=0; i--)
385  cr->add(currentGraceNotes.at(i));
386  currentGraceNotes.clear();
387  }
388  doTriplet(cr, triplet);
389  Ms::Fraction tickBefore = tick;
390  tick += Ms::Fraction::fromTicks(ticks);
391  Ms::Fraction nl(tick - currentMeasure->tick());
392  currentMeasure->setTicks(nl);
393  qDebug() << "MsScWriter::note()"
394  << "tickBefore:" << tickBefore
395  << "tick:" << tick
396  << "nl:" << nl.print()
397  ;
398  }
399  else {
400  currentGraceNotes.append(cr);
401  }
402  }

482 void MsScWriter::doTriplet(Ms::Chord* cr, StartStop triplet)

484  qDebug() << "MsScWriter::doTriplet(" << triplet << ")"
485  ;
486 
487  if (triplet == ST_START) {
488  tuplet = new Ms::Tuplet(score);
489  tuplet->setTrack(0);
490  tuplet->setRatio(Ms::Fraction(3, 2));
491 // tuplet->setTick(tick);
492  currentMeasure->add(tuplet);
493  }
494  else if (triplet == ST_STOP) {
495  if (tuplet) {
496  cr->setTuplet(tuplet);
497  tuplet->add(cr);
498  tuplet = 0;
499  }
500  else
501  qDebug("BWW::import: triplet stop without triplet start");
502  }
503  else if (triplet == ST_CONTINUE) {
504  if (!tuplet)
505  qDebug("BWW::import: triplet continue without triplet start");
506  }
507  else if (triplet == ST_NONE) {
508  if (tuplet)
509  qDebug("BWW::import: triplet none inside triplet");
510  }
511  else
512  qDebug("unknown triplet type %d", triplet);
513  if (tuplet) {
514  cr->setTuplet(tuplet);
515  tuplet->add(cr);
516  }
517  }
