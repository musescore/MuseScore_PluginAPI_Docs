90 bool doNotesOverlap(const MTrack &track)
91  {
92  const auto &chords = track.chords;
93  for (auto i1 = chords.begin(); i1 != chords.end(); ++i1) {
94  const auto &chord1 = i1->second;
95  for (auto noteIt1 = chord1.notes.begin();
96  noteIt1 != chord1.notes.end(); ++noteIt1) {
97  for (auto noteIt2 = std::next(noteIt1);
98  noteIt2 != chord1.notes.end(); ++noteIt2) {
99  if (noteIt2->pitch == noteIt1->pitch)
100  return true;
101  }
102  for (auto i2 = std::next(i1); i2 != chords.end(); ++i2) {
103  if (i2->first >= noteIt1->offTime)
104  break;
105  const auto &chord2 = i2->second;
106  if (chord1.voice != chord2.voice)
107  continue;
108  for (const auto &note2: chord2.notes) {
109  if (note2.pitch == noteIt1->pitch)
110  return true;
111  }
112  }
113  }
114  }
115  return false;
116  }

167 void findAllTupletsForDrums(
168  MTrack &mtrack,
169  TimeSigMap *sigmap,
170  const ReducedFraction &basicQuant)
171  {
172  const size_t drumVoiceCount = 2;
173  // drum track has 2 voices (stem up and stem down),
174  // and tuplet detection is applicable for single voice drum tracks,
175  // so split track chords into 2 voice groups,
176  // detect tuplets and merge chords (and found tuplets) back;
177  // it's a small hack due to the fact that tuplet detection
178  // is designed to work before voice setting
179 
180  std::vector<std::multimap<ReducedFraction, MidiChord> > chords(drumVoiceCount);
181  for (const auto &chord: mtrack.chords) {
182  const MidiChord &c = chord.second;
183 
184  Q_ASSERT(c.voice == 0 || c.voice == 1);
185 
186  chords[c.voice].insert({chord.first, c});
187  }
188 
189  std::vector<std::multimap<ReducedFraction,
190  MidiTuplet::TupletData> > tuplets(drumVoiceCount);
191  for (size_t voice = 0; voice < drumVoiceCount; ++voice) {
192  if (!chords[voice].empty())
193  MidiTuplet::findAllTuplets(tuplets[voice], chords[voice], sigmap, basicQuant);
194  }
195  mtrack.chords.clear();
196  for (size_t voice = 0; voice < drumVoiceCount; ++voice) {
197  for (auto &chord: chords[voice]) {
198  // correct voice because it can be changed during tuplet detection
199  setChordVoice(chord.second, int(voice));
200  mtrack.chords.insert({chord.first, chord.second});
201  }
202  }
203  mtrack.updateTupletsFromChords();
204  // note: temporary local tuplets and chords are deleted here
205  }

208  TimeSigMap *sigmap,

230  opers.data()->trackOpers.quantValue.value(mtrack.indexOfOperation));
231 
232  Q_ASSERT_X(MChord::isLastTickValid(lastTick, mtrack.chords),
233  "quantizeAllTracks", "Last tick is less than max note off time");
234 
235  MChord::setBarIndexes(mtrack.chords, basicQuant, lastTick, sigmap);
236 
237  if (mtrack.mtrack->drumTrack())
238  findAllTupletsForDrums(mtrack, sigmap, basicQuant);
239  else
240  MidiTuplet::findAllTuplets(mtrack.tuplets, mtrack.chords, sigmap, basicQuant);
241 
242  Q_ASSERT_X(!doNotesOverlap(track.second),
243  "quantizeAllTracks",
244  "There are overlapping notes of the same voice that is incorrect");
245 
246  // (4/3 of the smallest duration) tol is less sensitive
247  // to on time inaccuracies than 1/2 earlier
248  MChord::collectChords(mtrack, {2, 1}, {4, 3});
249  Quantize::quantizeChords(mtrack.chords, sigmap, basicQuant);
250  MidiTuplet::removeEmptyTuplets(mtrack);
251 
252  Q_ASSERT_X(MidiTuplet::areTupletRangesOk(mtrack.chords, mtrack.tuplets),

262 void MTrack::processMeta(int tick, const MidiEvent& mm)
263  {
264  if (!staff) {
265  qDebug("processMeta: no staff");
266  return;
267  }
268  const uchar* data = mm.edata();
269  Score* cs = staff->score();
270 
271  switch (mm.metaType()) {
272  case META_TEXT:
273  case META_LYRIC:
274  break; // lyric and text are added in importmidi_lyrics.cpp
275  case META_TRACK_NAME:
276  {
277  const std::string text = MidiCharset::fromUchar(data);
278 
279  auto &opers = midiImportOperations;
280  if (opers.data()->processingsOfOpenedFile == 0) {
281  const int currentTrack = indexOfOperation;
282  opers.data()->trackOpers.staffName.setValue(currentTrack, text);
283  }
284 
285  if (name.isEmpty())
286  name = MidiCharset::convertToCharset(text);
287  }
288  break;
289  case META_TEMPO: // add later, after adding of notes
290  break;
291  case META_KEY_SIGNATURE:
292  {
293  const signed char key = ((const signed char*)data)[0];
294  if (key < -7 || key > 7) {
295  qDebug("ImportMidi: illegal key %d", key);
296  break;
297  }
298  KeySigEvent ke;
299  ke.setKey(Key(key));
300  staff->setKey(Fraction::fromTicks(tick), ke);
301  hasKey = true;
302  }
303  break;
304  case META_COMPOSER: // mscore extension
305  case META_POET:
306  case META_TRANSLATOR:
307  case META_SUBTITLE:
308  case META_TITLE:
309  {
310  Tid ssid = Tid::DEFAULT;
311  switch(mm.metaType()) {
312  case META_COMPOSER:
313  ssid = Tid::COMPOSER;
314  break;
315  case META_TRANSLATOR:
316  ssid = Tid::TRANSLATOR;
317  break;
318  case META_POET:
319  ssid = Tid::POET;
320  break;
321  case META_SUBTITLE:
322  ssid = Tid::SUBTITLE;
323  break;
324  case META_TITLE:
325  ssid = Tid::TITLE;
326  break;
327  }
328  Text* text = new Text(cs, ssid);
329 
330  text->setPlainText((const char*)(mm.edata()));
331 
332  MeasureBase* measure = cs->first();
333  if (!measure->isVBox()) {
334  measure = new VBox(cs);
335  measure->setTick(Fraction(0,1));
336  measure->setNext(cs->first());
337  cs->measures()->add(measure);
338  }
339  measure->add(text);
340  }
341  break;
342  case META_COPYRIGHT:
343  cs->setMetaTag("copyright", QString((const char*)(mm.edata())));
344  break;
345  case META_TIME_SIGNATURE:
346  break; // added earlier
347  case META_PORT_CHANGE:
348  staff->part()->setMidiChannel(-1, (int)data[0]);
349  break;
350  default:
351  if (MScore::debugMode)
352  qDebug("unknown meta type 0x%02x", mm.metaType());
353  break;
354  }
355  }

357 QList<std::pair<ReducedFraction, TDuration> >
358 MTrack::toDurationList(const Measure *measure,
359  int voice,
360  const ReducedFraction &startTick,
361  const ReducedFraction &len,
362  Meter::DurationType durationType)
363  {
364  // find tuplets over which duration goes
365  auto barTick = ReducedFraction(measure->tick());
366  auto tupletsData = MidiTuplet::findTupletsInBarForDuration(
367  voice, barTick, startTick, len, tuplets);
368  struct {
369  bool operator()(const MidiTuplet::TupletData &d1,
370  const MidiTuplet::TupletData &d2)
371  {
372  return (d1.len > d2.len);
373  }
374  } comparator;
375  // sort by tuplet length in desc order
376  sort(tupletsData.begin(), tupletsData.end(), comparator);
377 
378  const ReducedFraction startTickInBar = startTick - barTick;
379  const ReducedFraction endTickInBar = startTickInBar + len;
380 
381  const auto &opers = midiImportOperations;
382  const bool useDots = opers.data()->trackOpers.useDots.value(indexOfOperation);
383  return Meter::toDurationList(startTickInBar, endTickInBar,
384  ReducedFraction(measure->timesig()), tupletsData,
385  durationType, useDots);
386  }

400 void MTrack::fillGapWithRests(Score* score,
401  int voice,
402  const ReducedFraction &startChordTickFrac,
403  const ReducedFraction &restLength,
404  int track)
405  {
406  ReducedFraction startChordTick = startChordTickFrac;
407  ReducedFraction restLen = restLength;
408  while (restLen > ReducedFraction(0, 1)) {
409  ReducedFraction len = restLen;
410  Measure* measure = score->tick2measure(startChordTick.fraction());
411  if (startChordTick >= ReducedFraction(measure->endTick())) {
412  qDebug("tick2measure: %d end of score?", startChordTick.ticks());
413  startChordTick += restLen;
414  restLen = ReducedFraction(0, 1);
415  break;
416  }
417  len = splitDurationOnBarBoundary(len, startChordTick, measure);
418 
419  if (len >= ReducedFraction(measure->ticks())) {
420  // rest to the whole measure
421  len = ReducedFraction(measure->ticks());
422  if (voice == 0) {
423  TDuration duration(TDuration::DurationType::V_MEASURE);
424  Rest* rest = new Rest(score, duration);
425  rest->setTicks(measure->ticks());
426  rest->setTrack(track);
427  Segment* s = measure->getSegment(SegmentType::ChordRest, startChordTick.fraction());
428  s->add(rest);
429  }
430  restLen -= len;
431  startChordTick += len;
432  }
433  else {
434  const auto dl = toDurationList(measure, voice, startChordTick, len,
435  Meter::DurationType::REST);
436  if (dl.isEmpty()) {
437  qDebug("cannot create duration list for len %d", len.ticks());
438  restLen = ReducedFraction(0, 1); // fake
439  break;
440  }
441  for (const auto &durationPair: dl) {
442  const TDuration &duration = durationPair.second;
443  const ReducedFraction &tupletRatio = durationPair.first;
444  len = ReducedFraction(duration.fraction()) / tupletRatio;
445  Rest* rest = new Rest(score, duration);
446  rest->setTicks(duration.fraction());
447  rest->setTrack(track);
448  Segment* s = measure->getSegment(SegmentType::ChordRest, startChordTick.fraction());
449  s->add(rest);
450  MidiTuplet::addElementToTuplet(voice, startChordTick, len, rest, tuplets);
451  restLen -= len;
452  startChordTick += len;
453  }
454  }
455 
456  }
457  }

514 void MTrack::processPendingNotes(QList<MidiChord> &midiChords,
515  int voice,
516  const ReducedFraction &startChordTickFrac,
517  const ReducedFraction &nextChordTick)
518  {
519  Score* score = staff->score();
520  const int track = staff->idx() * VOICES + voice;
521  const Drumset* drumset = staff->part()->instrument()->drumset();
522  const bool useDrumset = staff->part()->instrument()->useDrumset();
523 
524  const auto& opers = midiImportOperations.data()->trackOpers;
525  const int currentTrack = indexOfOperation;
526 
527  // all midiChords here should have the same onTime value
528  // and all notes in each midiChord should have the same duration
529  ReducedFraction startChordTick = startChordTickFrac;
530  while (!midiChords.isEmpty()) {
531  const ReducedFraction tick = startChordTick;
532  ReducedFraction len = nextChordTick - tick;
533  if (len <= ReducedFraction(0, 1))
534  break;
535  len = MChord::findMinDuration(tick, midiChords, len);
536  Measure* measure = score->tick2measure(tick.fraction());
537  len = splitDurationOnBarBoundary(len, tick, measure);
538 
539  const auto dl = toDurationList(measure, voice, tick, len, Meter::DurationType::NOTE);
540  if (dl.isEmpty())
541  break;
542  const TDuration &d = dl[0].second;
543  const ReducedFraction &tupletRatio = dl[0].first;
544  len = ReducedFraction(d.fraction()) / tupletRatio;
545 
546  Chord* chord = new Chord(score);
547  chord->setTrack(track);
548  chord->setDurationType(d);
549  chord->setTicks(d.fraction());
550 
551  if (opers.showStaccato.value(currentTrack)
552  && startChordTick == startChordTickFrac // first chord in tied chord sequence
553  && midiChords.begin()->isStaccato()) {
554  Articulation* a = new Articulation(chord->score());
555  a->setSymId(SymId::articStaccatoAbove);
556  chord->add(a);
557  }
558 
559  Segment* s = measure->getSegment(SegmentType::ChordRest, tick.fraction());
560  s->add(chord);
561  MidiTuplet::addElementToTuplet(voice, tick, len, chord, tuplets);
562 
563  for (int k = 0; k < midiChords.size(); ++k) {
564  MidiChord& midiChord = midiChords[k];
565  setMusicNotesFromMidi(score, midiChord.notes, chord, drumset, useDrumset);
566  if (!midiChord.notes.empty() && midiChord.notes.first().offTime - tick <= len) {
567  midiChords.removeAt(k);
568  --k;
569  continue;
570  }
571  setTies(chord, score, midiChord.notes);
572  }
573  startChordTick += len;
574  }
575  fillGapWithRests(score, voice, startChordTick, nextChordTick - startChordTick, track);
576  }

600  {

602  // startChordTick is onTime value of all simultaneous notes
603  // chords here are consist of notes with equal durations

609  const auto &nextChordTick = it->first;
610  const MidiChord& midiChord = it->second;
611  if (midiChord.voice != voice) {
612  ++it;
613  continue;
614  }
615  processPendingNotes(midiChords, voice, startChordTick, nextChordTick);
616  // now 'midiChords' list is empty
617  // so - fill it:
618  // collect all midiChords on current tick position
619  startChordTick = nextChordTick; // debug
620  for ( ; it != chords.end(); ++it) {
621  const MidiChord& midiChord1 = it->second;
622  if (it->first != startChordTick)
623  break;
624  if (midiChord1.voice != voice)
625  continue;

627  }
628  if (midiChords.isEmpty())
629  break;
