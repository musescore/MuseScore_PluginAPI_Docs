27 double findChordSalience1(
28  const std::pair<const ReducedFraction, MidiChord> &chord,
29  double ticksPerSec)
30  {
31  ReducedFraction duration(0, 1);
32  int pitch = std::numeric_limits<int>::max();
33  int velocity = 0;
34 
35  for (const MidiNote &note: chord.second.notes) {
36  if (note.offTime - chord.first > duration)
37  duration = note.offTime - chord.first;
38  if (note.pitch < pitch)
39  pitch = note.pitch;
40  velocity += note.velo;
41  }
42  const double durationInSeconds = duration.ticks() / ticksPerSec;
43 
44  const double c4 = 84;
45  const int pmin = 48;
46  const int pmax = 72;
47 
48  if (pitch < pmin)
49  pitch = pmin;
50  else if (pitch > pmax)
51  pitch = pmax;
52 
53  if (velocity <= 0)
54  velocity = 1;
55 
56  return durationInSeconds * (c4 - pitch) * log(velocity);

189  const ::EventList &events,
190  const std::vector<int> &levels,
191  int beatsInBar,
192  double ticksPerSec)
193  {
194  std::map<ReducedFraction, double> saliences;
195  for (const auto &e: events) {
196  saliences.insert({MidiTempo::time2Tick(e.time, ticksPerSec), e.salience});
197  }
198  std::vector<ReducedFraction> beatsOfBar;
199  double matchFrac = 0;
200  int matchCount = 0;
201  int beatCount = 0;
202 
203  for (const auto &beat: beatSet) {
204  beatsOfBar.push_back(beat);
205  ++beatCount;
206  if (beatCount == beatsInBar) {
207  beatCount = 0;
208  int relationCount = 0;
209  int relationMatches = 0;
210  for (size_t i = 0; i != beatsOfBar.size() - 1; ++i) {
211  const auto s1 = saliences.find(beatsOfBar[i]);
212  for (size_t j = i + 1; j != beatsOfBar.size(); ++j) {
213  ++relationCount; // before s1 search check
214  if (s1 == saliences.end())
215  continue;
216  const auto s2 = saliences.find(beatsOfBar[j]);
217  if (s2 == saliences.end())
218  continue;
219  if ((s1->second < s2->second) == (levels[i] < levels[j]))
220  ++relationMatches;
221  }
222  }
223  if (relationCount) {
224  matchFrac += relationMatches * 1.0 / relationCount;
225  ++matchCount;
226  }
227  beatsOfBar.clear();
228  }
229  }
230  if (matchCount)
231  matchFrac /= matchCount;
232 
233  return matchFrac;
