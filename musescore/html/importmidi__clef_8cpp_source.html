100 void createClef(ClefType clefType, Staff* staff, int tick, bool isSmall = false)
101  {
102  if (tick == 0) {
103  staff->setDefaultClefType(ClefTypeList(clefType, clefType));
104  }
105  else {
106  Clef* clef = new Clef(staff->score());
107  clef->setClefType(clefType);
108  const int track = staff->idx() * VOICES;
109  clef->setTrack(track);
110  clef->setGenerated(false);
111  clef->setMag(staff->mag(Fraction::fromTicks(tick)));
112  clef->setSmall(isSmall);
113  Measure* m = staff->score()->tick2measure(Fraction::fromTicks(tick));
114  Segment* seg = m->getSegment(SegmentType::Clef, Fraction::fromTicks(tick));
115  seg->add(clef);
116  }

120  {
121  AveragePitch averagePitch;

183 int findPitchPenaltyForClef(int pitch, int clefIndex)
184  {
185  const int farPitchPenalty = 10000;
186  const int veryFarPitchPenalty = 20000;
187  const int approxPitchPenalty = 1;
188  const int dx = 5;
189 
190  const int midPitch = clefMidPitch(); // all notes equal or upper - better in G clef
191  const int highPitch = midPitch + dx; // almost all notes equal or upper - in G clef
192  const int lowPitch = midPitch - dx; // almost all notes lower - in F clef
193  const int highestPitch = highPitch + dx; // all notes equal or upper - in G clef
194  const int lowestPitch = lowPitch - dx; // all notes lower - in F clef
195 
196  switch (clefIndex) {
197  case 0:
198  if (pitch < lowestPitch)
199  return veryFarPitchPenalty;
200  else if (pitch < lowPitch)
201  return farPitchPenalty;
202  else if (pitch < midPitch)
203  return approxPitchPenalty;
204  break;
205  case 1:
206  if (pitch >= highestPitch)
207  return veryFarPitchPenalty;
208  else if (pitch >= highPitch)
209  return farPitchPenalty;
210  else if (pitch >= midPitch)
211  return approxPitchPenalty;
212  break;
213  default:
214  Q_ASSERT_X(false, "MidiClef::pitchPenalty", "Unknown clef type");
215  break;
216  }
217  return 0;
218  }

220 std::pair<ElementType, ReducedFraction>
221 findChordRest(const Segment *seg, int strack)
222  {
223  ElementType elType = ElementType::INVALID;
224  ReducedFraction newRestLen(0, 1);
225  for (int voice = 0; voice < VOICES; ++voice) {
226  ChordRest *cr = static_cast<ChordRest *>(seg->element(strack + voice));
227  if (!cr)
228  continue;
229  if (cr->isChord()) {
230  elType = ElementType::CHORD;
231  break;
232  }
233  else if (cr->isRest()) {
234  elType = ElementType::REST;
235  newRestLen = qMax(newRestLen, ReducedFraction(cr->globalTicks()));
236  }
237  }
238  return {elType, newRestLen};
239  }

241 int findClefChangePenalty(
242  int pos,
243  int clefIndex,
244  const std::vector<std::vector<int>> &trebleBassPath,
245  const Segment *segment,
246  const Staff *staff)
247  {
248  if (pos == 0)
249  return 0;
250 
251  const int clefChangePenalty = 1000;
252  const int prevClefChangePenalty = 10001;
253  const int orphanChordPenalty = 2;
254  const int notesBetweenClefs = 5; // should be >= 2
255 
256  int j = pos;
257  ReducedFraction totalRestLen(0, 1);
258  int penalty = 0;
259  const int strack = staff->idx() * VOICES;
260  const auto barFraction = ReducedFraction(
261  staff->score()->sigmap()->timesig(segment->tick()).timesig());
262  const ReducedFraction beatLen = Meter::beatLength(barFraction);
263 
264  // find backward penalty
265  for (const Segment *segPrev = segment->prev1(SegmentType::ChordRest); ;
266  segPrev = segPrev->prev1(SegmentType::ChordRest)) {
267  if (!segPrev) {
268  penalty += clefChangePenalty;
269  break;
270  }
271  const auto el = findChordRest(segPrev, strack);
272  if (el.first == ElementType::CHORD) {
273  --j;
274  if (j == pos - notesBetweenClefs)
275  break;
276  if (j == 0 || trebleBassPath[clefIndex][j] != clefIndex) {
277  if (j == pos - 1)
278  penalty += prevClefChangePenalty;
279  else
280  penalty += clefChangePenalty;
281  break;
282  }
283  totalRestLen = {0, 1};
284  }
285  else if (el.first == ElementType::REST) {
286  totalRestLen += el.second;
287  if (totalRestLen >= beatLen) {
288  if (j != pos)
289  penalty += orphanChordPenalty;
290  break;
291  }
292  }
293  }
294  // find forward penalty
295  int chordCounter = 0;
296  for (const Segment *seg = segment; ; seg = seg->next1(SegmentType::ChordRest)) {
297  if (!seg) {
298  penalty += clefChangePenalty;
299  break;
300  }
301  const auto el = findChordRest(seg, strack);
302  if (el.first == ElementType::CHORD) {
303  ++chordCounter;
304  if (chordCounter == notesBetweenClefs)
305  break;
306  totalRestLen = {0, 1};
307  }
308  else if (el.first == ElementType::REST) {
309  totalRestLen += el.second;
310  if (totalRestLen >= beatLen) {
311  penalty += orphanChordPenalty;
312  break;
313  }
314  }
315  }
316 
317  return penalty;
318  }

325 void makeDynamicProgrammingStep(std::vector<std::vector<int>> &penalties,
326  std::vector<std::vector<int>> &optimalPaths,
327  int pos,
328  MidiTie::TieStateMachine::State tieState,
329  const MinMaxPitch &minMaxPitch,
330  const Segment *seg,
331  const Staff *staff)
332  {
333  for (int clefIndex = 0; clefIndex != 2; ++clefIndex)
334  optimalPaths[clefIndex].resize(pos + 1);
335 
336  for (int curClef = 0; curClef != 2; ++curClef) {
337  const int significantPitch = (curClef == 0)
338  ? minMaxPitch.minPitch() : minMaxPitch.maxPitch();
339  const int pitchPenalty = findPitchPenaltyForClef(significantPitch, curClef);
340  int minPenalty = std::numeric_limits<int>::max();
341  int minIndex = 0;
342  for (int prevClef = 0; prevClef != 2; ++prevClef) {
343  int penalty = pitchPenalty;
344  if (prevClef != curClef) {
345  if (tieState == MidiTie::TieStateMachine::State::TIED_BACK
346  || tieState == MidiTie::TieStateMachine::State::TIED_BOTH) {
347  continue; // there is a tie breakage that is incorrect
348  }
349  penalty += findClefChangePenalty(pos, prevClef, optimalPaths, seg, staff);
350  }
351  penalty += (pos > 0) ? penalties[prevClef][(pos + 1) % 2] : 0;
352  if ((prevClef != curClef && penalty < minPenalty)
353  || (prevClef == curClef && penalty <= minPenalty)) {
354  minPenalty = penalty;
355  minIndex = prevClef;
356  }
357  }
358 
359  penalties[curClef][pos % 2] = minPenalty;
360  if (pos > 0)
361  optimalPaths[curClef][pos] = minIndex;
362  }
363  }

367 // will be ignored),

371 bool createClefs(
372  Staff *staff,
373  const std::vector<std::vector<int>> &optimalPaths,
374  int lastClef,
375  const std::vector<Segment *> &segments)
376  {
377  std::vector<std::pair<ClefType, int>> clefsAndTicks;
378 
379  int currentClef = lastClef;
380  for (size_t i = optimalPaths[0].size() - 1; i; --i) {
381  const int prevClef = optimalPaths[currentClef][i];
382  if (prevClef != currentClef) {
383  clefsAndTicks.push_back({clefFromIndex(currentClef), segments[i]->tick().ticks()});
384  currentClef = prevClef;
385  }
386  }
387  if (clefsAndTicks.empty())
388  return false;
389 
390  createClef(clefFromIndex(currentClef), staff, 0); // main staff clef
391  for (size_t i = clefsAndTicks.size(); i; --i) // clef changes
392  createClef(clefsAndTicks[i - 1].first, staff, clefsAndTicks[i - 1].second, true);
393 
394  return true;
395  }

430 void createClefs(Staff *staff, int indexOfOperation, bool isDrumTrack)
431  {
432  const auto &opers = midiImportOperations.data()->trackOpers;
433  const auto &trackInstrList = opers.msInstrList.value(indexOfOperation);
434  const int msInstrIndex = opers.msInstrIndex.value(indexOfOperation);
435  const bool hasInstrument = !trackInstrList.empty() && trackInstrList[msInstrIndex];
436 
437  if (isDrumTrack && !hasInstrument) {
438  createClef(ClefType::PERC, staff, 0);
439  return;
440  }
441 
442  const int strack = staff->idx() * VOICES;
443  bool mainClefWasSet = false;
444  const bool canChangeClef = !hasInstrument || hasGFclefs(trackInstrList[msInstrIndex]);
445 
446  if (opers.changeClef.value(indexOfOperation) && canChangeClef) {
447  MidiTie::TieStateMachine tieTracker;
448 
449  // find optimal clef changes via dynamic programming
450  std::vector<std::vector<int>> penalties(2); // 0 - treble, 1 - bass
451  // remember only 2 last positions to save memory
452  for (size_t i = 0; i != penalties.size(); ++i)
453  penalties[i].resize(2); // 2 = current + prev
454  std::vector<std::vector<int>> optimalPaths(2); // first col is unused
455  std::vector<Segment *> segments;
456 
457  int pos = 0;
458  for (Segment *seg = staff->score()->firstSegment(SegmentType::ChordRest); seg;
459  seg = seg->next1(SegmentType::ChordRest)) {
460 
461  const auto minMaxPitch = findMinMaxSegPitch(seg, strack);
462  if (minMaxPitch.empty()) // no chords
463  continue;
464  tieTracker.addSeg(seg, strack);
465  segments.push_back(seg);
466 
467  makeDynamicProgrammingStep(penalties, optimalPaths, pos,
468  tieTracker.state(), minMaxPitch, seg, staff);
469  ++pos;
470  }
471 
472  if (!optimalPaths[0].empty()) {
473  const int lastClef = (penalties[1][(pos - 1) % 2] < penalties[0][(pos - 1) % 2])
474  ? 1 : 0;
475  // get the optimal clef changes found via dynamic programming
476  if (createClefs(staff, optimalPaths, lastClef, segments))
477  mainClefWasSet = true;
478  }
479  }
480  if (!mainClefWasSet && canChangeClef)
481  createMainClefFromAveragePitch(staff, strack);
482 
483  Q_ASSERT_X(!doesClefBreakTie(staff), "MidiClef::createClefs", "Clef breaks the tie");
484  }
