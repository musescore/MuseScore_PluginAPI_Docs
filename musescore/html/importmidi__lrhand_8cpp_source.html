183  const bool isHighOctave = isOctave(notes, splitPoint, notes.size());
184  const bool isPrevHighOctave = isOctave(prevNotes, prevSplitPoint, prevNotes.size());
185 
186  if (isHighOctave && isPrevHighOctave)
187  penalty -= 12;
188  else if (notes.size() - splitPoint > 1 && prevNotes.size() - prevSplitPoint > 1)
189  penalty -= 5;
190  }
191  // check for one-note melody
192  if (splitPoint - 0 == 1 && prevSplitPoint - 0 == 1)
193  penalty -= 12;
