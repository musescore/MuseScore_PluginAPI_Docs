100  const std::multimap<ReducedFraction, std::string> &lyricTrack,
101  const std::set<int> &usedTracks)
102  {
103  BestTrack bestTrack;
104  int maxMatches = 0;
105 
106  Q_ASSERT_X(areEqualIndexesSuccessive(tracks),
107  "MidiLyrics::findBestTrack", "Equal indexes of operations are not successive");
108 
109  for (int i = 0; i != tracks.size(); ++i) {
110  if (tracks[i].mtrack->drumTrack() || usedTracks.find(i) != usedTracks.end())
111  continue;
112  // due to staff split there can be equal indexes of operation
113  if (i > 0 && tracks[i].indexOfOperation == tracks[i - 1].indexOfOperation)
114  continue;

117  for (const auto &chord: tracks[i].chords) {

119  if (lyricTrack.find(note.origOnTime) != lyricTrack.end()) {
120  ++matches;
121  bestTrack.matchedLyricTimes.push_back({chord.first,
122  note.origOnTime});
123  break;
124  }
125  }
126  }
127  if (matches > maxMatches) {
128  maxMatches = matches;
129  bestTrack.index = tracks[i].indexOfOperation;
130  }
131  }
