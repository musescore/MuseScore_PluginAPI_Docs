57 DivisionInfo metricDivisionsOfBar(const ReducedFraction &barFraction)
58  {
59  DivisionInfo barDivInfo;
60  barDivInfo.len = barFraction;
61  // first value of each element in list is a length (in ticks) of every part of bar
62  // on which bar is subdivided on each level
63  // the level value is a second value of each element
64  auto &divLengths = barDivInfo.divLengths;
65  int level = 0;
66  divLengths.push_back({barFraction, level});
67  // pulse-level division
68  if (isDuple(barFraction))
69  divLengths.push_back({barFraction / 2, --level});
70  else if (isTriple(barFraction))
71  divLengths.push_back({barFraction / 3, --level});
72  else if (isQuadruple(barFraction)) {
73  divLengths.push_back({barFraction / 2, --level}); // additional central accent
74  divLengths.push_back({barFraction / 4, --level});
75  }
76  else if (isQuintuple(barFraction)) {
77  divLengths.push_back({barFraction / 5, --level});
78  }
79  else if (isSeptuple(barFraction)) {
80  divLengths.push_back({barFraction / 7, --level});
81  }
82  else { // if other - complex meter
83  divLengths.push_back({barFraction / barFraction.numerator(), --level});
84  }
85 
86  if (isCompound(barFraction)) {
87  --level; // additional min level for pulse divisions
88  // subdivide pulse of compound meter into 3 parts
89  divLengths.push_back({divLengths.back().len / 3, --level});
90  }
91 
92  while (divLengths.back().len >= MChord::minAllowedDuration() * 2)
93  divLengths.push_back({divLengths.back().len / 2, --level});
94 
95  return barDivInfo;
96  }

196  const ReducedFraction &endTickInBar,

198  {
199  MaxLevel level;
200  const auto startTickInDiv = startTickInBar - divInfo.onTime;
201  const auto endTickInDiv = endTickInBar - divInfo.onTime;
202  if (startTickInDiv >= endTickInDiv || startTickInDiv < ReducedFraction(0, 1)
203  || endTickInDiv > divInfo.len)
204  return level;

209  auto maxEndRaster = divLen * (ratio.numerator() / ratio.denominator());

212  if (startTickInDiv < maxEndRaster) {
213  // max level is found
214  const auto sratio = startTickInDiv / divLen;
215  const auto maxStartRaster = divLen * (sratio.numerator() / sratio.denominator());

217  level.pos = maxStartRaster + divLen + divInfo.onTime;
218  level.levelCount = count.numerator() / count.denominator();

471 QList<std::pair<ReducedFraction, TDuration> >
472 toDurationList(const ReducedFraction &startTickInBar,
473  const ReducedFraction &endTickInBar,
474  const ReducedFraction &barFraction,
475  const std::vector<MidiTuplet::TupletData> &tupletsInBar,
476  DurationType durationType,
477  bool useDots,
478  bool printRestRemains)
479  {
480  if (startTickInBar < ReducedFraction(0, 1)
481  || endTickInBar <= startTickInBar || endTickInBar > barFraction)
482  return QList<std::pair<ReducedFraction, TDuration>>();
483 
484  const auto divInfo = divisionInfo(barFraction, tupletsInBar); // mectric structure of bar
485  const auto minDuration = MChord::minAllowedDuration() * 2; // >= minAllowedDuration() after subdivision
486 
487  std::map<ReducedFraction, Node> nodes; // <onTime, Node>
488  {
489  int level = levelOfTick(startTickInBar, divInfo);
490  nodes.insert({startTickInBar, Node(level, level)});
491  level = levelOfTick(endTickInBar, divInfo);
492  nodes.insert({endTickInBar, Node(level, level)});
493  }
494 
495  QQueue<std::pair<ReducedFraction, ReducedFraction>> gapsToProcess;
496  gapsToProcess.enqueue({startTickInBar, endTickInBar});
497 
498  while (!gapsToProcess.isEmpty()) {
499  const auto gap = gapsToProcess.dequeue();
500  // don't split gap if its duration is less than minDuration
501  if (gap.second - gap.first < minDuration)
502  continue;
503  const auto splitPoint = findMaxLevelBetween(gap.first, gap.second, divInfo);
504  // sum levels if there are several positions (beats) with max level value
505  // for example, 8th + half duration + 8th in 3/4, and half is over two beats
506  if (splitPoint.pos == ReducedFraction(-1, 1)) // undefined
507  continue;
508  const int effectiveLevel = splitPoint.level + splitPoint.levelCount - 1;
509  const Node &startNode = nodes.find(gap.first)->second;
510  const Node &endNode = nodes.find(gap.second)->second;
511 
512  if (badLevelCondition(effectiveLevel - startNode.edgeLevel,
513  effectiveLevel - endNode.edgeLevel, 0)
514  || is2of3RestInTripleMeter(gap.first, gap.second, barFraction)
515  || (durationType == DurationType::REST
516  && is23EndOfBeatInCompoundMeter(gap.first, gap.second, barFraction)))
517  {
518  const int edgeLevel = adjustEdgeLevelIfTuplet(splitPoint, divInfo);
519  // split gap in splitPoint position
520  nodes.insert({splitPoint.pos, Node(edgeLevel, effectiveLevel)});
521  gapsToProcess.enqueue({gap.first, splitPoint.pos});
522  gapsToProcess.enqueue({splitPoint.pos, gap.second});
523  }
524  }
525 
526  const int tol = (durationType == DurationType::NOTE) ? 1 : 0;
527  excludeNodes(nodes, tol, useDots);
528 
529  return collectDurations(nodes, tupletsInBar, useDots, printRestRemains);
