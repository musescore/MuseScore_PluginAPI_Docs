360 bool isHumanPerformance(
361  const std::multimap<ReducedFraction, MidiChord> &chords,
362  const TimeSigMap *sigmap)
363  {
364  if (chords.empty())
365  return false;
366 
367  const auto basicQuant = ReducedFraction::fromTicks(MScore::division) / 4; // 1/16
368  int matches = 0;
369  int count = 0;
370 
371  std::set<ReducedFraction> usedOnTimes;
372 
373  for (const auto &chord: chords) {
374  const auto quant = qMin(basicQuant,
375  quantForLen(MChord::maxNoteLen(chord), basicQuant));
376  const auto onTime = quantizeValue(chord.first, quant);
377  int barIndex, beat, tick;
378  sigmap->tickValues(onTime.ticks(), &barIndex, &beat, &tick);
379 
380  const auto barStart = ReducedFraction::fromTicks(sigmap->bar2tick(barIndex, 0));
381  const auto barFraction = ReducedFraction(sigmap->timesig(barStart.ticks()).timesig());
382  const auto beatLen = Meter::beatLength(barFraction);
383 
384  if (((onTime - barStart) / beatLen).reduced().denominator() == 1
385  && usedOnTimes.find(onTime) == usedOnTimes.end()) {
386  usedOnTimes.insert(onTime);
387  ++count;
388  const auto diff = (onTime - chord.first).absValue();
389  if (diff < MChord::minAllowedDuration())
390  ++matches;
391  }
392  }
393 
394  const double TOL = 0.6;
395  const double matched = matches * 1.0 / count;
396 
397  return matched < TOL;
398  }

1038 void applyDynamicProgramming(std::vector<QuantData> &quantData)
1039  {
1040  const auto &opers = midiImportOperations.data()->trackOpers;
1041  const bool isHuman = opers.isHumanPerformance.value();
1042  const double MERGE_PENALTY_COEFF = 5.0;
1043 
1044  for (size_t chordIndex = 0; chordIndex != quantData.size(); ++chordIndex) {
1045  QuantData &d = quantData[chordIndex];
1046  for (size_t pos = 0; pos != d.positions.size(); ++pos) {
1047  QuantPos &p = d.positions[pos];
1048 
1049  const auto timePenalty = (d.chord->first - p.time).absValue().toDouble();
1050  const double levelDiff = qAbs(d.metricalLevelForLen - p.metricalLevel);
1051 
1052  if (isHuman) {
1053  if (p.metricalLevel <= d.metricalLevelForLen)
1054  p.penalty = timePenalty + levelDiff * d.quantForLen.toDouble();
1055  else
1056  p.penalty = timePenalty / (1 + levelDiff);
1057  }
1058  else {
1059  if (p.metricalLevel <= d.metricalLevelForLen)
1060  p.penalty = timePenalty * (1 + levelDiff);
1061  else
1062  p.penalty = timePenalty;
1063  }
1064 
1065  if (chordIndex == 0)
1066  continue;
1067 
1068  const QuantData &dPrev = quantData[chordIndex - 1];
1069  double minPenalty = std::numeric_limits<double>::max();
1070  int minPos = -1;
1071 
1072  for (size_t posPrev = 0; posPrev != dPrev.positions.size(); ++posPrev) {
1073  const QuantPos &pPrev = dPrev.positions[posPrev];
1074  if (pPrev.time > p.time)
1075  continue;
1076 
1077  double penalty = pPrev.penalty;
1078  if (pPrev.time == p.time) {
1079  if (!d.canMergeWithPrev)
1080  continue;
1081  penalty += d.quant.toDouble() * MERGE_PENALTY_COEFF;
1082  }
1083 
1084  if (penalty < minPenalty) {
1085  minPenalty = penalty;
1086  minPos = static_cast<int>(posPrev);
1087  }
1088  }
1089 
1090  Q_ASSERT_X(minPos != -1,
1091  "Quantize::applyDynamicProgramming", "Min pos was not found");
1092

1094  p.prevPos = minPos;
1095  }
