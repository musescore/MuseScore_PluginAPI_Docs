175  MidiNote &note,
176  const std::multimap<ReducedFraction, MidiChord>::iterator &it,
177  const std::multimap<ReducedFraction, MidiChord> &chords)
178  {
179  if (it->second.isInTuplet) { // if note on time is in tuplet -
180  // set note off time to be inside that tuplet
181  const auto &tuplet = it->second.tuplet->second;
182  const auto len = tuplet.len / tuplet.tupletNumber;
183  const auto newOffTime = qMin(it->first + len, tuplet.onTime + tuplet.len);
184  if (note.offTime > newOffTime) {
185  note.offTime = newOffTime;
186  note.isInTuplet = true;
187  note.tuplet = it->second.tuplet;
188  note.offTimeQuant = len;
189  }
190  }
191  else { // if note on time is outside tuplets -
192  // set note off time to be outside all tuplets
193  auto next = std::next(it);
194  while (next != chords.end() && next->second.voice != it->second.voice)
195  ++next;
196  if (next != chords.end()) {
197  const auto len = ReducedFraction::fromTicks(
198  MScore::division) / 8; // 1/32
199  auto newOffTime = it->first + len;
200  if (next->second.isInTuplet) {
201  const auto &tuplet = next->second.tuplet->second;
202  if (tuplet.onTime < newOffTime)
203  newOffTime = tuplet.onTime;
204  }
205  if (newOffTime < note.offTime) {
206  note.offTime = newOffTime;
207  note.isInTuplet = false;
