307 std::vector<TupletInfo> detectTuplets(
308  const std::multimap<ReducedFraction, MidiChord>::iterator &startBarChordIt,
309  const std::multimap<ReducedFraction, MidiChord>::iterator &endBarChordIt,
310  const ReducedFraction &startBarTick,
311  const ReducedFraction &barFraction,
312  std::multimap<ReducedFraction, MidiChord> &chords,
313  const ReducedFraction &basicQuant,
314  int barIndex)
315  {
316  const auto divLengths = Meter::divisionsOfBarForTuplets(barFraction);
317 
318  std::vector<TupletInfo> tuplets;
319  int id = 0;
320  const auto tol = basicQuant / 2;
321 
322  for (const auto &divLen: divLengths) {
323  const auto tupletNumbers = findTupletNumbers(divLen, barFraction);
324  const auto div = barFraction / divLen;
325  const int divCount = div.numerator() / div.denominator();
326 
327  for (int i = 0; i != divCount; ++i) {
328  const auto startDivTime = startBarTick + divLen * i;
329  const auto endDivTime = startBarTick + divLen * (i + 1);
330  // check which chords can be inside tuplet period
331  // [startDivTime - tol, endDivTime]
332  const auto startDivTimeWithTol = qMax(startBarTick, startDivTime - tol);
333  auto startDivChordIt = MChord::findFirstChordInRange(startDivTimeWithTol, endDivTime,
334  startBarChordIt, endBarChordIt);
335  if (startDivChordIt == endBarChordIt)
336  continue;
337 
338  Q_ASSERT_X(!startDivChordIt->second.isInTuplet,
339  "MIDI tuplets: findTuplets", "Tuplet chord has been already used");
340 
341  // end iterator, as usual, point to the next - invalid chord
342  auto endDivChordIt = chords.lower_bound(endDivTime);
343  if (!isNextBarOwnershipOk(startDivChordIt, endDivChordIt, chords, barIndex))
344  continue;
345  // try different tuplets, nested tuplets are not allowed
346  // here chords from next bar can be captured
347  // if their on time < next bar start
348  for (const auto &tupletNumber: tupletNumbers) {
349  if (!isTupletLenAllowed(divLen, tupletNumber, startDivChordIt, endDivChordIt,
350  basicQuant)) {
351  continue;
352  }
353  auto tupletInfo = findTupletApproximation(divLen, tupletNumber,
354  basicQuant, startDivTime, startDivChordIt, endDivChordIt);
355 
356  const auto &opers = midiImportOperations.data()->trackOpers;
357  const int currentTrack = midiImportOperations.currentTrack();
358 
359  if (opers.simplifyDurations.value(currentTrack)) {
360  if (!haveChordsInTheMiddleBetweenTupletChords(
361  startDivChordIt, endDivChordIt, tupletInfo)) {
362  detectStaccato(tupletInfo);
363  }
364  }
365  tupletInfo.sumLengthOfRests = findSumLengthOfRests(tupletInfo, startBarTick);
366 
367  if (!isTupletAllowed(tupletInfo))
368  continue;
369  tupletInfo.id = id++;

371  } // next tuplet type
372  }
373  }
