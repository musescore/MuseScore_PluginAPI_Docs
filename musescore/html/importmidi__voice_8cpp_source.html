705  const std::multimap<ReducedFraction, MidiChord>::iterator &chordIt,
706  const std::multimap<ReducedFraction, MidiChord> &chords)

733 bool splitChordToVoice(
734  std::multimap<ReducedFraction, MidiChord>::iterator &chordIt,
735  const QSet<int> &notesToMove,
736  int newVoice,
737  std::multimap<ReducedFraction, MidiChord> &chords,
738  std::multimap<ReducedFraction, MidiTuplet::TupletData> &tuplets,
739  std::multimap<ReducedFraction,
740  std::multimap<ReducedFraction, MidiTuplet::TupletData>::iterator> &insertedTuplets,
741  const ReducedFraction &maxChordLength,
742  bool allowParallelTuplets)
743  {
744  if (notesToMove.isEmpty())
745  return false;
746 
747  bool splitDone = true;
748  const ReducedFraction onTime = chordIt->first;
749  MidiChord &chord = chordIt->second;
750  auto &notes = chord.notes;
751 
752  Q_ASSERT_X(MidiTuplet::isTupletRangeOk(*chordIt, tuplets),
753  "MidiVoice::splitChordToVoice, before split",
754  "Tuplet chord/note is outside tuplet "
755  "or non-tuplet chord/note is inside tuplet before simplification");
756 
757  if (notesToMove.size() == notes.size()) {
758  // don't split chord, just move it to another voice
759  const int oldVoice = chord.voice; // remember for possible undo
760  chord.voice = newVoice;
761 
762  const bool success = updateChordTuplets(chord, onTime, insertedTuplets,
763  chords, tuplets, maxChordLength,
764  allowParallelTuplets);
765  if (!success) {
766  splitDone = false;
767  chord.voice = oldVoice; // rollback
768  }
769  }
770  else { // split chord
771  MidiChord newChord(chord);
772  newChord.notes.clear();
773  newChord.voice = newVoice;
774  QList<MidiNote> updatedOldNotes;
775 
776  for (int i = 0; i != notes.size(); ++i) {
777  if (notesToMove.contains(i))
778  newChord.notes.append(notes[i]);
779  else
780  updatedOldNotes.append(notes[i]);
781  }
782 
783  const auto rememberedNotes = notes; // to undo split if necessary
784  // update notes before tuplet update because there will be check
785  // for empty tuplets
786  notes = updatedOldNotes;
787 
788  const bool success = updateChordTuplets(newChord, onTime, insertedTuplets,
789  chords, tuplets, maxChordLength,
790  allowParallelTuplets);
791  if (success) {
792 
793  Q_ASSERT_X(!notes.isEmpty(),
794  "MidiVoice::splitChordToVoice", "Old chord notes are empty");
795  Q_ASSERT_X(!newChord.notes.isEmpty(),
796  "MidiVoice::splitChordToVoice", "New chord notes are empty");
797 
798  chordIt = chords.insert({onTime, newChord});
799  }
800  else {
801  notes = rememberedNotes; // rollback
802  splitDone = false;
803  }
804  }
805 
806  Q_ASSERT_X(MidiTuplet::isTupletRangeOk(*chordIt, tuplets),
807  "MidiVoice::splitChordToVoice, after split",
808  "Tuplet chord/note is outside tuplet "
809  "or non-tuplet chord/note is inside tuplet before simplification");
810 
811  return splitDone;
812  }

925  std::multimap<ReducedFraction, MidiChord>::iterator>> &voiceChords)
926  {

928  std::vector<std::pair<int, int>> pitchVoices;
929  for (const auto &v: voiceChords)
930  pitchVoices.push_back({averagePitchOfChords(v.second), v.first});
931 
932  struct {
933  bool operator()(const std::pair<int, int> &p1,
934  const std::pair<int, int> &p2) const
935  {
936  if (p1.first != p2.first)
937  return p1.first > p2.first;
938  return p1.second < p2.second;
939  }

941  std::sort(pitchVoices.begin(), pitchVoices.end(), comparator);
942

944  const int oldVoice = pitchVoices[newVoice].second;

956  c.tuplet->second.voice = newVoice;
957  for (auto &note: c.notes) {
