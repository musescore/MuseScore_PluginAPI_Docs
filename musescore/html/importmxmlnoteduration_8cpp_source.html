28 static Fraction noteTypeToFraction(const QString& type)
29  {
30  if (type == "1024th")
31  return Fraction(1, 1024);
32  else if (type == "512th")
33  return Fraction(1, 512);
34  else if (type == "256th")
35  return Fraction(1, 256);
36  else if (type == "128th")
37  return Fraction(1, 128);
38  else if (type == "64th")
39  return Fraction(1, 64);
40  else if (type == "32nd")
41  return Fraction(1, 32);
42  else if (type == "16th")
43  return Fraction(1, 16);
44  else if (type == "eighth")
45  return Fraction(1, 8);
46  else if (type == "quarter")
47  return Fraction(1, 4);
48  else if (type == "half")
49  return Fraction(1, 2);
50  else if (type == "whole")
51  return Fraction(1, 1);
52  else if (type == "breve")
53  return Fraction(2, 1);
54  else if (type == "long")
55  return Fraction(4, 1);
56  else if (type == "maxima")
57  return Fraction(8, 1);
58  else
59  return Fraction(0, 0);
60  }

66 /**
67  Convert note type, number of dots and actual and normal notes into a duration
68  */
69 
70 static Fraction calculateFraction(const QString& type, const int dots, const Fraction timeMod)
71  {
72  // type
73  Fraction f = noteTypeToFraction(type);
74  if (f.isValid()) {
75  // dot(s)
76  Fraction ff = f;
77  for (int i = 0; i < dots; ++i) {
78  ff = ff * Fraction(1,2);
79  f += ff;
80  }
81  // tuplet
82  if (timeMod.isValid())
83  f *= timeMod;
84  // clean up (just in case)
85  f.reduce();
86  }
87  return f;
88  }

94 /**
95  Do timing error checks.
96  Return empty string if OK, message in case of error.
97  */
98 
99 QString mxmlNoteDuration::checkTiming(const QString& type, const bool rest, const bool grace)
100  {
101  //qDebug("type %s rest %d grace %d", qPrintable(type), rest, grace);
102  QString errorStr;
103 
104  // normalize duration
105  if (_dura.isValid())
106  _dura.reduce();
107 
108  const auto calcDura = calculateFraction(type, _dots, _timeMod);
109  if (_dura.isValid() && calcDura.isValid()) {
110  if (_dura != calcDura) {
111  errorStr = QString("calculated duration (%1) not equal to specified duration (%2)")
112  .arg(calcDura.print(), _dura.print());
113  //qDebug("rest %d type '%s' timemod %s", rest, qPrintable(type), qPrintable(_timeMod.print()));
114 
115  if (rest && type == "whole" && _dura.isValid()) {
116  // Sibelius whole measure rest (not an error)
117  errorStr = "";
118  }
119  else if (grace && _dura == Fraction(0, 1)) {
120  // grace note (not an error)
121  errorStr = "";
122  }
123  else {
124  const int maxDiff = 3; // maximum difference considered a rounding error
125  if (qAbs(calcDura.ticks() - _dura.ticks()) <= maxDiff) {
126  errorStr += " -> assuming rounding error";
127  _dura = calcDura;
128  }
129  }
130 
131  // Special case:
132  // Encore generates rests in tuplets w/o <tuplet> or <time-modification>.
133  // Detect this by comparing the actual duration with the expected duration
134  // based on note type. If actual is 2/3 of expected, the rest is part
135  // of a tuplet.
136  if (rest) {
137  if (2 * calcDura.ticks() == 3 * _dura.ticks()) {
138  _timeMod = Fraction(2, 3);
139  errorStr += " -> assuming triplet";
140  }
141  }
142  }
143  }
144  else if (_dura.isValid()) {
145  // do not report an error for typeless (whole measure) rests
146  if (!(rest && type == ""))
147  errorStr = "calculated duration invalid, using specified duration";
148  }
149  else if (calcDura.isValid()) {
150  if (!grace) {
151  errorStr = "specified duration invalid, using calculated duration";
152  _dura = calcDura; // overrule dura
153  }
154  }
155  else {
156  errorStr = "calculated and specified duration invalid, using 4/4";
157  _dura = Fraction(4, 4);
158  }
159 
160  return errorStr;
161  }

171 void mxmlNoteDuration::duration(QXmlStreamReader& e)
172  {
173  Q_ASSERT(e.isStartElement() && e.name() == "duration");
174  _logger->logDebugTrace("MusicXMLParserPass1::duration", &e);
175 
176  _dura.set(0, 0); // invalid unless set correctly
177  int intDura = e.readElementText().toInt();
178  if (intDura > 0) {
179  if (_divs > 0) {
180  _dura.set(intDura, 4 * _divs);
181  _dura.reduce(); // prevent overflow in later Fraction operations
182  }
183  else
184  _logger->logError("illegal or uninitialized divisions", &e);
185  }
186  else
187  _logger->logError("illegal duration", &e);
188  //qDebug("duration %s valid %d", qPrintable(dura.print()), dura.isValid());
189  }

236  QString strNormal;
237 
238  while (e.readNextStartElement()) {
239  const QStringRef& tag(e.name());
240  if (tag == "actual-notes")
241  strActual = e.readElementText();
242  else if (tag == "normal-notes")
243  strNormal = e.readElementText();
244  else if (tag == "normal-type") {
245  // "measure" is not a valid normal-type,
246  // but would be accepted by setType()
247  QString strNormalType = e.readElementText();
248  if (strNormalType != "measure")
249  _normalType.setType(strNormalType);
250  }
251  else {
252  _logger->logDebugInfo(QString("skipping '%1'").arg(e.name().toString()), &e);
253  e.skipCurrentElement();
254  }
255  }
256 
257  intActual = strActual.toInt();
258  intNormal = strNormal.toInt();
259  if (intActual > 0 && intNormal > 0)
260  _timeMod.set(intNormal, intActual);
261  else {
262  _timeMod.set(1, 1);
263  _logger->logError(QString("illegal time-modification: actual-notes %1 normal-notes %2")
264  .arg(strActual, strNormal), &e);
265  }
266  }
