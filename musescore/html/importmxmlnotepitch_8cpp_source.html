30 static Accidental* accidental(QXmlStreamReader& e, Score* score)
31  {
32  Q_ASSERT(e.isStartElement() && e.name() == "accidental");
33 
34  bool cautionary = e.attributes().value("cautionary") == "yes";
35  bool editorial = e.attributes().value("editorial") == "yes";
36  bool parentheses = e.attributes().value("parentheses") == "yes";
37 
38  const auto s = e.readElementText();
39  const auto type = mxmlString2accidentalType(s);
40 
41  if (type != AccidentalType::NONE) {
42  auto a = new Accidental(score);
43  a->setAccidentalType(type);
44  if (editorial || cautionary || parentheses) {
45  a->setBracket(AccidentalBracket(cautionary || parentheses));
46  a->setRole(AccidentalRole::USER);
47  }
48  return a;
49  }
50 
51  return 0;
52  }

100 void mxmlNotePitch::pitch(QXmlStreamReader& e)
101  {
102  Q_ASSERT(e.isStartElement() && e.name() == "pitch");
103 
104  // defaults
105  _step = -1;
106  _alter = 0;
107  _octave = -1;
108 
109  while (e.readNextStartElement()) {
110  if (e.name() == "alter") {
111  const auto alter = e.readElementText();
112  bool ok;
113  _alter = MxmlSupport::stringToInt(alter, &ok); // fractions not supported by mscore
114  if (!ok || _alter < -2 || _alter > 2) {
115  _logger->logError(QString("invalid alter '%1'").arg(alter), &e);
116  bool ok2;
117  const auto altervalue = alter.toDouble(&ok2);
118  if (ok2 && (qAbs(altervalue) < 2.0) && (_accType == AccidentalType::NONE)) {
119  // try to see if a microtonal accidental is needed
120  _accType = microtonalGuess(altervalue);
121  }
122  _alter = 0;
123  }
124  }
125  else if (e.name() == "octave") {
126  const auto oct = e.readElementText();
127  bool ok;
128  _octave = oct.toInt(&ok);
129  if (!ok || _octave < 0 || _octave > 9) {
130  _logger->logError(QString("invalid octave '%1'").arg(oct), &e);
131  _octave = -1;
132  }
133  }
134  else if (e.name() == "step") {
135  const auto step = e.readElementText();
136  const auto pos = QString("CDEFGAB").indexOf(step);
137  if (step.size() == 1 && pos >=0 && pos < 7)
138  _step = pos;
139  else
140  _logger->logError(QString("invalid step '%1'").arg(step), &e);
141  }
142  else {
143  ; // TODO skipLogCurrElem();
144  }
145  }
146  //qDebug("pitch step %d alter %d oct %d accid %hhd", step, alter, oct, accid);
147  }
