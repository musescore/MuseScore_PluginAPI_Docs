188 void OmrState::importPdfPage(OmrPage* omrPage, qreal top)
189  {
190  TDuration d(TDuration::DurationType::V_MEASURE);
191 
192  int nsystems = omrPage->systems().size();
193  if(nsystems == 0) return;
194 
195  //add top margin for alignment
196  MeasureBase* first_measure = score->first();
197  if (first_measure == 0 || first_measure->isVBox()) {
198  VBox* vbox = new VBox(score);
199  vbox->setNext(score->first());
200  vbox->setTick(Fraction(0,1));
201  vbox->setBoxHeight(Spatium(top));
202  vbox->setBottomMargin(0);
203  vbox->setBottomGap(0);
204  score->measures()->add(vbox);
205  }
206 
207  for (int k = 0; k < nsystems; ++k) {
208  importPdfSystem(omrPage->system(nsystems - k - 1));
209  }
210 
211  Measure* measure = score->lastMeasure();
212  if (measure) {
213  LayoutBreak* b = new LayoutBreak(score);
214  b->setLayoutBreakType(LayoutBreak::Type::PAGE);
215  measure->add(b);
216  }
217 
218  measure = score->firstMeasure();
219  if (measure) {
220  if (!measure->vspacerUp(0)){
221  Spacer* spacer = new Spacer(score);
222  spacer->setSpacerType(SpacerType::UP);
223  spacer->setTrack(0);
224  measure->add(spacer);
225  }
226  Spacer* sp = measure->vspacerUp(0);
227  sp->layout();
228  sp->setPos(sp->rxpos(), top);
229  }
230  }

236 Score::FileError importPdf(MasterScore* score, const QString& path)
237  {
238  Omr* omr = new Omr(path, score);
239  if (!omr->readPdf()) {
240  delete omr;
241  return Score::FileError::FILE_BAD_FORMAT;
242  }
243 
244  score->setOmr(omr);
245  qreal sp = omr->spatiumMM();
246  if (sp == 0.0)
247  sp = 1.5;
248  score->setSpatium(sp * DPMM);
249  score->style().set(Sid::lastSystemFillLimit, 0.0);
250  score->style().set(Sid::staffLowerBorder, 0.0);
251  score->style().set(Sid::measureSpacing, 1.0);
252  score->style().set(Sid::frameSystemDistance, 0);
253  score->style().set(Sid::pageEvenLeftMargin, 5.0 * DPMM / DPI);
254  score->style().set(Sid::pageEvenTopMargin, 0);
255  score->style().set(Sid::pageEvenBottomMargin, 0);
256  score->style().set(Sid::pageOddLeftMargin, 5.0 * DPMM / DPI);
257  score->style().set(Sid::pageOddTopMargin, 0);
258  score->style().set(Sid::pageOddBottomMargin, 0);
259  score->style().set(Sid::minSystemDistance, Spatium(omr->systemDistance()));
260  score->style().set(Sid::maxSystemDistance, Spatium(omr->systemDistance()));
261  score->style().set(Sid::akkoladeDistance, Spatium(omr->staffDistance()));
262 
263  Part* part = new Part(score);
264  OmrPage* omrPage = omr->pages().front();
265 
266  //may need to identify maximal number of staff per system and then high some of those
267  if (omrPage->systems().size() > 0) {
268  for (int i = 0; i < omrPage->systems().front().staves().size(); i++) {
269  Staff* staff = new Staff(score);
270  staff->setPart(part);
271  part->insertStaff(staff, -1);
272  score->staves().append(staff);
273  }
274  }
275  score->appendPart(part);
276 
277  OmrState state;
278  state.score = score;
279  for (OmrPage* omrPage1 : omr->pages()) {
280  if (omrPage1->systems().count() == 0)
281  continue;
282  OmrSystem system = omrPage1->systems().last();
283  if (system.staves().count() == 0)
284  continue;
285  OmrStaff staff = system.staves().first();
286  qreal top = staff.top()/omr->spatium();
287  state.importPdfPage(omrPage1, top);
288  }
289 
290  //---create bracket
291 
292  //score->staff(0)->setBracket(0, BracketType::BRACE);
293  //score->staff(0)->setBracketSpan(0, 2);
294 
295  score->setShowOmr(true);
296  omr->page(0)->readHeader(score);
297  score->rebuildMidiMapping();
298  return Score::FileError::FILE_NO_ERROR;
299  }
