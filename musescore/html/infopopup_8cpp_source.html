139 void InfoPopup::updatePopup(ScoreView* view)
140  {
141  IF_ASSERT(view) {
142  return;
143  }
144 
145  move(view->width() - width() - POPUP_MARGIN, view->height() - height() - POPUP_MARGIN);
146 
147  //! NOTE When scrolling the view, only a small part is redrawn,

151  //! we will redraw the view completely.
