87 void InspectorGlissando::updateEvents() {

89  std::vector<int> body;
90  if (gs) {

93  glissandoPitchOffsets(gs->spanner(), body);
94  g.easeInOutCanvas->setBottomPitch(pitchEnd > pitchStart ? pitchStart : pitchEnd);
95  g.easeInOutCanvas->setPitchDelta(pitchEnd - pitchStart);
96  }
