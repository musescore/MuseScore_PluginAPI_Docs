81 void InspectorGroupElement::setColor()
82  {
83  if (inspector->el()->isEmpty())
84  return;

86  score->startCmd();
87  for (Element* e : *inspector->el()) {

166 void InspectorGroupElement::notesClicked()
167  {
168  Score* score = inspector->el()->front()->score();
169  QList<Element*> el = score->selection().elements();
170  QList<Element*> nel;
171  score->deselectAll();
172  for (Element* e : el) {
173  if (e->isNote()) {
174  Note* note = toNote(e);
175  //if note is not grace note, then add to selection
176  if (!note->chord()->isGrace()) {
177  nel.append(note);
178  score->selection().add(note);
179  }
180  }
181  }
182  score->update();
183  inspector->update();
184  }
