150 void StaffListItem::initStaffTypeCombo(bool forceRecreate)
151  {
152  if (_staffTypeCombo && !forceRecreate) // do not init more than once
153  return;
154 
155  // NOTE: DO NOT DELETE the old _staffTypeCombo if already created:
156  // a bug in Qt looses track of (and presumably deletes) the combo set into the item
157  // if the item is repositioned in the tree; in this case, the pointer to the combo
158  // is no longer valid and cannot be used to delete it
159  // Call initStaffTypeCombo(true) ONLY if the item has been repositioned
160  // or a memory leak may result
161 
162  bool canUseTabs = false; // assume only normal staves are applicable
163  int numFrettedStrings = 0;
164  bool canUsePerc = false;
165  PartListItem* part = static_cast<PartListItem*>(QTreeWidgetItem::parent());
166 
167  // PartListItem has different members filled out if used in New Score Wizard
168  // or in Instruments Wizard
169  if (part) {
170  const StringData* stringData = part->it ? &(part->it->stringData) :
171  ( (part->part && part->part->instrument()) ? part->part->instrument()->stringData() : 0);
172  canUseTabs = stringData && stringData->frettedStrings() > 0;
173  if (canUseTabs)
174  numFrettedStrings = stringData->frettedStrings();
175  canUsePerc = part->it ? part->it->useDrumset :
176  ( (part->part && part->part->instrument()) ? part->part->instrument()->useDrumset() : false);
177  }
178  _staffTypeCombo = new QComboBox();
179  _staffTypeCombo->setAutoFillBackground(true);
180  int idx = 0;
181  for (const StaffType& st : StaffType::presets()) {
182  if ( (st.group() == StaffGroup::STANDARD && (!canUsePerc)) // percussion excludes standard
183  || (st.group() == StaffGroup::PERCUSSION && canUsePerc)
184  || (st.group() == StaffGroup::TAB && canUseTabs && st.lines() <= numFrettedStrings)) {
185  _staffTypeCombo->addItem(st.name(), idx);
186  }
187  ++idx;
188  }
189  customStandardIdx = _staffTypeCombo->count();
190  _staffTypeCombo->addItem(tr("Custom Standard"), CUSTOM_STAFF_TYPE_IDX);
191  customPercussionIdx = _staffTypeCombo->count();
192  _staffTypeCombo->addItem(tr("Custom Percussion"), CUSTOM_STAFF_TYPE_IDX);
193  customTablatureIdx = _staffTypeCombo->count();
194  _staffTypeCombo->addItem(tr("Custom Tablature"), CUSTOM_STAFF_TYPE_IDX);
195 
196  treeWidget()->setItemWidget(this, 4, _staffTypeCombo);
197  connect(_staffTypeCombo, SIGNAL(currentIndexChanged(int)), SLOT(staffTypeChanged(int)) );
198  }

234 void StaffListItem::setStaffType(const StaffType* st)
235  {
236  if (!st) // if no staff type given, dault to stadard
237  _staffTypeCombo->setCurrentIndex(0); // staff type (at combo box index 0)
238  else {
239  // if staff type given, look into combo box item data for a preset equal to staff type
240  for (int i = 0; i < _staffTypeCombo->count(); ++i) {
241  const StaffType* _st = StaffType::preset(StaffTypes(_staffTypeCombo->itemData(i).toInt()));
242  if (*_st == *st) {
243  _staffTypeCombo->setCurrentIndex(i);
244  return;
245  }
246  }
247  // try harder
248  for (int i = 0; i < _staffTypeCombo->count(); ++i) {
249  const StaffType* _st = StaffType::preset(StaffTypes(_staffTypeCombo->itemData(i).toInt()));
250  if (_st->isSameStructure(*st)) {
251  _staffTypeCombo->setCurrentIndex(i);
252  return;
253  }
254  }
255  int idx = 0;
256  switch (st->group()) {
257  case StaffGroup::STANDARD:
258  idx = customStandardIdx;
259  break;
260  case StaffGroup::PERCUSSION:
261  idx = customPercussionIdx;
262  break;
263  case StaffGroup::TAB:
264  idx = customTablatureIdx;
265  break;
266  }
267  _staffTypeCombo->setCurrentIndex(idx);
268  }
269  }

706 
707 QTreeWidgetItem* InstrumentsWidget::movePartItem(int oldPos, int newPos)
708  {
709  if (oldPos == newPos)
710  return partiturList->topLevelItem(oldPos);
711 
712  QTreeWidgetItem* item = partiturList->takeTopLevelItem(oldPos);
713  // Qt looses the QComboBox set into StaffListItem's when they are re-inserted into the tree:
714  // get the currently selected staff type of each combo and re-insert
715  int numOfStaffListItems = item->childCount();
716 #if (!defined (_MSCVER) && !defined (_MSC_VER))
717  int staffIdx[numOfStaffListItems];
718 #else
719  // MSVC does not support VLA. Replace with std::vector. If profiling determines that the
720  // heap allocation is slow, an optimization might be used.
721  std::vector<int> staffIdx(numOfStaffListItems);
722 #endif
723  for (int itemIdx=0; itemIdx < numOfStaffListItems; ++itemIdx)
724  staffIdx[itemIdx] = (static_cast<StaffListItem*>(item->child(itemIdx)))->staffTypeIdx();
725  // do not consider hidden ones
726  const int diff = oldPos - newPos;
727  QTreeWidgetItem* prevParent = partiturList->topLevelItem(newPos);
728  while (prevParent && prevParent->isHidden()) {
729  newPos += diff;
730  prevParent = partiturList->topLevelItem(newPos);
731  }
