79  return;

81  qDebug()<<"JACK number of ports:"<<midiOutputPorts.size()<<", change to:"<<maxport;
82 
83  bool oldremember = preferences.getBool(PREF_IO_JACK_REMEMBERLASTCONNECTIONS);
84  preferences.setPreference(PREF_IO_JACK_REMEMBERLASTCONNECTIONS, true);
85 
86  if (maxport > midiOutputPorts.size()) {
87  for (int i = midiOutputPorts.size(); i < maxport; ++i)
88  registerPort(QString("mscore-midi-%1").arg(i+1), false, true);
89  restoreMidiConnections();
90  }
91  else if (maxport < midiOutputPorts.size()) {
92  rememberMidiConnections();
93  for(int i = midiOutputPorts.size() - 1; i >= maxport; --i) {
94  unregisterPort(midiOutputPorts[i]);
95  midiOutputPorts.removeAt(i);
96  }
97  }
98  preferences.setPreference(PREF_IO_JACK_REMEMBERLASTCONNECTIONS, oldremember);
99 
100  }

319 void JackAudio::timebase(jack_transport_state_t state, jack_nframes_t /*nframes*/, jack_position_t *pos, int /*new_pos*/, void *arg)
320  {
321  JackAudio* audio = (JackAudio*)arg;
322  if (!audio->seq->score()) {
323  if (state==JackTransportLooping || state==JackTransportRolling)
324  audio->stopTransport();
325  }
326  else if (audio->seq->isRunning()) {
327  if (!audio->seq->score()->masterScore())
328  return;
329 
330  pos->valid = JackPositionBBT;
331  int curTick = audio->seq->score()->repeatList().utick2tick(audio->seq->getCurTick());
332  int bar,beat,tick;
333  audio->seq->score()->sigmap()->tickValues(curTick, &bar, &beat, &tick);
334  // Providing the final tempo
335  pos->beats_per_minute = 60 * audio->seq->curTempo() * audio->seq->score()->tempomap()->relTempo();
336  pos->ticks_per_beat = MScore::division;
337  pos->tick = tick;
338  pos->bar = bar+1;
339  pos->beat = beat+1;
340 
341  if (audio->timeSigTempoChanged) {
342  Fraction timeSig = audio->seq->score()->sigmap()->timesig(curTick).nominal();
343  pos->beats_per_bar = timeSig.numerator();
344  pos->beat_type = timeSig.denominator();
345  audio->timeSigTempoChanged = false;
346  qDebug()<<"Time signature changed: "<< pos->beats_per_minute<<", bar: "<< pos->bar<<",beat: "<<pos->beat<<", tick:"<<pos->tick<<", time sig: "<<pos->beats_per_bar<<"/"<<pos->beat_type;
347  }
348  }
349  // TODO: Handle new_pos

356 int JackAudio::processAudio(jack_nframes_t frames, void* p)
357  {
358  JackAudio* audio = (JackAudio*)p;
359  // Prevent from crash if score not opened yet
360  if(!audio->seq->score())
361  return 0;
362 
363  float* l;
364  float* r;
365  if (preferences.getBool(PREF_IO_JACK_USEJACKAUDIO) && audio->ports.size() == 2) {
366  l = (float*)jack_port_get_buffer(audio->ports[0], frames);
367  r = (float*)jack_port_get_buffer(audio->ports[1], frames);
368  }
369  else {
370  l = 0;
371  r = 0;
372  }
373  if (preferences.getBool(PREF_IO_JACK_USEJACKMIDI)) {
374  foreach(jack_port_t* port, audio->midiOutputPorts) {
375  void* portBuffer = jack_port_get_buffer(port, frames);
376  jack_midi_clear_buffer(portBuffer);
377  }
378  foreach(jack_port_t* port, audio->midiInputPorts) {
379  void* portBuffer = jack_port_get_buffer(port, frames);
380  if (portBuffer) {
381  jack_nframes_t n = jack_midi_get_event_count(portBuffer);
382  for (jack_nframes_t i = 0; i < n; ++i) {
383  jack_midi_event_t event;
384  if (jack_midi_event_get(&event, portBuffer, i) != 0)
385  continue;
386  size_t nn = event.size;
387  int type = event.buffer[0];
388  if (nn && (type == ME_CLOCK || type == ME_SENSE))
389  continue;
390  Event e;
391  e.setChannel(type & 0xf);
392  type &= 0xf0;
393  e.setType(type);
394  if (type == ME_NOTEON || type == ME_NOTEOFF) {
395  e.setPitch(event.buffer[1]);
396  e.setVelo(event.buffer[2]);
397  audio->seq->eventToGui(e);
398  }
399  else if (type == ME_CONTROLLER) {
400  e.setController(event.buffer[1]);
401  e.setValue(event.buffer[2]);
402  audio->seq->eventToGui(e);
403  }
404  }
405  }
406  }
407  }
408  if (l && r) {
409 #if (!defined (_MSCVER) && !defined (_MSC_VER))
410  float buffer[frames * 2];
411 #else
412  // MSVC does not support VLA. Replace with std::vector. If profiling determines that the
413  // heap allocation is slow, an optimization might be used.
414  std::vector<float> vBuffer(frames * 2);
415  float* buffer = vBuffer.data();
416 #endif
417  audio->seq->process((unsigned)frames, buffer);
418  float* sp = buffer;
419  for (unsigned i = 0; i < frames; ++i) {
420  *l++ = *sp++;
421  *r++ = *sp++;
422  }
423  }
424  else {
425  // JACK MIDI only
426 #if (!defined (_MSCVER) && !defined (_MSC_VER))
427  float buffer[frames * 2];
428 #else
429  // MSVC does not support VLA. Replace with std::vector. If profiling determines that the
430  // heap allocation is slow, an optimization might be used.
431  std::vector<float> vBuffer(frames * 2);
432  float* buffer = vBuffer.data();
433 #endif
434  audio->seq->process((unsigned)frames, buffer);
435  }
436  return 0;
437  }

490  jack_set_freewheel_callback (client, freewheel_callback, this);
491  if (preferences.getBool(PREF_IO_JACK_TIMEBASEMASTER))

497  MScore::sampleRate = sampleRate();

503

558 void JackAudio::putEvent(const NPlayEvent& e, unsigned framePos)
559  {
560  if (!preferences.getBool(PREF_IO_JACK_USEJACKMIDI))
561  return;
562 
563  int portIdx = seq->score()->midiPort(e.channel());
564  int chan = seq->score()->midiChannel(e.channel());
565 
566 // qDebug("JackAudio::putEvent %d:%d pos %d(%d)", portIdx, chan, framePos, _segmentSize);
567 
568  if (portIdx < 0 || portIdx >= midiOutputPorts.size()) {
569  qDebug("JackAudio::putEvent: invalid port %d", portIdx);
570  return;
571  }
572  jack_port_t* port = midiOutputPorts[portIdx];
573  if (midiOutputTrace) {
574  const char* portName = jack_port_name(port);
575  int a = e.dataA();
576  int b = e.dataB();
577  qDebug("MidiOut<%s>: jackMidi: %02x %02x %02x, chan: %i", portName, e.type(), a, b, chan);
578  // e.dump();
579  }
580  void* pb = jack_port_get_buffer(port, _segmentSize);
581 
582  if (pb == NULL) {
583  qDebug()<<"jack_port_get_buffer failed, cannot send anything";
584  }
585 
586  if (framePos >= _segmentSize) {
587  qDebug("JackAudio::putEvent: time out of range %d(seg=%d)", framePos, _segmentSize);
588  if (framePos > _segmentSize)
589  framePos = _segmentSize - 1;
590  }
591 
592  switch(e.type()) {
593  case ME_NOTEON:
594  case ME_NOTEOFF:
595  case ME_POLYAFTER:
596  case ME_CONTROLLER:
597  // Catch CTRL_PROGRAM and let other ME_CONTROLLER events to go
598  if (e.dataA() == CTRL_PROGRAM) {
599  // Convert CTRL_PROGRAM event to ME_PROGRAM
600  unsigned char* p = jack_midi_event_reserve(pb, framePos, 2);
601  if (p == 0) {
602  qDebug("JackMidi: buffer overflow, event lost");
603  return;
604  }
605  p[0] = ME_PROGRAM | chan;
606  p[1] = less128(e.dataB());
607  break;
608  }
609  //fall through
610  case ME_PITCHBEND:
611  {
612  unsigned char* p = jack_midi_event_reserve(pb, framePos, 3);
613  if (p == 0) {
614  qDebug("JackMidi: buffer overflow, event lost");
615  return;
616  }
617  p[0] = e.type() | chan;
618  p[1] = less128(e.dataA());
619  p[2] = less128(e.dataB());
620  }
621  break;
622 
623  case ME_PROGRAM:
624  case ME_AFTERTOUCH:
625  {
626  unsigned char* p = jack_midi_event_reserve(pb, framePos, 2);
627  if (p == 0) {
628  qDebug("JackMidi: buffer overflow, event lost");
629  return;
630  }
631  p[0] = e.type() | chan;
632  p[1] = less128(e.dataA());
633  }
634  break;
635  // Do we really need to handle ME_SYSEX?
636  /* case ME_SYSEX:
637  {
638  const unsigned char* data = e.edata();
639  int len = e.len();
640  unsigned char* p = jack_midi_event_reserve(pb, framePos, len+2);
641  if (p == 0) {
642  qDebug("JackMidi: buffer overflow, event lost");
643  return;
644  }
645  p[0] = 0xf0;
646  p[len+1] = 0xf7;
647  memcpy(p+1, data, len);
648  }
649  break;*/
650  case ME_SONGPOS:
651  case ME_CLOCK:
652  case ME_START:
653  case ME_CONTINUE:
654  case ME_STOP:
655  qDebug("JackMidi: event type %x not supported", e.type());
656  break;
657  }

680 //---------------------------------------------------------
681 // checkTransportSeek
682 // The opposite of Timebase master:
683 // check JACK Transport for a new position or tempo.
684 //---------------------------------------------------------
685 
686 void JackAudio::checkTransportSeek(int cur_frame, int nframes, bool inCountIn)
687  {
688  if (!seq || !seq->score() || inCountIn)
689  return;
690 
691  // Obtaining the current JACK Transport position
692  jack_position_t pos;
693  jack_transport_query(client, &pos);
694 
695  if (preferences.getBool(PREF_IO_JACK_USEJACKTRANSPORT)) {
696  if (mscore->playPanelInterface() && mscore->playPanelInterface()->isSpeedSliderPressed())
697  return;
698  int cur_utick = seq->score()->utime2utick((qreal)cur_frame / MScore::sampleRate);
699  int utick = seq->score()->utime2utick((qreal)pos.frame / MScore::sampleRate);
700 
701  // Conversion is not precise, should check frames and uticks
702  if (labs((long int)cur_frame - (long int)pos.frame)>nframes + 1 && abs(utick - cur_utick)> seq->score()->utime2utick((qreal)nframes / MScore::sampleRate) + 1) {
703  if (MScore::debugMode)
