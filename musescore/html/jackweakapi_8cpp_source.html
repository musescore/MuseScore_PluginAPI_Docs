101 void *load_jack_function(const char *fn_name)
102 {
103  void *fn = 0;
104  if (!libjack_handle) {
105  qDebug("libjack not found, so do not try to load %s ffs !", fn_name);
106  return 0;
107  }
108 #ifdef WIN32
109  fn = (void*)GetProcAddress(libjack_handle, fn_name);
110 #else
111  fn = dlsym(libjack_handle, fn_name);
112 #endif
113  if (!fn) {
114 #ifdef WIN32
115  char* lpMsgBuf;
116  FormatMessage(FORMAT_MESSAGE_ALLOCATE_BUFFER | FORMAT_MESSAGE_FROM_SYSTEM | FORMAT_MESSAGE_IGNORE_INSERTS,NULL,GetLastError(),MAKELANGID(LANG_NEUTRAL, SUBLANG_DEFAULT),(LPTSTR) &lpMsgBuf,0,NULL );
117  qDebug("could not GetProcAddress( %s ), %s ", fn_name, lpMsgBuf) ;
118 #else
119  qDebug ("could not dlsym( %s ), %s ", fn_name, dlerror()) ;
120 #endif
121  }
122  return fn;
123 }
