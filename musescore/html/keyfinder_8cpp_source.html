23 /* _Important comment about how pitches and keys are represented_
24 
25  Notes may be inputted in either Note format (with just a pitch) or
26  TPCNote format (with a TPC). The user may also specify
27  "npc_or_tpc_profile": 0 for npc, 1 for tpc. (A TPC profile is like
28  an NPC profile except that a user-settable default value is used for
29  TPC's outside the range of b2 to #4.) We assume that Note format
30  will be used with an npc profile; if not, a fatal error is
31  reported. TPCNote format could be used with either; if you use
32  TPCNote input with an npc profile, the notes are simply mapped on to
33  one cycle of the LOF. (Scoring mode 0 or 3 requires an npc profile;
34  otherwise a fatal error is reported.)
35 
36  Pitches (e.g. note[].tpc values) are always represented in
37  line-of-fifths order, with C = 14. (This is done when the pitches
38  are first read in). (If there's Note input, or TPCNote input and
39  npc_or_tpc_profile = 0, all pitches are shifted into one cycle of
40  the line of fifths, from 9 to 20 inclusive; again, this is done when
41  the input is read.)
42 
43  Keys are also represented in line-of-fifths order. Major keys are 0
44  to 27, C=14; minor keys are 28-55, C minor = 42. If
45  npc_or_tpc_profile = 0, the search is nominally done on all keys,
46  but only keys from 9 to 20 (inclusive) and 37 to 48 (inclusive) are
47  looked at; others are given large negative values.
48 
49  Although key-profiles are read in in pitch-height order, they are
50  then adjusted (in generate_tpc/npc_profiles) to line-of-fifths
51  order, with the tonic as 5. */

172 static void create_segments()
173  {
174  segment.append(MidiSegment());
175 
176  segment[0].start = firstbeat; // Always start a segment at the very beginning of the piece (the first beat)
177  for (int b = 0; b < num_sbeats; b++) {
178  if (b == 0 && (sbeat[0].time - firstbeat) < ((sbeat[1].time-firstbeat) - (sbeat[0].time-firstbeat))/2)
179  continue;
180  /* If it's the first beat of the piece, and the upbeat is
181  less than half of the first beat interval, don't start a segment
182  */
183  else {
184  MidiSegment seg;
185  seg.start = sbeat[b].time;
186  segment.back().end = sbeat[b].time;
187  segment.append(seg);
188  }
189  }
190  int s = segment.size() - 1;
191 
192  /* If final segment starts at or after final timepoint of piece, ignore it,
193  decrementing number of segments by 1; if not, set that segment's ending
194  to final timepoint of piece
195  */
196  if (segment[s].start >= final_timepoint) {
197  s--;
198  }
199  else {
200  segment[s].end = final_timepoint;
201  /* qDebug("Final segment ends at %d", segment[s].end); */
202  }
203  segtotal = s; // index of final segment
204  }

234  }
235  if (ontime < start && offtime > start && offtime <= end) {
236  // note ends, doesn't begin in segment
237  Event sn(ME_NOTE);

244  // note doesn't begin or end in segment
245  Event sn(ME_NOTE);
246  sn.setDataA(n.dataA());
247  sn.setTpc(n.tpc());

357  {
358  int key, shift, tpc, i;
359  float majp[12];
360  float minp[12];
361 
362  /* First we rearrange the key profile values (inputted in pitch height order) into lof order, C = 5 */
363  for(i=0; i<12; i++) {
364  majp[((((i * 7) % 12) + 5) % 12)] = major_profile[i];

373  }
374  if (tpc-shift < -5 || tpc-shift > 6) {
375  key_profile[key][tpc] = default_profile_value;
376  }
377  }
378  }
379  for (key=28, shift=0; key<56; ++key, ++shift) {
380  for (tpc=0; tpc<28; ++tpc) {
381  if (tpc-shift >= -5 && tpc-shift <= 6 ) {
382  key_profile[key][tpc] = minp[5 + (tpc-shift)];
383  }
384  if (tpc-shift < -5 || tpc-shift > 6) {
385  key_profile[key][tpc] = default_profile_value;

399 //-----------------------------------------------------------------------------
400 // generate_npc_profiles
401 // This is an alternative function for generating profiles given NPC
402 // profile. It's similar to the TPC version, except that only keys on
403 // a certain range of the line are assigned non-zero values. (This is
404 // really an unnecessary step, as steps outside the range will be
405 // disqualified in "match_profiles" in any case.) Also, for those keys
406 // considered, only key profile slots within the 9-to-20 range are
407 // assigned non-zero values. (This is necessary, since the input profile
408 // is within this range as well.) We begin by assigning default values
409 // of ZERO to everything.
410 //-----------------------------------------------------------------------------
411 
412 static void generate_npc_profiles()
413  {
414  int key, shift, tpc, tpc_to_use, i;
415  float majp[12];
416  float minp[12];
417 
418  /* First we rearrange the key profile values (inputted in pitch height order) into lof order, C = 5 */
419  for(i=0; i<12; i++) {
420  majp[((((i * 7) % 12) + 5) % 12)] = major_profile[i];
421  minp[((((i * 7) % 12) + 5) % 12)] = minor_profile[i];
422  }
423  for (key=0; key<56; ++key) {
424  for(tpc=0; tpc<28; ++tpc) {
425  key_profile[key][tpc]=0;
426  }
427  }
428 
429  for (key=9, shift=9; key<21; ++key, ++shift) {
430  for (tpc=0; tpc<28; ++tpc) {
431  /* tpc_to_use is the profile slot to use for a given value of the key-profile. In this way we keep
432  all profile slots within the 9-to-20 range
433  */
434  if(tpc<9)
435  tpc_to_use=tpc+12;
436  else if(tpc>20)
437  tpc_to_use=tpc-12;
438  else
439  tpc_to_use = tpc;
440  if (tpc-shift >= -5 && tpc-shift <= 6) {
441  /* For example: for key 14 (C major) and tpc 17 (A),
442  read from profile step 5 + (17-14) = 8. For degree
443  6 of B major (key 19), degree 6 (22) is outside the
444  9-to-20 range, so tpc_to_use is 22-12=10; still
445  read from profile step 5 + (22-19) = 8. */
446 
447  key_profile[key][tpc_to_use] = majp[5 + (tpc-shift)];
448  }
449  }
450  }
451  for (key = 37, shift = 9; key < 49; ++key, ++shift) {
452  for (tpc = 0; tpc < 28; ++tpc) {
453  if (tpc<9)
454  tpc_to_use = tpc + 12;
455  else if (tpc > 20)
456  tpc_to_use = tpc - 12;
457  else
458  tpc_to_use = tpc;
459  if (tpc - shift >= -5 && tpc - shift <= 6 ) {
460  key_profile[key][tpc_to_use] = minp[5 + (tpc-shift)];
461  }
462  }
463  }

474 //---------------------------------------------------------
475 // match_profiles
476 // Here we generate the "key scores" - the local score for each key
477 //
478 // Notice that we generate profiles for all 56 keys and all 28
479 // tpc's, even in the case where npc profiles are being used. In
480 // this case, though, all keys outside the allowable NPC range are
481 // given large negative values. And within both the key profiles
482 // and the segment profiles, only TPC's within the range 9 to 20
483 // have been given nonzero values.
484 //---------------------------------------------------------
485 
486 static void match_profiles()
487  {
488  int key, tpc, s, best_key, i;
489  qreal major_sumsq, minor_sumsq, input_sumsq;
490  qreal kprob[56];
491 
492  for (key = 0; key < 56; ++key) {
493  for (s = 0; s <= segtotal; ++s)
494  keyscore()[key].append(0.0);
495  }
496 
497  if (scoring_mode==0) {
498  major_sumsq = 0.0;
499  minor_sumsq = 0.0;
500  for(i=0; i<12; i++)
501  major_sumsq += major_profile[i]*major_profile[i];
502  for(i=0; i<12; i++)
503  minor_sumsq += minor_profile[i]*minor_profile[i];
504  if (verbosity==3)
505  qDebug("major_sumsq = %6.3f, minor_sumsq = %6.3f", major_sumsq, minor_sumsq);
506  }
507 
508  qreal total_prob[segtotal + 1];
509 
510  for (s = 0; s <= segtotal; ++s) {
511  if (scoring_mode==0) {
512  input_sumsq = 0.0;
513  for (i = 9; i <= 20; i++) {
514  input_sumsq += pow((seg_prof[i][s]-segment[s].average_dur), 2.0);
515  /* qDebug("%d X %6.3f squared is %6.3f", seg_prof[i][s], segment[s].average_dur, pow((seg_prof[i][s]-segment[s].average_dur), 2.0)); */
516  }
517  if (verbosity==3)
518  qDebug("For segment %d: average_dur = %6.3f; input_sumsq = %6.3f", s, segment[s].average_dur, input_sumsq);
519  }
520  best_key=0;
521 
522  for (key=0; key<56; ++key) {
523  kprob[key] = 0.0;
524  keyscore()[key][s] = -1000000.0;
525  if (npc_or_tpc_profile==0 && (key<9 || (key>20 && key<37) || key>48))
526  continue;
527  kprob[key] = 1.0;
528  keyscore()[key][s] = 0.0;
529  for (tpc=0; tpc<28; ++tpc) {
530 
531  /*
532  If scoring mode is 0, this is the K-S algorithm (this works for npc mode only). Segment
533  profile values represent total duration of each pc (in all other cases, they're just 1
534  for present pc's and 0 for absent ones). Key-profiles have been normalized linearly
535  around the average key-profile value. We normalize the input values similarly by taking
536  (seg_prof[tpc][s]-segment[s].average_dur). Then we multiply each normalized KP value by
537  the normalized input value, and sum these products; this gives us the numerator of the
538  correlation expression (as commented below). We've summed the squares of the normalized
539  key-profile value (major_sumsq and minor_sumsq above) and the normalized input values
540  (input_sumsq above), so this allows us to calculate the denominator also.
541 
542  If scoring_mode is 1, the key score is the sum of key-profile values for all pc's present
543  (this is the algorithm used in CBMS)
544 
545  If scoring_mode is 2, calculate key scores as above, but divide each one by the number
546  of pc's in the segment
547 
548  If scoring_mode is 3: for each key, add the log of the key-profile value for all present pc's;
549  subtract values for all absent pc's. (This is the Bayesian approach; assume key-profiles
550  represent pc distribution's in a corpus, i.e. the number of segments containing each scale
551  degree)
552  */
553 
554  if(scoring_mode == 0) {
555  if(tpc<9 || tpc>20)
556  continue;
557  /* calculate numerator */
558  keyscore()[key][s] += key_profile[key][tpc] * (seg_prof[tpc][s]-segment[s].average_dur);
559  /* qDebug("x-X=%6.3f, y-Y=%6.3f, product=%6.3f, new total=%6.3f", key_profile[key][tpc], seg_prof[tpc][s]-segment[s].average_dur, key_profile[key][tpc] * (seg_prof[tpc][s]-segment[s].average_dur), keyscore()[key][s]); */
560  }
561 
562  if(scoring_mode==1 || scoring_mode==2)
563  keyscore()[key][s] += (key_profile[key][tpc] * seg_prof[tpc][s]);
564 
565  if(scoring_mode == 3) {
566  /* if(tpc>11) continue; */
567  /* if(tpc<9 || tpc>20) continue; */
568 
569  if(seg_prof[tpc][s]==0) {
570  keyscore()[key][s] += log(1.000 - key_profile[key][tpc]);
571  /* qDebug("kp value = %6.3f: log(1-p) = %6.3f: score = %6.3f", key_profile[key][tpc], log(1.000 - key_profile[key][tpc]), keyscore()[key][s]); */
572  if(tpc>=9 && tpc<=20)
573  kprob[key] *= (1.000 - key_profile[key][tpc]);
574  }
575  else {
576  keyscore()[key][s] += log(key_profile[key][tpc]);
577  if(tpc>=9 && tpc<=20)
578  kprob[key] *= key_profile[key][tpc];
579  }
580 
581  /* qDebug("kp value = %6.3f: log(p) = %6.3f: score = %6.3f", key_profile[key][tpc], log(key_profile[key][tpc]), keyscore()[key][s]); */
582  }
583  }
584 
585  if(scoring_mode == 0) {
