81 void LayoutBreak::read(XmlReader& e)
82  {
83  while (e.readNextStartElement()) {
84  const QStringRef& tag(e.name());
85  if (tag == "subtype")
86  readProperty(e, Pid::LAYOUT_BREAK);
87  else if (tag == "pause")
88  readProperty(e, Pid::PAUSE);
89  else if (tag == "startWithLongNames")
90  readProperty(e, Pid::START_WITH_LONG_NAMES);
91  else if (tag == "startWithMeasureOne")
92  readProperty(e, Pid::START_WITH_MEASURE_ONE);
93  else if (tag == "firstSystemIdentation")
94  readProperty(e, Pid::FIRST_SYSTEM_INDENTATION);
95  else if (!Element::readProperties(e))
96  e.unknown();
97  }
98  layout0();
99  }

106  {

114

120 
121  painter->fillPath(stroke, selected() ? MScore::selectColor[0] : MScore::layoutBreakColor);
122 
123  painter->setPen(QPen(selected() ? MScore::selectColor[0] : MScore::layoutBreakColor,

125  painter->setBrush(Qt::NoBrush);

133 void LayoutBreak::layout0()
134  {
135  qreal _spatium = spatium();
136  path = QPainterPath();
137  path2 = QPainterPath();
138  qreal h = _spatium * 2.5;
139  qreal w = _spatium * 2.5;
140 
141  QRectF rect(0.0, 0.0, w, h);
142  path.addRect(rect);
143 
144  switch (layoutBreakType()) {
145  case Type::LINE:
146  path2.moveTo(w * .8, h * .3);
147  path2.lineTo(w * .8, h * .6);
148  path2.lineTo(w * .3, h * .6);
149 
150  path2.moveTo(w * .4, h * .5);
151  path2.lineTo(w * .25, h * .6);
152  path2.lineTo(w * .4, h * .7);
153  path2.lineTo(w * .4, h * .5);
154  break;
155 
156  case Type::PAGE:
157  path2.moveTo(w*.25, h*.2);
158  path2.lineTo(w*.60, h*.2);
159  path2.lineTo(w*.75, h*.35);
160  path2.lineTo(w*.75, h*.8);
161  path2.lineTo(w*.25, h*.8);
162  path2.lineTo(w*.25, h*.2);
163 
164  path2.moveTo(w*.55, h*.21); // 0.01 to avoid overlap
165  path2.lineTo(w*.55, h*.40);
166  path2.lineTo(w*.74, h*.40);
167  break;
168 
169  case Type::SECTION:
170  path2.moveTo(w*.25, h*.2);
171  path2.lineTo(w*.75, h*.2);
172  path2.lineTo(w*.75, h*.8);
173  path2.lineTo(w*.25, h*.8);
174 
175  path2.moveTo(w*.55, h*.21); // 0.01 to avoid overlap
176  path2.lineTo(w*.55, h*.79);
177  break;
178 
179  case Type::NOBREAK:
180  path2.moveTo(w * .1, h * .5);
181  path2.lineTo(w * .9, h * .5);
182 
183  path2.moveTo(w * .7, h * .3);
184  path2.lineTo(w * .5, h * .5);
185  path2.lineTo(w * .7, h * .7);
186  path2.lineTo(w * .7, h * .3);
187 
188  path2.moveTo(w * .3, h * .3);
189  path2.lineTo(w * .5, h * .5);
190  path2.lineTo(w * .3, h * .7);
191  path2.lineTo(w * .3, h * .3);
192  break;
193 
194  default:
195  qDebug("unknown layout break symbol");
196  break;
197  }
198  QRectF bb(0, 0, w, h);
199  bb.adjust(-lw, -lw, lw, lw);
200  setbbox(bb);
201  }
