50 void Score::resetSystems(bool layoutAll, LayoutContext& lc)
51  {
52  Page* page = 0;
53  if (layoutAll) {
54  for (System* s : qAsConst(_systems)) {
55  for (SpannerSegment* ss : s->spannerSegments())
56  ss->setParent(0);
57  }
58  qDeleteAll(_systems);
59  _systems.clear();
60  qDeleteAll(pages());
61  pages().clear();
62  if (!firstMeasure()) {
63  qDebug("no measures");
64  return;
65  }
66 
67  for (MeasureBase* mb = first(); mb; mb = mb->next())
68  mb->setSystem(0);
69 
70  page = new Page(this);
71  pages().push_back(page);
72  page->bbox().setRect(0.0, 0.0, loWidth(), loHeight());
73  page->setNo(0);
74 
75  System* system = new System(this);
76  _systems.push_back(system);
77  page->appendSystem(system);
78  system->adjustStavesNumber(nstaves());
79  }
80  else {
81  if (pages().isEmpty())
82  return;
83  page = pages().front();
84  System* system = systems().front();
85  system->clear();
86  system->adjustStavesNumber(nstaves());
87  }
88  lc.page = page;
89  }

96  void Score::collectLinearSystem(LayoutContext& lc)
97  {
98  System* system = systems().front();
99  system->setInstrumentNames(/* longNames */ true);
100 
101  QPointF pos;
102  bool firstMeasure = true; //lc.startTick.isZero();
103 
104  //set first measure to lc.nextMeasures for following
105  //utilizing in getNextMeasure()
106  lc.nextMeasure = _measures.first();
107  lc.tick = Fraction(0, 1);
108  getNextMeasure(lc);
109 
110  while (lc.curMeasure) {
111  qreal ww = 0.0;
112  if (lc.curMeasure->isVBox() || lc.curMeasure->isTBox()) {
113  lc.curMeasure->setParent(nullptr);
114  getNextMeasure(lc);
115  continue;
116  }
117  system->appendMeasure(lc.curMeasure);
118  if (lc.curMeasure->isMeasure()) {
119  Measure* m = toMeasure(lc.curMeasure);
120  if (m->mmRest()) {
121  m->mmRest()->setSystem(nullptr);
122  }
123  if (firstMeasure) {
124  system->layoutSystem(pos.rx());
125  if (m->repeatStart()) {
126  Segment* s = m->findSegmentR(SegmentType::StartRepeatBarLine, Fraction(0,1));
127  if (!s->enabled())
128  s->setEnabled(true);
129  }
130  m->addSystemHeader(true);
131  pos.rx() += system->leftMargin();
132  firstMeasure = false;
133  }
134  else if (m->header())
135  m->removeSystemHeader();
136  if (m->trailer())
137  m->removeSystemTrailer();
138  if (m->tick() >= lc.startTick && m->tick() <= lc.endTick) {
139  // for measures in range, do full layout
140  m->createEndBarLines(false);
141  m->computeMinWidth();
142  ww = m->width();
143  m->stretchMeasure(ww);
144  }
145  else {
146  // for measures not in range, use existing layout
147  ww = m->width();
148  if (m->pos() != pos) {
149  // fix beam positions
150  // other elements with system as parent are processed in layoutSystemElements()
151  // but full beam processing is expensive and not needed if we adjust position here
152  QPointF p = pos - m->pos();
153  for (const Segment& s : m->segments()) {
154  if (!s.isChordRestType())
155  continue;
156  for (int track = 0; track < ntracks(); ++track) {
157  Element* e = s.element(track);
158  if (e) {
159  ChordRest* cr = toChordRest(e);
160  if (cr->beam() && cr->beam()->elements().front() == cr)
161  cr->beam()->rpos() += p;
162  }
163  }
164  }
165  }
166  }
167  m->setPos(pos);
168  m->layoutStaffLines();
169  }
170  else if (lc.curMeasure->isHBox()) {
171  lc.curMeasure->setPos(pos + QPointF(toHBox(lc.curMeasure)->topGap(), 0.0));
172  lc.curMeasure->layout();
173  ww = lc.curMeasure->width();
174  }
175  pos.rx() += ww;
176 
177  getNextMeasure(lc);
178  }
179 
180  system->setWidth(pos.x());
181  }

202 void LayoutContext::layoutLinear()
203  {
204  System* system = score->systems().front();
205 
206  score->layoutSystemElements(system, *this);
207 
208  system->layout2(); // compute staff distances
209 
210  for (MeasureBase* mb : system->measures()) {
211  if (!mb->isMeasure())
212  continue;
213  Measure* m = toMeasure(mb);
214 
215  for (int track = 0; track < score->ntracks(); ++track) {
216  for (Segment* segment = m->first(); segment; segment = segment->next()) {
217  Element* e = segment->element(track);
218  if (!e)
219  continue;
220  if (e->isChordRest()) {
221  if (m->tick() < startTick || m->tick() > endTick)
222  continue;
223  if (!score->staff(track2staff(track))->show())
224  continue;
225  ChordRest* cr = toChordRest(e);
226  if (notTopBeam(cr)) // layout cross staff beams
227  cr->beam()->layout();
228  if (notTopTuplet(cr)) {
229  // fix layout of tuplets
230  DurationElement* de = cr;
231  while (de->tuplet() && de->tuplet()->elements().front() == de) {
232  Tuplet* t = de->tuplet();
233  t->layout();
234  de = de->tuplet();
235  }
236  }
237 
238  if (cr->isChord()) {
239  Chord* c = toChord(cr);
240  for (Chord* cc : c->graceNotes()) {
241  if (cc->beam() && cc->beam()->elements().front() == cc)
242  cc->beam()->layout();
243  cc->layoutSpanners();
244  for (Element* element : cc->el()) {
245  if (element->isSlur())
246  element->layout();
247  }
248  }
249  c->layoutArpeggio2();
250  c->layoutSpanners();
251  if (c->tremolo()) {
252  Tremolo* t = c->tremolo();
253  Chord* c1 = t->chord1();
254  Chord* c2 = t->chord2();
255  if (t->twoNotes() && c1 && c2 && (c1->staffMove() || c2->staffMove()))
256  t->layout();
257  }
258  }
259  }
260  else if (e->isBarLine())
261  toBarLine(e)->layout2();
262  }
263  }
264  m->layout2();
265  }
266  page->setPos(0, 0);
267  system->setPos(page->lm(), page->tm() + score->styleP(Sid::staffUpperBorder));
268  page->setWidth(system->width() + system->pos().x());
269  // Set buffer space after the last system to avoid problems with mouse input.
270  // Mouse input divides space between systems equally (see Score::searchSystem),
271  // hence the choice of the value.
272  const qreal buffer = 0.5 * score->styleS(Sid::maxSystemDistance).val() * score->spatium();
273  page->setHeight(system->height() + system->pos().y() + buffer);
274  page->rebuildBspTree();
275  }
