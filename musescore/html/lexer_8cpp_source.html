342 
343  void Lexer::getSym()
344  {
345  qDebug() << "Lexer::getSym()";
346 
347  // if unparsed words remaining, use these
348  if (list.size() > 0)
349  {
350  categorizeWord(list.at(0));
351  list.removeFirst();
352  return;
353  }
354 
355  // read the next non-empty line
356  do
357  {
358  line = in.readLine();
359  ++lineNumber;
360  if (line.isNull())
361  {
362  // end of file
363  qDebug() << "-> end of file";
364  type = NONE;
365  value = "";
366  return;
367  }
368  }
369  while (line == "");
370 
371  qDebug() << "getSym: read line" << line;
372  QRegExp rHeaderIgnore("^Bagpipe Reader|^MIDINoteMappings|^FrequencyMappings"
373  "|^InstrumentMappings|^GracenoteDurations|^FontSizes"
374  "|^TuneFormat");
375  QRegExp rTuneTempo("^TuneTempo");
376  if (rHeaderIgnore.indexIn(line) == 0)
377  {
378  type = COMMENT;
379  value = "";
380  qDebug()
381  << "-> header ignore,"
382  << "type:" << symbolToString(type)
383  << "value:" << value
384  ;
385  line = "";
386  return;
387  }
388  else if (rTuneTempo.indexIn(line) == 0)
389  {
390  type = TEMPO;
391  value = line;
392  qDebug()
393  << "-> tempo,"
394  << "type:" << symbolToString(type)
395  << "value:" << value
396  ;
397  line = "";
398  return;
399  }
400  else if (line.at(0) == '"')
401  {
402  type = STRING;
403  value = line;
404  qDebug()
405  << "-> quoted string,"
406  << "type:" << symbolToString(type)
407  << "value:" << value
408  ;
409  line = "";
410  }
411  else
412  {
413  // split line into space-separated words
414  list = line.trimmed().split(QRegExp("\\s+"));
415  qDebug()
416  << "-> words"
417  << list
418  ;
419  categorizeWord(list.at(0));
420  list.removeFirst();
421  }
422  line = "";
423  }

447  void Lexer::categorizeWord(QString word)
448  {
449  qDebug() << "Lexer::categorizeWord(" << word << ")";
450 
451  // default values
452  type = NONE;
453  value = word;
454 
455  QRegExp rClef("&");
456  QRegExp rKey("sharp[cf]");
457  QRegExp rTSig("\\d+_(1|2|4|8|16|32)");
458  QRegExp rPart("I!''|I!|''!I|!I|'intro|[2-9]|'[12]|_'");
459  QRegExp rBar("!|!t|!!t");
460  QRegExp rNote("(LG|LA|[B-F]|HG|HA)[lr]?_(1|2|4|8|16|32)");
461  QRegExp rTie("\\^t[es]");
462  QRegExp rTriplet("\\^3[es]");
463  QRegExp rDot("'([hl][ag]|[b-f])");
464 
465  if (rClef.exactMatch(word))
466  type = CLEF;
467  else if (rKey.exactMatch(word))
468  type = KEY;
469  else if (rTSig.exactMatch(word))
470  type = TSIG;
471  else if (rPart.exactMatch(word))
472  type = PART;
473  else if (rBar.exactMatch(word))
474  type = BAR;
475  else if (rNote.exactMatch(word))
476  type = NOTE;
477  else if (rTie.exactMatch(word))
478  type = TIE;
479  else if (rTriplet.exactMatch(word))
480  type = TRIPLET;
481  else if (rDot.exactMatch(word))
482  type = DOT;
483  else if (graceMap.contains(word))
484  {
485  type = GRACE;
486  value = graceMap.value(word);
487  }
488  else
489  type = UNKNOWN;
490 
491  qDebug()
492  << " type: " << qPrintable(symbolToString(type))
493  << " value: '" << qPrintable(value) << "'"
494  ;
495  }
