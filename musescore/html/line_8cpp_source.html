140  {

144  if (!system() || grip == Grip::APERTURE)
145  return result;

148  qreal y;
149  if (spanner()->anchor() == Spanner::Anchor::NOTE)

207 //---------------------------------------------------------
208 // edit
209 // return true if event is accepted
210 //---------------------------------------------------------
211 
212 bool LineSegment::edit(EditData& ed)
213  {
214  const bool moveStart = ed.curGrip == Grip::START;
215  const bool moveEnd = ed.curGrip == Grip::END || ed.curGrip == Grip::MIDDLE;
216 
217  if (!((ed.modifiers & Qt::ShiftModifier)
218  && ((isSingleBeginType() && moveStart) || (isSingleEndType() && moveEnd))
219  ))
220  return false;
221 
222  LineSegment* ls = 0;
223  SpannerSegmentType st = spannerSegmentType(); // may change later
224  SLine* l = line();
225  int track = l->track();
226  int track2 = l->track2(); // assumed to be same as track
227 
228  switch (l->anchor()) {
229  case Spanner::Anchor::SEGMENT:
230  {
231  Segment* s1 = spanner()->startSegment();
232  Segment* s2 = spanner()->endSegment();
233  // check for line going to end of score
234  if (spanner()->tick2() >= score()->lastSegment()->tick()) {
235  // endSegment calculated above will be the last chord/rest of score
236  // but that is not correct - it should be an imaginary note *after* the end of the score
237  // best we can do is set s2 to lastSegment (probably the end barline)
238  s2 = score()->lastSegment();
239  }
240  if (!s1 && !s2) {
241  qDebug("LineSegment::edit: no start/end segment");
242  return true;
243  }
244  if (ed.key == Qt::Key_Left) {
245  if (moveStart)
246  s1 = prevSeg1(s1, track);
247  else if (moveEnd)
248  s2 = prevSeg1(s2, track2);
249  }
250  else if (ed.key == Qt::Key_Right) {
251  if (moveStart)
252  s1 = nextSeg1(s1, track);
253  else if (moveEnd) {
254  Segment* ns2 = nextSeg1(s2, track2);
255  if (ns2)
256  s2 = ns2;
257  else
258  s2 = score()->lastSegment();
259  }
260  }
261  if (s1 == 0 || s2 == 0 || s1->tick() >= s2->tick())
262  return true;
263  spanner()->undoChangeProperty(Pid::SPANNER_TICK, s1->tick());
264  spanner()->undoChangeProperty(Pid::SPANNER_TICKS, s2->tick() - s1->tick());
265  }
266  break;
267  case Spanner::Anchor::NOTE:
268  {
269  Note* note1 = toNote(l->startElement());
270  Note* note2 = toNote(l->endElement());
271  Note* oldNote1 = note1;
272  Note* oldNote2 = note2;
273  if (!note1 && !note2) {
274  qDebug("LineSegment::edit: no start/end note");
275  return true; // accept the event without doing anything
276  }
277 
278  switch (ed.key) {
279  case Qt::Key_Left:
280  if (moveStart)
281  note1 = prevChordNote(note1);
282  else if (moveEnd)
283  note2 = prevChordNote(note2);
284  break;
285  case Qt::Key_Right:
286  if (moveStart)
287  note1 = nextChordNote(note1);
288  else if (moveEnd)
289  note2 = nextChordNote(note2);
290  break;
291  case Qt::Key_Up:
292  if (moveStart)
293  note1 = toNote(score()->upAlt(note1));
294  else if (moveEnd)
295  note2 = toNote(score()->upAlt(note2));
296  break;
297  case Qt::Key_Down:
298  if (moveStart)
299  note1 = toNote(score()->downAlt(note1));
300  else if (moveEnd)
301  note2 = toNote(score()->downAlt(note2));
302  break;
303  default:
304  return true;
305  }
306 
307  // check prevChordNote() and nextchordNote() didn't return null
308  // OR Score::upAlt() and Score::downAlt() didn't return non-Note (notably rests)
309  // OR spanner duration is > 0
310  // OR note1 and note2 didn't end up in different instruments
311  // if this is the case, accepts the event and return without doing nothing
312  if (!note1 || !note2
313  || !note1->isNote() || !note2->isNote()
314  || note1->chord()->tick() >= note2->chord()->tick()
315  || note1->chord()->staff()->part()->instrument(note1->chord()->tick())
316  != note2->chord()->staff()->part()->instrument(note2->chord()->tick()) )
317  return true;
318  if (note1 != oldNote1 || note2 != oldNote2)
319  score()->undoChangeSpannerElements(spanner(), note1, note2);
320  }
321  break;
322  case Spanner::Anchor::MEASURE:
323  case Spanner::Anchor::CHORD:
324  {
325  Measure* m1 = l->startMeasure();
326  Measure* m2 = l->endMeasure();
327 
328  if (ed.key == Qt::Key_Left) {
329  if (moveStart) {
330  if (m1->prevMeasure())
331  m1 = m1->prevMeasure();
332  }
333  else if (moveEnd) {
334  Measure* m = m2->prevMeasure();
335  if (m)
336  m2 = m;
337  }
338  }
339  else if (ed.key == Qt::Key_Right) {
340  if (moveStart) {
341  if (m1->nextMeasure())
342  m1 = m1->nextMeasure();
343  }
344  else if (moveEnd) {
345  if (m2->nextMeasure())
346  m2 = m2->nextMeasure();
347  }
348  }
349  if (m1->tick() > m2->tick())
350  return true;
351  if (l->startElement() != m1) {
352  spanner()->undoChangeProperty(Pid::SPANNER_TICK, m1->tick());
353  spanner()->undoChangeProperty(Pid::SPANNER_TICKS, m2->endTick() - m1->tick());
354  }
355  else if (l->endElement() != m2) {
356  spanner()->undoChangeProperty(Pid::SPANNER_TICKS, m2->endTick() - m1->tick());
357  }
358  }
359  }
360  triggerLayout();
361  l->layout(); // recompute segment list, segment type may change
362 
363  LineSegment* nls = 0;
364  if (st == SpannerSegmentType::SINGLE) {
365  if (moveStart)
366  nls = l->frontSegment();
367  else if (moveEnd)
368  nls = l->backSegment();
369  }
370  else if (st == SpannerSegmentType::BEGIN)
371  nls = l->frontSegment();
372  else if (st == SpannerSegmentType::END)
373  nls = l->backSegment();
374  else
375  qDebug("spannerSegmentType %d", int(spannerSegmentType()));
376 
377  if (nls && (nls != this))
378  ed.view->changeEditElement(nls);
379  if (ls)
380  score()->undoRemoveElement(ls);
381 
382  triggerLayout();
383  return true;
384  }

390 Segment* LineSegment::findSegmentForGrip(Grip grip, QPointF pos) const
391  {
392  if (grip != Grip::START && grip != Grip::END)
393  return nullptr;
394 
395  SLine* l = line();
396  const bool left = (grip == Grip::START);
397 
398  Segment* const oldSeg = left ? l->startSegment() : score()->tick2leftSegmentMM(l->tick2() - Fraction::eps());
399  const int oldStaffIndex = left ? staffIdx() : track2staff(l->effectiveTrack2());
400 
401  const qreal spacingFactor = left ? 0.5 : 1.0; // defines the point where canvas is divided between segments, systems etc.
402 
403  System* sys = oldSeg->system();
404  const QList<System*> foundSystems = score()->searchSystem(pos, sys, spacingFactor);
405 
406  if (!foundSystems.empty() && !foundSystems.contains(sys))
407  sys = foundSystems[0];
408 
409  // Restrict searching segment to the correct staff
410  pos.setY(sys->staffCanvasYpage(oldStaffIndex));
411 
412  Segment* seg = nullptr; // don't prefer any segment while searching line position
413  int staffIndex = oldStaffIndex;
414  score()->dragPosition(pos, &staffIndex, &seg, spacingFactor);
415 
416  return seg;
417  }

456 //---------------------------------------------------------
457 // rebaseAnchor
458 /// If system has changed, returns the new line segment
459 /// (may appear to be reused \c this segment), otherwise
460 /// returns nullptr.
461 //---------------------------------------------------------
462 
463 LineSegment* LineSegment::rebaseAnchor(Grip grip, Segment* newSeg)
464  {
465  switch (grip) {
466  case Grip::START:
467  if (!isSingleBeginType())
468  return nullptr;
469  break;
470  case Grip::END:
471  if (!isSingleEndType())
472  return nullptr;
473  break;
474  default:
475  return nullptr;
476  }
477 
478  SLine* l = line();
479  const bool left = (grip == Grip::START);
480  Segment* const oldSeg = left ? l->startSegment() : score()->tick2leftSegmentMM(l->tick2() - Fraction::eps());
481  System* const oldSystem = system();
482 
483  if (!newSeg || oldSeg == newSeg)
484  return nullptr;
485 
486  Fraction startTick = left ? newSeg->tick() : l->tick();
487  Fraction endTick = left ? l->tick2() : lastSegmentEndTick(newSeg, l);
488 
489  if (endTick <= startTick) {
490  if (left)
491  endTick = lastSegmentEndTick(newSeg, l);
492  else
493  startTick = newSeg->tick();
494  }
495 
496  bool anchorChanged = false;
497 
498  if (l->tick() != startTick) {
499  l->undoChangeProperty(Pid::SPANNER_TICK, startTick);
500  anchorChanged = true;
501  }
502 
503  const Fraction ticksLength = endTick - startTick;
504 
505  if (ticksLength != l->ticks()) {
506  l->undoChangeProperty(Pid::SPANNER_TICKS, ticksLength);
507  anchorChanged = true;
508  }
509 
510  if (newSeg->system() != oldSystem) {
511  l->layout();
512  return left ? l->frontSegment() : l->backSegment();
513  }
514  else if (anchorChanged) {
515  const QPointF delta = left ? deltaRebaseLeft(oldSeg, newSeg) : deltaRebaseRight(oldSeg, newSeg, track2staff(l->effectiveTrack2()));
516  if (left) {
517  setOffset(offset() + delta);
518  _offset2 -= delta;
519  setOffsetChanged(true);
520  }
521  else {
522  _offset2 += delta;
523  }
524  }
525 
526  return nullptr;
527  }

533 void LineSegment::rebaseAnchors(EditData& ed, Grip grip)
534  {
535  if (line()->anchor() != Spanner::Anchor::SEGMENT)
536  return;
537  // don't change anchors on keyboard adjustment or if Ctrl is pressed
538  // (Ctrl+Left/Right is handled elsewhere!)
539  if (ed.key == Qt::Key_Left || ed.key == Qt::Key_Right || ed.modifiers & Qt::ControlModifier)
540  return;
541 
542  switch (grip) {
543  case Grip::START:
544  case Grip::END: {
545  const bool left = (grip == Grip::START);
546 
547  if (left && !isSingleBeginType())
548  return;
549  if (!left && !isSingleEndType())
550  return;
551 
552  // Find an appropriate segment to bind from actual mouse
553  // position. This allows changing systems while dragging
554  // while not setting line position to something
555  // inappropriate.
556  Segment* seg = findSegmentForGrip(grip, ed.pos);
557  LineSegment* newLineSegment = rebaseAnchor(grip, seg);
558 
559  if (newLineSegment) {
560  // If new line segment is not the same as this one,
561  // switch to dragging that new segment.
562  if (newLineSegment != this) {
563  // Reset offset for the old line segment
564  if (left) {
565  _offset2.rx() -= offset().x();
566  setOffset(QPointF());
567  }
568  else {
569  setUserOff2(QPointF());
570  }
571 
572  // Switch to dragging the new line segment
573  ed.view->changeEditElement(newLineSegment);
574  }
575 
576  // Set offset for the new line segment for grip to appear under the mouse cursor
577  System* sys;
578  const QPointF lp = line()->linePos(grip, &sys);
579  const qreal xoff = sys->mapFromCanvas(ed.pos).x() - lp.x();
580 
581  if (left) {
582  newLineSegment->rxoffset() = xoff;
583  newLineSegment->setUserOff2(QPointF(-xoff, 0.0));
584  }
585  else {
586  newLineSegment->setUserXoffset2(xoff);
587  }
588  }
589  }
590  break;
591  case Grip::MIDDLE: {
592  if (!isSingleType())
593  return;
594 
595  SLine* l = line();
596 
597  // If dragging middle grip (or the entire hairpin), mouse position
598  // does not directly correspond to any sensible position, so use
599  // actual line coordinates instead. This method doesn't allow for
600  // system changes, but that seems OK when dragging the entire line:
601  // the line will just push away other systems according to autoplacement
602  // rules if necessary.
603  QPointF cpos = canvasPos();
604  cpos.setY(l->startSegment()->system()->staffCanvasYpage(l->staffIdx())); // prevent cross-system move
605 
606  Segment* seg1 = findSegmentForGrip(Grip::START, cpos);
607  Segment* seg2 = findSegmentForGrip(Grip::END, cpos + pos2());
608 
609  if (!(seg1 && seg2 && seg1->system() == seg2->system() && seg1->system() == system()))
610  return;
611 
612  rebaseAnchor(Grip::START, seg1);
613  rebaseAnchor(Grip::END, seg2);
614  }
615  default:
616  break;
617  }
618  }

624 void LineSegment::editDrag(EditData& ed)
625  {
626  // Only for resizing according to the diagonal properties
627  const QPointF deltaResize(ed.evtDelta.x(), line()->diagonal() ? ed.evtDelta.y() : 0.0);
628 
629  switch (ed.curGrip) {
630  case Grip::START: // Resize the begin of element (left grip)
631  setOffset(offset() + deltaResize);
632  _offset2 -= deltaResize;
633 
634  if (isStyled(Pid::OFFSET))
635  setPropertyFlags(Pid::OFFSET, PropertyFlags::UNSTYLED);
636 
637  rebaseAnchors(ed, ed.curGrip);
638  break;
639  case Grip::END: // Resize the end of element (right grip)
640  _offset2 += deltaResize;
641  rebaseAnchors(ed, ed.curGrip);
642  break;
643  case Grip::MIDDLE: { // Move the element (middle grip)
644  // Only for moving, no y limitation
645  const QPointF deltaMove(ed.evtDelta);
646  setOffset(offset() + deltaMove);
647  setOffsetChanged(true);
648  if (isStyled(Pid::OFFSET))
649  setPropertyFlags(Pid::OFFSET, PropertyFlags::UNSTYLED);
650  rebaseAnchors(ed, ed.curGrip);
651  }
652  break;
653  default:
654  break;
655  }
656  if (line()->anchor() == Spanner::Anchor::NOTE && ed.isStartEndGrip()) {
657  //
658  // if we touch a different note, change anchor
659  //
660  Element* e = ed.view->elementNear(ed.pos);
661  if (e && e->isNote()) {
662  SLine* l = line();
663  if (ed.curGrip == Grip::END && e != line()->endElement()) {
664  qDebug("LineSegment: move end anchor");
665  Note* noteOld = toNote(l->endElement());
666  Note* noteNew = toNote(e);
667  Note* sNote = toNote(l->startElement());
668  // do not change anchor if new note is before start note
669  if (sNote && sNote->chord() && noteNew->chord() && sNote->chord()->tick() < noteNew->chord()->tick()) {
670  score()->undoChangeSpannerElements(l, sNote, noteNew);
671 
672  _offset2 += noteOld->canvasPos() - noteNew->canvasPos();
673  }
674  }
675  else if (ed.curGrip == Grip::START && e != l->startElement())
676  qDebug("LineSegment: move start anchor (not impl.)");
677  }
678  }
679  triggerLayout();
680  }

766 /// - Anchor::NOTE: return anchor note position in system
767 /// coordinates
768 /// - Anchor::CHORD: not implemented
769 /// - Other: return (x position in system coordinates, 0)

771 
772 QPointF SLine::linePos(Grip grip, System** sys) const
773  {
774  qreal x = 0.0;
775  qreal sp = staff()->spatium(tick());
776  switch (anchor()) {
777  case Spanner::Anchor::SEGMENT:
778  {
779  ChordRest* cr;
780  if (grip == Grip::START) {
781  cr = toChordRest(startElement());
782  if (cr && type() == ElementType::OTTAVA) {
783  // some sources say to center the text over the notehead
784  // others say to start the text just to left of notehead
785  // some say to include accidental, others don't
786  // our compromise - left align, but account for accidental
787  if (cr->durationType() == TDuration::DurationType::V_MEASURE && !cr->measure()->hasVoices(cr->staffIdx()))
788  x = cr->x(); // center for measure rests
789 //TODO else if (cr->spaceLw > 0.0)
790 // x = -cr->spaceLw; // account for accidentals, etc
791  }
792  }
793  else {
794  cr = toChordRest(endElement());
795  if (isOttava()) {
796  if (cr && cr->durationType() == TDuration::DurationType::V_MEASURE) {
797  x = cr->x() + cr->width() + sp;
798  }
799  else if (cr) {
800  // lay out just past right edge of all notes for this segment on this staff
801 
802  Segment* s = cr->segment();
803 
804  int startTrack = staffIdx() * VOICES;
805  int endTrack = startTrack + VOICES;
806  qreal width = 0.0;
807 
808  // don’t consider full measure rests, which are centered
809  // (TODO: what if there is only a full measure rest?)
810 
811  for (int track = startTrack; track < endTrack; ++track) {
812  ChordRest* cr1 = toChordRest(s->element(track));
813  if (!cr1)
814  continue;
815  if (cr1->isChord()) {
816  for (Note* n : toChord(cr1)->notes())
817  width = qMax(width, n->shape().right() + n->pos().x() + cr1->pos().x());
818  }
819  else if (cr1->isRest() && (cr1->actualDurationType() != TDuration::DurationType::V_MEASURE))
820  width = qMax(width, cr1->bbox().right() + cr1->pos().x());
821  }
822 
823  x = width + sp;
824 
825  // extend past chord/rest
826  // but don't overlap next chord/rest
827 
828  bool crFound = false;
829  int n = staffIdx() * VOICES;
830  Segment* ns = s->next();
831  while (ns) {
832  for (int i = 0; i < VOICES; ++i) {
833  if (ns->element(n + i)) {
834  crFound = true;
835  break;
836  }
837  }
838  if (crFound)
839  break;
840  ns = ns->next();
841  }
842  if (crFound) {
843  qreal nextNoteDistance = ns->x() - s->x() + lineWidth();
844  if (x > nextNoteDistance)
845  x = qMax(width, nextNoteDistance);
846  }
847  }
848  }
849  else if (isLyricsLine() && toLyrics(parent())->ticks() > Fraction(0,1)) {
850  // melisma line
851  // it is possible CR won't be in correct track
852  // prefer element in current track if available
853  if (!cr)
854  qDebug("no end for lyricsline segment - start %d, ticks %d", tick().ticks(), ticks().ticks());
855  else if (cr->track() != track()) {
856  Element* e = cr->segment()->element(track());
857  if (e)
858  cr = toChordRest(e);
859  }
860 
861  // layout to right edge of CR
862  // except if CR is start element, in which case use a nominal length
863  if (cr && cr != toChordRest(startElement()))
864  x = cr->rightEdge();
865  else
866  x = spatium() - score()->styleP(Sid::minNoteDistance);
867  }
868  else if (isHairpin() || isTrill() || isVibrato() || isTextLine() || isLyricsLine()) {
869  // (for LYRICSLINE, this is hyphen; melisma line is handled above)
870  // lay out to just before next chordrest on this staff, or barline
871  // tick2 actually tells us the right chordrest to look for
872  if (cr && endElement()->parent() && endElement()->parent()->type() == ElementType::SEGMENT) {
873  qreal x2 = cr->x() /* TODO + cr->space().rw() */;
874  Segment* currentSeg = toSegment(endElement()->parent());
875  Segment* seg = score()->tick2segmentMM(tick2(), false, SegmentType::ChordRest);
876  if (!seg) {
877  // no end segment found, use measure width
878  x2 = endElement()->parent()->parent()->width() - sp;
879  }
880  else if (currentSeg->measure() == seg->measure()) {
881  // next chordrest found in same measure;
882  // end line 1sp to left
883  x2 = qMax(x2, seg->x() - sp);
884  }
885  else {
886  // next chordrest is in next measure
887  // lay out to end (barline) of current measure instead
888  seg = currentSeg->next(SegmentType::EndBarLine);
889  if (!seg)
890  seg = currentSeg->measure()->last();
891  // allow lyrics hyphen to extend to barline
892  // other lines stop 1sp short
893  qreal gap = (type() == ElementType::LYRICSLINE) ? 0.0 : sp;
894  qreal x3 = seg->enabled() ? seg->x() : seg->measure()->width();
895  x2 = qMax(x2, x3 - gap);
896  }
897  x = x2 - endElement()->parent()->x();
898  }
899  }
900  }
901 
902  Fraction t = grip == Grip::START ? tick() : tick2();
903  Measure* m = cr ? cr->measure() : score()->tick2measure(t);
904 
905  if (m) {
906  x += cr ? cr->segment()->pos().x() + m->pos().x() : m->tick2pos(t);
907  *sys = m->system();
908  }
909  else
910  *sys = 0;
911  }
912  break;
913 
914  case Spanner::Anchor::MEASURE:
915  {
916  // anchor() == Anchor::MEASURE
917  const Measure* m;
918  if (grip == Grip::START) {
919  m = startMeasure();
920  // start after clef/keysig/timesig/barline
921  qreal offset = 0.0;
922  Segment* s = m->first(SegmentType::ChordRest);
923  if (s) {
924  s = s->prev();
925  if (s && s->enabled()) {
926  offset = s->x();
927  Element* e = s->element(staffIdx() * VOICES);
928  if (e)
929  offset += e->width();
930  }
931  }
932  x = m->pos().x() + offset;
933  if (score()->styleB(Sid::createMultiMeasureRests) && m->hasMMRest()) {
934  x = m->mmRest()->pos().x();
935  }
936  }
937  else {
938  qreal _spatium = spatium();
939 
940  if (score()->styleB(Sid::createMultiMeasureRests)) {
941  // find the actual measure where the volta should stop
942  m = startMeasure();
943  if (m->hasMMRest())
944  m = m->mmRest();
945  while (m->nextMeasureMM() && (m->endTick() < tick2()))
946  m = m->nextMeasureMM();
947  }
948  else {
949  m = endMeasure();
950  }
951 
952  // back up to barline (skip courtesy elements)
953  Segment* seg = m->last();
954  while (seg && seg->segmentType() != SegmentType::EndBarLine)
955  seg = seg->prev();
956  if (!seg || !seg->enabled()) {
957  // no end bar line; look for BeginBarLine or StartRepeatBarLine of next measure
958  Measure* nm = m->nextMeasure();
959  if (nm->system() == m->system())
960  seg = nm->first(SegmentType::BeginBarLine|SegmentType::StartRepeatBarLine);
961  }
962  qreal mwidth = seg && seg->measure() == m ? seg->x() : m->bbox().right();
963  x = m->pos().x() + mwidth;
964  // align to barline
965  if (seg && (seg->segmentType() & SegmentType::BarLineType)) {
966  Element* e = seg->element(0);
967  if (e && e->type() == ElementType::BAR_LINE) {
968  BarLineType blt = toBarLine(e)->barLineType();
969  switch (blt) {
970  case BarLineType::END_REPEAT:
971  // skip dots
972  x += symWidth(SymId::repeatDot);
973  x += score()->styleS(Sid::endBarDistance).val() * _spatium;
974  // fall through
975  case BarLineType::DOUBLE:
976  // center on leftmost (thinner) barline
977  x += score()->styleS(Sid::doubleBarWidth).val() * _spatium * 0.5;
978  break;
979  case BarLineType::START_REPEAT:
980  // center on leftmost (thicker) barline
981  x += score()->styleS(Sid::endBarWidth).val() * _spatium * 0.5;
982  break;
983  default:
984  // center on barline
985  x += score()->styleS(Sid::barWidth).val() * _spatium * 0.5;
986  break;
987  }
988  }
989  }
990  }
991  if (score()->styleB(Sid::createMultiMeasureRests))
992  m = m->mmRest1();
993  Q_ASSERT(m->system());
994  *sys = m->system();
995  }
996  break;
997 
998  case Spanner::Anchor::NOTE: {
999  Element* e = grip == Grip::START ? startElement() : endElement();
1000  if (!e)
1001  return QPointF();
1002  Note* n = toNote(e);
1003  System* s = n->chord()->segment()->system();
1004  if (s == 0) {
1005  qDebug("no system: %s start %s chord parent %s\n", name(), n->name(), n->chord()->parent()->name());
1006  return QPointF();
1007  }
1008  *sys = s;
1009  // return the position of the anchor note relative to the system
1010 // QPointF elemPagePos = e->pagePos(); // DEBUG
1011 // QPointF systPagePos = s->pagePos();
1012 // qreal staffYPage = s->staffYpage(e->staffIdx());
1013  QPointF p = n->pagePos() - s->pagePos();
1014  if (!isGlissando())
1015  p.rx() += n->headWidth() * 0.5;
1016  return p;
1017  }
1018 
1019  case Spanner::Anchor::CHORD:
1020  qFatal("Sline::linePos(): anchor not implemented");
1021  break;
1022  }
1023  return QPointF(x, 0.0);
1024  }

1047  }

1075  lineSegm->setPos2(QPointF(x2 - p1.x(), 0.0));

1083  lineSegm->setPos(QPointF(x1, p1.y()));
1084  lineSegm->setPos2(QPointF(x2 - x1, 0.0));
1085  }
1086  break;

1089  QPointF p2 = linePos(Grip::END, &s);
1090  qreal x1 = system->firstNoteRestSegmentX(true);
1091  qreal len = p2.x() - x1;
1092  lineSegm->setPos(QPointF(p2.x() - len, p2.y()));
1093  lineSegm->setPos2(QPointF(len, 0.0));
1094  }
1095  break;
