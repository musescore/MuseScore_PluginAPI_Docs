131 void Location::fillPositionForElement(const Element* e, bool absfrac)
132  {
133  Q_ASSERT(isAbsolute());
134  if (!e) {
135  qWarning("Location::fillPositionForElement: element is nullptr");
136  return;
137  }
138  if (track() == absDefaults.track())

237 int Location::note(const Element* e)
238  {
239  if (e->isNote()) {
240  const Note* n = toNote(e);
241  const std::vector<Note*>& notes = n->chord()->notes();
242  if (notes.size() == 1)
243  return 0;
244  size_t noteIdx;
245  for (noteIdx = 0; noteIdx < notes.size(); ++noteIdx) {
246  if (n == notes.at(noteIdx))
247  break;
248  }
249  return static_cast<int>(noteIdx);
250  }
251  return absDefaults.note();
252  }

258 QVariant Location::getLocationProperty(Pid pid, const Element* start, const Element* end)
259  {
260  switch(pid) {
261  case Pid::LOCATION_STAVES:
262  return (track(start) / VOICES) - (track(end) / VOICES);
263  case Pid::LOCATION_VOICES:
264  return (track(start) % VOICES) - (track(end) / VOICES);
265  case Pid::LOCATION_MEASURES:
266  return measure(end) - measure(start);
267  case Pid::LOCATION_FRACTIONS:
268  return end->rtick() - start->rtick();
269  case Pid::LOCATION_GRACE:
270  return graceIndex(start) - graceIndex(end);
271  case Pid::LOCATION_NOTE:
272  return note(start) - note(end);
273  default:
274  return QVariant();
275  }
276  }
