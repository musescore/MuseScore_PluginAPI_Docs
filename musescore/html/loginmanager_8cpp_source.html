46 #if (QT_VERSION >= QT_VERSION_CHECK(5, 11, 0))

53  qDebug() << "randBytes =" << randBytes << "sizeof(randId)" << sizeof(randId);
54  for (size_t bytes = randBytes; bytes < sizeof(randId); bytes += randBytes) {

207 void LoginManager::onReplyFinished(ApiRequest* request, RequestType requestType)
208  {
209  if (!request)
210  return;
211  QNetworkReply* reply = request->reply();
212  if (!reply)
213  return;
214 
215  const int code = reply->attribute(QNetworkRequest::HttpStatusCodeAttribute).toInt();
216 
217  if (code == HTTP_UNAUTHORIZED && requestType != RequestType::LOGIN && requestType != RequestType::LOGIN_REFRESH) {
218  if (request->retryCount() < MAX_REFRESH_LOGIN_RETRY_COUNT) {
219  ApiRequest* refreshRequest = buildLoginRefreshRequest();
220  refreshRequest->setParent(this);
221  connect(refreshRequest, &ApiRequest::replyFinished, this, [this, request, requestType](ApiRequest* refreshRequest) {
222  _accessToken.clear();
223  onReplyFinished(refreshRequest, RequestType::LOGIN_REFRESH);
224  if (!_accessToken.isEmpty()) {
225  // try to execute the request once more with the new token
226  request->setToken(_accessToken);
227  request->executeRequest(_networkManager);
228  }
229  else {
230  handleReply(request->reply(), requestType);
231  request->deleteLater();
232  }
233  });
234  refreshRequest->executeRequest(_networkManager);
235  return;
236  }
237  }
238 
239  handleReply(reply, requestType);
240 
241  request->deleteLater();
242  }

248 void LoginManager::handleReply(QNetworkReply* reply, RequestType requestType)
249  {
250  if (!reply)
251  return;
252 
253  const int code = reply->attribute(QNetworkRequest::HttpStatusCodeAttribute).toInt();
254 
255  QByteArray ba(reply->readAll());
256  QJsonObject obj;
257  if (!ba.isEmpty()) {
258  QJsonDocument jsonResponse = QJsonDocument::fromJson(ba);
259  if (jsonResponse.isObject())
260  obj = jsonResponse.object();
261  }
262 
263  switch (requestType) {
264  case RequestType::LOGIN:
265  onLoginReply(reply, code, obj);
266  break;
267  case RequestType::LOGIN_REFRESH:
268  onLoginRefreshReply(reply, code, obj);
269  break;
270  case RequestType::GET_USER_INFO:
271  onGetUserReply(reply, code, obj);
272  break;
273  case RequestType::GET_SCORE_INFO:
274  onGetScoreInfoReply(reply, code, obj);
275  break;
276  case RequestType::UPLOAD_SCORE:
277  onUploadReply(reply, code, obj);
278  break;
279  case RequestType::GET_MEDIA_URL:
280  onGetMediaUrlReply(reply, code, obj);
281  break;
282  }
283  }
