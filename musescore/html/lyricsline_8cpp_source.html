32 static Lyrics* searchNextLyrics(Segment* s, int staffIdx, int verse, Placement p)
33  {
34  Lyrics* l = 0;
35  while ((s = s->next1(SegmentType::ChordRest))) {
36  int strack = staffIdx * VOICES;
37  int etrack = strack + VOICES;
38  // search through all tracks of current staff looking for a lyric in specified verse
39  for (int track = strack; track < etrack; ++track) {
40  ChordRest* cr = toChordRest(s->element(track));
41  if (cr) {
42  // cr with lyrics found, but does it have a syllable in specified verse?
43  l = cr->lyrics(verse, p);
44  if (l)
45  break;
46  }
47  }
48  if (l)
49  break;
50  }
51  return l;
52  }

87 void LyricsLine::layout()
88  {
89  bool tempMelismaTicks = (lyrics()->ticks() == Fraction::fromTicks(Lyrics::TEMP_MELISMA_TICKS));
90  if (isEndMelisma()) { // melisma
91  setLineWidth(score()->styleP(Sid::lyricsLineThickness));
92  // if lyrics has a temporary one-chord melisma, set to 0 ticks (just its own chord)
93  if (tempMelismaTicks)
94  lyrics()->setTicks(Fraction(0,1));
95 
96  // Lyrics::_ticks points to the beginning of the last spanned segment,
97  // but the line shall include it:
98  // include the duration of this last segment in the melisma duration
99  Segment* lyricsSegment = lyrics()->segment();
100  Fraction lyricsStartTick = lyricsSegment->tick();
101  Fraction lyricsEndTick = lyrics()->endTick();
102  int lyricsTrack = lyrics()->track();
103 
104  // find segment with tick >= endTick
105  Segment* s = lyricsSegment;
106  while (s && s->tick() < lyricsEndTick)
107  s = s->nextCR(lyricsTrack, true);
108  if (!s) {
109  // user probably deleted measures at end of score, leaving this melisma too long
110  // set s to last segment and reset lyricsEndTick to trigger FIXUP code below
111  s = score()->lastSegment();
112  lyricsEndTick = Fraction(-1,1);
113  }
114  Element* se = s->element(lyricsTrack);
115  // everything is OK if we have reached a chord at right tick on right track
116  if (s->tick() == lyricsEndTick && se && se->type() == ElementType::CHORD) {
117  // advance to next CR, or last segment if no next CR
118  s = s->nextCR(lyricsTrack, true);
119  if (!s)
120  s = score()->lastSegment();
121  }
122  else {
123  // FIXUP - lyrics tick count not valid
124  // this happens if edits to score have removed the original end segment
125  // so let's fix it here
126  // s is already pointing to segment past endTick (or to last segment)
127  // we should shorten the lyrics tick count to make this work
128  Segment* ns = s;
129  Segment* ps = s->prev1(SegmentType::ChordRest);
130  while (ps && ps != lyricsSegment) {
131  Element* pe = ps->element(lyricsTrack);
132  // we're looking for an actual chord on this track
133  if (pe && pe->type() == ElementType::CHORD)
134  break;
135  s = ps;
136  ps = ps->prev1(SegmentType::ChordRest);
137  }
138  if (!ps || ps == lyricsSegment) {
139  // no valid previous CR, so try to lengthen melisma instead
140  ps = ns;
141  s = ps->nextCR(lyricsTrack, true);
142  Element* e = s ? s->element(lyricsTrack) : nullptr;
143  // check to make sure we have a chord
144  if (!e || e->type() != ElementType::CHORD) {
145  // nothing to do but set ticks to 0
146  // this will result in melisma being deleted later
147  lyrics()->undoChangeProperty(Pid::LYRIC_TICKS, 0);
148  setTicks(Fraction(0,1));
149  return;
150  }
151  }
152  lyrics()->undoChangeProperty(Pid::LYRIC_TICKS, ps->tick() - lyricsStartTick);
153  }
154  // Spanner::computeEndElement() will actually ignore this value and use the (earlier) lyrics()->endTick() instead
155  // still, for consistency with other lines, we should set the ticks for this to the computed (later) value
156  if (s)
157  setTicks(s->tick() - lyricsStartTick);
158  }
159  else { // dash(es)
160  _nextLyrics = searchNextLyrics(lyrics()->segment(), staffIdx(), lyrics()->no(), lyrics()->placement());
161  setTick2(_nextLyrics ? _nextLyrics->segment()->tick() : tick());
162  }
163  if (ticks().isNotZero()) { // only do layout if some time span
164  // do layout with non-0 duration
165  if (tempMelismaTicks)
166  lyrics()->setTicks(Fraction::fromTicks(Lyrics::TEMP_MELISMA_TICKS));
167  }
168  }

174 SpannerSegment* LyricsLine::layoutSystem(System* system)
175  {
176  Fraction stick = system->firstMeasure()->tick();
177  Fraction etick = system->lastMeasure()->endTick();
178 
179  LyricsLineSegment* lineSegm = toLyricsLineSegment(getNextLayoutSystemSegment(system, [this]() { return createLineSegment(); }));
180 
181  SpannerSegmentType sst;
182  if (tick() >= stick) {
183  layout();
184  if (ticks().isZero()) // only do layout if some time span
185  return nullptr;
186  SLine::layout();
187  //
188  // this is the first call to layoutSystem,
189  // processing the first line segment
190  //
191  computeStartElement();
192  computeEndElement();
193  sst = tick2() <= etick ? SpannerSegmentType::SINGLE : SpannerSegmentType::BEGIN;
194  }
195  else if (tick() < stick && tick2() > etick) {
196  sst = SpannerSegmentType::MIDDLE;
197  }
198  else {
199  //
200  // this is the last call to layoutSystem
201  // processing the last line segment
202  //
203  sst = SpannerSegmentType::END;
204  }
205  lineSegm->setSpannerSegmentType(sst);
206 
207  switch (sst) {
208  case SpannerSegmentType::SINGLE: {
209  System* s;
210  QPointF p1 = linePos(Grip::START, &s);
211  QPointF p2 = linePos(Grip::END, &s);
212  qreal len = p2.x() - p1.x();
213  lineSegm->setPos(p1);
214  lineSegm->setPos2(QPointF(len, p2.y() - p1.y()));
215  }
216  break;
217  case SpannerSegmentType::BEGIN: {
218  System* s;
219  QPointF p1 = linePos(Grip::START, &s);
220  lineSegm->setPos(p1);
221  qreal x2 = system->bbox().right();
222  lineSegm->setPos2(QPointF(x2 - p1.x(), 0.0));
223  }
224  break;
225  case SpannerSegmentType::MIDDLE: {
226  bool leading = (anchor() == Anchor::SEGMENT || anchor() == Anchor::MEASURE);
227  qreal x1 = system->firstNoteRestSegmentX(leading);
228  qreal x2 = system->bbox().right();
229  System* s;
230  QPointF p1 = linePos(Grip::START, &s);
231  lineSegm->setPos(QPointF(x1, p1.y()));
232  lineSegm->setPos2(QPointF(x2 - x1, 0.0));
233  }
234  break;
235  case SpannerSegmentType::END: {
236  System* s;
237  QPointF p2 = linePos(Grip::END, &s);
238  bool leading = (anchor() == Anchor::SEGMENT || anchor() == Anchor::MEASURE);
239  qreal x1 = system->firstNoteRestSegmentX(leading);
240  qreal len = p2.x() - x1;
241  lineSegm->setPos(QPointF(p2.x() - len, p2.y()));
242  lineSegm->setPos2(QPointF(len, 0.0));
243  }
244  break;
245  }
246  lineSegm->layout();
247  // if temp melisma extend the first line segment to be
248  // after the lyrics syllable (otherwise the melisma segment
249  // will be too short).
250  const bool tempMelismaTicks = (lyrics()->ticks() == Fraction::fromTicks(Lyrics::TEMP_MELISMA_TICKS));
251  if (tempMelismaTicks && spannerSegments().size() > 0 && spannerSegments().front() == lineSegm)
252  lineSegm->rxpos2() += lyrics()->width();
253  // avoid backwards melisma
254  if (lineSegm->pos2().x() < 0)
255  lineSegm->rxpos2() = 0;
256  return lineSegm;
257  }

324 void LyricsLineSegment::layout()
325  {
326  ryoffset() = 0.0;
327 
328  bool endOfSystem = false;
329  bool isEndMelisma = lyricsLine()->isEndMelisma();
330  Lyrics* lyr = 0;
331  Lyrics* nextLyr = 0;
332  qreal fromX = 0;
333  qreal toX = 0; // start and end point of intra-lyrics room
334  qreal sp = spatium();
335  System* sys;
336 
337  if (lyricsLine()->ticks() <= Fraction(0,1)) { // if no span,
338  _numOfDashes = 0; // nothing to draw
339  return; // and do nothing
340  }
341 
342  // HORIZONTAL POSITION
343  // A) if line precedes a syllable, advance line end to right before the next syllable text
344  // if not a melisma and there is a next syllable;
345  if (!isEndMelisma && lyricsLine()->nextLyrics() && isSingleEndType()) {
346  lyr = nextLyr = lyricsLine()->nextLyrics();
347  sys = lyr->segment()->system();
348  endOfSystem = (sys != system());
349  // if next lyrics is on a different system, this line segment is at the end of its system:
350  // do not adjust for next lyrics position
351  if (sys && !endOfSystem) {
352  qreal lyrX = lyr->bbox().x();
353  qreal lyrXp = lyr->pagePos().x();
354  qreal sysXp = sys->pagePos().x();
355  toX = lyrXp - sysXp + lyrX; // syst.rel. X pos.
356  qreal offsetX = toX - pos().x() - pos2().x() - score()->styleP(Sid::lyricsDashPad);
357  // delta from current end pos.| ending padding
358  rxpos2() += offsetX;
359  }
360  }
361  // B) if line follows a syllable, advance line start to after the syllable text
362  lyr = lyricsLine()->lyrics();
363  sys = lyr->segment()->system();
364  if (sys && isSingleBeginType()) {
365  qreal lyrX = lyr->bbox().x();
366  qreal lyrXp = lyr->pagePos().x();
367  qreal lyrW = lyr->bbox().width();
368  qreal sysXp = sys->pagePos().x();
369  fromX = lyrXp - sysXp + lyrX + lyrW;
370  // syst.rel. X pos. | lyr.advance
371  qreal offsetX = fromX - pos().x();
372  offsetX += score()->styleP(isEndMelisma ? Sid::lyricsMelismaPad : Sid::lyricsDashPad);
373 
374  // delta from curr.pos. | add initial padding
375  rxpos() += offsetX;
376  rxpos2() -= offsetX;
377  }
378 
379  // VERTICAL POSITION: at the base line of the syllable text
380  if (!isEndType()) {
381  rypos() = lyr->ipos().y();
382  ryoffset() = lyr->offset().y();
383  }
384  else {
385  // use Y position of *next* syllable if there is one on same system
386  Lyrics* nextLyr1 = searchNextLyrics(lyr->segment(), lyr->staffIdx(), lyr->no(), lyr->placement());
387  if (nextLyr1 && nextLyr1->segment()->system() == system()) {
388  rypos() = nextLyr1->ipos().y();
389  ryoffset() = nextLyr1->offset().y();
390  }
391  else {
392  rypos() = lyr->ipos().y();
393  ryoffset() = lyr->offset().y();
394  }
395  }
396 
397  // MELISMA vs. DASHES
398  if (isEndMelisma) { // melisma
399  _numOfDashes = 1;
400  rypos() -= lyricsLine()->lineWidth() * .5; // let the line 'sit on' the base line
401  // if not final segment, shorten it
402  if (isBeginType() || isMiddleType())
403  rxpos2() -= score()->styleP(Sid::minNoteDistance) * mag();
404  }
405  else { // dash(es)
406  // set conventional dash Y pos
407  rypos() -= MScore::pixelRatio * lyr->fontMetrics().xHeight() * score()->styleD(Sid::lyricsDashYposRatio);
408  _dashLength = score()->styleP(Sid::lyricsDashMaxLength) * mag(); // and dash length
409  qreal len = pos2().x();
410  qreal minDashLen = score()->styleS(Sid::lyricsDashMinLength).val() * sp;
411  qreal maxDashDist = score()->styleS(Sid::lyricsDashMaxDistance).val() * sp;
412  if (len < minDashLen) { // if no room for a dash
413  // if at end of system or dash is forced
414  if (endOfSystem || score()->styleB(Sid::lyricsDashForce)) {
415  rxpos2() = minDashLen; // draw minimal dash
416  _numOfDashes = 1;
417  _dashLength = minDashLen;
418  }
419  else // if within system or dash not forced
420  _numOfDashes = 0; // draw no dash
421  }
422  else if (len < (maxDashDist * 1.5)) { // if no room for two dashes
423  _numOfDashes = 1; // draw one dash
424  if (_dashLength > len) // if no room for a full dash
425  _dashLength = len; // shorten it
426  }
427  else
428  _numOfDashes = len / maxDashDist + 1; // draw several dashes
429 
430  // adjust next lyrics horiz. position if too little a space forced to skip the dash
431  if (_numOfDashes == 0 && nextLyr != nullptr && len > 0)
432  nextLyr->rxpos() -= (toX - fromX);
433  }
434 
435  // apply yoffset for staff type change (keeps lyrics lines aligned with lyrics)
436  if (staffType())
437  rypos() += staffType()->yoffset().val() * spatium();
438 
439  // set bounding box
440  QRectF r = QRectF(0.0, 0.0, pos2().x(), pos2().y()).normalized();
441  qreal lw = lyricsLine()->lineWidth() * .5;
442  setbbox(r.adjusted(-lw, -lw, lw, lw));
443  }

449 void LyricsLineSegment::draw(QPainter* painter) const
450  {
451  if (_numOfDashes < 1) // nothing to draw
452  return;
453 
454  QPen pen(lyricsLine()->lyrics()->curColor());
455  pen.setWidthF(lyricsLine()->lineWidth());
456  pen.setCapStyle(Qt::FlatCap);
457  painter->setPen(pen);
458  if (lyricsLine()->isEndMelisma()) // melisma
459  painter->drawLine(QPointF(), pos2());
460  else { // dash(es)
461  qreal step = pos2().x() / _numOfDashes;
462  qreal x = step * .5 - _dashLength * .5;
463  for (int i = 0; i < _numOfDashes; i++, x += step)
464  painter->drawLine(QPointF(x, 0.0), QPointF(x + _dashLength, 0.0));
465  }
466  }
