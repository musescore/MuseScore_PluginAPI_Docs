68 Chord* MCursor::addChord(int pitch, const TDuration& duration)
69  {
70  createMeasures();
71  Measure* measure = _score->tick2measure(_tick);
72  Segment* segment = measure->getSegment(SegmentType::ChordRest, _tick);
73  Chord* chord = toChord(segment->element(_track));
74  if (chord == 0) {
75  chord = new Chord(_score);
76  chord->setTrack(_track);
77  chord->setDurationType(duration);
78  chord->setTicks(duration.fraction());
79  segment->add(chord);
80  }
81  Note* note = new Note(_score);
82  chord->add(note);
83  note->setPitch(pitch);
84  note->setTpcFromPitch();
85  _tick += duration.ticks();
