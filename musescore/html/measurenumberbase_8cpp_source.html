109 void MeasureNumberBase::layout()
110  {
111  setPos(QPointF());
112  if (!parent())
113  setOffset(0.0, 0.0);
114 
115  // TextBase::layout1() needs to be called even if there's no measure attached to it.
116  // This happens for example in the palettes.
117  TextBase::layout1();
118  // this could be if (!measure()) but it is the same as current and slower
119  // See implementation of MeasureNumberBase::measure().
120  if (!parent())
121  return;
122 
123  if (placeBelow()) {
124  qreal yoff = bbox().height();
125 
126  // If there is only one line, the barline spans outside the staff lines, so the default position is not correct.
127  if (staff()->constStaffType(measure()->tick())->lines() == 1)
128  yoff += 2.0 * spatium();
129  else
130  yoff += staff()->height();
131 
132  rypos() = yoff;
133  }
134  else {
135  qreal yoff = 0.0;
136 
137  // If there is only one line, the barline spans outside the staff lines, so the default position is not correct.
138  if (staff()->constStaffType(measure()->tick())->lines() == 1)
139  yoff -= 2.0 * spatium();
140 
141  rypos() = yoff;
142  }
143 
144  if (hPlacement() == HPlacement::CENTER) {
145  // measure numbers should be centered over where there can be notes.
146  // This means that header and trailing segments should be ignored,
147  // which includes all timesigs, clefs, keysigs, etc.
148  // This is how it should be centered:
149  // |bb 4/4 notes-chords #| other measure |
150  // | ------18------ | other measure |
151 
152  // x1 - left measure position of free space
153  // x2 - right measure position of free space
154 
155  const Measure* mea = measure();
156 
157  // find first chordrest
158  Segment* chordRest = mea->first(SegmentType::ChordRest);
159 
160  Segment* s1 = chordRest->prevActive();
161  // unfortunately, using !s1->header() does not work
162  while (s1 && (s1->isChordRestType()
163  || s1->isBreathType()
164  || s1->isClefType()
165  || s1->isBarLineType()
166  || !s1->element(staffIdx() * VOICES)))
167  s1 = s1->prevActive();
168 
169  Segment* s2 = chordRest->next();
170  // unfortunately, using !s1->trailer() does not work
171  while (s2 && (s2->isChordRestType()
172  || s2->isBreathType()
173  || s2->isClefType()
174  || s2->isBarLineType()
175  || !s2->element(staffIdx() * VOICES)))
176  s2 = s2->nextActive();
177 
178  // if s1/s2 does not exist, it means there is no header/trailer segment. Align with start/end of measure.
179  qreal x1 = s1 ? s1->x() + s1->minRight() : 0;
180  qreal x2 = s2 ? s2->x() - s2->minLeft() : mea->width();
181 
182  rxpos() = (x1 + x2) * 0.5;
183  }
184  else if (hPlacement() == HPlacement::RIGHT)
185  rxpos() = measure()->width();
186  }
187
