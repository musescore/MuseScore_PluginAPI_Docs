120 void MeasureProperties::setMeasure(Measure* _m)
121  {
122  m = _m;
123  setWindowTitle(tr("Measure Properties for Measure %1").arg(m->no()+1));
124  m->score()->deselectAll();
125  m->score()->select(m, SelectType::ADD, 0);
126 
127  actualZ->setValue(m->ticks().numerator());
128  int index = actualN->findText(QString::number(m->ticks().denominator()));
129  if (index == -1)
130  index = 2;
131  actualN->setCurrentIndex(index);
132  nominalZ->setNum(m->timesig().numerator());
133  nominalN->setNum(m->timesig().denominator());
134 
135  irregular->setChecked(m->irregular());
136  breakMultiMeasureRest->setChecked(m->breakMultiMeasureRest());
137  int n = m->repeatCount();
138  count->setValue(n);
139  bool enableCount = m->repeatEnd();
140  count->setEnabled(enableCount);
141  count->setVisible(enableCount);
142  labelCount->setVisible(enableCount);
143  layoutStretch->setValue(m->userStretch());
144  measureNumberMode->setCurrentIndex(int(m->measureNumberMode()));
145  measureNumberOffset->setValue(m->noOffset());
146 
147  Score* score = m->score();
148  int rows = score->nstaves();
149  staves->setRowCount(rows);
150  staves->setColumnCount(3);
151 
152  for (int staffIdx = 0; staffIdx < rows; ++staffIdx) {
153  QTableWidgetItem* item = new QTableWidgetItem(QString("%1").arg(staffIdx+1));
154  staves->setItem(staffIdx, 0, item);
155 
156  item = new QTableWidgetItem(tr("visible"));
157  item->setFlags(item->flags() | Qt::ItemIsUserCheckable | Qt::ItemIsEnabled);
158  item->setCheckState(m->visible(staffIdx) ? Qt::Checked : Qt::Unchecked);
159  if (rows == 1) // cannot be invisible if only one row
160  item->setFlags(item->flags() & ~Qt::ItemIsEnabled);
161  staves->setItem(staffIdx, 1, item);
162 
163  item = new QTableWidgetItem(tr("stemless"));
164  item->setFlags(item->flags() | Qt::ItemIsUserCheckable | Qt::ItemIsEnabled);
165  item->setCheckState(m->stemless(staffIdx) ? Qt::Checked : Qt::Unchecked);
166  staves->setItem(staffIdx, 2, item);
167  }
168  }

247 void MeasureProperties::apply()
248  {
249  Score* score = m->score();
250 
251  bool propertiesChanged = false;
252  for (int staffIdx = 0; staffIdx < score->nstaves(); ++staffIdx) {
253  bool v = visible(staffIdx);
254  bool s = stemless(staffIdx);
255  if (m->visible(staffIdx) != v || m->stemless(staffIdx) != s) {
256  score->undo(new ChangeMStaffProperties(m, staffIdx, v, s));
257  propertiesChanged = true;
258  }
259  }
260 
261  m->undoChangeProperty(Pid::REPEAT_COUNT, repeatCount());
262  m->undoChangeProperty(Pid::BREAK_MMR, breakMultiMeasureRest->isChecked());
263  m->undoChangeProperty(Pid::USER_STRETCH, layoutStretch->value());
264  m->undoChangeProperty(Pid::MEASURE_NUMBER_MODE, measureNumberMode->currentIndex());
265  m->undoChangeProperty(Pid::NO_OFFSET, measureNumberOffset->value());
266  m->undoChangeProperty(Pid::IRREGULAR, isIrregular());
267 
268  if (m->ticks() != len()) {
269  ScoreRange range;
270  range.read(m->first(), m->last());
271  m->adjustToLen(len());
272 #if 0
273  // handled by endCmd():
274  else if (!MScore::noGui) {
275  QMessageBox::warning(0,
276  QT_TRANSLATE_NOOP("MeasureProperties", "MuseScore"),
277  QT_TRANSLATE_NOOP("MeasureProperties", "Cannot change measure length:\n"
278  "tuplet would cross measure")
279  );
280  }
281 #endif
282  }
283 
284  if (propertiesChanged) {
285  m->triggerLayout();
286  }
287 
288  score->select(m, SelectType::SINGLE, 0);
289  score->update();
290  mscore->timeline()->updateGridFull();
291  }
