173 static const QRectF getBounds(const QSize outerSize, const QSize innerSize)
174  {
175  // Horizontal Offset
176  qreal hOffset = 0.0;
177  if (innerSize.width() < outerSize.width())
178  hOffset = (outerSize.width() - innerSize.width()) * 0.5;
179 
180  // Vertical Offset
181  qreal vOffset = 0.0;
182  if (innerSize.height() < outerSize.height())
183  vOffset = (outerSize.height() - innerSize.height()) * 0.5;
184 
185  return QRectF(hOffset, vOffset, innerSize.width(), innerSize.height());

192 QPixmap MIconEngine::pixmap(const QSize &size, QIcon::Mode mode, QIcon::State state)
193  {
194  QPixmap pm;
195 
196  QString pmckey(d->pmcKey(size, mode, state));
197  pmckey.prepend("Ms");
198 
199  if (QPixmapCache::find(pmckey, &pm))
200  return pm;
201 
202  if (d->addedPixmaps) {
203  pm = d->addedPixmaps->value(d->hashKey(mode, state));
204  if (!pm.isNull() && pm.size() == size)
205  return pm;
206  }
207 
208  QSvgRenderer renderer;
209  d->loadDataForModeAndState(&renderer, mode, state);
210  if (!renderer.isValid())
211  return pm;
212 
213  QSize actualSize = renderer.defaultSize();
214  if (!actualSize.isNull())
215  actualSize.scale(size, Qt::KeepAspectRatio);
216 
217  // Generate an image of the requested size, but render the
218  // the SVG with the correct aspect ratio centered in the image
219  // to prevent scaling issues when setting non square icon size.
220  QImage img(size, QImage::Format_ARGB32);
221  img.fill(0x00000000);
222  QPainter p(&img);
223  renderer.render(&p, getBounds(size, actualSize));
224  p.end();
225  pm = QPixmap::fromImage(img);
226 
227  if (!pm.isNull())
228  QPixmapCache::insert(pmckey, pm);
229  return pm;
230  }

248 void MIconEngine::addFile(const QString &fileName, const QSize &, QIcon::Mode mode, QIcon::State state)
249  {
250  if (!fileName.isEmpty()) {
251  QString abs = fileName;
252  if (fileName.at(0) != QLatin1Char(':'))
253  abs = QFileInfo(fileName).absoluteFilePath();
254  if (abs.endsWith(QLatin1String(".svg"), Qt::CaseInsensitive)
255  || abs.endsWith(QLatin1String(".svgz"), Qt::CaseInsensitive)
256  || abs.endsWith(QLatin1String(".svg.gz"), Qt::CaseInsensitive))
257  {
258  QSvgRenderer renderer(abs);
259  if (renderer.isValid()) {
260  d->stepSerialNum();
261  d->svgFiles.insert(d->hashKey(mode, state), abs);
262  }
263  }
264  else {
265  QPixmap pm(abs);
266  if (!pm.isNull())
267  addPixmap(pm, mode, state);
268  }
269  }
