45 void MasterScore::checkMidiMapping()
46  {
47  isSimpleMidiMaping = true;
48  rebuildMidiMapping();
49 
50  QList<bool> drum;
51  drum.reserve(int(_midiMapping.size()));
52  for (Part* part : parts()) {
53  const InstrumentList* il = part->instruments();
54  for (auto i = il->begin(); i != il->end(); ++i) {
55  const Instrument* instr = i->second;
56  for (int j = 0; j < instr->channel().size(); ++j)
57  drum.append(instr->useDrumset());
58  }
59  }
60  int lastChannel = -1; // port*16+channel
61  int lastDrumPort = -1;
62  int index = 0;
63  for (const MidiMapping& m : _midiMapping) {
64  if (index >= drum.size())
65  break;
66  if (drum[index]) {
67  lastDrumPort++;
68  if (m.port() != lastDrumPort) {
69  isSimpleMidiMaping = false;
70  return;
71  }
72  }
73  else {
74  lastChannel++;
75  if (lastChannel % 16 == 9)
76  lastChannel++;
77  int p = lastChannel / 16;
78  int c = lastChannel % 16;
79  if (m.port() != p || m.channel() != c) {
80  isSimpleMidiMaping = false;
81  return;
82  }
83  }
84  index++;
85  }
86  }

92 int MasterScore::getNextFreeMidiMapping(int p, int ch)
93  {
94  if (ch != -1 && p != -1)
95  return p*16+ch;
96 
97  else if (ch != -1 && p == -1) {
98  for (int port = 0;; port++) {
99  if (!occupiedMidiChannels.contains(port*16+ch)) {
100  occupiedMidiChannels.insert(port*16+ch);
101  return port*16+ch;
102  }
103  }
104  }
105  else if (ch == -1 && p != -1) {
106  for (int channel = 0; channel < 16; channel++) {
107  if (channel != 9 && !occupiedMidiChannels.contains(p*16+channel)) {
108  occupiedMidiChannels.insert(p*16+channel);
109  return p*16+channel;
110  }
111  }
112  }
113 
114  for (;;searchMidiMappingFrom++) {
115  if (searchMidiMappingFrom % 16 != 9 && !occupiedMidiChannels.contains(searchMidiMappingFrom)) {
116  occupiedMidiChannels.insert(searchMidiMappingFrom);
117  return searchMidiMappingFrom;
118  }
119  }
120  }

140 void MasterScore::rebuildExcerptsMidiMapping()
141  {
142  for (Excerpt* ex : excerpts()) {
143  for (Part* p : ex->partScore()->parts()) {
144  const Part* masterPart = p->masterPart();
145  if (!masterPart->score()->isMaster()) {
146  qWarning() << "rebuildExcerptsMidiMapping: no part in master score is linked with " << p->partName();
147  continue;
148  }
149  Q_ASSERT(p->instruments()->size() == masterPart->instruments()->size());
150  for (const auto& item : *masterPart->instruments()) {
151  const Instrument* iMaster = item.second;
152  const int tick = item.first;
153  Instrument* iLocal = p->instrument(Fraction::fromTicks(tick));
154  const int nchannels = iMaster->channel().size();
155  if (iLocal->channel().size() != nchannels) {
156  // may happen, e.g., if user changes an instrument
157  (*iLocal) = (*iMaster);
158  continue;
159  }
160  for (int c = 0; c < nchannels; ++c) {
161  Channel* cLocal = iLocal->channel(c);
162  const Channel* cMaster = iMaster->channel(c);
163  cLocal->setChannel(cMaster->channel());
164  }
165  }
166  }
167  }
168  }

175 void MasterScore::reorderMidiMapping()
176  {
177  using std::swap;
178  int sequenceNumber = 0;
179  for (Part* part : parts()) {
180  const InstrumentList* il = part->instruments();
181  for (auto i = il->begin(); i != il->end(); ++i) {
182  const Instrument* instr = i->second;
183  for (Channel* channel : instr->channel()) {
184  if (!(_midiMapping[sequenceNumber].part() == part
185  && _midiMapping[sequenceNumber].masterChannel == channel)) {
186  int shouldBe = channel->channel();
187  swap(_midiMapping[sequenceNumber], _midiMapping[shouldBe]);
188  _midiMapping[sequenceNumber].articulation()->setChannel(sequenceNumber);
189  channel->setChannel(sequenceNumber);
190  _midiMapping[shouldBe].articulation()->setChannel(shouldBe);
191  }
192  sequenceNumber++;
193  }
194  }
195  }
196  }

203 void MasterScore::removeDeletedMidiMapping()
204  {
205  int removeOffset = 0;
206  int mappingSize = int(_midiMapping.size());
207  for (int index = 0; index < mappingSize; index++) {
208  Part* p = midiMapping(index)->part();
209  if (!parts().contains(p)) {
210  removeOffset++;
211  continue;
212  }
213  // Not all channels could exist
214  bool channelExists = false;
215  const InstrumentList* il = p->instruments();
216  for (auto i = il->begin(); i != il->end() && !channelExists; ++i) {
217  const Instrument* instr = i->second;
218  channelExists = (_midiMapping[index].articulation()->channel() != -1
219  && instr->channel().contains(_midiMapping[index].masterChannel)
220  && !(_midiMapping[index].port() == -1 && _midiMapping[index].channel() == -1));
221  if (channelExists)
222  break;
223  }
224  if (!channelExists) {
225  removeOffset++;
226  continue;
227  }
228  // Let's do a left shift by 'removeOffset' items if necessary
229  if (index != 0 && removeOffset != 0) {
230  _midiMapping[index-removeOffset] = std::move(_midiMapping[index]);
231 
232  const int chanVal = _midiMapping[index-removeOffset].articulation()->channel();
233  _midiMapping[index-removeOffset].articulation()->setChannel(chanVal - removeOffset);
234  }
235  }
236  // We have 'removeOffset' deleted instruments, let's remove their mappings
237  for (int index = 0; index < removeOffset; index++)
238  _midiMapping.pop_back();
239  }

246 int MasterScore::updateMidiMapping()
247  {
248  int maxport = 0;
249  occupiedMidiChannels.clear();
250  searchMidiMappingFrom = 0;
251  occupiedMidiChannels.reserve(int(_midiMapping.size())); // Bringing down the complexity of insertion to amortized O(1)
252 
253  for (const MidiMapping& mm :_midiMapping) {
254  if (mm.port() == -1 || mm.channel() == -1)
255  continue;
256  occupiedMidiChannels.insert(static_cast<int>(mm.port())*16+(int)mm.channel());
257  if (maxport < mm.port())
258  maxport = mm.port();
259  }
260 
261  for (Part* part : parts()) {
262  const InstrumentList* il = part->instruments();
263  for (auto i = il->begin(); i != il->end(); ++i) {
264  const Instrument* instr = i->second;
265  bool drum = instr->useDrumset();
266  for (Channel* channel : instr->channel()) {
267  bool channelExists = false;
268  for (const MidiMapping& mapping: _midiMapping) {
269  if (channel == mapping.masterChannel && channel->channel() != -1) {
270  channelExists = true;
271  break;
272  }
273  }
274  // Channel could already exist, but have unassigned port or channel. Repair and continue
275  if (channelExists) {
276  if (_midiMapping[channel->channel()].port() == -1) {
277  const int nm = getNextFreeMidiMapping(-1, _midiMapping[channel->channel()].channel());
278  _midiMapping[channel->channel()]._port = nm / 16;
279  }
280  else if (_midiMapping[channel->channel()].channel() == -1) {
281  if (drum) {
282  _midiMapping[channel->channel()]._port = getNextFreeDrumMidiMapping() / 16;
283  _midiMapping[channel->channel()]._channel = 9;
284  continue;
285  }
286  int nm = getNextFreeMidiMapping(_midiMapping[channel->channel()].port());
287  _midiMapping[channel->channel()]._port = nm / 16;
288  _midiMapping[channel->channel()]._channel = nm % 16;
289  }
290  continue;
291  }
292 
293  int midiPort;
294  int midiChannel;
295  if (drum) {
296  midiPort = getNextFreeDrumMidiMapping() / 16;
297  midiChannel = 9;
298  }
299  else {
300  int nm = getNextFreeMidiMapping();
301  midiPort = nm / 16;
302  midiChannel = nm % 16;
303  }
304 
305  if (midiPort > maxport)
306  maxport = midiPort;
307 
308  addMidiMapping(channel, part, midiPort, midiChannel);
309  }
310  }
311  }
312  return maxport;
313  }

319 void MasterScore::addMidiMapping(Channel* channel, Part* part, int midiPort, int midiChannel)
320  {
321  if (!part->score()->isMaster())
322  return;
323 
324  MidiMapping mm;
325  mm._part = part;
326  mm.masterChannel = channel;
327  mm._articulation.reset(new Channel(*channel));
328  mm.link = PartChannelSettingsLink(mm.articulation(), mm.masterChannel, /* excerpt */ false);
329 
330  mm._port = midiPort;
331  mm._channel = midiChannel;
332 
333  const int mscoreChannel = int(_midiMapping.size());
334  mm._articulation->setChannel(mscoreChannel);
335  mm.masterChannel->setChannel(mscoreChannel);
336 
337  _midiMapping.push_back(std::move(mm));
338  }
