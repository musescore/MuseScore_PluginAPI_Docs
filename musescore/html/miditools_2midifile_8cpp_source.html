40 bool MidiFile::read(QIODevice* f)
41  {
42  fp = f;
43  _tracks.clear();
44  curPos = 0;
45 
46  char tmp[4];
47 
48  read(tmp, 4);
49  int len = readLong();
50  if (memcmp(tmp, "MThd", 4) || len < 6)
51  throw(QString("bad midifile: MThd expected"));
52 
53  _format = readShort();
54  int ntracks = readShort();
55  _division = readShort();
56 
57  if (_division < 0)
58  _division = (-(_division/256)) * (_division & 0xff);
59  if (len > 6)
60  skip(len-6); // skip the excess
61 
62  switch (_format) {
63  case 0:
64  if (readTrack())
65  return false;
66  break;
67  case 1:
68  for (int i = 0; i < ntracks; i++) {
69  if (readTrack())
70  return false;
71  }
72  break;
73  default:
74  throw(QString("midi file format %1 not implemented").arg(_format));
75  }
76  return true;
77  }

84 bool MidiFile::readTrack()
85  {
86  char tmp[4];
87  read(tmp, 4);
88  if (memcmp(tmp, "MTrk", 4))
89  throw(QString("bad midifile: MTrk expected"));
90  int len = readLong(); // len
91  qint64 endPos = curPos + len;
92  status = -1;
93  sstatus = -1; // running status, will not be reset on meta or sysex
94  click = 0;
95  MidiTrack* track = new MidiTrack(this);
96  _tracks.push_back(track);
97 
98  for (;;) {
99  MidiEvent event;
100  int rv = readEvent(&event);
101  if (rv == 0)
102  track->events().insert(std::pair<int, MidiEvent>(click, event));
103  else if (rv == 2)
104  break;
105  }
106  if (curPos != endPos) {
107  qWarning("bad track len: %lld != %lld, %lld bytes too much\n", endPos, curPos, endPos - curPos);
108  if (curPos < endPos) {
109  qWarning(" skip %lld\n", endPos-curPos);
110  skip(endPos - curPos);
111  }
112  }
113  return false;
114  }

217 int MidiFile::readEvent(MidiEvent* event)
218  {
219  uchar me, a, b;
220 
221  int nclick = getvl();
222  if (nclick == -1)
223  throw(QString("readEvent: error 1(getvl)"));
224  click += nclick;
225  for (;;) {
226  read(&me, 1);
227  if (me >= 0xf1 && me <= 0xfe && me != 0xf7)
228  qDebug("Midi: Unknown Message 0x%02x", me & 0xff);
229  else
230  break;
231  }
232 
233  uchar* data;
234  int dataLen;
235 
236  if (me == 0xf0 || me == 0xf7) {
237  status = -1; // no running status
238  int len = getvl();
239  if (len == -1)
240  throw(QString("readEvent: error 3"));
241  data = new unsigned char[len+1];
242  dataLen = len;
243  read(data, len);
244  data[dataLen] = 0; // always terminate with zero
245  if (data[len-1] != 0xf7) {
246  qDebug("SYSEX does not end with 0xf7!");
247  // more to come?
248  }
249  else
250  dataLen--; // don't count 0xf7
251  delete[] data;
252 #if 0
253  event->setType(MidiEventType::SYSEX);
254  event->setData(data);
255  event->setLen(dataLen);
256  event->setOntime(click);
257 #endif
258  return 1;
259  }
260 
261  if (me == 0xff) { // MidiEventType::META) {
262  status = -1; // no running status
263  uchar type;
264  read(&type, 1);
265  dataLen = getvl(); // read len
266  if (dataLen == -1)
267  throw(QString("readEvent: error 6"));
268  data = new unsigned char[dataLen + 1];
269  if (dataLen)
270  read(data, dataLen);
271  data[dataLen] = 0; // always terminate with zero so we get valid C++ strings
272  if (type == META_TEMPO) {
273  unsigned tempo = data[2] + (data[1] << 8) + (data[0] << 16);
274  double t = 1000000.0 / double(tempo);
275  _tempoMap.insert(std::pair<const int, qreal>(click, t));
276  }
277 //else
278 //printf("META %02x\n", type);
279  delete[] data;
280  if (type == META_EOT)
281  return 2;
282  return 1;
283  }
284 
285  if (me & 0x80) { // status byte
286  status = me;
287  sstatus = status;
288  read(&a, 1);
289  }
290  else {
291  if (status == -1) {
292  qDebug("readEvent: no running status, read 0x%02x", me);
293  qDebug("sstatus ist 0x%02x", sstatus);
294  if (sstatus == -1)
295  return 0;
296  status = sstatus;
297  }
298  a = me;
299  }
300  int channel = status & 0x0f;
301  b = 0;
302  MidiEventType t = MidiEventType(status & 0xf0);
303  switch (t) {
304  case MidiEventType::NOTEOFF:
305  case MidiEventType::NOTEON:
306  case MidiEventType::POLYAFTER:
307  case MidiEventType::CONTROLLER: // controller
308  case MidiEventType::PITCHBEND: // pitch bend
309  read(&b, 1);
310  default:
311  break;
312  }
313  switch (t) {
314  case MidiEventType::NOTEOFF:
315  case MidiEventType::NOTEON:
316  case MidiEventType::CONTROLLER:
317  case MidiEventType::PITCHBEND:
318  case MidiEventType::POLYAFTER:
319  event->set(t, channel, a, b);
320  break;
321  case MidiEventType::PROGRAM:
322  case MidiEventType::AFTERTOUCH:
323  event->set(t, channel, a, 0);
324  break;
325  default: // f1 f2 f3 f4 f5 f6 f7 f8 f9
326  throw(QString("BAD STATUS")); // 0x%02x, me 0x%02x", status, me);
327  }
328 
329  if ((a & 0x80) || (b & 0x80)) {
330  qDebug("8't bit in data set(%02x %02x): tick %d read 0x%02x status:0x%02x",
331  a & 0xff, b & 0xff, click, me, status);
332  qDebug("readEvent: error 16");
333  if (b & 0x80) {
334  // Try to fix: interpret as channel byte
335  status = b;
336  sstatus = status;
337  return 0;
338  }
339  throw(QString("readEvent: error 16"));
340  }
341  return 0;
342  }

427 bool MidiFile::writeTrack(const MidiTrack* t)
428  {
429  write("MTrk", 4);
430  qint64 lenpos = fp->pos();
431  writeLong(0); // dummy len
432 
433  status = -1;
434  int tick = 0;
435  for (auto i : t->events()) {
436  int ntick = i.first;
437  putvl(ntick - tick); // write tick delta
438  writeEvent(i.second);
439  tick = ntick;
440  }
441 
442  //---------------------------------------------------
443  // write "End Of Track" Meta
444  // write Track Len
445  //
446 
447  putvl(1);
448  put(0xff); // Meta
449  put(0x2f); // EOT
450  putvl(0); // len 0
451  qint64 endpos = fp->pos();
452  fp->seek(lenpos);
453  writeLong(endpos-lenpos-4); // tracklen
454  fp->seek(endpos);
455  return false;
456  }
