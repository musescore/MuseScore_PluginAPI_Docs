132 void XmlWriter::tag(const QString& name, QVariant data)
133  {
134  QString ename(name.split(' ')[0]);
135 
136  putLevel();
137  switch(data.type()) {
138  case QVariant::Bool:
139  case QVariant::Char:
140  case QVariant::Int:
141  case QVariant::UInt:
142  *this << "<" << name << ">";
143  *this << data.toInt();
144  *this << "</" << ename << ">\n";
145  break;
146  case QVariant::Double:
147  *this << "<" << name << ">";
148  *this << data.value<double>();
149  *this << "</" << ename << ">\n";
150  break;
151  case QVariant::String:
152  *this << "<" << name << ">";
153  *this << xmlString(data.value<QString>());
154  *this << "</" << ename << ">\n";
155  break;
156  case QVariant::Color:
157  {
158  QColor color(data.value<QColor>());
159  *this << QString("<%1 r=\"%2\" g=\"%3\" b=\"%4\" a=\"%5\"/>\n")
160  .arg(name).arg(color.red()).arg(color.green()).arg(color.blue()).arg(color.alpha());
161  }
162  break;
163  case QVariant::Rect:
164  {
165  QRect r(data.value<QRect>());
166  *this << QString("<%1 x=\"%2\" y=\"%3\" w=\"%4\" h=\"%5\"/>\n").arg(name).arg(r.x()).arg(r.y()).arg(r.width()).arg(r.height());
167  }
168  break;
169  case QVariant::RectF:
170  {
171  QRectF r(data.value<QRectF>());
172  *this << QString("<%1 x=\"%2\" y=\"%3\" w=\"%4\" h=\"%5\"/>\n").arg(name).arg(r.x()).arg(r.y()).arg(r.width()).arg(r.height());
173  }
174  break;
175  case QVariant::PointF:
176  {
177  QPointF p(data.value<QPointF>());
178  *this << QString("<%1 x=\"%2\" y=\"%3\"/>\n").arg(name).arg(p.x()).arg(p.y());
179  }
180  break;
181  case QVariant::SizeF:
182  {
183  QSizeF p(data.value<QSizeF>());
184  *this << QString("<%1 w=\"%2\" h=\"%3\"/>\n").arg(name).arg(p.width()).arg(p.height());
185  }
186  break;
187  default:
188  qDebug("XmlWriter::tag: unsupported type %d", data.type());
189  // abort();
190  break;
191  }
192  }

199  {

214  case '\"':

216  break;

218  case 0x01:
