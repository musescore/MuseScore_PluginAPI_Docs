358 void Mixer::updateTracks()
359  {
360  MixerTrackItem* oldSel = mixerDetails->track().get();
361 
362  Part* selPart = oldSel ? oldSel->part() : 0;
363  Channel* selChan = oldSel ? oldSel->chan() : 0;
364 
365  if (_score && !selPart) {
366  //If nothing selected, select first available track
367  if (!_score->parts().isEmpty())
368  {
369  selPart = _score->parts()[0]->masterPart();
370  selChan = selPart->instrument(Fraction(0,1))->playbackChannel(0, _score->masterScore());
371  }
372 
373  }
374 
375 
376  if (trackHolder) {
377  trackAreaLayout->removeWidget(trackHolder);
378  trackHolder->deleteLater();
379  trackHolder = 0;
380  }
381 
382  trackList.clear();
383  mixerDetails->setTrack(0);
384 
385 
386  if (!_score)
387  return;
388 
389  trackHolder = new QWidget();
390  QHBoxLayout* holderLayout = new QHBoxLayout();
391  holderLayout->setContentsMargins(0, 0, 0, 0);
392  holderLayout->setSpacing(0);
393  trackHolder->setLayout(holderLayout);
394 
395  trackAreaLayout->addWidget(trackHolder);
396 
397  for (Part* localPart : _score->parts()) {
398  Part* part = localPart->masterPart();
399  //Add per part tracks
400  bool expanded = expandedParts.contains(part);
401  const InstrumentList* il = part->instruments();
402  Instrument* proxyInstr = nullptr;
403  Channel* proxyChan = nullptr;
404  if (!il->empty()) {
405  proxyInstr = il->begin()->second;
406  proxyChan = proxyInstr->playbackChannel(0, _score->masterScore());
407  }
408 
409  MixerTrackItemPtr mti = std::make_shared<MixerTrackItem>(
410  MixerTrackItem::TrackType::PART, part, proxyInstr, proxyChan);
411 
412  MixerTrackPart* track = new MixerTrackPart(this, mti, expanded);
413  track->setGroup(this);
414  trackList.append(track);
415  holderLayout->addWidget(track);
416 
417  if (selPart == part &&
418  (selChan == 0 || !expanded)) {
419  track->setSelected(true);
420  mixerDetails->setTrack(mti);
421  }
422 
423  if (expanded) {
424  //Add per channel tracks
425  const InstrumentList* il1 = part->instruments();
426  for (auto it = il1->begin(); it != il1->end(); ++it) {
427  Instrument* instr = it->second;
428  for (int i = 0; i < instr->channel().size(); ++i) {
429  Channel* chan = instr->playbackChannel(i, _score->masterScore());
430  MixerTrackItemPtr mti1 = std::make_shared<MixerTrackItem>(
431  MixerTrackItem::TrackType::CHANNEL, part, instr, chan);
432 // MixerTrackItemPtr mti = new MixerTrackItem(
433 // MixerTrackItem::TrackType::CHANNEL, part, instr, chan);
434  MixerTrackChannel* track1 = new MixerTrackChannel(this, mti1);
435  track1->setGroup(this);
436  trackList.append(track1);
437  holderLayout->addWidget(track1);
438 
439  if (selPart == part &&
440  selChan == chan) {
441  track1->setSelected(true);
442  mixerDetails->setTrack(mti1);
443  }
444  }
445  }
446  }
447  }
448 
449  holderLayout->addSpacerItem(new QSpacerItem(1, 1, QSizePolicy::Expanding, QSizePolicy::Fixed));
450  keepScrollPosition();
451  }
