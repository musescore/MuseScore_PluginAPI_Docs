509 void MixerDetails::drumkitToggled(bool val)
510  {
511  if (_mti == 0)
512  return;
513 
514  Part* part = _mti->part();
515  Channel* channel = _mti->focusedChan();
516 
517 
518  Instrument *instr;
519  if (_mti->trackType() == MixerTrackItem::TrackType::CHANNEL)
520  instr = _mti->instrument();
521  else
522  instr = part->instrument(Fraction(0,1));
523 
524  if (instr->useDrumset() == val)
525  return;
526 
527  const MidiPatch* newPatch = 0;
528  const QList<MidiPatch*> pl = synti->getPatchInfo();
529  for (const MidiPatch* p : pl) {
530  if (p->drum == val) {
531  newPatch = p;
532  break;
533  }
534  }
535 
536  Score* score = part->score();
537  if (newPatch) {
538  score->startCmd();
539  part->undoChangeProperty(Pid::USE_DRUMSET, val);
540  score->undo(new ChangePatch(score, channel, newPatch));
541  score->setLayoutAll();
542  score->endCmd();
543  }
544  }
