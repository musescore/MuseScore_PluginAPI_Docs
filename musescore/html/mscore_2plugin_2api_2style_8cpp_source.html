86 void MStyle::setValue(const QString& key, QVariant value) {
87  const Sid sid = keyToSid(key);
88 
89  if (sid == Sid::NOSTYLE)
90  return;
91 
92  if (!strcmp(Ms::MStyle::valueType(sid), "Ms::Spatium"))
93  value = QVariant::fromValue(Ms::Spatium(value.toReal()));
94 
95  if (_score) {
96  // Style belongs to actual score: change style value in undoable way
97  _score->undoChangeStyleVal(sid, value);
98  }
99  else {
100  // Style is not bound to a score: change the value directly
101  _style->set(sid, value);
102  }
103  }
