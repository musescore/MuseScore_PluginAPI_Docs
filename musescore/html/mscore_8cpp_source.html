163 Direction toDirection(const QString& s)
164  {
165  Direction val;
166  if (s == "up")
167  val = Direction::UP;
168  else if (s == "down")
169  val = Direction::DOWN;
170  else if (s == "auto")
171  val = Direction::AUTO;
172  else
173  val = Direction(s.toInt());
174  return val;
175  }
