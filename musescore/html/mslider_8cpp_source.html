106 void MeterSlider::resizeEvent(QResizeEvent* /*ev*/)
107  {
108  int h = height();
109  int kh = sliderSize().height();
110  int mh = h - kh;
111  int mw = _meterWidth / _channel;
112 
113  onPm = QPixmap(mw, mh);
114  offPm = QPixmap(mw, mh);
115 
116  double range = maxValue() - minValue();
117  int h1 = mh - lrint((maxValue() - redScale) * mh / range);
118  int h2 = mh - lrint((maxValue() - yellowScale) * mh / range);
119 
120  QColor yellowRed;
121  yellowRed.setHsv(QColor(Qt::yellow).hue() - 8,
122  QColor(Qt::yellow).saturation(),
123  QColor(Qt::yellow).value());
124  QColor yellRedRed;
125  yellRedRed.setHsv(QColor(Qt::yellow).hue() - 16,
126  QColor(Qt::yellow).saturation(),
127  QColor(Qt::yellow).value());
128 
129  QLinearGradient linearGrad(QPointF(0, 0), QPointF(0, mh));
130  linearGrad.setColorAt(0, Qt::red);
131  linearGrad.setColorAt(1 - (double)(h1 - 5) / (double)mh, yellRedRed);
132  linearGrad.setColorAt(1 - (double)(h1 - 6) / (double)mh, yellowRed);
133  linearGrad.setColorAt(1 - (double)h2 / (double)mh, Qt::yellow);
134  linearGrad.setColorAt(1, Qt::green);
135 
136  QColor darkYellowRed;
137  darkYellowRed.setHsv(QColor(Qt::darkYellow).hue()-8,
138  QColor(Qt::darkYellow).saturation(),
139  QColor(Qt::darkYellow).value());
140  QColor darkYellRedRed;
141  darkYellRedRed.setHsv(QColor(Qt::darkYellow).hue()-16,
142  QColor(Qt::darkYellow).saturation(),
143  QColor(Qt::darkYellow).value());
144  QLinearGradient linearDarkGrad(QPointF(0, 0), QPointF(0, mh));
145  linearDarkGrad.setColorAt(0, Qt::darkRed);
146  linearDarkGrad.setColorAt(1 - (double)(h1 - 5) / (double)mh, darkYellRedRed);
147  linearDarkGrad.setColorAt(1 - (double)(h1 - 6) / (double)mh, darkYellowRed);
148  linearDarkGrad.setColorAt(1 - (double)h2 / (double)mh, Qt::darkYellow);
149  linearDarkGrad.setColorAt(1, Qt::darkGreen);

152  p.begin(&onPm);

164 void MeterSlider::paintEvent(QPaintEvent* ev)
165  {
166  int h = height();
167  int kh = sliderSize().height();
168  int mh = h - kh;
169 
170  double range = maxValue() - minValue();
171  int ppos = int(mh * (_value - minValue()) / range);
172  if (_invert)
173  ppos = mh - ppos;
174 
175  QPainter p(this);
176  p.setRenderHint(QPainter::Antialiasing, false);
177 
178  //---------------------------------------------------
179  // draw meter
180  //---------------------------------------------------
181 
182  int mw = _meterWidth / _channel;
183  int x = METER_LEFT_EDGE;
184  int y1 = kh / 2;
185  int y3 = h - y1;
186 
187  p.setPen(QPen(Qt::white, 2));
188 
189  for (int i = 0; i < _channel; ++i) {
190  int h1 = mh - (lrint(fast_log10(meterval[i]) * -20.0f * mh / range));
191  if (h1 < 0)
192  h1 = 0;
193  else if (h1 > mh)
194  h1 = mh;
195 
196  p.drawPixmap(x, y1 + mh - h1, mw, h1, onPm, 0, mh - h1, mw, h1);
197  p.drawPixmap(x, y1, mw, mh - h1, offPm, 0, 0, mw, mh - h1);
198 
199  //---------------------------------------------------
200  // draw peak line
201  //---------------------------------------------------
202 
203  h1 = mh - (lrint(fast_log10(meterPeak[i]) * -20.0f * mh / range));
204  if (h1 > mh)
205  h1 = mh;
206  if (h1 > 0)
207  p.drawLine(x, y3 - h1, x + mw, y3 - h1);
208 
209  x += mw;
210  }
211  x += 4;
212 
213  // optimize common case:
214  if (ev->rect() == QRect(METER_LEFT_EDGE, kh / 2, _meterWidth - 1, mh))
215  return;
216 
217  QColor sc(isEnabled() ? _scaleColor : Qt::gray);
218  QColor svc(isEnabled() ? _scaleValueColor : Qt::gray);
219  p.setBrush(svc);
220 
221  //---------------------------------------------------
222  // draw scale
223  //---------------------------------------------------
224 
225  int y2 = h - (ppos + y1);
226  p.fillRect(x, y1, _scaleWidth, y2 - y1, sc);
227  p.fillRect(x, y2, _scaleWidth, y3 - y2, svc);
228 
229  //---------------------------------------------------
230  // draw tick marks
231  //---------------------------------------------------
232 
233  QFont f(p.font());
234  f.setPointSize(6);
235  p.setFont(f);
236  p.setPen(QPen(Qt::darkGray, 2));
237 
238  QString s;
239  for (int i = 0; i <= range; i += 10) {
240  h = y1 + lrint(i * mh / range);
241  s.setNum(i);
242  if (i == 0) {
243  p.drawText(QRect(0, h - 3, METER_LEFT_EDGE - 3, 9), Qt::AlignRight, QString("dB"));
244  p.drawLine(METER_LEFT_EDGE - 1, h + 1, METER_LEFT_EDGE - 1, h + 1);
245  continue;
246  }
247  else if (i == range)
248  h -= 2;
249  p.drawText(QRect(0, h - 3, METER_LEFT_EDGE - 3, 9), Qt::AlignRight, QString("-") + s);
250  p.drawLine(METER_LEFT_EDGE - 1, h + 1, METER_LEFT_EDGE - 1, h + 1);
251  }
252 
253  //---------------------------------------------------
254  // draw slider
255  //---------------------------------------------------
256 
257  x += _scaleWidth / 2;
258  p.setPen(QPen(svc, 0));
259  p.translate(QPointF(x, y2));
260  p.setRenderHint(QPainter::Antialiasing, true);
261  p.drawPath(*points);
262  }
