40 void MasterSynthesizer::init()
41  {
42  SynthesizerState state;
43  QString s(dataPath + "/synthesizer.xml");
44  QFile f(s);
45  if (!f.open(QIODevice::ReadOnly)) {
46  // qDebug("cannot read synthesizer settings <%s>", qPrintable(s));
47  setState(defaultState);
48  return;
49  }
50  XmlReader e(&f);
51  while (e.readNextStartElement()) {
52  if (e.name() == "Synthesizer")
53  state.read(e);
54  else
55  e.unknown();
56  }
57  if (!setState(state)) {
58  f.remove();
59  setState(defaultState);
60  }
61  }

278 void MasterSynthesizer::process(unsigned n, float* p)
279  {
280  if (lock2)
281  return;
282  lock1 = true;
283  if (lock2) {
284  lock1 = false;
285  return;
286  }
287  // avoid overflow
288  if (n > MAX_BUFFERSIZE / 2)
289  return;
290  for (Synthesizer* s : _synthesizer) {
291  if (s->active())
292  s->process(n, p, effect1Buffer, effect2Buffer);
293  }
294 
295  if (_effect[0] && _effect[1]) {
296  memset(effect1Buffer, 0, n * sizeof(float) * 2);
297  _effect[0]->process(n, p, effect1Buffer);
298  _effect[1]->process(n, effect1Buffer, p);
299  }
300  else if (_effect[0] || _effect[1]) {
301  memcpy(effect1Buffer, p, n * sizeof(float) * 2);
302  if (_effect[0])
303  _effect[0]->process(n, effect1Buffer, p);
304  else
305  _effect[1]->process(n, effect1Buffer, p);
306  }
307  float g = _gain * _boost;
308  for (unsigned i = 0; i < n * 2; ++i)
309  *p++ *= g;
310  lock1 = false;
311  }

340 bool MasterSynthesizer::setState(const SynthesizerState& ss)
341  {
342  bool result = true;
343  for (const SynthesizerGroup& g : ss) {
344  if (g.name() == "master") {
345  for (const IdValue& v : g) {
346  switch (v.id) {
347  case 0:
348  setEffect(0, indexOfEffect(0, v.data));
349  break;
350  case 1:
351  setEffect(1, indexOfEffect(1, v.data));
352  break;
353  case 2: {
354  float f = v.data.toDouble();
355  setGain(f);
356  }
357  break;
358  case 3:
359  setMasterTuning(v.data.toDouble());
360  break;
361  case 4:
362  setDynamicsMethod(v.data.toInt());
363  break;
364  case 5:
365  setCcToUseIndex(v.data.toInt());
366  break;
367  default:
368  qDebug("MasterSynthesizer::setState: unknown master id <%d>", v.id);
369  }
370  }
371  }
372  else {
373  Synthesizer* s = synthesizer(g.name());
374  if (s) {
375  bool r = s->setState(g);
376  result = result && r;
377  }
378  else {
379  if (effect(0) && effect(0)->name() == g.name())
380  effect(0)->setState(g);
381  else if (effect(1) && effect(1)->name() == g.name())
382  effect(1)->setState(g);
383  else
384  qDebug("MasterSynthesizer::setState: unknown <%s>", qPrintable(g.name()));
385  }
386  }
387  }
388  return result;
389  }

395 SynthesizerState MasterSynthesizer::state() const
396  {
397  SynthesizerState ss;
398  SynthesizerGroup g;
399  g.setName("master");
400  g.push_back(IdValue(0, QString("%1").arg(_effect[0] ? _effect[0]->name() : "NoEffect")));
401  g.push_back(IdValue(1, QString("%1").arg(_effect[1] ? _effect[1]->name() : "NoEffect")));
402  g.push_back(IdValue(2, QString("%1").arg(gain())));
403  g.push_back(IdValue(3, QString("%1").arg(masterTuning())));
404  g.push_back(IdValue(4, QString("%1").arg(dynamicsMethod())));
405  g.push_back(IdValue(5, QString("%1").arg(ccToUseIndex())));
406  ss.push_back(g);
407  for (Synthesizer* s : _synthesizer)
408  ss.push_back(s->state());
409  if (_effect[0])
410  ss.push_back(_effect[0]->state());
411  if (_effect[1])
412  ss.push_back(_effect[1]->state());
413  return ss;
414  }
