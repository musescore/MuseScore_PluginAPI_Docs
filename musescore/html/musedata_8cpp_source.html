116 void MuseData::readChord(Part*, const QString& s)
117  {
118  // a b c d e f g
119  static int table[7] = { 9, 11, 0, 2, 4, 5, 7 };
120 
121  int step = s[1].toLatin1() - 'A';
122  int alter = 0;
123  int octave = 0;
124  for (int i = 2; i < 4; ++i) {
125  if (s[i] == '#')
126  alter += 1;
127  else if (s[i] == 'f')
128  alter -= 1;
129  else if (s[i].isDigit()) {
130  octave = s.midRef(i,1).toInt();
131  break;
132  }
133  }
134  int staffIdx = 0;
135  if (s.size() >= 24) {
136  if (s[23].isDigit())
137  staffIdx = s.midRef(23,1).toInt() - 1;
138  }
139  int pitch = table[step] + alter + (octave + 1) * 12;
140  if (pitch < 0)
141  pitch = 0;
142  if (pitch > 127)
143  pitch = 127;
144 
145  Chord* chord = (Chord*)chordRest;
146  Note* note = new Note(score);
147  note->setPitch(pitch);
148  note->setTpcFromPitch();
149  note->setTrack(staffIdx * VOICES + voice);
150  chord->add(note);
151  }

190 void MuseData::readNote(Part* part, const QString& s)
191  {
192  // a b c d e f g
193  static int table[7] = { 9, 11, 0, 2, 4, 5, 7 };
194 
195  int step = s[0].toLatin1() - 'A';
196  int alter = 0;
197  int octave = 0;
198  for (int i = 1; i < 3; ++i) {
199  if (s[i] == '#')
200  alter += 1;
201  else if (s[i] == 'f')
202  alter -= 1;
203  else if (s[i].isDigit()) {
204  octave = s.midRef(i,1).toInt();
205  break;
206  }
207  }
208  Direction dir = Direction::AUTO;
209  if (s.size() >= 23) {
210  if (s[22] == 'u')
211  dir = Direction::UP;
212  else if (s[22] == 'd')
213  dir = Direction::DOWN;
214  }
215 
216  int staffIdx = 0;
217  if (s.size() >= 24) {
218  if (s[23].isDigit())
219  staffIdx = s.midRef(23,1).toInt() - 1;
220  }
221  Staff* staff = part->staff(staffIdx);
222  int gstaff = staff->idx();
223 
224  int pitch = table[step] + alter + (octave + 1) * 12;
225  if (pitch < 0)
226  pitch = 0;
227  if (pitch > 127)
228  pitch = 127;
229  Fraction ticks = Fraction::fromTicks((s.midRef(5, 3).toInt() * MScore::division + _division/2) / _division);
230  Fraction tick = curTick;
231  curTick += ticks;
232 
233  Tuplet* tuplet = 0;
234  if (s.size() >= 22) {
235  int a = 1;
236  int b = 1;
237  if (s[19] != ' ') {
238  a = s[19].toLatin1() - '0';
239  if (a == 3 && s[20] != ':')
240  b = 2;
241  else {
242  b = s[21].toLatin1() - '0';
243  }
244  }
245  if (a == 3 && b == 2) { // triplet
246  if (chordRest && chordRest->tuplet() && ntuplet) {
247  tuplet = chordRest->tuplet();
248  }
249  else {
250  tuplet = new Tuplet(score);
251  tuplet->setTrack(gstaff * VOICES);
252  tuplet->setTick(tick);
253  ntuplet = a;
254  tuplet->setRatio(Fraction(a, b));
255  measure->add(tuplet);
256  }
257  }
258  else if (a == 1 && b == 1)
259  ;
260  else
261  qDebug("unsupported tuple %d/%d", a, b);
262  }
263 
264  Chord* chord = new Chord(score);
265  chordRest = chord;
266  chord->setTrack(gstaff * VOICES);
267  chord->setStemDirection(dir);
268  if (tuplet) {
269  chord->setTuplet(tuplet);
270  tuplet->add(chord);
271  --ntuplet;
272  }
273  TDuration d;
274  d.setVal(ticks.ticks());
275  chord->setDurationType(d);
276 
277  Segment* segment = measure->getSegment(SegmentType::ChordRest, tick);
278 
279  voice = 0;
280  for (; voice < VOICES; ++voice) {
281  Element* e = segment->element(gstaff * VOICES + voice);
282  if (e == 0) {
283  chord->setTrack(gstaff * VOICES + voice);
284  segment->add(chord);
285  break;
286  }
287  }
288  if (voice == VOICES) {
289  qDebug("cannot allocate voice");
290  delete chord;
291  return;
292  }
293  Note* note = new Note(score);
294  note->setPitch(pitch);
295  note->setTpcFromPitch();
296  note->setTrack(gstaff * VOICES + voice);
297  chord->add(note);
298 
299  QString dynamics;
300  QString an = s.mid(31, 11);
301  for (int i = 0; i < an.size(); ++i) {
302  if (an[i] == '(')
303  openSlur(0, tick, staff, voice);
304  else if (an[i] == ')')
305  closeSlur(0, tick, staff, voice);
306  else if (an[i] == '[')
307  openSlur(1, tick, staff, voice);
308  else if (an[i] == ']')
309  closeSlur(1, tick, staff, voice);
310  else if (an[i] == '{')
311  openSlur(2, tick, staff, voice);
312  else if (an[i] == '}')
313  closeSlur(2, tick, staff, voice);
314  else if (an[i] == 'z')
315  openSlur(3, tick, staff, voice);
316  else if (an[i] == 'x')
317  closeSlur(3, tick, staff, voice);
318  else if (an[i] == '.') {
319  Articulation* atr = new Articulation(score);
320  atr->setSymId(SymId::articStaccatoAbove);
321  chord->add(atr);
322  }
323  else if (an[i] == '_') {
324  Articulation* atr = new Articulation(score);
325  atr->setSymId(SymId::articTenutoAbove);
326  chord->add(atr);
327  }
328  else if (an[i] == 'v') {
329  Articulation* atr = new Articulation(score);
330  atr->setSymId(SymId::stringsUpBow);
331  chord->add(atr);
332  }
333  else if (an[i] == 'n') {
334  Articulation* atr = new Articulation(score);
335  atr->setSymId(SymId::stringsDownBow);
336  chord->add(atr);
337  }
338  else if (an[i] == 't') {
339  Articulation* atr = new Articulation(score);
340  atr->setSymId(SymId::ornamentTrill);
341  chord->add(atr);
342  }
343  else if (an[i] == 'F') {
344  Articulation* atr = new Articulation(score);
345  atr->setUp(true);
346  atr->setSymId(SymId::fermataAbove);
347  chord->add(atr);
348  }
349  else if (an[i] == 'E') {
350  Articulation* atr = new Articulation(score);
351  atr->setUp(false);
352  atr->setSymId(SymId::fermataBelow);
353  chord->add(atr);
354  }
355  else if (an[i] == 'O') {
356  // Articulation* atr = new Articulation(score);
357  // atr->setArticulationType(ArticulationType::Downbow);
358  // chord->add(atr);
359  qDebug("%06d: open string '%c' not implemented", tick.ticks(), an[i].toLatin1());
360  }
361  else if (an[i] == '&') {
362  // skip editorial level
363  if (i <= an.size() && an[i+1].isDigit())
364  ++i;
365  }
366  else if (an[i] == 'p')
367  dynamics += "p";
368  else if (an[i] == 'm')
369  dynamics += "m";
370  else if (an[i] == 'f')
371  dynamics += "f";
372  else if (an[i] == '-') // tie
373  ;
374  else if (an[i] == '*') // start tuplet
375  ;
376  else if (an[i] == '!') // stop tuplet
377  ;
378  else if (an[i] == '+') // cautionary accidental
379  ;
380  else if (an[i] == 'X') // ???
381  ;
382  else if (an[i] == ' ')
383  ;
384  else {
385  qDebug("%06d: notation '%c' not implemented", tick.ticks(), an[i].toLatin1());
386  }
387  }
388  if (!dynamics.isEmpty()) {
389  Dynamic* dyn = new Dynamic(score);
390  dyn->setDynamicType(dynamics);
391  dyn->setTrack(gstaff * VOICES);
392  Segment* seg = measure->getSegment(SegmentType::ChordRest, tick);
393  seg->add(dyn);
394  }
395 
396  QString txt = s.mid(43, 36);
397  if (!txt.isEmpty()) {
398  QStringList sl = txt.split("|");
399  int no = 0;
400  foreach(QString w, sl) {
401  w = diacritical(w);
402  Lyrics* l = new Lyrics(score);
403  l->setPlainText(w);
404  l->setNo(no++);
405  l->setTrack(gstaff * VOICES);
406  Segment* seg = measure->tick2segment(tick);
407  seg->add(l);
408  }
409  }
410  }

544 void MuseData::readPart(QStringList sl, Part* part)
545  {
546  int line = 10;
547  QString s;
548  for (; line < sl.size(); ++line) {
549  s = sl[line];
550  if (!s.isEmpty() && s[0] == '$')
551  break;
552  }
553  if (line >= sl.size()) {
554  qDebug(" $ not found in part");
555  return;
556  }
557  curTick = Fraction(0,1);
558  slur[0] = 0;
559  slur[1] = 0;
560  slur[2] = 0;
561  slur[3] = 0;
562  measure = 0;
563  measure = createMeasure();
564  for (; line < sl.size(); ++line) {
565  s = sl[line];
566 // qDebug("%6d: <%s>", curTick.ticks(), qPrintable(s));
567  char c = s[0].toLatin1();
568  switch(c) {
569  case 'A':
570  case 'B':
571  case 'C':
572  case 'D':
573  case 'E':
574  case 'F':
575  case 'G':
576  readNote(part, s);
577  break;
578  case ' ': // chord
579  readChord(part, s);
580  break;
581  case 'r':
582  readRest(part, s);
583  break;
584  case 'g': // grace note
585  case 'c': // cue note
586  case 'f': // basso continuo
587  break;
588  case 'b': // backspace
589  case 'i': // forward space
590  readBackup(s);
591  break;
592  case 'm': // measure line / bar line
593  measure = createMeasure();
594  break;
595  case '*': // musical direction
596  break;
597  case 'P': // print suggestion
598  break;
599  case 'S': // sound record
600  break;
601  case '$':
602  musicalAttribute(s, part);
603  break;
604  default:
605  qDebug("unknown record <%s>", qPrintable(s));
606  break;
607  }
608  }
609  }

650 bool MuseData::read(const QString& name)
651  {
652  QFile fp(name);
653  if (!fp.open(QIODevice::ReadOnly)) {
654  qDebug("Cannot open file <%s>", qPrintable(name));
655  return false;
656  }
657  QTextStream ts(&fp);
658  QStringList part;
659  bool commentMode = false;
660  for (;;) {
661  QString s(ts.readLine());
662  if (s.isNull())
663  break;
664  if (s.isEmpty()) {
665  if (!commentMode)
666  part.append(QString(""));
667  continue;
668  }
669  if (s[0] == '&') {
670  commentMode = !commentMode;
671  continue;
672  }
673  if (commentMode)
674  continue;
675  if (s[0] == '@')
676  continue;
677  if (s[0] == '/') {
678  parts.append(part);
679 
680  Part* mpart = new Part(score);
681  int staves = countStaves(part);
682  for (int i = 0; i < staves; ++i) {
683  Staff* staff = new Staff(score);
684  staff->setPart(mpart);
685  mpart->insertStaff(staff, i);
686  score->staves().push_back(staff);
687  if ((staves == 2) && (i == 0)) {
688  staff->setBracketType(0, BracketType::BRACE);
689  staff->setBracketSpan(0, 2);
690  }
691  }
692  score->appendPart(mpart);
693  if(part.size() > 8)
694  mpart->setPlainLongName(part[8]);
695  part.clear();
696  continue;
697  }
698  if (s[0] == 'a') {
699  part.back().append(s.midRef(1));
700  continue;
701  }
702  part.append(s);
703  }
704  fp.close();
705  return true;
706  }
