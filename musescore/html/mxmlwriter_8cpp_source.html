68  void MxmlWriter::beginMeasure(const Bww::MeasureBeginFlags mbf)
69  {
70  // qDebug() << "MxmlWriter::beginMeasure()";
71  QString implicit;
72  QString strMeasureNumber;
73  if (mbf.irregular)
74  {
75  implicit = " implicit=\"yes\"";
76  if (irregularMeasureNumber == 0)
77  strMeasureNumber = "0";
78  else
79  strMeasureNumber = QString("X%1").arg(irregularMeasureNumber);
80  ++irregularMeasureNumber;
81  }
82  else
83  {
84  ++regularMeasureNumber;
85  strMeasureNumber.setNum(regularMeasureNumber);
86  }
87  out << " <measure number=\"" << strMeasureNumber << "\"" << implicit << ">" << endl;
88  if ((regularMeasureNumber + irregularMeasureNumber) != 1 && mbf.firstOfSystem)
89  {
90  out << " <print new-system=\"yes\"/>" << endl;
91  }
92  if (mbf.repeatBegin || mbf.endingFirst || mbf.endingSecond)
93  {
94  out << " <barline location=\"left\">" << endl;
95  if (mbf.repeatBegin)
96  {
97  out << " <bar-style>heavy-light</bar-style>" << endl;
98  out << " <repeat direction=\"forward\"/>" << endl;
99  }
100  if (mbf.endingFirst)
101  {
102  out << " <ending number=\"1\" type=\"start\"/>" << endl;
103  ending = 1;
104  }
105  if (mbf.endingSecond)
106  {
107  out << " <ending number=\"2\" type=\"start\"/>" << endl;
108  ending = 2;
109  }
110  out << " </barline>" << endl;
111  }
112  if ((regularMeasureNumber + irregularMeasureNumber) == 1)
113  {
114  out << " <attributes>" << endl;
115  out << " <divisions>" << wholeDur() / 4 << "</divisions>" << endl;
116 // out << " <key print-object=\"no\">" << endl;
117  out << " <key>" << endl;
118  out << " <fifths>2</fifths>" << endl;
119  out << " <mode>major</mode>" << endl;
120  out << " </key>" << endl;
121  out << " <time>" << endl;
122  out << " <beats>" << beats << "</beats>" << endl;
123  out << " <beat-type>" << beat << "</beat-type>" << endl;
124  out << " </time>" << endl;
125  out << " <clef>" << endl;
126  out << " <sign>G</sign>" << endl;
127  out << " <line>2</line>" << endl;
128  out << " </clef>" << endl;
129  out << " </attributes>" << endl;
130  if (tempo)
131  {
132  out << " <direction placement=\"above\">" << endl;
133  out << " <direction-type>" << endl;
134  out << " <metronome parentheses=\"no\">" << endl;
135  out << " <beat-unit>quarter</beat-unit>" << endl;
136  out << " <per-minute>" << tempo << "</per-minute>" << endl;
137  out << " </metronome>" << endl;
138  out << " </direction-type>" << endl;
139  out << " <sound tempo=\"" << tempo << "\"/>" << endl;
140  out << " </direction>" << endl;
141  }
142  }
143  }

149  void MxmlWriter::endMeasure(const Bww::MeasureEndFlags mef)
150  {
151  // qDebug() << "MxmlWriter::endMeasure()";
152  if (mef.repeatEnd || mef.endingEnd || mef.lastOfPart || mef.doubleBarLine)
153  {
154  out << " <barline location=\"right\">" << endl;
155  if (mef.repeatEnd || mef.lastOfPart)
156  {
157  out << " <bar-style>light-heavy</bar-style>" << endl;
158  }
159  else if (mef.doubleBarLine)
160  {
161  out << " <bar-style>light-light</bar-style>" << endl;
162  }
163  if (mef.endingEnd)
164  {
165  if (ending == 1)
166  out << " <ending number=\"1\" type=\"stop\"/>" << endl;
167  else if (ending == 2)
168  out << " <ending number=\"2\" type=\"discontinue\"/>" << endl;
169  else
170  {
171  // TODO: error message
172  }
173  }
174  if (mef.repeatEnd)
175  {
176  out << " <repeat direction=\"backward\"/>" << endl;
177  }
178  out << " </barline>" << endl;
179  }
180  out << " </measure>" << endl;
181  }

187  void MxmlWriter::note(const QString pitch, QVector<BeamType> beamList,
188  const QString type, const int dots,
189  bool tieStart, bool tieStop,
190  StartStop triplet,
191  bool grace)
192  {
193  // qDebug() << "MxmlWriter::note()";
194 
195  if (!stepAlterOctMap.contains(pitch)
196  || !typeMap.contains(type))
197  {
198  // TODO: error message
199  return;
200  }
201  StepAlterOct sao = stepAlterOctMap.value(pitch);
202 
203  int dur = wholeDur() / type.toInt();
204  if (dots == 1) dur = 3 * dur / 2;
205  if (triplet != ST_NONE) dur = 2 * dur / 3;
206  out << " <note>" << endl;
207  if (grace) out << " <grace/>" << endl;
208  out << " <pitch>" << endl;
209  out << " <step>" << sao.s << "</step>" << endl;
210  if (sao.a) out << " <alter>" << sao.a << "</alter>" << endl;
211  out << " <octave>" << sao.o << "</octave>" << endl;
212  out << " </pitch>" << endl;
213  if (!grace)
214  out << " <duration>" << dur << "</duration>" << endl;
215  if (tieStart)
216  out << " <tie type=\"start\"/>" << endl;
217  if (tieStop)
218  out << " <tie type=\"stop\"/>" << endl;
219  out << " <type>" << typeMap.value(type) << "</type>" << endl;
220  if (dots == 1) out << " <dot/>" << endl;
221  if (triplet != ST_NONE)
222  {
223  out << " <time-modification>" << endl;
224  out << " <actual-notes>3</actual-notes>" << endl;
225  out << " <normal-notes>2</normal-notes>" << endl;
226  out << " </time-modification>" << endl;
227  }
228  if (grace)
229  out << " <stem>up</stem>" << endl;
230  else
231  {
232  if (type != "1")
233  out << " <stem>down</stem>" << endl;
234  }
235  for (int i = 0; i < maxBeamLevel; ++i)
236  {
237  QString s;
238  switch (beamList.at(i))
239  {
240  case BM_BEGIN: s = "begin"; break;
241  case BM_CONTINUE: s = "continue"; break;
242  case BM_END: s = "end"; break;
243  case BM_FORWARD_HOOK: s = "forward hook"; break;
244  case BM_BACKWARD_HOOK: s = "backward hook"; break;
245  default: s = ""; break;
246  }
247  if (s != "")
248  out << " <beam number=\"" << i + 1 << "\">"
249  << s << "</beam>" << endl;
250  else
251  break;
252  }
253  if (tieStart || tieStop || triplet == ST_START || triplet == ST_STOP)
254  {
255  out << " <notations>" << endl;
256  if (tieStart)
257  out << " <tied type=\"start\"/>" << endl;
258  if (tieStop)
259  out << " <tied type=\"stop\"/>" << endl;
260  if (triplet == ST_START)
261  {
262  if (type == "1" || type == "2" || type == "4")
263  out << " <tuplet type=\"start\" bracket=\"yes\"/>" << endl;
264  else
265  out << " <tuplet type=\"start\" bracket=\"no\"/>" << endl;
266  }
267  if (triplet == ST_STOP)
268  out << " <tuplet type=\"stop\"/>" << endl;
269  out << " </notations>" << endl;
270  }
271  out << " </note>" << endl;
272  }
