36 QMenu* QmlNativeMenu::createMenu() const
37  {
38  QMenu* menu = new QMenu();
39 
40  for (QObject* obj : _contentData) {
41  if (qobject_cast<QmlMenuSeparator*>(obj))
42  menu->addSeparator();
43  else if (QmlMenuItem* m = qobject_cast<QmlMenuItem*>(obj)) {
44  QAction* a = menu->addAction(m->text());
45  a->setCheckable(m->checkable());
46  a->setChecked(m->checked());
47  a->setEnabled(m->enabled());
48  connect(a, &QAction::triggered, m, &QmlMenuItem::triggered, Qt::QueuedConnection);
49  }
50  }
51 
52  return menu;
53  }
