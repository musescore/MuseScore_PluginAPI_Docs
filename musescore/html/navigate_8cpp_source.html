35 // nextChordRest
36 // return next Chord or Rest

39 ChordRest* nextChordRest(ChordRest* cr, bool skipGrace)
40  {
41  if (!cr)
42  return 0;
43 
44  if (cr->isGrace()) {
45  //
46  // cr is a grace note
47 
48  Chord* c = toChord(cr);
49  Chord* pc = toChord(cr->parent());
50 
51  if (skipGrace) {
52  cr = toChordRest(cr->parent());
53  }
54  else if (cr->isGraceBefore()) {
55  QVector<Chord*> cl = pc->graceNotesBefore();
56  auto i = std::find(cl.begin(), cl.end(), c);
57  if (i == cl.end())
58  return 0; // unable to find self?
59  ++i;
60  if (i != cl.end())
61  return *i;
62  // if this was last grace note before, return parent
63  return pc;
64  }
65  else {
66  QVector<Chord*> cl = pc->graceNotesAfter();
67  auto i = std::find(cl.begin(), cl.end(), c);
68  if (i == cl.end())
69  return 0; // unable to find self?
70  ++i;
71  if (i != cl.end())
72  return *i;
73  // if this was last grace note after, fall through to find next main note
74  cr = pc;
75  }
76  }
77  else {
78  //
79  // cr is not a grace note
80  if (cr->isChord() && !skipGrace) {
81  Chord* c = toChord(cr);
82  if (!c->graceNotes().empty()) {
83  QVector<Chord*> cl = c->graceNotesAfter();
84  if (!cl.empty())
85  return cl.first();
86  }
87  }
88  }
89 
90  int track = cr->track();
91  SegmentType st = SegmentType::ChordRest;
92 
93  for (Segment* seg = cr->segment()->next1MM(st); seg; seg = seg->next1MM(st)) {
94  ChordRest* e = toChordRest(seg->element(track));
95  if (e) {
96  if (e->isChord() && !skipGrace) {
97  Chord* c = toChord(e);
98  if (!c->graceNotes().empty()) {
99  QVector<Chord*> cl = c->graceNotesBefore();
100  if (!cl.empty())
101  return cl.first();
102  }
103  }
104  return e;
105  }
106  }
107 
108  return 0;
109  }

112 // prevChordRest
113 // return previous Chord or Rest
114 // if grace is true, include grace notes
115 //---------------------------------------------------------
116 
117 ChordRest* prevChordRest(ChordRest* cr, bool skipGrace)
118  {
119  if (!cr)
120  return 0;
121 
122  if (cr->isGrace()) {
123  //
124  // cr is a grace note
125 
126  Chord* c = toChord(cr);
127  Chord* pc = toChord(cr->parent());
128 
129  if (skipGrace) {
130  cr = toChordRest(cr->parent());
131  }
132  else if (cr->isGraceBefore()) {
133  QVector<Chord*> cl = pc->graceNotesBefore();
134  auto i = std::find(cl.begin(), cl.end(), c);
135  if (i == cl.end())
136  return 0; // unable to find self?
137  if (i != cl.begin())
138  return *--i;
139  // if this was first grace note before, fall through to find previous main note
140  cr = pc;
141  }
142  else {
143  QVector<Chord*> cl = pc->graceNotesAfter();
144  auto i = std::find(cl.begin(), cl.end(), c);
145  if (i == cl.end())
146  return 0; // unable to find self?
147  if (i != cl.begin())
148  return *--i;
149  // if this was first grace note after, return parent
150  return pc;
151  }
152  }
153  else {
154  //
155  // cr is not a grace note
156  if (cr->isChord() && !skipGrace) {
157  Chord* c = toChord(cr);
158  QVector<Chord*> cl = c->graceNotesBefore();
159  if (!cl.empty())
160  return cl.last();
161  }
162  }
163 
164  int track = cr->track();
165  SegmentType st = SegmentType::ChordRest;
166  for (Segment* seg = cr->segment()->prev1MM(st); seg; seg = seg->prev1MM(st)) {
167  ChordRest* e = toChordRest(seg->element(track));
168  if (e) {
169  if (e->type() == ElementType::CHORD && !skipGrace) {
170  QVector<Chord*> cl = toChord(e)->graceNotesAfter();
171  if (!cl.empty())
172  return cl.last();
173  }
174  return e;
175  }
176  }
177 
178  return 0;
179  }

190 Element* Score::upAlt(Element* element)
191  {
192  Element* re = 0;
193  if (element->isRest())
194  re = prevTrack(toRest(element));
195  else if (element->isNote()) {
196  Note* note = toNote(element);
197  Chord* chord = note->chord();
198  const std::vector<Note*>& notes = chord->notes();
199  auto i = std::find(notes.begin(), notes.end(), note);
200  ++i;
201  if (i != notes.end())
202  re = *i;
203  else {
204  re = prevTrack(chord);
205  if (re->track() == chord->track())
206  re = element;
207  }
208  }
209  if (re == 0)
210  return 0;
211  if (re->isChord())
212  re = toChord(re)->notes().front();
213  return re;
214  }

232 Element* Score::downAlt(Element* element)
233  {
234  Element* re = 0;
235  if (element->isRest())
236  re = nextTrack(toRest(element));
237  else if (element->isNote()) {
238  Note* note = toNote(element);
239  Chord* chord = note->chord();
240  const std::vector<Note*>& notes = chord->notes();
241  auto i = std::find(notes.begin(), notes.end(), note);
242  if (i != notes.begin()) {
243  --i;
244  re = *i;
245  }
246  else {
247  re = nextTrack(chord);
248  if (re->track() == chord->track())
249  re = element;
250  }
251  }
252  if (re == 0)
253  return 0;
254  if (re->isChord())
255  re = toChord(re)->notes().back();
256  return re;
257  }

288 Element* Score::lastElement(bool frame)
289  {
290  if (frame) {
291  MeasureBase* mb = measures()->last();
292  if (mb && mb->isBox())
293  return mb;
294  }
295  Element* re = 0;
296  Segment* seg = lastSegmentMM();
297  if (!seg)
298  return nullptr;
299  while (true) {
300  for (int i = (staves().size() -1) * VOICES; i < staves().size() * VOICES; i++) {
301  if (seg->element(i))
302  re = seg->element(i);
303  }
304  if (re) {
305  if (re->isChord()) {
306  return toChord(re)->notes().front();
307  }
308  return re;
309  }
310  seg = seg->prev1MM(SegmentType::All);
311  }

362 // nextTrack
363 // returns note at or just before current (cr) position
364 // in next track for this measure
365 // that contains such an element
366 //---------------------------------------------------------
367 
368 ChordRest* Score::nextTrack(ChordRest* cr)
369  {
370  if (!cr)
371  return 0;
372 
373  ChordRest* el = 0;
374  Measure* measure = cr->measure();
375  int track = cr->track();
376  int tracks = nstaves() * VOICES;
377 
378  while (!el) {
379  // find next non-empty track
380  while (++track < tracks) {
381  if (measure->hasVoice(track))
382  break;
383  }
384  // no more tracks, return original element
385  if (track == tracks)
386  return cr;
387  // find element at same or previous segment within this track
388  for (Segment* segment = cr->segment(); segment; segment = segment->prev(SegmentType::ChordRest)) {
389  el = toChordRest(segment->element(track));
390  if (el)
391  break;
392  }
393  }
394  return el;
395  }

436 ChordRest* Score::nextMeasure(ChordRest* element, bool selectBehavior, bool mmRest)
437  {
438  if (!element)
439  return 0;
440 
441  Measure* measure = 0;
442  if (mmRest)
443  measure = element->measure()->nextMeasureMM();
444  else
445  measure = element->measure()->nextMeasure();
446 
447  if (!measure)
448  return 0;
449 
450  Fraction endTick = element->measure()->last()->nextChordRest(element->track(), true)->tick();
451  bool last = false;
452 
453  if (selection().isRange()) {
454  if (element->tick() != endTick && selection().tickEnd() <= endTick) {
455  measure = element->measure();
456  last = true;
457  }
458  else if (element->tick() == endTick && selection().isEndActive())
459  last = true;
460  }
461  else if (element->tick() != endTick && selectBehavior) {
462  measure = element->measure();
463  last = true;
464  }
465  if (!measure) {
466  measure = element->measure();
467  last = true;
468  }
469  int staff = element->staffIdx();
470 
471  Segment* startSeg = last ? measure->last() : measure->first();
472  for (Segment* seg = startSeg; seg; seg = last ? seg->prev() : seg->next()) {
473  int etrack = (staff+1) * VOICES;
474  for (int track = staff * VOICES; track < etrack; ++track) {
475  Element* pel = seg->element(track);
476 
477  if (pel && pel->isChordRest())
478  return toChordRest(pel);
479  }
480  }
481  return 0;
482  }

531 Element* Score::nextElement()
532  {
533  Element* e = getSelectedElement();
534  if (!e)
535  return nullptr;
536  int staffId = e->staffIdx();
537  while (e) {
538  switch (e->type()) {
539  case ElementType::NOTE:
540  case ElementType::REST:
541  case ElementType::CHORD: {
542  Element* next = e->nextElement();
543  if (next)
544  return next;
545  else
546  break;
547  }
548  case ElementType::SEGMENT: {
549  Segment* s = toSegment(e);
550  Element* next = s->nextElement(staffId);
551  if (next)
552  return next;
553  else
554  break;
555  }
556  case ElementType::MEASURE: {
557  Measure* m = toMeasure(e);
558  Element* next = m->nextElementStaff(staffId);
559  if (next)
560  return next;
561  else
562  break;
563  }
564  case ElementType::CLEF:
565  case ElementType::KEYSIG:
566  case ElementType::TIMESIG:
567  case ElementType::BAR_LINE: {
568  for (; e && e->type() != ElementType::SEGMENT; e = e->parent()) {
569  ;
570  }
571  Segment* s = toSegment(e);
572  Element* next = s->nextElement(staffId);
573  if (next)
574  return next;
575  else
576  return score()->firstElement();
577  }
578 #if 1
579  case ElementType::VOLTA_SEGMENT:
580 #else
581  case ElementType::VOLTA_SEGMENT: {
582  // TODO: see Spanner::nextSpanner()
583  System* sys = toSpannerSegment(e)->system();
584  if (sys)
585  staffId = sys->firstVisibleStaff();
586  }
587  // fall through
588 #endif
589  case ElementType::SLUR_SEGMENT:
590  case ElementType::TEXTLINE_SEGMENT:
591  case ElementType::HAIRPIN_SEGMENT:
592  case ElementType::OTTAVA_SEGMENT:
593  case ElementType::TRILL_SEGMENT:
594  case ElementType::VIBRATO_SEGMENT:
595  case ElementType::LET_RING_SEGMENT:
596  case ElementType::PALM_MUTE_SEGMENT:
597  case ElementType::PEDAL_SEGMENT: {
598  SpannerSegment* s = toSpannerSegment(e);
599  Spanner* sp = s->spanner();
600  Spanner* nextSp = sp->nextSpanner(sp, staffId);
601  if (nextSp)
602  return nextSp->spannerSegments().front();
603 
604  Segment* seg = tick2segment(sp->tick());
605  if (seg) {
606  Segment* nextSegment = seg->next1();
607  while (nextSegment) {
608  Element* nextEl = nextSegment->firstElementOfSegment(nextSegment, staffId);
609  if (nextEl)
610  return nextEl;
611  nextSegment = nextSegment->next1MM();
612  }
613  }
614  break;
615  }
616  case ElementType::GLISSANDO_SEGMENT:
617  case ElementType::TIE_SEGMENT: {
618  SpannerSegment* s = toSpannerSegment(e);
619  Spanner* sp = s->spanner();
620  Element* elSt = sp->startElement();
621  Note* n = toNote(elSt);
622  Element* next = n->nextElement();
623  if (next)
624  return next;
625  else
626  break;
627  }
628  case ElementType::VBOX:
629  case ElementType::HBOX:
630  case ElementType::TBOX: {
631  MeasureBase* mb = toMeasureBase(e)->nextMM();
632  if (!mb) {
633  break;
634  }
635  else if (mb->isMeasure()) {
636  ChordRest* cr = selection().currentCR();
637  int si = cr ? cr->staffIdx() : 0;
638  return toMeasure(mb)->nextElementStaff(si);
639  }
640  else {
641  return mb;
642  }
643  }
644  case ElementType::LAYOUT_BREAK: {
645  staffId = 0; // otherwise it will equal -1, which breaks the navigation
646  }
647  default:
648  break;
649  }
650  e = e->parent();
651  }
652  return score()->lastElement();
653  }

659 Element* Score::prevElement()
660  {
661  Element* e = getSelectedElement();
662  if (!e)
663  return nullptr;
664  int staffId = e->staffIdx();
665  while (e) {
666  switch (e->type()) {
667  case ElementType::NOTE:
668  case ElementType::REST:
669  case ElementType::CHORD: {
670  Element* prev = e->prevElement();
671  if (prev)
672  return prev;
673  else
674  break;
675  }
676  case ElementType::SEGMENT: {
677  Segment* s = toSegment(e);
678  Element* prev = s->prevElement(staffId);
679  if (prev)
680  return prev;
681  else
682  break;
683  }
684  case ElementType::MEASURE: {
685  Measure* m = toMeasure(e);
686  return m->prevElementStaff(staffId);
687  }
688  case ElementType::CLEF:
689  case ElementType::KEYSIG:
690  case ElementType::TIMESIG:
691  case ElementType::BAR_LINE: {
692  for (; e && e->type() != ElementType::SEGMENT; e = e->parent()) {
693  ;
694  }
695  Segment* s = toSegment(e);
696  return s->prevElement(staffId);
697  }
698 #if 1
699  case ElementType::VOLTA_SEGMENT:
700 #else
701  case ElementType::VOLTA_SEGMENT: {
702  // TODO: see Spanner::nextSpanner()
703  System* sys = toSpannerSegment(e)->system();
704  if (sys)
705  staffId = sys->firstVisibleStaff();
706  }
707  // fall through
708 #endif
709  case ElementType::SLUR_SEGMENT:
710  case ElementType::TEXTLINE_SEGMENT:
711  case ElementType::HAIRPIN_SEGMENT:
712  case ElementType::OTTAVA_SEGMENT:
713  case ElementType::TRILL_SEGMENT:
714  case ElementType::VIBRATO_SEGMENT:
715  case ElementType::PEDAL_SEGMENT: {
716  SpannerSegment* s = toSpannerSegment(e);
717  Spanner* sp = s->spanner();
718  Element* stEl = sp->startElement();
719  Spanner* prevSp = sp->prevSpanner(sp, staffId);
720  if (prevSp)
721  return prevSp->spannerSegments().front();
722  else {
723  Segment* startSeg = sp->startSegment();
724  if (!startSeg->annotations().empty()) {
725  Element* last = startSeg->lastAnnotation(startSeg, staffId);
726  if (last)
727  return last;
728  }
729  Element* el = startSeg->lastElementOfSegment(startSeg, staffId);
730  if (stEl->type() == ElementType::CHORD || stEl->type() == ElementType::REST
731  || stEl->type() == ElementType::REPEAT_MEASURE || stEl->type() == ElementType::NOTE) {
732  ChordRest* cr = startSeg->cr(stEl->track());
733  if (cr) {
734  Element* elCr = cr->lastElementBeforeSegment();
735  if (elCr) {
736  return elCr;
737  }
738  }
739  }
740  if (el->isChord())
741  return toChord(el)->lastElementBeforeSegment();
742  else if (el->isNote()) {
743  Chord* c = toNote(el)->chord();
744  return c->lastElementBeforeSegment();
745  }
746  else {
747  return el;
748  }
749  }
750  }
751  case ElementType::GLISSANDO_SEGMENT:
752  case ElementType::TIE_SEGMENT: {
753  SpannerSegment* s = toSpannerSegment(e);
754  Spanner* sp = s->spanner();
755  Element* elSt = sp->startElement();
756  Q_ASSERT(elSt->type() == ElementType::NOTE);
757  Note* n = toNote(elSt);
758  Element* prev = n->prevElement();
759  if(prev)
760  return prev;
761  else
762  break;
763  }
764  case ElementType::VBOX:
765  case ElementType::HBOX:
766  case ElementType::TBOX: {
767  MeasureBase* mb = toMeasureBase(e)->prevMM();
768  if (!mb) {
769  break;
770  }
771  else if (mb->isMeasure()) {
772  ChordRest* cr = selection().currentCR();
773  int si = cr ? cr->staffIdx() : 0;
774  Segment* s = toMeasure(mb)->last();
775  if (s)
776  return s->lastElement(si);
777  }
778  else {
779  return mb;
780  }
781  }
782  break;
783  case ElementType::LAYOUT_BREAK: {
784  staffId = 0; // otherwise it will equal -1, which breaks the navigation
785  }
786  default:
787  break;
788  }
789  e = e->parent();
790  }
791  return score()->firstElement();
792  }
