188 void Navigator::rescale()
189  {
190  if (!_score || _score->pages().isEmpty()) {
191  setMaximumSize(QWIDGETSIZE_MAX, QWIDGETSIZE_MAX);
192  setMinimumSize(0, 0);
193  return;
194  }
195  Page* lp = _score->pages().back();
196 
197  // reset the layout before setting fix size
198  setMaximumSize(QWIDGETSIZE_MAX, QWIDGETSIZE_MAX);
199  setMinimumSize(0, 0);
200 
201  if (MScore::verticalOrientation() && !_previewOnly) {
202  qreal scoreWidth = lp->width();
203  qreal scoreHeight = lp->y() + lp->height();
204  qreal m = width() / scoreWidth;
205  setFixedHeight(int(scoreHeight * m));
206  matrix = QTransform(m, 0, 0, m, 0, 0);
207  }
208  else {
209  qreal scoreWidth = lp->x() + lp->width();
210  qreal scoreHeight = lp->height();
211  if (_previewOnly)
212  scoreWidth = lp->width() * _score->pages().size();
213  qreal m = height() / scoreHeight;
214  setFixedWidth(int(scoreWidth * m));
215  matrix = QTransform(m, 0, 0, m, 0, 0);
216  }
217  }

235 void Navigator::mousePressEvent(QMouseEvent* ev)
236  {
237  if (_cv == 0)
238  return;
239  startMove = ev->pos();

241  QPointF p = matrix.inverted().map(QPointF(ev->pos()));
242  QRectF r(_cv->toLogical(QRectF(0.0, 0.0, _cv->width(), _cv->height())));
243  double dx = p.x() - (r.x() + (r.width() * .5));
244  double dy = p.y() - (r.y() + (r.height() * .5));
245  r.translate(dx, dy);
246  setViewRect(r);
247  emit viewRectMoved(matrix.inverted().mapRect(viewRect->geometry()));

266  else if (r.right() > width())
267  r.moveRight(width());

270  if (r.right() < width())
271  r.moveRight(width());
272  else if (r.left() > 0)
273  r.moveLeft(0);
274  }
275 
276  if (r.height() == height())
277  r.moveTop(0);
278  else if (r.height() < height()) {
279  if (r.y() < 0)
280  r.moveTop(0);
281  else if (r.bottom() > height())
282  r.moveBottom(height());
283  }
284  else {
285  if (r.bottom() < height())
286  r.moveBottom(height());
287  else if (r.top() > 0)

345 void Navigator::paintEvent(QPaintEvent* ev)
346  {
347  QPainter p(this);
348  QRect r(ev->rect());
349  p.fillRect(r, palette().color(QPalette::Window));
350 
351 // qDebug("navigator paint x %d w %d h %d", r.x(), r.width(), r.height());
352 
353  if (!_score)
354  return;
355  if (_score->pages().size() <= 0)
356  return;
357 
358  // compute optimal size of page number
359  QFont font("FreeSans", 4000);
360  QFontMetrics fm (font);
361  Page* firstPage = _score->pages()[0];
362  qreal factor = (firstPage->width() * 0.5) / fm.width(QString::number(_score->pages().size()));
363  font.setPointSizeF(font.pointSizeF() * factor);
364 
365  p.setTransform(matrix);
366  QRectF fr = matrix.inverted().mapRect(QRectF(r));
367  int i = 0;
368  for (Page* page : _score->pages()) {
369  QPointF pos(page->pos());
370  if (_previewOnly)
371  pos = QPointF(i * page->width(), 0);
372  QRectF pr(page->abbox().translated(pos));
373  if (pr.right() < fr.left())
374  continue;
375  if (pr.left() > fr.right())
376  break;
377 
378  p.fillRect(pr, Qt::white);
379  p.translate(pos);
380  for (System* s : page->systems()) {
381  for (MeasureBase* m : s->measures())
382  m->scanElements(&p, paintElement, false);
383  }
384  page->scanElements(&p, paintElement, false);
385  if (page->score()->layoutMode() == LayoutMode::PAGE) {
386  p.setFont(font);
387  p.setPen(MScore::layoutBreakColor);
388  p.drawText(page->bbox(), Qt::AlignCenter, QString("%1").arg(page->no() + 1 + _score->pageNumberOffset()));
389  }
390  p.translate(-pos);
391  i++;
392  }
393  }
