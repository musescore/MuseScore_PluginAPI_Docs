37 void NoteDot::draw(QPainter* p) const
38  {
39  if (note() && note()->dotsHidden()) // don't draw dot if note is hidden
40  return;
41  Note* n = note();
42  Fraction tick = n ? n->chord()->tick() : rest()->tick();
43  // always draw dot for non-tab
44  // for tab, draw if on a note and stems through staff or on a rest and rests shown
45  if (!staff()->isTabStaff(tick)
46  || (n && staff()->staffType(tick)->stemThrough())
47  || (!n && staff()->staffType(tick)->showRests())) {
48  p->setPen(curColor());
49  drawSymbol(SymId::augmentationDot, p);
50  }
51  }
