67 void NoteTweakerDialog::setNoteOffTime()
68  {
69  if (!_staff)
70  return;
71 
72  QString s = ui->comboOffTimeFrac->currentText();
73  QStringList parts = s.split("/");
74  int num = parts[0].toInt();
75  double denom = parts[1].toInt();
76  double gapTicks = MScore::division * num / denom;
77 
78  Score* score = _staff->score();
79 
80  score->startCmd();
81 
82  for (Note* note: noteList) {
83  if (!note->selected())
84  continue;
85 
86  Chord* chord = note->chord();
87  int ticks = chord->ticks().ticks();
88 
89  int scalar = 1000 * (ticks - gapTicks) / ticks;
90  if (scalar <= 0)
91  scalar = 1;
92 
93  for (NoteEvent& e : note->playEvents()) {
94  NoteEvent ne = e;
95  ne.setLen(scalar);
96  score->undo(new ChangeNoteEvent(note, &e, ne));
97  }
98  }
99 
100  score->endCmd();
101 
102  emit notesChanged();
103  }

137 void NoteTweakerDialog::updateNotes()
138  {
139  clearNoteData();
140 
141  if (!_staff) {
142  return;
143  }
144 
145  int staffIdx = _staff->idx();
146  if (staffIdx == -1)
147  return;
148 
149  SegmentType st = SegmentType::ChordRest;
150  for (Segment* s = _staff->score()->firstSegment(st); s; s = s->next1(st)) {
151  for (int voice = 0; voice < VOICES; ++voice) {
152  int track = voice + staffIdx * VOICES;
153  Element* e = s->element(track);
154  if (e && e->isChord())
