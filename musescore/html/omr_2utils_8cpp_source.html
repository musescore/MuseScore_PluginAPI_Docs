48 // mean_(n) = mean(n-1) + (data[n] - mean(n-1))/(n+1)

64 double covariance(const double data1[],

66  {
67  long double covariance = 0.0;
68 
69  /* find the sum of the squares */
70  for (size_t i = 0; i < (size_t)n; i++) {
71  const long double delta1 = (data1[i] - mean1);
72  const long double delta2 = (data2[i] - mean2);
73  covariance += (delta1 * delta2 - covariance) / (i + 1);
74  }
75  return covariance;
76  }
