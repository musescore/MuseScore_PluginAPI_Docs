131 float OmrPage::searchBarLines(int start_staff, int end_staff)
132  {
133  OmrStaff& r1 = staves[start_staff];
134  OmrStaff& r2 = staves[end_staff];
135 
136  int x1 = r1.x();
137  int x2 = x1 + r1.width();
138  int y1 = r1.y();
139  int y2 = r2.y() + r2.height();
140 
141  int th = 0;
142  for (int i = start_staff; i <= end_staff; ++i) {
143  th += staves[i].height() / 2;
144  }
145 
146  int vpw = x2 - x1;
147 #if (!defined (_MSCVER) && !defined (_MSC_VER))
148  float vp[vpw];
149  memset(vp, 0, sizeof(float) * vpw);
150 
151  //using note constraints
152  //searchNotes();
153 
154  int note_constraints[x2 - x1];
155 #else
156  // MSVC does not support VLA. Replace with std::vector. If profiling determines that the
157  // heap allocation is slow, an optimization might be used.
158  std::vector<float> vp(vpw); // Default-initialized, doesn't need to be cleared
159  std::vector<int> note_constraints(x2 - x1);
160 #endif
161  for (OmrNote* n : r1.notes()) {
162  for(int x = n->x(); x <= n->x() + n->width(); ++x)
163  note_constraints[x - x1] = 1;
164  }
165  for (OmrNote* n : r2.notes()) {
166  for(int x = n->x(); x <= n->x() + n->width(); ++x)
167  note_constraints[x - x1] = 1;
168  }
169 
170  //
171  // compute vertical projections
172  //
173  for (int x = x1; x < x2; ++x) {
174  int dots = 0;
175  for (int y = y1; y < y2; ++y) {
176  if (this->dot(x, y))
177  ++dots;
178  }
179  if (!note_constraints[x - x1])
180  vp[x - x1] = dots - th;
181  else
182  vp[x - x1] = -HUGE_VAL;
183  }
184 
185 #if (!defined (_MSCVER) && !defined (_MSC_VER))
186  float scores[x2 - x1 + 1][2];
187  BAR_STATE pred[x2 - x1 + 1][2];
188 #else
189  // MSVC does not support VLA. Replace with std::vector. If profiling determines that the
190  // heap allocation is slow, an optimization might be used.
191  std::vector<float[2]> scores(x2 - x1 + 1);
192  std::vector<BAR_STATE[2]> pred(x2 - x1 + 1);
193 #endif
194  BAR_STATE bs;
195 
196  //initialization
197  bs.x = -1; bs.status = -1;
198  for (int x = x1; x <= x2; ++x) {
199  int i = x - x1;
200  for (int status = 0; status <= 1; ++status) {
201  scores[i][status] = -HUGE_VAL;
202  pred[i][status] = bs;
203  }
204  }
205  scores[0][0] = 0;
206 
207  //forward pass
208  for (int x = x1; x <= x2; ++x) {
209  int i = x - x1;
210  for (int cur = 0; cur <= 1; ++cur) {
211  //current state
212  if (scores[i][cur] < -1000)
213  continue;
214 
215  if (cur) {
216  scores[i][cur] += vp[i];
217  int next = 0;
218  int step = 3 * _spatium;//constraints between adjacent barlines
219  if (step + i <= x2 - x1) {
220  if (scores[step + i][next] < scores[i][cur]) {
221  scores[step + i][next] = scores[i][cur];
222  bs.x = x; bs.status = cur;
223  pred[step + i][next] = bs;
224  }
225  }
226  }
227  else {
228  for (int next = 0; next <= 1; ++next) {
229  int step = 1;
230  if (step + i <= x2 - x1) {
231  if (scores[step + i][next] < scores[i][cur]) {
232  scores[step + i][next] = scores[i][cur];
233  bs.x = x; bs.status = cur;
234  pred[step + i][next] = bs;
235  }
236  }
237  }
238  }
239  }
240  }
241  return(scores[x2][0]);
242  }

323  }
324  else {

334  }
335 
336  //forward pass
337  if (scores[cur_staff][status] + cur_score > scores[next_staff][1 - status]) {
338  scores[next_staff][1 - status] = scores[cur_staff][status] + cur_score;
339  ss.status = status; ss.index = cur_staff;
340  pred[next_staff][1 - status] = ss;
341  }
342  }
343 
344  status = 1;//end_staff
345  next_staff = cur_staff + 1;
346  if (next_staff < numStaves) {
347  //forward pass
348  if (scores[cur_staff][status] > scores[next_staff][1 - status]) {
349  scores[next_staff][1 - status] = scores[cur_staff][status];
350  ss.status = status; ss.index = cur_staff;

352  }
353  }
354  }

361

363  //add system here
364  OmrSystem omrSystem(this);
365  for (int i = ss.index; i <= cur_staff; ++i) {
366  omrSystem.staves().append(staves[i]);
367  }
368  omrSystem.searchBarLinesvar(cur_staff - ss.index + 1, note_labels + ss.index);
369  _systems.append(omrSystem);
370  }
371  cur_staff = ss.index;
372  status = ss.status;
373  }
374 
375  int systems = _systems.size();
376  for (int i = 0; i < systems; ++i) {
377  OmrSystem* system = &_systems[i];
378  int n = system->barLines.size();
379  for (int k = 0; k < n - 1; ++k) {
380  const QLine& l1 = system->barLines[k];
381  const QLine& l2 = system->barLines[k + 1];
382  OmrMeasure m(l1.x1(), l2.x1());
383  system->measures().append(m);
384  }
385  }
386 
387  //delete allocated space
388  for (int i = 0; i < numStaves; i++) {
389  delete[] scores[i];
390  delete[] pred[i];
391  delete[] temp_scores[i];
392  delete[] hashed[i];
393  delete[] note_labels[i];
394  }
395 
396  delete[] note_labels;
397  delete[] scores;
398  delete[] pred;
399  delete[] temp_scores;
400  delete[] hashed;
401  }
