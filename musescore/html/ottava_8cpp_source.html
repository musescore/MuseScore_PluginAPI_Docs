180  Sid::ottava22MAPlacement,
181  Sid::ottava22MAText,
182  Sid::ottava22MAContinueText,
183  Sid::ottava22MBPlacement,
184  Sid::ottava22MBText,
185  Sid::ottava22MBContinueText,
186  };
187 
188  size_t idx = size_t(_ottavaType) * 3 + (_numbersOnly ? 0 : ss.size()/2);
189  switch (pid) {
190  case Pid::OFFSET:
191  return placeAbove() ? Sid::ottavaPosAbove : Sid::ottavaPosBelow;
192  case Pid::PLACEMENT:
193  return ss[idx];
194  case Pid::BEGIN_TEXT:
195  return ss[idx+1]; // BEGIN_TEXT
196  case Pid::CONTINUE_TEXT:
197  return ss[idx+2]; // CONTINUE_TEXT
198  case Pid::END_HOOK_HEIGHT:
199  if (isStyled(Pid::PLACEMENT))
200  return score()->styleI(ss[idx]) == int(Placement::ABOVE) ? Sid::ottavaHookAbove : Sid::ottavaHookBelow;
201  else
202  return placeAbove() ? Sid::ottavaHookAbove : Sid::ottavaHookBelow;
203  default:
204  return TextLineBase::getPropertyStyle(pid);
205  }

282 void Ottava::read(XmlReader& e)
283  {
284  eraseSpannerSegments();
285  if (score()->mscVersion() < 301)

290  styleChanged();

297 bool Ottava::readProperties(XmlReader& e)
298  {
299  const QStringRef& tag(e.name());
300  if (tag == "subtype") {
301  QString s = e.readElementText();
302  bool ok;
303  int idx = s.toInt(&ok);
304  if (!ok) {
305  _ottavaType = OttavaType::OTTAVA_8VA;
306  for (OttavaDefault d : ottavaDefault) {
307  if (s == d.name) {
308  _ottavaType = d.type;
309  break;
310  }
311  }
312  }
313  else if (score()->mscVersion() <= 114) {
314  //subtype are now in a different order...
315  if (idx == 1)
316  idx = 2;
317  else if (idx == 2)
318  idx = 1;
319  _ottavaType = OttavaType(idx);
320  }
321  else
322  _ottavaType = OttavaType(idx);
323  }
324  else if (readStyledProperty(e, tag))
325  return true;
326  else if (!TextLineBase::readProperties(e)) {
327  e.unknown();
328  return false;
329  }
330  return true;
331  }
