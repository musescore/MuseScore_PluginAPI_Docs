147  {
148  filterActive = false;
149  setMouseTracking(true);
150  QString t = text.toLower();
151  bool res = true;
152  dragCells.clear();
153  // if palette name is searched for, display all elements in the palette
154  if (_name.startsWith(t, Qt::CaseInsensitive)) {
155  PaletteCell* c = cells.first();
156  for (PaletteCell* cell : cells)
157  dragCells.append(cell);
158 
159  bool contains = t.isEmpty() || c;
160  if (!contains)
161  filterActive = true;
162  if (contains && res)
163  res = false;
164  }
165 
166  for (PaletteCell* cell : cells) {
167  QStringList h = cell->name.toLower().split(" ");
168  bool c = false;
169  QStringList n = t.split(" ");
170  for (QString hs : h) {
171  for (QString ns : n) {
172  if (!ns.trimmed().isEmpty())
173  c = hs.trimmed().startsWith(ns.trimmed());
174  }
175  if (c)
176  break;
177  }
178  if (t.isEmpty() || c)
179  dragCells.append(cell);
180  bool contains = t.isEmpty() || c;
181  if (!contains)
182  filterActive = true;
183  if (contains && res)
184  res = false;
185  }
186  setFixedHeight(heightForWidth(width()));
187  updateGeometry();
188  return res;

498 bool Palette::applyPaletteElement(Element* element, Qt::KeyboardModifiers modifiers)
499  {
500  Score* score = mscore->currentScore();
501  if (score == 0)
502  return false;
503  const Selection sel = score->selection(); // make a copy of selection state before applying the operation.
504  if (sel.isNone())
505  return false;
506 
507 // Element* element = 0;
508 // if (cell)
509 // element = cell->element.get();
510  if (element == 0)
511  return false;
512 
513  if (element->isSpanner())
514  TourHandler::startTour("spanner-drop-apply");
515 
516 #ifdef MSCORE_UNSTABLE
517  if (ScriptRecorder* rec = mscore->getScriptRecorder()) {
518  if (modifiers == 0)
519  rec->recordPaletteElement(element);
520  }
521 #endif
522 
523  ScoreView* viewer = mscore->currentScoreView();
524 
525  // exit edit mode, to allow for palette element to be applied properly
526  if (viewer && viewer->editMode() && !(viewer->mscoreState() & STATE_ALLTEXTUAL_EDIT))
527  viewer->changeState(ViewState::NORMAL);
528 
529  if (viewer->mscoreState() != STATE_EDIT
530  && viewer->mscoreState() != STATE_LYRICS_EDIT
531  && viewer->mscoreState() != STATE_HARMONY_FIGBASS_EDIT) { // Already in startCmd in this case
532  score->startCmd();
533  }
534  if (sel.isList()) {
535  ChordRest* cr1 = sel.firstChordRest();
536  ChordRest* cr2 = sel.lastChordRest();
537  bool addSingle = false; // add a single line only
538  if (cr1 && cr2 == cr1) {
539  // one chordrest selected, ok to add line
540  addSingle = true;
541  }
542  else if (sel.elements().size() == 2 && cr1 && cr2 && cr1 != cr2) {
543  // two chordrests selected
544  // must be on same staff in order to add line, except for slur
545  if (element->isSlur() || cr1->staffIdx() == cr2->staffIdx())
546  addSingle = true;
547  }
548  if (viewer->mscoreState() == STATE_NOTE_ENTRY_STAFF_DRUM && element->isChord()) {
549  InputState& is = score->inputState();
550  Element* e = nullptr;
551  if (!(modifiers & Qt::ShiftModifier)) {
552  // shift+double-click: add note to "chord"
553  // use input position rather than selection if possible
554  // look for a cr in the voice predefined for the drum in the palette
555  // back up if necessary
556  // TODO: refactor this with similar code in putNote()
557  if (is.segment()) {
558  Segment* seg = is.segment();
559  while (seg) {
560  if (seg->element(is.track()))
561  break;
562  seg = seg->prev(SegmentType::ChordRest);
563  }
564  if (seg)
565  is.setSegment(seg);
566  else
567  is.setSegment(is.segment()->measure()->first(SegmentType::ChordRest));
568  }
569  score->expandVoice();
570  e = is.cr();
571  }
572  if (!e)
573  e = sel.elements().first();
574  if (e) {
575  // get note if selection was full chord
576  if (e->isChord())
577  e = toChord(e)->upNote();
578  applyDrop(score, viewer, e, element, modifiers, QPointF(), true);
579  // note has already been played (and what would play otherwise may be *next* input position)
580  score->setPlayNote(false);
581  score->setPlayChord(false);
582  // continue in same track
583  is.setTrack(e->track());
584  }
585  else
586  qDebug("nowhere to place drum note");
587  }
588  else if (element->isLayoutBreak()) {
589  LayoutBreak* breakElement = toLayoutBreak(element);
590  score->cmdToggleLayoutBreak(breakElement->layoutBreakType());
591  }
592  else if (element->isSlur() && addSingle) {
593  viewer->cmdAddSlur(toSlur(element));
594  }
595  else if (element->isSLine() && !element->isGlissando() && addSingle) {
596  Segment* startSegment = cr1->segment();
597  Segment* endSegment = cr2->segment();
598  if (element->type() == ElementType::PEDAL && cr2 != cr1)
599  endSegment = endSegment->nextCR(cr2->track());
600  // TODO - handle cross-voice selections
601  int idx = cr1->staffIdx();
602 
603  QByteArray a = element->mimeData(QPointF());
604 //printf("<<%s>>\n", a.data());
605  XmlReader e(a);
606  Fraction duration; // dummy
607  QPointF dragOffset;
608  ElementType type = Element::readType(e, &dragOffset, &duration);
609  Spanner* spanner = static_cast<Spanner*>(Element::create(type, score));
610  spanner->read(e);
611  spanner->styleChanged();
612  score->cmdAddSpanner(spanner, idx, startSegment, endSegment);
613  }
614  else {
615  for (Element* e : sel.elements())
616  applyDrop(score, viewer, e, element, modifiers);
617  }
618  }
619  else if (sel.isRange()) {
620  if (element->type() == ElementType::BAR_LINE
621  || element->type() == ElementType::MARKER
622  || element->type() == ElementType::JUMP
623  || element->type() == ElementType::SPACER
624  || element->type() == ElementType::VBOX
625  || element->type() == ElementType::HBOX
626  || element->type() == ElementType::TBOX
627  || element->type() == ElementType::MEASURE
628  || element->type() == ElementType::BRACKET
629  || element->type() == ElementType::STAFFTYPE_CHANGE
630  || (element->type() == ElementType::ICON
631  && (toIcon(element)->iconType() == IconType::VFRAME
632  || toIcon(element)->iconType() == IconType::HFRAME
633  || toIcon(element)->iconType() == IconType::TFRAME
634  || toIcon(element)->iconType() == IconType::MEASURE
635  || toIcon(element)->iconType() == IconType::BRACKETS))) {
636  Measure* last = sel.endSegment() ? sel.endSegment()->measure() : nullptr;
637  for (Measure* m = sel.startSegment()->measure(); m; m = m->nextMeasureMM()) {
638  QRectF r = m->staffabbox(sel.staffStart());
639  QPointF pt(r.x() + r.width() * .5, r.y() + r.height() * .5);
640  pt += m->system()->page()->pos();
641  applyDrop(score, viewer, m, element, modifiers, pt);
642  if (m == last)
643  break;
644  }
645  }
646  else if (element->type() == ElementType::LAYOUT_BREAK) {
647  LayoutBreak* breakElement = static_cast<LayoutBreak*>(element);
648  score->cmdToggleLayoutBreak(breakElement->layoutBreakType());
649  }
650  else if (element->isClef() || element->isKeySig() || element->isTimeSig()) {
651  Measure* m1 = sel.startSegment()->measure();
652  Measure* m2 = sel.endSegment() ? sel.endSegment()->measure() : nullptr;
653  if (m2 == m1 && sel.startSegment()->rtick().isZero())
654  m2 = nullptr; // don't restore original if one full measure selected
655  else if (m2)
656  m2 = m2->nextMeasureMM();
657  // for clefs, apply to each staff separately
658  // otherwise just apply to top staff
659  int staffIdx1 = sel.staffStart();
660  int staffIdx2 = element->type() == ElementType::CLEF ? sel.staffEnd() : staffIdx1 + 1;
661  for (int i = staffIdx1; i < staffIdx2; ++i) {
662  // for clefs, use mid-measure changes if appropriate
663  Element* e1 = nullptr;
664  Element* e2 = nullptr;
665  // use mid-measure clef changes as appropriate
666  if (element->type() == ElementType::CLEF) {
667  if (sel.startSegment()->isChordRestType() && sel.startSegment()->rtick().isNotZero()) {
668  ChordRest* cr = static_cast<ChordRest*>(sel.startSegment()->nextChordRest(i * VOICES));
669  if (cr && cr->isChord())
670  e1 = static_cast<Chord*>(cr)->upNote();
671  else
672  e1 = cr;
673  }
674  if (sel.endSegment() && sel.endSegment()->segmentType() == SegmentType::ChordRest) {
675  ChordRest* cr = static_cast<ChordRest*>(sel.endSegment()->nextChordRest(i * VOICES));
676  if (cr && cr->isChord())
677  e2 = static_cast<Chord*>(cr)->upNote();
678  else
679  e2 = cr;
680  }
681  }
682  if (m2 || e2) {
683  // restore original clef/keysig/timesig
684  Staff* staff = score->staff(i);
685  Fraction tick1 = sel.startSegment()->tick();
686  Element* oelement = nullptr;
687  switch (element->type()) {
688  case ElementType::CLEF:
689  {
690  Clef* oclef = new Clef(score);
691  oclef->setClefType(staff->clef(tick1));
692  oelement = oclef;
693  break;
694  }
695  case ElementType::KEYSIG:
696  {
697  KeySig* okeysig = new KeySig(score);
698  okeysig->setKeySigEvent(staff->keySigEvent(tick1));
699  if (!score->styleB(Sid::concertPitch) && !okeysig->isCustom() && !okeysig->isAtonal()) {
700  Interval v = staff->part()->instrument(tick1)->transpose();
701  if (!v.isZero()) {
702  Key k = okeysig->key();
703  okeysig->setKey(transposeKey(k, v, staff->part()->preferSharpFlat()));
704  }
705  }
706  oelement = okeysig;
707  break;
708  }
709  case ElementType::TIMESIG:
710  {
711  TimeSig* otimesig = new TimeSig(score);
712  otimesig->setFrom(staff->timeSig(tick1));
713  oelement = otimesig;
714  break;
715  }
716  default:
717  break;
718  }
719  if (oelement) {
720  if (e2) {
721  applyDrop(score, viewer, e2, oelement, modifiers);
722  }
723  else {
724  QRectF r = m2->staffabbox(i);
725  QPointF pt(r.x() + r.width() * .5, r.y() + r.height() * .5);
726  pt += m2->system()->page()->pos();
727  applyDrop(score, viewer, m2, oelement, modifiers, pt);
728  }
729  delete oelement;
730  }
731  }
732  // apply new clef/keysig/timesig
733  if (e1) {
734  applyDrop(score, viewer, e1, element, modifiers);
735  }
736  else {
737  QRectF r = m1->staffabbox(i);
738  QPointF pt(r.x() + r.width() * .5, r.y() + r.height() * .5);
739  pt += m1->system()->page()->pos();
740  applyDrop(score, viewer, m1, element, modifiers, pt);
741  }
742  }
743  }
744  else if (element->isSlur()) {
745  viewer->cmdAddSlur(toSlur(element));
746  }
747  else if (element->isSLine() && element->type() != ElementType::GLISSANDO) {
748  Segment* startSegment = sel.startSegment();
749  Segment* endSegment = sel.endSegment();
750  bool firstStaffOnly = (element->isVolta() || (element->isTextLine() && toTextLine(element)->systemFlag())) && !(modifiers & Qt::ControlModifier);
751  int startStaff = firstStaffOnly ? 0 : sel.staffStart();
752  int endStaff = firstStaffOnly ? 1 : sel.staffEnd();
753  for (int i = startStaff; i < endStaff; ++i) {
754  Spanner* spanner = static_cast<Spanner*>(element->clone());
755  spanner->setScore(score);
756  spanner->styleChanged();
757  score->cmdAddSpanner(spanner, i, startSegment, endSegment);
758  }
759  }
760  else {
761  int track1 = sel.staffStart() * VOICES;
762  int track2 = sel.staffEnd() * VOICES;
763  Segment* startSegment = sel.startSegment();
764  Segment* endSegment = sel.endSegment(); //keep it, it could change during the loop
765 
766  for (Segment* s = startSegment; s && s != endSegment; s = s->next1()) {
767  for (int track = track1; track < track2; ++track) {
768  Element* e = s->element(track);
769  if (e == 0 || !score->selectionFilter().canSelect(e) || !score->selectionFilter().canSelectVoice(track))
770  continue;
771  if (e->isChord()) {
772  Chord* chord = toChord(e);
773  for (Note* n : chord->notes()) {
774  applyDrop(score, viewer, n, element, modifiers);
775  if (!(element->isAccidental() || element->isNoteHead())) // only these need to apply to every note
776  break;
777  }
778  }
779  else {
780  // do not apply articulation to barline in a range selection
781  if (!e->isBarLine() || !element->isArticulation())
782  applyDrop(score, viewer, e, element, modifiers);
783  }
784  }
785  if (!element->placeMultiple())
786  break;
787  }
788  }
789  }
790  else
791  qDebug("unknown selection state");
792 
793  if (viewer->mscoreState() != STATE_EDIT
794  && viewer->mscoreState() != STATE_LYRICS_EDIT
795  && viewer->mscoreState() != STATE_HARMONY_FIGBASS_EDIT) { //Already in startCmd mode in this case
796  score->endCmd();
797  if (viewer->mscoreState() == STATE_NOTE_ENTRY_STAFF_DRUM)
798  viewer->moveCursor();
799  }
800  else if (viewer->mscoreState() & STATE_ALLTEXTUAL_EDIT)
801  viewer->setFocus();
802  viewer->setDropTarget(0);
803 // mscore->endCmd();
804  return true;
805  }
