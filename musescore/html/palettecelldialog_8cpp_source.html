113 void PaletteCellPropertiesDialog::reject()
114  {
115  //if nothing is changed, just close the palette properties dialog
116  if (areInitialPropertiesChanged()) {
117  applyInitialPropertiesToThePalette();
118  QDialog::reject();
119  return;
120  }
121 
122  QMessageBox msgBox;
123  msgBox.setText(tr("The palette cell properties have been modified."));
124  msgBox.setInformativeText(tr("Do you want to save your changes?"));
125  msgBox.setStandardButtons(QMessageBox::Save | QMessageBox::Discard | QMessageBox::Cancel);
126  msgBox.setDefaultButton(QMessageBox::Discard);
127  msgBox.setIcon(QMessageBox::Question);
128  int ret = msgBox.exec();
129  switch (ret) {
130  case QMessageBox::Save:
131  QDialog::accept();
132  break;
133  case QMessageBox::Discard:
134  applyInitialPropertiesToThePalette();
135  QDialog::reject();
136  break;
137  case QMessageBox::Cancel:
138  break;
139  default:
140  Q_ASSERT(false && "Should never be reached");
141  }
142  }
