136 void PalettePropertiesDialog::reject()
137  {
138  //if nothing is changed, just close the palette properties dialog
139  if (areInitialPropertiesChanged()) {
140  applyInitialPropertiesToThePalette();
141  QDialog::reject();
142  return;
143  }
144 
145  QMessageBox msgBox;
146  msgBox.setText(tr("The palette properties have been modified."));
147  msgBox.setInformativeText(tr("Do you want to save your changes?"));
148  msgBox.setStandardButtons(QMessageBox::Save | QMessageBox::Discard | QMessageBox::Cancel);
149  msgBox.setDefaultButton(QMessageBox::Discard);
150  msgBox.setIcon(QMessageBox::Question);
151  int ret = msgBox.exec();
152  switch (ret) {
153  case QMessageBox::Save:
154  QDialog::accept();
155  break;
156  case QMessageBox::Discard:
157  applyInitialPropertiesToThePalette();
158  QDialog::reject();
159  break;
160  case QMessageBox::Cancel:
161  break;
162  default:
163  Q_ASSERT(false && "Should never be reached");
164  }
165  }
