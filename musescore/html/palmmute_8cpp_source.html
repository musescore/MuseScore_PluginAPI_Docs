204 QPointF PalmMute::linePos(Grip grip, System** sys) const
205  {
206  qreal x = 0.0;
207  qreal nhw = score()->noteHeadWidth();
208  System* s = nullptr;
209  if (grip == Grip::START) {
210  ChordRest* c = toChordRest(startElement());
211  if (!c)
212  return QPointF();
213  s = c->segment()->system();
214  x = c->pos().x() + c->segment()->pos().x() + c->segment()->measure()->pos().x();
215  if (c->isRest() && c->durationType() == TDuration::DurationType::V_MEASURE)
216  x -= c->x();
217  }
218  else {
219  Element* e = endElement();
220  ChordRest* c = toChordRest(endElement());
221  if (!e || e == startElement() || (endHookType() == HookType::HOOK_90)) {
222  // palmMute marking on single note or ends with non-angled hook:
223  // extend to next note or end of measure
224  Segment* seg = nullptr;
225  if (!e)
226  seg = startSegment();
227  else
228  seg = c->segment();
229  if (seg) {
230  seg = seg->next();
231  for ( ; seg; seg = seg->next()) {
232  if (seg->isChordRestType()) {
233  // look for a chord/rest in any voice on this staff
234  bool crFound = false;
235  int track = staffIdx() * VOICES;
236  for (int i = 0; i < VOICES; ++i) {
237  if (seg->element(track + i)) {
238  crFound = true;
239  break;
240  }
241  }
242  if (crFound)
243  break;
244  }
245  else if (seg->segmentType() == SegmentType::EndBarLine) {
246  break;
247  }
248  }
249  }
250  if (seg) {
251  s = seg->system();
252  x = seg->pos().x() + seg->measure()->pos().x() - nhw * 2;
253  }
254  }
255  else if (c) {
256  s = c->segment()->system();
257  x = c->pos().x() + c->segment()->pos().x() + c->segment()->measure()->pos().x();
258  if (c->type() == ElementType::REST && c->durationType() == TDuration::DurationType::V_MEASURE)
259  x -= c->x();
260  }
261  if (!s) {
262  Fraction t = tick2();
263  Measure* m = score()->tick2measure(t);
264  s = m->system();
265  x = m->tick2pos(t);
266  }
267  if (endHookType() == HookType::HOOK_45)
268  x += nhw * .5;
269  else
270  x += nhw;
271  }
272 
273  *sys = s;
274  return QPointF(x, 0);
275  }
