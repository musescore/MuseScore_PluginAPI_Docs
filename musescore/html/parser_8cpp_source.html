70 static void dumpBeams(QList<Bww::MeasureDescription> const& measures)
71 {
72  for (int j = 0; j < measures.size(); ++j)
73  {
74  QString beams;
75  QString beamStates;
76  QVector<QString> beamList(3);
77  for (int i = 0; i < measures.at(j).notes.size(); ++i)
78  {
79  QString beam = measures.at(j).notes.at(i).beam;
80  if (beam == "")
81  beam = " ";
82 
83  beams += beam;
84  switch (measures.at(j).notes.at(i).beamState)
85  {
86  case Bww::ST_NONE: beamStates += " "; break;
87  case Bww::ST_START: beamStates += "["; break;
88  case Bww::ST_CONTINUE: beamStates += "_"; break;
89  case Bww::ST_STOP: beamStates += "]"; break;
90  default: beamStates += " ";
91  }
92  for (int k = 0; k < 3; k++)
93  switch (measures.at(j).notes.at(i).beamList.at(k))
94  {
95  case Bww::BM_NONE: beamList[k] += " "; break;
96  case Bww::BM_BEGIN: beamList[k] += "b"; break;
97  case Bww::BM_CONTINUE: beamList[k] += "c"; break;
98  case Bww::BM_END: beamList[k] += "e"; break;
99  case Bww::BM_FORWARD_HOOK: beamList[k] += ">"; break;
100  case Bww::BM_BACKWARD_HOOK: beamList[k] += "<"; break;
101  default: beamList[k] += "?";
102  }
103  }
104  qDebug() << "beams measure #" << j + 1 << beams;
105  qDebug() << "beams measure #" << j + 1 << beamStates;
106  for (int k = 0; k < 3; k++)
107  qDebug() << "beams measure #" << j + 1 << beamList.at(k);
108  }
109 }

156 
157 static void calculateMeasureDurations(QList<Bww::MeasureDescription> & measures)
158 {
159  for (int j = 0; j < measures.size(); ++j)
160  {
161  int measureDuration = 0;
162  for (int i = 0; i < measures.at(j).notes.size(); ++i)
163  {
164  int ticks = WHOLE_MEASURE_DURATION / measures.at(j).notes.at(i).type.toInt();
165  if (measures.at(j).notes.at(i).dots) ticks = 3 * ticks / 2;
166  if (measures.at(j).notes.at(i).triplet != Bww::ST_NONE) ticks = 2 * ticks / 3;
167  if (measures.at(j).notes.at(i).grace) ticks = 0; // grace notes don't count
168  measureDuration += ticks;
169  qDebug()
170  << measures.at(j).notes.at(i).pitch
171  << measures.at(j).notes.at(i).beam
172  << measures.at(j).notes.at(i).type
173  << measures.at(j).notes.at(i).dots
174  << measures.at(j).notes.at(i).tieStart
175  << measures.at(j).notes.at(i).tieStop
176  << measures.at(j).notes.at(i).triplet
177  << measures.at(j).notes.at(i).grace
178  << "->" << ticks
179  ;
180  }
181  qDebug() << "measureDuration:" << measureDuration;
182  measures[j].duration = measureDuration;
183  }
184 }

190 static void determineTimesig(QList<Bww::MeasureDescription> const& measures, int & beats, int & beat)
191 {
192  QMap<int, int> map;
193  for (int j = 0; j < measures.size(); ++j)
194  {
195  int dur = measures[j].duration;
196  if (map.contains(dur))
197  map[dur]++;
198  else
199  map.insert(dur, 1);
200  }
201  // determine most common duration
202  int commonDur = 0;
203  int max = 0;
204  QMap<int, int>::const_iterator i = map.constBegin();
205  while (i != map.constEnd())
206  {
207  qDebug() << "measureDurations:" << i.key() << i.value();
208  if (i.value() > max)
209  {
210  commonDur = i.key();
211  max = i.value();
212  }
213  ++i;
214  }
215  qDebug() << "measureDuration commonDur:" << commonDur << "max:" << max;
216  // determine time signature
217  beat = 4;
218  beats = 0;
219  int divisor = WHOLE_MEASURE_DURATION / 4;
220  for (; beat < 64; beat *= 2, divisor /= 2)
221  {
222  if ((commonDur % divisor) == 0)
223  {
224  beats = commonDur / divisor;
225  qDebug()
226  << "measureDuration found beat:" << beat
227  << "beats:" << beats
228  << "divisor:" << divisor
229  ;
230  return;
231  }
232  }
233  // could not determine time signature, set default
234  beat = 4;
235  beats = 4;
236 }

238 /**
239  Find irregular measures
240  */
241 
242 static void findIrregularMeasures(QList<Bww::MeasureDescription> & measures, int beats, int beat)
243 {
244  qDebug() << "findIrregularMeasures" << measures.size()
245  << "beats" << beats
246  << "beat" << beat
247  ;
248 
249  int normalDuration = WHOLE_MEASURE_DURATION * beats / beat;
250 
251  // need at least one measure
252  if (measures.size() == 0) return;
253 
254  // if the first measure is shorter that normal, it is irregular
255  if (measures.at(0).duration < normalDuration) measures[0].mbf.irregular = true;
256 
257  for (int j = 1; j < measures.size(); ++j)
258  {
259  // the second measure of a pair where the sum of their duration adds up
260  // to the normal duration is also irregular
261  const int d1 = measures.at(j - 1).duration;
262  const int d2 = measures.at(j).duration;
263  if (d1 > 0 && d2 > 0 && (d1 + d2) == normalDuration)
264  measures[j].mbf.irregular = true;
265  }
266 }

305 static void calculateHigherBeamStates(Bww::MeasureDescription & m)
306 {
307  qDebug() << "calculateHigherBeamStates";
308  for (int i = 0; i < m.notes.size(); ++i)
309  {
310  qDebug()
311  << m.notes.at(i).pitch
312  << m.notes.at(i).beam
313  << m.notes.at(i).type
314  << m.notes.at(i).dots
315  << m.notes.at(i).tieStart
316  << m.notes.at(i).tieStop
317  << m.notes.at(i).triplet
318  << m.notes.at(i).grace
319  ;
320 
321  if (m.notes.at(i).grace)
322  {
323  // just copy beamList[0] into beamList[1] and beamList[2]
324  m.notes[i].beamList[1] = m.notes.at(i).beamList[0];
325  m.notes[i].beamList[2] = m.notes.at(i).beamList[0];
326  }
327  else
328  {
329  int blp = -1; // beam level previous chord
330  int blc = -1; // beam level current chord
331  int bln = -1; // beam level next chord
332 
333  // find beam level current note
334  blc = type2beams(m.notes.at(i).type);
335  if (blc == 0)
336  continue; // note does not have a beam
337 
338  // find beam level previous note
339  if (m.notes.at(i).beamList[0] == Bww::BM_CONTINUE
340  || m.notes.at(i).beamList[0] == Bww::BM_END)
341  {
342  for (int j = i - 1; blp == -1 && j >= 0; --j)
343  {
344  if (m.notes.at(j).grace)
345  continue; // ignore grace notes
346  blp = type2beams(m.notes.at(j).type);
347  }
348  }
349 
350  // find beam level next note
351  if (m.notes.at(i).beamList[0] == Bww::BM_BEGIN
352  || m.notes.at(i).beamList[0] == Bww::BM_CONTINUE)
353  {
354  for (int j = i + 1; bln == -1 && j < m.notes.size(); ++j)
355  {
356  if (m.notes.at(j).grace)
357  continue; // ignore grace notes
358  bln = type2beams(m.notes.at(j).type);
359  }
360  }
361 
362  qDebug()
363  << "blp" << blp
364  << "blc" << blc
365  << "bln" << bln;
366  for (int j = 2; j <= blc; ++j)
367  {
368  Bww::BeamType bt = Bww::BM_NONE;
369  if (blp < j && bln >= j) bt = Bww::BM_BEGIN;
370  else if (blp < j && bln < j) {
371  if (bln > 0) bt = Bww::BM_FORWARD_HOOK;
372  else if (blp > 0) bt = Bww::BM_BACKWARD_HOOK;
373  }
374  else if (blp >= j && bln < j) bt = Bww::BM_END;
375  else if (blp >= j && bln >= j) bt = Bww::BM_CONTINUE;
376  m.notes[i].beamList[j - 1] = bt;
377  qDebug() << "beamList" << j - 1 << "=" << bt;
378  }
379  } // else
380  } // for (int i = 0; i < m.notes.size(); ++i)
381 }
382 
383 /**
384  Determine all beam states.
385  First for normal notes convert "r" and "l" in notes.beam into BM_BEGIN,
386  BM_CONTINUE and BM_END in notes.beamList[0]. For grace notes "b", "c" and
387  "e" can be directly converted into the corresponding beam states in
388  notes.beamList[0].
389  Then calculate the higher level beams.
390  */
391 
392 static void determineBeamStates(QList<Bww::MeasureDescription> & measures)
393 {
394  enum State { NONE, LEFT, RIGHT };
395  for (int j = 0; j < measures.size(); ++j)
396  {
397  State state = NONE;
398  for (int i = 0; i < measures.at(j).notes.size(); ++i)
399  {
400  QString beam = measures.at(j).notes.at(i).beam;
401  // handle normal notes
402  if (beam == "")
403  {
404  measures[j].notes[i].beamState = Bww::ST_NONE;
405  measures[j].notes[i].beamList[0] = Bww::BM_NONE;
406  state = NONE;
407  }
408  else if (beam == "r")
409  {
410  if (state == NONE)
411  {
412  measures[j].notes[i].beamState = Bww::ST_START;
413  measures[j].notes[i].beamList[0] = Bww::BM_BEGIN;
414  state = LEFT; // now in left part of beam
415  }
416  else if (state == LEFT)
417  {
418  measures[j].notes[i].beamState = Bww::ST_CONTINUE;
419  measures[j].notes[i].beamList[0] = Bww::BM_CONTINUE;
420  }
421  else if (state == RIGHT)
422  {
423  // shouldn't happen TODO report (internal?) error
424  }
425  }
426  else if (beam == "l")
427  {
428  if (state == NONE)
429  {
430  // shouldn't happen TODO report error
431  }
432  else if (state == LEFT || state == RIGHT)
433  {
434  // if the beam does not end here (next note has beam "l")
435  // then beamState is CONTINUE else STOP
436  if (findNextNextNoteBeam(measures, j, i) == "l")
437  {
438  measures[j].notes[i].beamState = Bww::ST_CONTINUE;
439  measures[j].notes[i].beamList[0] = Bww::BM_CONTINUE;
440  state = RIGHT; // now in right part of beam
441  }
442  else
443  {
444  measures[j].notes[i].beamState = Bww::ST_STOP;
445  measures[j].notes[i].beamList[0] = Bww::BM_END;
446  state = NONE; // now in right part of beam
447  }
448  }
449  }
450  // handle grace notes
451  else if (beam == "b") measures[j].notes[i].beamList[0] = Bww::BM_BEGIN;
452  else if (beam == "c") measures[j].notes[i].beamList[0] = Bww::BM_CONTINUE;
453  else if (beam == "e") measures[j].notes[i].beamList[0] = Bww::BM_END;
454  }
455  calculateHigherBeamStates(measures[j]);
456  }
457 }

482  /**
483  Parse the input stream and write result.
484  */
485 
486  void Parser::parse()
487  {
488  // read the header, handling only strings and tune tempo
489  while (lex.symType() == COMMENT
490  || lex.symType() == STRING
491  || lex.symType() == TEMPO)
492  {
493  if (lex.symType() == STRING)
494  parseString();
495  else if (lex.symType() == TEMPO)
496  parseTempo();
497  else if (lex.symType() == COMMENT)
498  lex.getSym();
499  }
500  qDebug() << "Parser::parse()"
501  << "title:" << title
502  << "type:" << type
503  << "composer:" << composer
504  << "footer:" << footer
505  ;
506  wrt.header(title, type, composer, footer, tempo);
507 
508  // read the actual music
509  if (lex.symType() != CLEF)
510  errorHandler("clef ('&') expected");
511  while (lex.symType() != NONE)
512  {
513  if (isNonNote(lex.symType()))
514  parseSeqNonNotes();
515  else if (isNote(lex.symType()))
516  parseSeqNotes();
517  else if (lex.symType() == UNKNOWN)
518  {
519  errorHandler("unknown symbol '" + lex.symValue() + "'");
520  lex.getSym();
521  }
522  else
523  {
524  ; // others not implemented yet: silently ignored
525  lex.getSym();
526  }
527  }
528 
529  qDebug() << "Parser::parse() finished, #measures" << measures.size()
530  ;
531 
532  calculateMeasureDurations(measures);
533  if (!tsigFound)
534  {
535  determineTimesig(measures, beats, beat);
536  wrt.tsig(beats, beat);
537  }
538  findIrregularMeasures(measures, beats, beat);
539  determineBeamStates(measures);
540  setLastOfPart(measures);
541  dumpMeasures(measures);
542  dumpBeams(measures);
543 
544  for (int j = 0; j < measures.size(); ++j)
545  {
546  wrt.beginMeasure(measures.at(j).mbf);
547  for (int i = 0; i < measures.at(j).notes.size(); ++i)
548  {
549  wrt.note(measures.at(j).notes.at(i).pitch,
550  measures.at(j).notes.at(i).beamList,
551  measures.at(j).notes.at(i).type,
552  measures.at(j).notes.at(i).dots,
553  measures.at(j).notes.at(i).tieStart,
554  measures.at(j).notes.at(i).tieStop,
555  measures.at(j).notes.at(i).triplet,
556  measures.at(j).notes.at(i).grace
557  );
558  }
559  wrt.endMeasure(measures.at(j).mef);
560  }
561 
562  // trailer
563  wrt.trailer();
564  }

592  /**
593  Parse a bww note.
594  */
595 
596  void Parser::parseNote()
597  {
598  qDebug() << "Parser::parseNote() value:" << qPrintable(lex.symValue());
599 
600  QRegExp rNotes("(LG|LA|[B-F]|HG|HA)([lr]?)_(1|2|4|8|16|32)");
601 
602  QStringList caps;
603  if (rNotes.exactMatch(lex.symValue()))
604  {
605  caps = rNotes.capturedTexts();
606  qDebug() << " match" << caps.size();
607  if (caps.size() == 4)
608  {
609  qDebug()
610  << "caps[1]" << caps.at(1)
611  << "caps[2]" << caps.at(2)
612  << "caps[3]" << caps.at(3)
613  ;
614  }
615  }
616  lex.getSym();
617 
618  int dots = 0;
619  bool tieStop = false;
620  if (tieStart) inTie = true;
621  bool tripletStop = false;
622  if (tripletStart) inTriplet = true;
623  if (lex.symType() == DOT)
624  {
625  qDebug() << " dot" << qPrintable(lex.symValue());
626  ++dots;
627  lex.getSym();
628  }
629  else if (lex.symType() == TIE)
630  {
631  qDebug() << " tie" << qPrintable(lex.symValue());
632  if (lex.symValue() == "^ts")
633  {
634  if (inTie) errorHandler("tie start ('^ts') unexpected");
635  }
636  else
637  {
638  if (!inTie) errorHandler("tie end ('^te') unexpected");
639  else
640  {
641  tieStop = true;
642  inTie = false;
643  lex.getSym();
644  }
645  }
646  }
647  else if (lex.symType() == TRIPLET)
648  {
649  qDebug() << " triplet" << qPrintable(lex.symValue());
650  if (lex.symValue() == "^3s")
651  {
652  if (inTriplet) errorHandler("triplet start ('^3s') unexpected");
653  }
654  else
655  {
656  if (!inTriplet) errorHandler("triplet end ('^3e') unexpected");
657  else
658  {
659  tripletStop = true;
660  }
661  }
662  lex.getSym();
663  }
664 
665  StartStop triplet = ST_NONE;
666  if (inTriplet)
667  {
668  if (tripletStart) triplet = ST_START;
669  else if (tripletStop) triplet = ST_STOP;
670  else triplet = ST_CONTINUE;
671  }
672  qDebug() << " tie start" << tieStart << " tie stop" << tieStop;
673  qDebug() << " triplet start" << tripletStart << " triplet stop" << tripletStop;
674  NoteDescription noteDesc(caps[1], caps[2], caps[3], dots, tieStart, tieStop, triplet);
675 
676  if (measures.isEmpty())
677  {
678  errorHandler("cannot append note: no measure");
679  }
680  else
681  {
682  measures.last().notes.append(noteDesc);
683  }
684  tieStart = false;
685  tripletStart = false;
686  if (tripletStop)
687  {
688  inTriplet = false;
689  tripletStop = false;
690  }
691  }

694  Determine beam for grace note \a index in a group of \a size grace notes.

696 
697  static QString graceBeam(const int size, const int index)
