81 void PartEdit::setPart(Part* p, Channel* a)
82  {
83  Channel dummy;
84  channel = a;
85  part = p;
86  QString s = part->partName();
87  if (!a->name().isEmpty()) {
88  if (a->name() != Channel::DEFAULT_NAME) {
89  s += "-";
90  s += qApp->translate("InstrumentsXML", a->name().toUtf8().data());
91  }
92  }
93  partName->setText(s);
94  _setValue(volume, a->volume());
95  volume->setDclickValue1(dummy.volume());
96  volume->setDclickValue2(dummy.volume());
97  _setValue(reverb, a->reverb());
98  reverb->setDclickValue1(dummy.reverb());
99  reverb->setDclickValue2(dummy.reverb());
100  _setValue(chorus, a->chorus());
101 
102  chorus->setDclickValue1(dummy.chorus());
103  chorus->setDclickValue2(dummy.chorus());
104  _setValue(pan, a->pan());
105 
106  pan->setDclickValue1(0);
107  pan->setDclickValue2(0);
108  for (int i = 0; i < patch->count(); ++i) {
109  MidiPatch* mp = (MidiPatch*)patch->itemData(i, Qt::UserRole).value<void*>();
110  if (a->synti() == mp->synti && a->program() == mp->prog && a->bank() == mp->bank) {
111  patch->setCurrentIndex(i);
112  break;
113  }
114  }
115  _setChecked(drumset, p->instrument()->useDrumset());
116  _setValue(portSpinBox, part->masterScore()->midiMapping(a->channel())->port() + 1);
117  _setValue(channelSpinBox, part->masterScore()->midiMapping(a->channel())->channel() + 1);
118 
119  QHBoxLayout* hb = voiceButtonBox;
120  int idx = 0;
121  for (Staff* staff : *part->staves()) {
122  for (int voice = 0; voice < VOICES; ++voice) {
123  if (!voiceButtons.value(idx)) {
124  QToolButton* tb = new QToolButton;
125  tb->setText(QString("%1").arg(voice+1));
126  tb->setCheckable(true);
127  tb->setChecked(staff->playbackVoice(voice));
128  tb->setFocusPolicy(Qt::ClickFocus);
129  tb->setToolButtonStyle(Qt::ToolButtonTextOnly);
130  QPalette pal(tb->palette());
131  pal.setColor(QPalette::Base, MScore::selectColor[voice]);
132  tb->setPalette(pal);
133  hb->addWidget(tb);
134  voiceButtons.append(tb);
135  connect(tb, SIGNAL(clicked()), SLOT(playbackVoiceChanged()));
136  }
137  ++idx;
138  }
139  hb->addStretch(5);
140  }
141  while (voiceButtons.value(idx)) {
142  QToolButton* tb = voiceButtons.value(idx);
143  if (!tb)
144  break;
145  voiceButtons.takeAt(idx);
146  delete tb;
147  }
148  }

154 void PartEdit::playbackVoiceChanged()

163  if (val != staff->playbackVoice(voice)) {
164  switch (voice) {
165  case 0:
166 //printf("undo\n");
167  staff->undoChangeProperty(Pid::PLAYBACK_VOICE1, val);
168  break;
169  case 1:
170  staff->undoChangeProperty(Pid::PLAYBACK_VOICE2, val);
171  break;
172  case 2:
173  staff->undoChangeProperty(Pid::PLAYBACK_VOICE3, val);
174  break;
175  case 3:
176  staff->undoChangeProperty(Pid::PLAYBACK_VOICE4, val);
177  break;
178  }

181  }

274 void PartEdit::soloToggled(bool val, bool syncControls)
275  {
276  channel->setSolo(val);
277  channel->setSoloMute(!val);
278  if (val) {
279  mute->setChecked(false);
280  for (Part* p : part->score()->parts()) {
281  const InstrumentList* il = p->instruments();
282  for (auto i = il->begin(); i != il->end(); ++i) {
283  const Instrument* instr = i->second;
284  for (Channel* instrChan : instr->channel()) {
285  Channel* a = part->masterScore()->playbackChannel(instrChan);
286  a->setSoloMute((channel != a && !a->solo()));
287  a->setSolo(channel == a || a->solo());
288  if (a->soloMute())
289  seq->stopNotes(a->channel());
290  }
291  }
292  }
293  emit soloChanged(true);
294  }
295  else { //do nothing except if it's the last solo to be switched off
296  bool found = false;
297  for (Part* p : part->score()->parts()) {
298  const InstrumentList* il = p->instruments();
299  for (auto i = il->begin(); i != il->end(); ++i) {
300  const Instrument* instr = i->second;
301  for (Channel* instrChan : instr->channel()) {
302  Channel* a = part->masterScore()->playbackChannel(instrChan);
303  if (a->solo()){
304  found = true;
305  break;
306  }
307  }
308  }
309  }
310  if (!found){
311  foreach(Part* p, part->score()->parts()) {
312  const InstrumentList* il = p->instruments();
313  for (auto i = il->begin(); i != il->end(); ++i) {
314  const Instrument* instr = i->second;
315  for (Channel* instrChan : instr->channel()) {
316  Channel* a = part->masterScore()->playbackChannel(instrChan);
317  a->setSoloMute(false);
318  a->setSolo(false);
319  }
320  }
321  }
322  emit soloChanged(false);
323  }
324  }
325  sync(syncControls);
326  }

365 void PartEdit::sync(bool syncControls)
366  {
367  if (!syncControls)
368  return;
369  int count = this->parentWidget()->layout()->count();
370  for(int i = 0; i < count; i++) {
371  QWidgetItem* wi = (QWidgetItem*)(this->parentWidget()->layout()->itemAt(i));
372  PartEdit* pe = (PartEdit*)(wi->widget());
373  if (pe != 0 && pe != this
374  && this->channelSpinBox->value() == pe->channelSpinBox->value()
375  && this->portSpinBox->value() == pe->portSpinBox->value()) {
376 
377  if (volume->value() != pe->volume->value()) {
378  _setValue(pe->volume, this->volume->value());
379  emit pe->volChanged(this->volume->value(), false);
380  }
381  if (pan->value() != pe->pan->value()) {
382  _setValue(pe->pan, this->pan->value());
383  emit pe->panChanged(this->pan->value(), false);
384  }
385  if (reverb->value() != pe->reverb->value()) {
386  _setValue(pe->reverb, this->reverb->value());
387  emit pe->reverbChanged(this->reverb->value(), false);
388  }
389  if (chorus->value() != pe->chorus->value()) {
390  _setValue(pe->chorus, this->chorus->value());
391  emit pe->chorusChanged(this->chorus->value(), false);
392  }
393  if (mute->isChecked() != pe->mute->isChecked()) {
394  _setChecked(pe->mute, channel->mute());
395  emit pe->muteChanged(channel->mute(), false);
396  }
397  if (solo->isChecked() != pe->solo->isChecked()) {
398  _setChecked(pe->solo, channel->solo());
399  emit pe->soloToggled(channel->solo(), false);
400  }
401 
402  if (drumset->isChecked() != pe->drumset->isChecked()) {
403  _setChecked(pe->drumset, drumset->isChecked());
404  emit pe->drumsetToggled(drumset->isChecked(), false);
405  }
406  if (patch->currentIndex() != pe->patch->currentIndex()) {
407  pe->patch->blockSignals(true);
408  pe->patch->setCurrentIndex(this->patch->currentIndex());
409  pe->patch->blockSignals(false);
410  }
411  }
412  }

420 void PartEdit::midiChannelChanged(int)
421  {
422  if (part == 0)
423  return;
424  seq->stopNotes(channel->channel());
425  int p = portSpinBox->value() - 1;
426  int c = channelSpinBox->value() - 1;
427 
428  // 1 is for going up, -1 for going down
429  int direction = copysign(1, c - part->masterScore()->midiMapping(channel->channel())->channel());
430 
431  // Channel 9 is special for drums
432  if (part->instrument()->useDrumset() && c != 9) {
433  _setValue(channelSpinBox, 10);
434  return;
435  }
436  else if (!part->instrument()->useDrumset() && c == 9) {
437  c = 9 + direction;
438  }
439 
440  if (c == 16) {
441  c = 0;
442  p++;
443  }
444 
445  int newChannel = p*16+c;
446 
447  // If there is an instrument with the same MIDI port and channel, sync this instrument to a found one
448  bool needSync = true;
449  int elementsInMixer = parentWidget()->layout()->count();
450  for (int i = 0; i < elementsInMixer; i++) {
451  QWidgetItem* wi = (QWidgetItem*)(this->parentWidget()->layout()->itemAt(i));
452  PartEdit* pe = (PartEdit*)(wi->widget());
453  if (pe != 0 && pe != this
454  && pe->channelSpinBox->value() == this->channelSpinBox->value()
455  && pe->portSpinBox->value() == this->portSpinBox->value()) {
456  // Show datails if parameters are different
457  QString detailedText;
458  if (patch->currentIndex() != pe->patch->currentIndex())
459  detailedText += QString(tr("Sound: '%1' vs. '%2'\n")).arg(patch->itemText(patch->currentIndex()), pe->patch->itemText(pe->patch->currentIndex()));
460  if (this->volume->value() != pe->volume->value())
461  detailedText += QString(tr("Volume: %1 vs. %2\n")).arg(QString::number(volume->value()), QString::number(pe->volume->value()));
462  if (this->pan->value() != pe->pan->value())
463  detailedText += QString(tr("Pan: %1 vs. %2\n")).arg(QString::number(pan->value()), QString::number(pe->pan->value()));
464  if (this->reverb->value() != pe->reverb->value())
465  detailedText += QString(tr("Reverb: %1 vs. %2\n")).arg(QString::number(reverb->value()), QString::number(pe->reverb->value()));
466  if (this->chorus->value() != pe->chorus->value())
467  detailedText += QString(tr("Chorus: %1 vs. %2\n")).arg(QString::number(chorus->value()), QString::number(pe->chorus->value()));
468  if (this->solo->isChecked() != pe->solo->isChecked())
469  detailedText += QString(tr("Solo: %1 vs. %2\n")).arg(solo->isChecked()?"Yes":"No", pe->solo->isChecked()?"Yes":"No");
470  if (this->mute->isChecked() != pe->mute->isChecked())
471  detailedText += QString(tr("Mute: %1 vs. %2\n")).arg(mute->isChecked()?"Yes":"No", pe->mute->isChecked()?"Yes":"No");
472 
473  if (!detailedText.isEmpty())
474  detailedText = QString(tr("Instrument '%1'\tInstrument '%2'\n")).arg(this->partName->text(), pe->partName->text())+ detailedText;
475  QMessageBox msgBox;
476  msgBox.setIcon(QMessageBox::Warning);
477  QString text = QString(tr("There is already an instrument '%1' with MIDI port = %2 and channel = %3.")).arg(pe->partName->text(), QString::number(pe->portSpinBox->value()), QString::number(pe->channelSpinBox->value()));
478  msgBox.setText(text);
479  msgBox.setInformativeText(tr("Do you want to synchronize the current instrument with an existing one?"));
480  msgBox.setDetailedText(detailedText);
481  msgBox.setStandardButtons(QMessageBox::Ok | QMessageBox::Cancel);
482  QPushButton *assignFreeChannel = msgBox.addButton(tr("Assign next free MIDI channel"), QMessageBox::HelpRole);
483  msgBox.setDefaultButton(QMessageBox::Ok);
484  if (msgBox.exec() == QMessageBox::Cancel) {
485  _setValue(channelSpinBox, part->masterScore()->midiMapping(channel->channel())->channel() + 1);
486  _setValue(portSpinBox, part->masterScore()->midiMapping(channel->channel())->port() + 1);
487  needSync = false;
488  break;
489  }
490 
491  if (msgBox.clickedButton() == assignFreeChannel) {
492  newChannel = part->masterScore()->getNextFreeMidiMapping();
493  break;
494  }
495  // Sync
496  const int port = newChannel / 16;
497  const int channelNo = newChannel % 16;
498  _setValue(channelSpinBox, channelNo + 1);
499  _setValue(portSpinBox, port + 1);
500  MasterScore* ms = part->masterScore();
501  ms->updateMidiMapping(ms->playbackChannel(channel), part, port, channelNo);
502  channel->setVolume(lrint(pe->volume->value()));
503  channel->setPan(lrint(pe->pan->value()));
504  channel->setReverb(lrint(pe->reverb->value()));
505  channel->setChorus(lrint(pe->chorus->value()));
506  channel->setMute(pe->mute->isChecked());
507  channel->setSolo(pe->solo->isChecked());
508 
509  MidiPatch* newPatch = (MidiPatch*)pe->patch->itemData(pe->patch->currentIndex(), Qt::UserRole).value<void*>();
510  if (newPatch == 0)
511  return;
512  patch->setCurrentIndex(pe->patch->currentIndex());
513  channel->setProgram(newPatch->prog);
514  channel->setBank(newPatch->bank);
515  channel->setSynti(newPatch->synti);
516 
517  part->masterScore()->setSoloMute();
518  part->score()->setInstrumentsChanged(true);
519  part->score()->setLayoutAll();
520  break;
521  }
522  }
523  if (needSync) {
524  const int port = newChannel / 16;
525  const int channelNo = newChannel % 16;
526  _setValue(channelSpinBox, channelNo + 1);
527  _setValue(portSpinBox, port + 1);
528  MasterScore* ms = part->masterScore();
529  ms->updateMidiMapping(ms->playbackChannel(channel), part, port, channelNo);
530  ms->setInstrumentsChanged(true);
531  ms->setLayoutAll();
532  seq->initInstruments();
533  }
534  else {
535  // Initializing an instrument with new channel
536  foreach(const MidiCoreEvent& e, channel->initList()) {
537  if (e.type() == ME_INVALID)
538  continue;
539  NPlayEvent event(e.type(), channel->channel(), e.dataA(), e.dataB());
540  seq->sendEvent(event);
541  }
542  }
543 
544  // Update MIDI Out ports
545  int maxPort = std::max(p, part->score()->masterScore()->midiPortCount());
546  part->score()->masterScore()->setMidiPortCount(maxPort);
547  if (seq->driver() && (preferences.getBool(PREF_IO_JACK_USEJACKMIDI) || preferences.getBool(PREF_IO_ALSA_USEALSAAUDIO)))
548  seq->driver()->updateOutPortCount(maxPort + 1);
549  }
