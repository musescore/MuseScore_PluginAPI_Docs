74 QImage Pdf::binarization(QImage image){
75  QImage bw = QImage(image.width(), image.height(), QImage::Format_MonoLSB);
76  QVector<QRgb> ct(2);
77  ct[0] = qRgb(255, 255, 255);
78  ct[1] = qRgb(0, 0, 0);
79  bw.setColorTable(ct);
80  bw.fill(0);
81  float thresh = 128;
82  float new_thresh = 0;
83 
84  while (thresh != new_thresh) {
85  float sum_black = 0;
86  float sum_white = 0;
87  int num_black = 0;
88  int num_white = 0;
89  new_thresh = thresh;
90  for (int x = 0; x < image.width(); x++){
91  for (int y = 0; y < image.height(); y++) {
92  QRgb c = image.pixel(x, y);
93  float g = qGray(c);
94  if (g < thresh) {
95  sum_black += g;
96  num_black++;
97  }
98  else {
99  sum_white += g;
100  num_white++;
101  }
102 
103  }
104  }
105  thresh = (sum_black/num_black + sum_white/num_white)/2.0;
106  }
107 
108  int stride = (bw.width() + 7) / 8;
109  uchar* p = bw.bits();
110 
111  for (int y = 0; y < bw.height(); ++y) {
112  p = bw.scanLine(y);
113  for (int x = 0; x < stride; ++x) {
114  int temp = 0;
115  for (int i = 0; i < 8; i++) {
116  if (x*8 + i >= bw.width()) continue;
117  QRgb c = image.pixel(x*8 + i, y);
118  float g = qGray(c);
119  temp += ((g<thresh) ? 1:0)<<i;
120  }
121  *p++ = temp;
122  }
123  }
124  return bw;
125 }
