62 void PianoKeyboard::paintEvent(QPaintEvent* /*event*/)
63  {
64  QPainter p(this);
65  p.setRenderHints(QPainter::Antialiasing | QPainter::SmoothPixmapTransform | QPainter::TextAntialiasing);
66 
67  const int fontSize = 8;
68  QFont f("FreeSans", fontSize);
69  p.setFont(f);
70 
71  //Check for drumset, if any
72  Drumset* ds = nullptr;
73  Interval transp;
74  if (_staff) {
75  Part* part = _staff->part();
76  ds = part->instrument()->drumset();
77  transp = part->instrument()->transpose();
78  }
79 
80  p.setPen(QPen(Qt::black, 2));
81 
82  int keyboardLen = 128 * noteHeight;
83  const int blackKeyLen = BLACK_KEY_WIDTH;
84 
85  const qreal whiteKeyOffset[] = {0, 1.5, 3.5, 5, 6.5, 8.5, 10.5, 12};
86  const int whiteKeyDegree[] = {0, 2, 4, 5, 7, 9, 11};
87  const qreal blackKeyOffset[] = {1.5, 3.5, 6.5, 8.5, 10.5};
88  const int blackKeyDegree[] = {1, 3, 6, 8, 10};
89 
90  //White keys
91  p.setPen(QPen(Qt::black));
92  for (int midiPitch = 0; midiPitch < 128; ++midiPitch) {
93  int instrPitch = midiPitch - transp.chromatic;
94 
95  int octave = instrPitch / 12;
96  int degree = (instrPitch + 60) % 12;
97 
98  int key = -1;
99  for (int i = 0; i < 7; ++i)
100  if (whiteKeyDegree[i] == degree) {
101  key = i;
102  break;
103  }
104 
105  if (key == -1)
106  continue;
107 
108  QString noteName = qApp->translate("utils", pitchNames[degree]) + QString::number(octave - 1);
109  if (ds)
110  noteName = qApp->translate("drumset", ds->name(instrPitch).toUtf8().constData());
111 
112  p.setBrush(curPitch == midiPitch ? colKeySelect : Qt::white);
113 
114  qreal off1 = whiteKeyOffset[key] * noteHeight + (octave * 12 + transp.chromatic) * noteHeight;
115  qreal off2 = whiteKeyOffset[key + 1] * noteHeight + (octave * 12 + transp.chromatic) * noteHeight;
116  if (_orientation == PianoOrientation::HORIZONTAL) {
117  QRectF rect(-_ypos + off2, 0, off2 - off1, height());
118  p.drawRect(rect);
119 
120  if (preferences.getBool(PREF_UI_PIANO_SHOWPITCHHELP) && degree == 0 && noteHeight > fontSize + 2) {
121  QRectF rectText(rect.x() + 1, rect.y(), rect.height() - 2, rect.width() - 2);
122  QTransform xform = p.transform();
123  p.rotate(90);
124  p.drawText(rectText,
125  ds ? Qt::AlignLeft | Qt::AlignBottom
126  : Qt::AlignRight | Qt::AlignBottom,
127  noteName);
128  p.setTransform(xform);
129  }
130  }
131  else {
132  QRectF rect(0, -_ypos + keyboardLen - off2, width(), off2 - off1);
133 
134  p.drawRect(rect);
135 
136  if (noteHeight > fontSize + 2) {
137  if (ds) {
138  QRectF rectText(rect.x() + 1, -_ypos + keyboardLen - (instrPitch + 1) * noteHeight, rect.width() - 1, noteHeight);
139 
140  p.drawText(rectText, Qt::AlignBottom | Qt::AlignLeft, noteName);
141  }
142  else if (preferences.getBool(PREF_UI_PIANO_SHOWPITCHHELP) && degree == 0) {
143  QRectF rectText(rect.x(), rect.y(), rect.width() - 4, rect.height() - 1);
144  p.drawText(rectText, Qt::AlignRight | Qt::AlignBottom, noteName);
145  }
146  }
147  }
148  }
149 
150  //Black keys
151  for (int midiPitch = 0; midiPitch < 128; ++midiPitch) {
152  int instrPitch = midiPitch - transp.chromatic;
153 
154  int octave = instrPitch / 12;
155  int degree = instrPitch % 12;
156 
157  int key = -1;
158  for (int i = 0; i < 5; ++i)
159  if (blackKeyDegree[i] == degree) {
160  key = i;
161  break;
162  }
163 
164  if (key == -1)
165  continue;
166 
167  QString noteName = qApp->translate("utils", pitchNames[blackKeyDegree[key]]) + QString::number(octave);
168  if (ds)
169  noteName = qApp->translate("drumset", ds->name(instrPitch).toUtf8().constData());
170 
171  qreal center = blackKeyOffset[key] * noteHeight;
172  qreal offset = center - noteHeight / 2.0 + (octave * 12 + transp.chromatic) * noteHeight;
173 
174  p.setPen(QPen(Qt::black));
175  p.setBrush(curPitch == midiPitch ? colKeySelect : Qt::black);
176 
177  if (_orientation == PianoOrientation::HORIZONTAL) {
178  QRectF rect(offset, 0, noteHeight, blackKeyLen);
179 
180  p.drawRect(rect);
181 
182  if (ds) {
183  if (noteHeight > fontSize + 2) {
184  rect.setWidth(blackKeyLen);
185  rect.setHeight(noteHeight);
186 
187  p.setPen(QPen(Qt::white));
188  p.drawText(rect, Qt::AlignLeft | Qt::AlignBottom, noteName);
189  }
190  }
191  }
192  else {
193  QRectF rect(0, -_ypos + keyboardLen - offset - noteHeight, blackKeyLen, noteHeight);
194 
195  p.drawRect(rect);
196 
197  if (noteHeight > fontSize + 2) {
198  if (ds) {
199  p.setPen(QPen(Qt::white));
200  p.drawText(rect, Qt::AlignLeft | Qt::AlignBottom, noteName);
201  }
202  }
203  }
204  }
205  }

247 void PianoKeyboard::mousePressEvent(QMouseEvent* event)
248  {
249  int offset = _orientation == PianoOrientation::HORIZONTAL
250  ? event->pos().x()
251  : 128 * noteHeight - (event->y() + _ypos);
252 
253  int pitch = curKeyPressed = offset / noteHeight;
254  if (pitch < 0 || pitch > 127)
255  pitch = -1;
256 
257  uint modifiers = QGuiApplication::keyboardModifiers();
258  bool bnCtrl = modifiers & Qt::ControlModifier;
259  if (bnCtrl)
260  emit pitchHighlightToggled(curKeyPressed);
261  else {
262  curKeyPressed = pitch;
263  emit keyPressed(curKeyPressed);
264  }
265  }

283 void PianoKeyboard::mouseMoveEvent(QMouseEvent* event)
284  {
285  int offset = _orientation == PianoOrientation::HORIZONTAL
286  ? event->pos().x()
287  : 128 * noteHeight - (event->y() + _ypos);
288 
289  int pitch = offset / noteHeight;
290 
291  if (pitch != curPitch) {
292  curPitch = pitch;
293 
294  //Set tooltip
295  int degree = curPitch % 12;
296  int octave = curPitch / 12;
297  QString text = qApp->translate("utils", pitchNames[degree]) + QString::number(octave - 1);
298  Part* part = _staff->part();
299  Drumset* ds = part->instrument()->drumset();
300  if (ds)
301  text += " - " + qApp->translate("drumset", ds->name(curPitch).toUtf8().constData());
302 
303  setToolTip(text);
304 
305  //Send event
306  emit pitchChanged(curPitch);
307  if ((curKeyPressed != -1) && (curKeyPressed != pitch)) {
308  emit keyReleased(curKeyPressed);
309  curKeyPressed = pitch;
310  emit keyPressed(curKeyPressed);
311  }
312  update();
313  }
314  }
