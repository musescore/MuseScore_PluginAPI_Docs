563 void PianorollEditor::setStaff(Staff* st)
564  {
565  if (staff == st)
566  return;
567 
568  if (st)
569  partLabel->setText(tr("Part: %1").arg(st->partName()));
570 
571  if ((st && st->score() != _score) || (!st && _score)) {
572  if (_score) {
573  _score->removeViewer(this);
574  disconnect(_score, SIGNAL(posChanged(POS,unsigned)), this, SLOT(posChanged(POS,unsigned)));
575  disconnect(_score, SIGNAL(playlistChanged()), this, SLOT(playlistChanged()));
576  }
577  _score = st ? st->score() : nullptr;
578  if (_score) {
579  _score->addViewer(this);
580  setLocator(POS::CURRENT, _score->pos(POS::CURRENT).ticks());
581  setLocator(POS::LEFT, _score->pos(POS::LEFT).ticks());
582  setLocator(POS::RIGHT, _score->pos(POS::RIGHT).ticks());
583  connect(_score, SIGNAL(posChanged(POS,unsigned)), SLOT(posChanged(POS,unsigned)));
584  connect(_score, SIGNAL(playlistChanged()), SLOT(playlistChanged()));
585  }
586  }
587  staff = st;

741 void PianorollEditor::veloTypeChanged(int val)
742  {
743  QList<PianoItem*> items = pianoView->getSelectedItems();
744  if (!items.size())
745  return;
746 
747  _score->startCmd();
748  for (int i = 0; i < items.size(); i++) {
749  PianoItem* item = items[i];
750  Note* note = item->note();
751  if (Note::ValueType(val) == note->veloType())
752  return;
753 
754  int newVelocity = note->veloOffset();
755  int dynamicsVel = staff->velocities().val(note->tick());
756 
757  // change velocity to equivalent in new metric
758  switch (Note::ValueType(val)) {
759  case Note::ValueType::USER_VAL:
760  newVelocity = static_cast<int>(dynamicsVel * (1 + newVelocity / 100.0));
761  break;
762  case Note::ValueType::OFFSET_VAL:
763  newVelocity = static_cast<int>((newVelocity / (qreal)dynamicsVel - 1) * 100);
764  break;
765  }
766 
767  _score->undo(new ChangeVelocity(note, Note::ValueType(val), newVelocity));
768  updateVelocity(note);
769  }
770  _score->endCmd();
771  }
