173 void PianoRuler::paintEvent(QPaintEvent* e)
174  {
175  QPainter p(this);
176  const QRect& r = e->rect();
177 
178  int x = r.x();
179  int w = r.width();
180  int y = pianoRulerHeight - 16;
181  int h = 16; // 14;
182  int y1 = r.y();
183  int rh = r.height();
184  if (y1 < pianoRulerHeight) {
185  rh -= pianoRulerHeight - y1;
186  y1 = pianoRulerHeight;
187  }
188  int y2 = y1 + rh;
189 
190  if (x < (-_xpos))
191  x = -_xpos;
192 
193  if (!_score)
194  return;
195 
196  Pos pos1 = pix2pos(x);
197  Pos pos2 = pix2pos(x+w);
198 
199  //---------------------------------------------------
200  // draw lines
201  //---------------------------------------------------
202 
203  int bar1, bar2, beat, tick;
204 
205  pos1.mbt(&bar1, &beat, &tick);
206  pos2.mbt(&bar2, &beat, &tick);
207 
208  const int minBarGapSize = 48;
209  const int minBeatGapSize = 30;
210 
211  //Estimate bar width since changing time signatures can make this inconsistent.
212  // Assuming 480 ticks per beat, 4 beats per bar
213  qreal pixPerBar = MScore::division * 4 * _xZoom;
214  qreal pixPerBeat = MScore::division * _xZoom;
215 
216  int barSkip = ceil(minBarGapSize / pixPerBar);
217  barSkip = (int)pow(2, ceil(log(barSkip)/log(2)));
218 
219  int beatSkip = ceil(minBeatGapSize / pixPerBeat);
220  beatSkip = (int)pow(2, ceil(log(beatSkip)/log(2)));
221 
222  //Round down to first bar to be a multiple of barSkip
223  bar1 = (bar1 / barSkip) * barSkip;
224 
225  for (int bar = bar1; bar <= bar2; bar += barSkip) {
226  Pos stick(_score->tempomap(), _score->sigmap(), bar, 0, 0);
227 
228  SigEvent sig = stick.timesig();
229  int z = sig.timesig().numerator();
230  for (int beat1 = 0; beat1 < z; beat1 += beatSkip) {
231  Pos xx(_score->tempomap(), _score->sigmap(), bar, beat1, 0);
232  int xp = pos2pix(xx);
233  if (xp < 0)
234  continue;
235  QString s;
236  QRect r1(xp+2, y + 1, 1000, h);
237  int y3;
238  int num;
239  if (beat1 == 0) {
240  num = bar + 1;
241  y3 = y + 2;
242  p.setFont(_font2);
243  }
244  else {
245  num = beat1 + 1;
246  y3 = y + 8;
247  p.setFont(_font1);
248  r1.moveTop(r1.top() + 1);
249  }
250  s.setNum(num);
251  p.setPen(Qt::black);
252  p.drawLine(xp, y3, xp, y+h);
253  p.drawText(r1, Qt::AlignLeft | Qt::AlignVCenter, s);
254  p.setPen(beat1 == 0 ? Qt::lightGray : Qt::gray);
255  if (xp > 0)
256  p.drawLine(xp, y1, xp, y2);
257  }
258 
259  }
260  //
261  // draw mouse cursor marker
262  //
263  p.setPen(Qt::black);
264  if (_cursor.valid()) {
265  int xp = pos2pix(_cursor);
266  if (xp >= x && xp < x+w)
267  p.drawLine(xp, 0, xp, pianoRulerHeight);
268  }
269  static const QColor lcColors[3] = { Qt::red, Qt::blue, Qt::blue };
270  for (int i = 0; i < 3; ++i) {
271  if (!_locator[i].valid())
272  continue;
273  p.setPen(lcColors[i]);
274  int xp = pos2pix(_locator[i]);
275  QPixmap* pm = markIcon[i];
276  int pw = pm->width() / 2;
277  int x1 = x - pw;
278  int x2 = x + w + pw;
279  if (xp >= x1 && xp < x2)
280  p.drawPixmap(xp - pw, y-2, *pm);
281  }
282  }
