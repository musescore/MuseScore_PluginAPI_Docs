119 void HPiano::setScale(qreal s)
120  {
121  if (s > 16.0)
122  s = 16.0;
123  else if (s < .5)
124  s = .5;
125  if (s != scaleVal) {
126  scaleVal = s;
127  int margin = 16;
128  QDockWidget* par = static_cast<QDockWidget*>(parent());
129  if (par) {
130  if (!par->isFloating())
131  setMaximumSize(QSize((KEY_WIDTH * 52 + margin/2) * scaleVal, 1000));
132  else
133  setMaximumSize(QSize((KEY_WIDTH * 52 + margin/2) * scaleVal, (KEY_HEIGHT + margin) * scaleVal));
134  }
135  else
136  setMaximumSize(QSize((KEY_WIDTH * 52 + margin/2) * scaleVal, (KEY_HEIGHT + margin) * scaleVal));
137  setMinimumSize(QSize(100 * scaleVal, (KEY_HEIGHT + margin) * scaleVal));
138  QTransform t;
139  t.scale(scaleVal, scaleVal);
140  setTransform(t, false);
141  }
142  }

187 void HPiano::changeSelection(const Selection& selection)
188  {
189  for (PianoKeyItem* key : qAsConst(keys)) {
190  key->setHighlighted(false);
191  key->setSelected(false);
192  }
193  for (Note* n : selection.noteList()) {
194  if (n->epitch() >= _firstKey && n->epitch() <= _lastKey)
195  keys[n->epitch() - _firstKey]->setSelected(true);
196  for (Note* other : n->chord()->notes())
197  if (other->epitch() >= _firstKey && other->epitch() <= _lastKey)
198  keys[other->epitch() - _firstKey]->setHighlighted(true);
199  }
200  for (PianoKeyItem* key : qAsConst(keys))
201  key->update();
202  }

471 void HPiano::wheelEvent(QWheelEvent* event)
472  {
473  static int deltaSum = 0;
474  deltaSum += event->angleDelta().y();
475  int step = deltaSum / 120;
476  deltaSum %= 120;
477  qreal mag = scaleVal;
478  if (event->modifiers() & Qt::ControlModifier) {
479  if (step > 0) {
480  for (int i = 0; i < step; ++i)
481  mag *= 1.1;
482  }
483  else {
484  for (int i = 0; i < -step; ++i)
485  mag /= 1.1;
486  }
487  setScale(mag);
488  }
489  }

509 bool HPiano::gestureEvent(QGestureEvent *event)
510  {
511  if (QGesture *gesture = event->gesture(Qt::PinchGesture)) {
512  // Zoom in/out when receiving a pinch gesture
513  QPinchGesture *pinch = static_cast<QPinchGesture *>(gesture);
514 
515  static qreal magStart = 1.0;
516  if (pinch->state() == Qt::GestureStarted) {
517  magStart = scaleVal;
518  }
519  if (pinch->changeFlags() & QPinchGesture::ScaleFactorChanged) {
520  // On Windows, totalScaleFactor() contains the net magnification.
521  // On OS X, totalScaleFactor() is 1, and scaleFactor() contains the net magnification.
522  qreal value = pinch->totalScaleFactor();
523  if (value == 1) {
524  value = pinch->scaleFactor();
525  }
526  // Qt 5.4 doesn't report pinch->centerPoint() correctly
527  setScale(magStart*value);
528  }
529  }
530  return true;
531  }
