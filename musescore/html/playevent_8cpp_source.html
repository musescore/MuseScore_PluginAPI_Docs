26 void PlayEvent::setPitch(int v)
27  {
28  if (!ne || ne->pitch() == v)
29  return; // Value hasn't changed so no need to do more.
30 
31  if (parentNote == nullptr)
32  {
33  // We can't check against a parent note we don't have so
34  // we check against the absolute range. When this NoteEvent
35  // is added to an actual note the range should be checked
36  // again.
37  if (pitchIsValid(abs(v))) {
38  ne->setPitch(v); // Set new ontTime value
39  }
40  else
41  {
42  qWarning("PluginAPI::PlayEvent::setPitch: Invalid relative pitch value when added to parent note pitch.");
43  return;
44  }
45  }
46  else
47  {
48  Ms::Score* score = parentNote->note()->score();
49  if (!score) {
50  qWarning("PluginAPI::PlayEvent::setOntime: A score is required.");
51  return;
52  }
53  int parentPitch = parentNote->note()->pitch();
54  // A NoteEvent's pitch is actually summed with the parent pitch. This
55  // check ensures that it doesn't result with an illegal pitch.
56  if (!pitchIsValid(v + parentPitch)) {
57  qWarning("PluginAPI::PlayEvent::setPitch: Invalid relative pitch value when added to parent note pitch.");
58  return;
59  }
60  Ms::NoteEvent e = *ne; // Make copy of NoteEvent value
61  e.setPitch(v); // Set new ontTime value
62  score->undo(new ChangeNoteEvent(parentNote->note(), ne, e));
63  }
64  }

70 void PlayEvent::setOntime(int v)
71  {
72  if (!ne || ne->ontime() == v)
73  return; // Value hasn't changed so no need to do more.
74  // Note that onTime can be negative so a note can play earlier.
75  // See: https://musescore.org/en/node/74651
76  if (v < -2 * Ms::NoteEvent::NOTE_LENGTH || v > 2 * Ms::NoteEvent::NOTE_LENGTH) {
77  qWarning("PluginAPI::PlayEvent::setOntime: Invalid value.");
78  return;
79  }
80  if (parentNote == nullptr)
81  {
82  // Unowned floating value in QML context.
83  ne->setOntime(v); // Set new ontTime value
84  }
85  else
86  {
87  Ms::Score* score = parentNote->note()->score();
88  if (!score) {
89  qWarning("PluginAPI::PlayEvent::setOntime: A score is required.");
90  return;
91  }
92  Ms::NoteEvent e = *ne; // Make copy of NoteEvent value
93  e.setOntime(v); // Set new ontTime value
94  score->undo(new ChangeNoteEvent(parentNote->note(), ne, e));
