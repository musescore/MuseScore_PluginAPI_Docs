117 void PluginCreator::setState(PCState newState)
118  {
119  if (state == newState)
120  return;
121  switch(state) {
122  case PCState::INIT:
123  switch(newState) {
124  case PCState::INIT:
125  break;
126  case PCState::EMPTY:
127  setTitle("");
128  actionSave->setEnabled(false);
129  actionSaveAs->setEnabled(false);
130  run->setEnabled(false);
131  stop->setEnabled(false);
132  textEdit->setEnabled(false);
133  break;
134  case PCState::CLEAN:
135  case PCState::DIRTY:
136  break;
137  }
138  break;
139  case PCState::EMPTY:
140  switch(newState) {
141  case PCState::INIT:
142  case PCState::EMPTY:
143  break;
144  case PCState::CLEAN:
145  setTitle(path);
146  actionSaveAs->setEnabled(true);
147  run->setEnabled(true);
148  textEdit->setEnabled(true);
149  break;
150  case PCState::DIRTY:
151  return;
152  }
153  break;
154  case PCState::CLEAN:
155  switch(newState) {
156  case PCState::INIT:
157  case PCState::EMPTY:
158  case PCState::CLEAN:
159  break;
160  case PCState::DIRTY:
161  actionSave->setEnabled(true);
162  actionSaveAs->setEnabled(true);
163  break;
164  }
165  break;
166  case PCState::DIRTY:
167  switch(newState) {
168  case PCState::INIT:
169  case PCState::EMPTY:
170  case PCState::CLEAN:
171  actionSave->setEnabled(false);
172  actionSave->setEnabled(true);
173  case PCState::DIRTY:
174  break;
175  }
176  break;
177  }
178  state = newState;
179  }

253 static void qmlMsgHandler(QtMsgType type, const QMessageLogContext &, const QString & msg)
254  {
255  QString s;
256  switch(type) {
257  case QtDebugMsg:
258  s = QString("Debug: %1\n").arg(msg);
259  break;
260  case QtWarningMsg:
261  s = QString("Warning: %1\n").arg(msg);
262  break;
263  case QtCriticalMsg:
264  s = QString("Critical: %1\n").arg(msg);
265  break;
266  case QtFatalMsg:
267  s = QString("Fatal: %1\n").arg(msg);
268  break;
269  default:
270  s = QString("Info: %1\n").arg(msg);
271  break;
272  }
273  mscore->pluginCreator()->msg(s);

280 void PluginCreator::runClicked()
281  {
282  log->clear();
283  msg(tr("Running\n"));
284  QmlPluginEngine* qml = mscore->getPluginEngine();
285  connect(qml, SIGNAL(warnings(const QList<QQmlError>&)),
286  SLOT(qmlWarnings(const QList<QQmlError>&)));
287 
288  item = 0;
289  QQmlComponent component(qml);
290  const QUrl url = created ? QUrl() : QUrl::fromLocalFile(path);
291  component.setData(textEdit->toPlainText().toUtf8(), url);
292  QObject* obj = component.create();
293  if (obj == 0) {
294  msg(tr("Creating component failed\n"));
295  for (QQmlError e : component.errors())
296  msg(" " + tr("line %1: %2\n").arg(e.line()).arg(e.description()));
297  stop->setEnabled(false);
298  return;
299  }
300 
301  item = qobject_cast<QmlPlugin*>(obj);
302 
303  if (!item) {
304  msg(tr("Component is not a MuseScore plugin") + '\n');
305  delete obj;
306  return;
307  }
308 
309  qInstallMessageHandler(qmlMsgHandler);
310  stop->setEnabled(true);
311  run->setEnabled(false);
312 
313  msg(tr("Plugin Details:") + "\n");
314  msg(" " + tr("Menu Path:") + " " + item->menuPath() + "\n");
315  msg(" " + tr("Version:") + " " + item->version() + "\n");
316  msg(" " + tr("Description:") + " " + item->description() + "\n");
317  if (item->requiresScore()) msg(" " + tr("Requires Score\n"));
318  if(MuseScoreCore::mscoreCore->currentScore() == nullptr && item->requiresScore() == true) {
319  QMessageBox::information(0,
320  tr("MuseScore"),
321  tr("No score open.\n"
322  "This plugin requires an open score to run.\n"),
323  QMessageBox::Ok, QMessageBox::NoButton);
324  delete obj;
325  item = nullptr;
326  closePlugin();
327  return;
328  }
329  item->setFilePath(path.isEmpty() ? QString() : path.section('/', 0, -2));
330 
331  if (item->pluginType() == "dock" || item->pluginType() == "dialog") {
332  view = new QQuickView(qml, 0);
333  view->setTitle(item->menuPath().mid(item->menuPath().lastIndexOf(".") + 1));
334  view->setColor(QApplication::palette().color(QPalette::Window));
335  view->setResizeMode(QQuickView::SizeRootObjectToView);
336  view->setWidth(item->width());
337  view->setHeight(item->height());
338  item->setParentItem(view->contentItem());
339 
340  if (item->pluginType() == "dock") {
341  dock = new QDockWidget(view->title(), 0);
342  dock->setAttribute(Qt::WA_DeleteOnClose);
343  dock->setWidget(QWidget::createWindowContainer(view));
344  dock->widget()->setSizePolicy(QSizePolicy(QSizePolicy::Expanding, QSizePolicy::Expanding));
345  Qt::DockWidgetArea area = Qt::RightDockWidgetArea;
346  if (item->dockArea() == "left")
347  area = Qt::LeftDockWidgetArea;
348  else if (item->dockArea() == "top")
349  area = Qt::TopDockWidgetArea;
350  else if (item->dockArea() == "bottom")
351  area = Qt::BottomDockWidgetArea;
352  addDockWidget(area, dock);
353  connect(dock, SIGNAL(destroyed()), SLOT(closePlugin()));
354  dock->widget()->setAttribute(Qt::WA_DeleteOnClose);
355  dock->show();
356  }
357  view->show();
358  connect(view, SIGNAL(destroyed()), SLOT(closePlugin()));
359  }
360 
361  connect(qml, SIGNAL(quit()), SLOT(closePlugin()));
362  connect(qml, &QmlPluginEngine::endCmd, item, &QmlPlugin::endCmd);
363 
364  if (mscore->currentScore() && item->pluginType() != "dock")
365  mscore->currentScore()->startCmd();
366  item->runPlugin();
367  if (mscore->currentScore() && item->pluginType() != "dock")
368  mscore->currentScore()->endCmd();
369  mscore->endCmd();
370  // Main window is on top at this point. Make sure correct view is on top.
371  if (item->pluginType() == "dock") {
372  raise(); // Screen needs to be on top to see docked panel.
373  }
374  else if (view) {
375  view->raise();
376  }
377  else {
378  raise(); // Console only, bring to top to see results.
379  }
380  }
