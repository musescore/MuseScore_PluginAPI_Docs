67 bool PluginManager::readPluginList()
68  {
69  QFile f(dataPath + "/plugins.xml");
70  if (!f.exists())
71  return false;
72  if (!f.open(QIODevice::ReadOnly)) {
73  qDebug("Cannot open plugins file <%s>", qPrintable(f.fileName()));
74  return false;
75  }
76  XmlReader e(&f);
77  while (e.readNextStartElement()) {
78  if (e.name() == "museScore") {
79  while (e.readNextStartElement()) {
80  const QStringRef& tag(e.name());
81  if (tag == "Plugin") {
82  PluginDescription d;
83  while (e.readNextStartElement()) {
84  const QStringRef& t(e.name());
85  if (t == "path")
86  d.path = e.readElementText();
87  else if (t == "load")
88  d.load = e.readInt();
89  else if (t == "SC")
90  d.shortcut.read(e);
91  else if (t == "version")
92  d.version = e.readElementText();
93  else if (t == "description")
94  d.description = e.readElementText();
95  else
96  e.unknown();
97  }
98  d.shortcut.setState(STATE_NORMAL | STATE_NOTE_ENTRY | STATE_EDIT |
99  STATE_ALLTEXTUAL_EDIT | STATE_PLAY | STATE_FOTO | STATE_LOCK );
100  d.shortcut.setKey(pluginShortcutActionName);
101  if (d.path.endsWith(".qml"))
102  _pluginList.append(d);
103  }
104  else
105  e.unknown();
106  }
107  }
108  else
109  e.unknown();
110  }
111  return true;
112 }

118 void PluginManager::writePluginList()
119  {
120  QDir dir;
121  dir.mkpath(dataPath);
122  QFile f(dataPath + "/plugins.xml");
123  if (!f.open(QIODevice::WriteOnly)) {
124  qDebug("cannot create plugin file <%s>", qPrintable(f.fileName()));
125  return;
126  }
127  XmlWriter xml(0, &f);
128  xml.header();
129  xml.stag("museScore version=\"" MSC_VERSION "\"");
130  foreach(const PluginDescription& d, _pluginList) {
131  xml.stag("Plugin");
132  xml.tag("path", d.path);
133  xml.tag("load", d.load);
134  xml.tag("version", d.version);
135  xml.tag("description", d.description);
136  if (!d.shortcut.keys().isEmpty())
137  d.shortcut.write(xml);
138  xml.etag();
139  }
140  xml.etag();
141  f.close();
142 }

150 #ifdef SCRIPT_INTERFACE
151 static void updatePluginList(QList<QString>& pluginPathList, const QString& pluginPath,
152  QList<PluginDescription>& pluginList)
153  {
154  QDirIterator it(pluginPath, QDir::NoDot|QDir::NoDotDot|QDir::Dirs|QDir::Files,
155  QDirIterator::Subdirectories);
156  while (it.hasNext()) {
157  it.next();
158  QFileInfo fi = it.fileInfo();
159  QString path(fi.absoluteFilePath());
160  if (fi.isFile()) {
161  if (path.endsWith(".qml")) {
162  bool alreadyInList = false;
163  foreach (const PluginDescription& p, pluginList) {
164  if (p.path == path) {
165  alreadyInList = true;
166  break;
167  }
168  }
169  if (!alreadyInList) {
170  PluginDescription p;
171  p.path = path;
172  p.load = false;
173  p.shortcut.setKey(pluginShortcutActionName);
174  if (collectPluginMetaInformation(&p))
175  pluginList.append(p);
176  }
177  }
178  }
179  else
180  updatePluginList(pluginPathList, path, pluginList);
181  }
182  }

185 void PluginManager::updatePluginList(bool forceRefresh)
186  {
187 #ifdef SCRIPT_INTERFACE
188  QList<QString> pluginPathList;
189  pluginPathList.append(dataPath + "/plugins");
190  pluginPathList.append(mscoreGlobalShare + "plugins");
191  pluginPathList.append(preferences.getString(PREF_APP_PATHS_MYPLUGINS));
192  if (forceRefresh) {
193  _pluginList.clear();
194  QmlPluginEngine* engine = mscore->getPluginEngine();
195  engine->clearComponentCache(); //TODO: Check this doesn't have unwanted side effects.
196  }
197 
198  for (QString _pluginPath : pluginPathList) {
199  Ms::updatePluginList(pluginPathList, _pluginPath, _pluginList);
200  }
201  //remove non existing files
202  auto i = _pluginList.begin();
203  while (i != _pluginList.end()) {
204  PluginDescription d = *i;
205  QFileInfo fi(d.path);
206  if (!fi.exists())
207  i = _pluginList.erase(i);
208  else
209  ++i;
210  }
211 #endif
212  }

218 void PluginManager::loadList(bool forceRefresh)
219  {
220  QStringList saveLoaded; // If forcing a refresh, the load flags are lost. Keep a copy and reapply.
221  int n = _pluginList.size();
222  if (forceRefresh && n > 0) {
223  for (int i = 0; i < n; i++) {
224  PluginDescription& d = _pluginList[i];
225  if (d.load) {
226  saveLoaded.append(d.path);
227  mscore->unregisterPlugin(&d); // This will force the menu to rebuild.
228  }
229  }
230  }
231  updatePluginList(forceRefresh);
232  n = _pluginList.size();
233  pluginListWidget->clear();
234  for (int i = 0; i < n; ++i) {
235  PluginDescription& d = _pluginList[i];
236  Shortcut* s = &d.shortcut;
237  localShortcuts[s->key()] = new Shortcut(*s);
238  if (saveLoaded.contains(d.path)) d.load = true;
239  QListWidgetItem* item = new QListWidgetItem(QFileInfo(d.path).completeBaseName(), pluginListWidget);
240  item->setFlags(item->flags() | Qt::ItemIsEnabled);
241  item->setCheckState(d.load ? Qt::Checked : Qt::Unchecked);
242  item->setData(Qt::UserRole, i);
243  }
244 
245  if (n) {
246  pluginListWidget->setCurrentRow(0);
247  pluginListWidgetItemChanged(pluginListWidget->item(0), 0);
248  }
249  }

255 void PluginManager::accept()
256  {
257  if (shortcutsChanged) {
258  shortcutsChanged = false;
259  foreach(const Shortcut* s, localShortcuts) {
260  Shortcut* os = Shortcut::getShortcut(s->key());
261  if (os) {
262  if (!os->compareKeys(*s))
263  os->setKeys(s->keys());
264  }
265  }
266  Shortcut::dirty = true;
267  }
268  int n = _pluginList.size();
269  for (int i = 0; i < n; ++i) {
270  PluginDescription& d = _pluginList[i];
271  if (d.load)
272  mscore->registerPlugin(&d);
273  else
274  mscore->unregisterPlugin(&d);
275  }
276 
277  writePluginList();
278  if (Shortcut::dirty)
279  Shortcut::save();
280  Shortcut::dirty = false;
281 
282  disconnect(pluginListWidget, SIGNAL(itemChanged(QListWidgetItem*)));
283  disconnect(pluginListWidget, SIGNAL(currentItemChanged(QListWidgetItem*, QListWidgetItem*)));
284  QDialog::accept();
285  }
