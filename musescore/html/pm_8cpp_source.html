65 bool PortMidiDriver::init()
66  {
67  inputId = getDeviceIn(preferences.getString(PREF_IO_PORTMIDI_INPUTDEVICE));
68  if (inputId == -1)
69  inputId = Pm_GetDefaultInputDeviceID();
70 
71  if (inputId == pmNoDevice)
72  return false;
73 
74  outputId = getDeviceOut(preferences.getString(PREF_IO_PORTMIDI_OUTPUTDEVICE)); // Note: allow init even if outputId == pmNoDevice, since input is more important than output.
75 
76  static const int DRIVER_INFO = 0;
77  static const int TIME_INFO = 0;
78 
79  Pt_Start(20, 0, 0); // timer started, 20 millisecond accuracy
80 
81  PmError error = Pm_OpenInput(&inputStream,
82  inputId,
83  (void*)DRIVER_INFO,
84  preferences.getInt(PREF_IO_PORTMIDI_INPUTBUFFERCOUNT),
85  ((PmTimeProcPtr) Pt_Time),
86  (void*)TIME_INFO);
87  if (error != pmNoError) {
88  const char* p = Pm_GetErrorText(error);
89  qDebug("PortMidi: open input (id=%d) failed: %s", int(inputId), p);
90  Pt_Stop();
91  return false;
92  }
93 
94  Pm_SetFilter(inputStream, PM_FILT_ACTIVE | PM_FILT_CLOCK | PM_FILT_SYSEX);
95 
96  PmEvent buffer[1];
97  while (Pm_Poll(inputStream))
98  Pm_Read(inputStream, buffer, 1);
99 
100  if (outputId != pmNoDevice) {
101  error = Pm_OpenOutput(&outputStream,
102  outputId,
103  (void*)DRIVER_INFO,
104  preferences.getInt(PREF_IO_PORTMIDI_OUTPUTBUFFERCOUNT),
105  ((PmTimeProcPtr) Pt_Time),
106  (void*)TIME_INFO,
107  preferences.getInt(PREF_IO_PORTMIDI_OUTPUTLATENCYMILLISECONDS));
108  if (error != pmNoError) {
109  const char* p = Pm_GetErrorText(error);
110  qDebug("PortMidi: open output (id=%d) failed: %s", int(outputId), p);
111  Pt_Stop();
112  return false;
113  }
114  }
115 
116  timer = new QTimer();
117  timer->setInterval(20); // 20 msec
118  timer->start();
119  timer->connect(timer, SIGNAL(timeout()), seq, SLOT(midiInputReady()));
120  return true;
121  }

163 void PortMidiDriver::read()
164  {
165  if (!inputStream)
166  return;
167  PmEvent buffer[1];
168  while (Pm_Poll(inputStream)) {
169  int n = Pm_Read(inputStream, buffer, 1);
170  if (n > 0) {
171  int status = Pm_MessageStatus(buffer[0].message);
172  int type = status & 0xF0;
173  int channel = status & 0x0F;
174  if (type == ME_NOTEON) {
175  int pitch = Pm_MessageData1(buffer[0].message);
176  int velo = Pm_MessageData2(buffer[0].message);
177  mscore->midiNoteReceived(channel, pitch, velo);
178  }
179  else if (type == ME_NOTEOFF) {
180  int pitch = Pm_MessageData1(buffer[0].message);
181  (void)Pm_MessageData2(buffer[0].message); // read but ignore
182  mscore->midiNoteReceived(channel, pitch, 0);
183  }
184  else if (type == ME_CONTROLLER) {
185  int param = Pm_MessageData1(buffer[0].message);
186  int value = Pm_MessageData2(buffer[0].message);
187  mscore->midiCtrlReceived(param, value);
188  }
189  }
190  }
191  }

269 // determines if both the input and output devices are the same shared CoreMIDI "IAC" bus
270 //---------------------------------------------------------
271 
272 bool PortMidiDriver::isSameCoreMidiIacBus(const QString& inInterfaceAndName, const QString& outInterfaceAndName)
273  {
274  int interf = Pm_CountDevices();
275  const PmDeviceInfo* inInfo = 0;
276  const PmDeviceInfo* outInfo = 0;
277  for (PmDeviceID id = 0; id < interf; id++) {
278  const PmDeviceInfo* info = Pm_GetDeviceInfo((PmDeviceID)id);
279  if (info->input && inInterfaceAndName.contains(info->name))
280  inInfo = info;
281  if (info->output && outInterfaceAndName.contains(info->name))
282  outInfo = info;
283  }
284 
285  if (inInfo && outInfo &&
286  QString(inInfo->interf) == "CoreMIDI" && QString(outInfo->interf) == "CoreMIDI" &&
287  inInterfaceAndName.contains("IAC") && outInterfaceAndName == inInterfaceAndName)
288  return true;
289  else
290  return false;
291  }
292 }
