29 //! Note: see musescore.h getSharePath(). This copy is needed for build without MuseScore
30 QString sharePath()
31  {
32 #ifdef Q_OS_WIN
33  QDir dir(QCoreApplication::applicationDirPath() + QString("/../" INSTALL_NAME));
34  return dir.absolutePath() + "/";
35 #else
36 #ifdef Q_OS_MAC
37  QDir dir(QCoreApplication::applicationDirPath() + QString("/../Resources"));
38  return dir.absolutePath() + "/";
39 #else
40  // Try relative path (needed for portable AppImage and non-standard installations)
41  QDir dir(QCoreApplication::applicationDirPath() + QString("/../share/" INSTALL_NAME));
42  if (dir.exists())
43  return dir.absolutePath() + "/";
44  // Otherwise fall back to default location (e.g. if binary has moved relative to share)
45  return QString( INSTPREFIX "/share/" INSTALL_NAME);
46 #endif
47 #endif
48  }

61 #if defined(Q_OS_WIN)
62  // Use the correct default GUI font on Windows. Qt 5 and below use the deprecated function GetStockObject() with
63  // DEFAULT_GUI_FONT, which returns MS Shell Dlg 2 in 8 pt. MS Shell Dlg 2 is a virtual font that maps to Tahoma, which has
64  // not been the default Windows GUI font since 2006.
65  //
66  // The correct way to determine the default GUI font is to call SystemParametersInfoW() with SPI_GETNONCLIENTMETRICS and use
67  // the returned lfMessageFont structure. On all versions of Windows from Windows Vista through Windows 10, this typically
68  // returns Segoe UI in 9 pt.
69  //
70  // This problem is slated to be fixed in Qt 6. For details, see: https://bugreports.qt.io/browse/QTBUG-58610
71  //
72  // In the meantime, we can work around the problem by having Qt use the "QMessageBox" font instead, which is already being
73  // initialized the correct way.
74  QApplication::setFont(QApplication::font("QMessageBox"));
75 
76  // If the current settings file is using the deprecated default GUI font, reset it so that the correct default font can be
77  // picked up automatically.
78  if (_settings && _settings->value(PREF_UI_THEME_FONTFAMILY) == "MS Shell Dlg 2") {
79  _settings->remove(PREF_UI_THEME_FONTFAMILY);
