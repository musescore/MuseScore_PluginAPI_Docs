427 
428 QVariant propertyFromString(Pid id, QString value)
429  {
430  switch (propertyType(id)) {
431  case P_TYPE::BOOL:
432  return QVariant(bool(value.toInt()));
433  case P_TYPE::ZERO_INT:
434  case P_TYPE::INT:
435  return QVariant(value.toInt());
436  case P_TYPE::REAL:
437  case P_TYPE::SPATIUM:
438  case P_TYPE::SP_REAL:
439  case P_TYPE::TEMPO:
440  return QVariant(value.toDouble());
441  case P_TYPE::FRACTION:
442  return Fraction::fromString(value);
443  case P_TYPE::COLOR:
444  // not used by MSCX
445  return QColor(value);
446  case P_TYPE::POINT:
447  case P_TYPE::POINT_SP:
448  case P_TYPE::POINT_SP_MM: {
449  // not used by MSCX
450  const int i = value.indexOf(';');
451  return QPointF(value.leftRef(i).toDouble(), value.midRef(i+1).toDouble());
452  }
453  case P_TYPE::SCALE:
454  case P_TYPE::SIZE: {
455  // not used by MSCX
456  const int i = value.indexOf('x');
457  return QSizeF(value.leftRef(i).toDouble(), value.midRef(i+1).toDouble());
458  }
459  case P_TYPE::FONT:
460  case P_TYPE::STRING:
461  return value;
462  case P_TYPE::GLISS_STYLE: {
463  if ( value == "whitekeys")
464  return QVariant(int(GlissandoStyle::WHITE_KEYS));
465  else if ( value == "blackkeys")
466  return QVariant(int(GlissandoStyle::BLACK_KEYS));
467  else if ( value == "diatonic")
468  return QVariant(int(GlissandoStyle::DIATONIC));
469  else if ( value == "portamento")
470  return QVariant(int(GlissandoStyle::PORTAMENTO));
471  else // e.g., normally "Chromatic"
472  return QVariant(int(GlissandoStyle::CHROMATIC));
473  }
474  break;
475  case P_TYPE::ORNAMENT_STYLE: {
476  if ( value == "baroque")
477  return QVariant(int(MScore::OrnamentStyle::BAROQUE));
478  return QVariant(int(MScore::OrnamentStyle::DEFAULT));
479  }
480 
481  case P_TYPE::DIRECTION:
482  return QVariant::fromValue<Direction>(toDirection(value));
483 
484  case P_TYPE::DIRECTION_H:
485  {
486  if (value == "left" || value == "1")
487  return QVariant(int(MScore::DirectionH::LEFT));
488  else if (value == "right" || value == "2")
489  return QVariant(int(MScore::DirectionH::RIGHT));
490  else if (value == "auto")
491  return QVariant(int(MScore::DirectionH::AUTO));
492  }
493  break;
494  case P_TYPE::LAYOUT_BREAK: {
495  if (value == "line")
496  return QVariant(int(LayoutBreak::LINE));
497  if (value == "page")
498  return QVariant(int(LayoutBreak::PAGE));
499  if (value == "section")
500  return QVariant(int(LayoutBreak::SECTION));
501  if (value == "nobreak")
502  return QVariant(int(LayoutBreak::NOBREAK));
503  qDebug("getProperty: invalid P_TYPE::LAYOUT_BREAK: <%s>", qPrintable(value));
504  }
505  break;
506  case P_TYPE::VALUE_TYPE: {
507  if (value == "offset")
508  return QVariant(int(Note::ValueType::OFFSET_VAL));
509  else if (value == "user")
510  return QVariant(int(Note::ValueType::USER_VAL));
511  }
512  break;
513  case P_TYPE::PLACEMENT: {
514  if (value == "above")
515  return QVariant(int(Placement::ABOVE));
516  else if (value == "below")
517  return QVariant(int(Placement::BELOW));
518  }
519  break;
520  case P_TYPE::HPLACEMENT: {
521  if (value == "left")
522  return QVariant(int(HPlacement::LEFT));
523  else if (value == "center")
524  return QVariant(int(HPlacement::CENTER));
525  else if (value == "right")
526  return QVariant(int(HPlacement::RIGHT));
527  }
528  break;
529  case P_TYPE::TEXT_PLACE: {
530  if (value == "auto")
531  return QVariant(int(PlaceText::AUTO));
532  else if (value == "above")
533  return QVariant(int(PlaceText::ABOVE));
534  else if (value == "below")
535  return QVariant(int(PlaceText::BELOW));
536  else if (value == "left")
537  return QVariant(int(PlaceText::LEFT));
538  }
539  break;
540  case P_TYPE::BARLINE_TYPE: {
541  bool ok;
542  const int ct = value.toInt(&ok);
543  if (ok)
544  return QVariant(ct);
545  else {
546  BarLineType t = BarLine::barLineType(value);
547  return QVariant::fromValue(t);
548  }
549  }
550  break;
551  case P_TYPE::BEAM_MODE: // TODO
552  return QVariant(int(0));
553 
554  case P_TYPE::GROUPS:
555  // unsupported
556  return QVariant();
557  case P_TYPE::SYMID:
558  return QVariant::fromValue(Sym::name2id(value));
559  case P_TYPE::HEAD_SCHEME:
560  return QVariant::fromValue(NoteHead::name2scheme(value));
561  case P_TYPE::HEAD_GROUP:
562  return QVariant::fromValue(NoteHead::name2group(value));
563  case P_TYPE::HEAD_TYPE:
564  return QVariant::fromValue(NoteHead::name2type(value));
565  case P_TYPE::POINT_MM: // not supported
566  case P_TYPE::TDURATION:
567  case P_TYPE::SIZE_MM:
568  case P_TYPE::INT_LIST:
569  return QVariant();
570  case P_TYPE::SUB_STYLE:
571  return int(textStyleFromName(value));
572  case P_TYPE::ALIGN: {
573  QStringList sl = value.split(',');
574  if (sl.size() != 2) {
575  qDebug("bad align text <%s>", qPrintable(value));
576  return QVariant();
577  }
578  Align align = Align::LEFT;
579  if (sl[0] == "center")
580  align = align | Align::HCENTER;
581  else if (sl[0] == "right")
582  align = align | Align::RIGHT;
583  else if (sl[0] == "left")
584  ;
585  else {
586  qDebug("bad align text <%s>", qPrintable(sl[0]));
587  return QVariant();
588  }
589  if (sl[1] == "center")
590  align = align | Align::VCENTER;
