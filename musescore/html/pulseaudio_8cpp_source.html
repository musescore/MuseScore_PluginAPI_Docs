96 static void pa_state_cb(pa_context* c, void* data)
97  {
98  int* pa_ready = (int*)data;
99  switch (pa_context_get_state(c)) {
100  // These are just here for reference
101  case PA_CONTEXT_UNCONNECTED:
102  case PA_CONTEXT_CONNECTING:
103  case PA_CONTEXT_AUTHORIZING:
104  case PA_CONTEXT_SETTING_NAME:
105  default:
106  break;
107  case PA_CONTEXT_FAILED:
108  case PA_CONTEXT_TERMINATED:
109  *pa_ready = 2;
110  break;
111  case PA_CONTEXT_READY:
112  *pa_ready = 1;
113  break;
114  }

122 bool PulseAudio::init(bool)
123  {
124  pa_ml = pa_mainloop_new();
125  pa_mainloop_api* pa_mlapi = pa_mainloop_get_api(pa_ml);
126  pa_context* pa_ctx = pa_context_new(pa_mlapi, "MuseScore");
127  if (pa_context_connect(pa_ctx, NULL, pa_context_flags_t(0), NULL) != 0) {
128  qDebug("PulseAudio Context Connect Failed with Error: %s", pa_strerror(pa_context_errno(pa_ctx)));
129  return false;
130  }
131 
132  int pa_ready = 0;
133  pa_context_set_state_callback(pa_ctx, pa_state_cb, &pa_ready);
134 
135  while (pa_ready == 0)
136  pa_mainloop_iterate(pa_ml, 1, NULL);
137  if (pa_ready == 2)
138  return false;
139 
140  ss.rate = _sampleRate;
141  ss.channels = 2;
142  ss.format = PA_SAMPLE_FLOAT32LE;
143 
144  pa_stream* playstream = pa_stream_new(pa_ctx, "Playback", &ss, NULL);
145  if (!playstream) {
146  qDebug("pa_stream_new failed: %s", pa_strerror(pa_context_errno(pa_ctx)));
147  return false;
148  }
149  pa_stream_set_write_callback(playstream, paCallback, this);
150 
151  bufattr.fragsize = (uint32_t)-1;
152  bufattr.maxlength = FRAMES * 2 * sizeof(float);
153  bufattr.minreq = FRAMES * 1 * sizeof(float); // pa_usec_to_bytes(0, &ss);
154  bufattr.prebuf = (uint32_t)-1;
155  bufattr.tlength = bufattr.maxlength;
156 
157  int r = pa_stream_connect_playback(playstream, nullptr, &bufattr,
158  PA_STREAM_NOFLAGS, nullptr, nullptr);
159 
160  if (r < 0) {
161  qDebug("pa_stream_connect_playback failed");
162  pa_context_disconnect(pa_ctx);
163  pa_context_unref(pa_ctx);
164  pa_mainloop_free(pa_ml);
165  pa_ml = 0;
166  return false;
167  }
168  return true;
169  }
