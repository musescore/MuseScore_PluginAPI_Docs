289 void QmlDockWidget::changeEvent(QEvent* evt)
290  {
291  QDockWidget::changeEvent(evt);
292  switch (evt->type()) {
293  case QEvent::StyleChange:
294  setupStyle();
295  break;
296 #if (QT_VERSION < QT_VERSION_CHECK(5, 10, 0))
297  case QEvent::LanguageChange:
298  // In the absence of QQmlEngine::retranslate() function, just
299  // reload the entire QML view. Qt 5.10 and above seem to
300  // handle retranslation in Qt Quick itself so no explicit
301  // retranslation is required.
302  if (_view)
303  _view->setSource(_view->source());
304  break;
305 #endif
306  default:
307  break;
308  }
309  }
