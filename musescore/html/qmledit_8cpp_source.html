96 void JSHighlighter::highlightBlock(const QString& text)
97  {
98  // parsing state
99  enum class State : char {
100  Start = 0,
101  Number = 1,
102  Identifier = 2,
103  String = 3,
104  Comment = 4,
105  Regex = 5
106  };
107 
108  QList<int> bracketPositions;
109  int blockState = previousBlockState();
110  int bracketLevel = blockState >> 4;
111  State state = State(blockState & 15);
112  if (blockState < 0) {
113  bracketLevel = 0;
114  state = State::Start;
115  }
116  int start = 0;
117  int i = 0;
118  while (i <= text.length()) {
119  QChar ch = (i < text.length()) ? text.at(i) : QChar();
120  QChar next = (i < text.length() - 1) ? text.at(i + 1) : QChar();
121  switch (state) {
122  case State::Start:
123  start = i;
124  if (ch.isSpace()) {
125  ++i;
126  }
127  else if (ch.isDigit()) {
128  ++i;
129  state = State::Number;
130  }
131  else if (ch.isLetter() || ch == '_') {
132  ++i;
133  state = State::Identifier;
134  }
135  else if (ch == '\'' || ch == '\"') {
136  ++i;
137  state = State::String;
138  }
139  else if (ch == '/' && next == '*') {
140  ++i;
141  ++i;
142  state = State::Comment;
143  }
144  else if (ch == '/' && next == '/') {
145  i = text.length();
146  setFormat(start, text.length(), m_colors[QmlEdit::Comment]);
147  }
148  else if (ch == '/' && next != '*') {
149  ++i;
150  state = State::Regex;
151  }
152  else {
153  if (!QString("(){}[]").contains(ch))
154  setFormat(start, 1, m_colors[QmlEdit::Operator]);
155  if (ch =='{' || ch == '}') {
156  bracketPositions += i;
157  if (ch == '{')
158  bracketLevel++;
159  else
160  bracketLevel--;
161  }
162  ++i;
163  state = State::Start;
164  }
165  break;
166  case State::Number:
167  if (ch.isSpace() || !ch.isDigit()) {
168  setFormat(start, i - start, m_colors[QmlEdit::Number]);
169  state = State::Start;
170  }
171  else {
172  ++i;
173  }
174  break;
175  case State::Identifier:
176  if (ch.isSpace() || !(ch.isDigit() || ch.isLetter() || ch == '_')) {
177  QString token = text.mid(start, i - start).trimmed();
178  if (m_keywords.contains(token))
179  setFormat(start, i - start, m_colors[QmlEdit::Keyword]);
180  else if (m_knownIds.contains(token))
181  setFormat(start, i - start, m_colors[QmlEdit::BuiltIn]);
182  state = State::Start;
183  }
184  else {
185  ++i;
186  }
187  break;
188  case State::String:
189  if (ch == text.at(start)) {
190  QChar prev = (i > 0) ? text.at(i - 1) : QChar();
191  if (prev != '\\') {
192  ++i;
193  setFormat(start, i - start, m_colors[QmlEdit::String]);
194  state = State::Start;
195  }
196  else {
197  ++i;
198  }
199  }
200  else {
201  ++i;
202  }
203  break;
204  case State::Comment:
205  if (ch == '*' && next == '/') {
206  ++i;
207  ++i;
208  setFormat(start, i - start, m_colors[QmlEdit::Comment]);
209  state = State::Start;
210  }
211  else {
212  ++i;
213  }
214  break;
215  case State::Regex:
216  if (ch == '/') {
217  QChar prev = (i > 0) ? text.at(i - 1) : QChar();
218  if (prev != '\\') {
219  ++i;
220  setFormat(start, i - start, m_colors[QmlEdit::String]);
221  state = State::Start;
222  }
223  else {
224  ++i;
225  }
226  }
227  else {
228  ++i;
229  }
230  break;
231  default:
232  state = State::Start;
233  break;
234  }
235  }
236 
237  if (state == State::Comment)
238  setFormat(start, text.length(), m_colors[QmlEdit::Comment]);
239  else
240  state = State::Start;
241 
242  if (!m_markString.isEmpty()) {
243  int pos = 0;
244  int len = m_markString.length();
245  QTextCharFormat markerFormat;
246  markerFormat.setBackground(m_colors[QmlEdit::Marker]);
247  markerFormat.setForeground(m_colors[QmlEdit::Normal]);
248  for (;;) {
249  pos = text.indexOf(m_markString, pos, m_markCaseSensitivity);
250  if (pos < 0)
251  break;
252  setFormat(pos, len, markerFormat);
253  ++pos;
254  }
255  }
256  if (!bracketPositions.isEmpty()) {
257  JSBlockData *blockData = reinterpret_cast<JSBlockData*>(currentBlock().userData());
258  if (!blockData) {
259  blockData = new JSBlockData;
260  currentBlock().setUserData(blockData);
261  }
262  blockData->bracketPositions = bracketPositions;
263  }
264  blockState = (int(state) & 15) | (bracketLevel << 4);
265  setCurrentBlockState(blockState);
266  }

424 int QmlEdit::lineNumberAreaWidth()
425  {
426  QString num { QString::number(blockCount()) };
427 
428  // HACK: boundingRect() subtracts internal whitespace. Internal whitespace
429  // amounts to roughly half the width, so I multiply by 2. Then I also
430  // subtract twice the string's length px to combat the left margin growing
431  // each time the string grows.
432  int space { 6 + (fontMetrics().boundingRect(num).width() - num.size()) * 2 };
433  return space;
434  }
