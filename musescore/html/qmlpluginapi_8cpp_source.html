135 /// Creates a new element with the given type. The
136 /// element can be then added to a score via Cursor::add.
137 /// \param elementType Element type, should be the value
138 /// from PluginAPI::PluginAPI::Element enumeration.
139 //---------------------------------------------------------
140 
141 Element* PluginAPI::newElement(int elementType)
142  {
143  Ms::Score* score = msc()->currentScore();
144  if (!score)
145  return nullptr;
146  if (elementType <= int(ElementType::INVALID) || elementType >= int(ElementType::MAXTYPE)) {
147  qWarning("PluginAPI::newElement: Wrong type ID: %d", elementType);
148  return nullptr;
149  }
150  const ElementType type = ElementType(elementType);
151  Ms::Element* e = Ms::Element::create(type, score);
152  return wrap(e, Ownership::PLUGIN);
153  }

172 Score* PluginAPI::newScore(const QString& name, const QString& part, int measures)
173  {
174  if (msc()->currentScore())
175  msc()->currentScore()->endCmd();
176  MasterScore* score = new MasterScore(MScore::defaultStyle());
177  score->setName(name);
178  score->appendPart(Score::instrTemplateFromName(part));
179  score->appendMeasures(measures);
180  score->doLayout();
181  const int view = msc()->appendScore(score);
182  msc()->setCurrentView(0, view);
183  qApp->processEvents();
184  Q_ASSERT(msc()->currentScore() == score);
185  score->startCmd();
186  return wrap<Score>(score, Ownership::SCORE);
187  }
