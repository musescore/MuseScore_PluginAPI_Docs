177  /**
178  * Notifies plugin about changes in score state.
179  * Called after each user (or plugin) action which may have changed a
180  * score. Implement \p onScoreStateChanged() function in your plugin to
181  * handle this signal.
182  *
183  * \p state variable is available within the handler with following
184  * fields:
185  *
186  * - \p selectionChanged
187  * - \p excerptsChanged
188  * - \p instrumentsChanged
189  * - \p startLayoutTick
190  * - \p endLayoutTick
191  * - \p undoRedo - whether this onScoreStateChanged invokation results
192  * from user undo/redo action. It is usualy not recommended to modify
193  * score from plugins in this case. Available since MuseScore 3.5.
194  *
195  * If a plugin modifies score in this handler, then it should:
196  * 1. enclose all modifications within Score::startCmd() / Score::endCmd()
197  * 2. take care of preventing an infinite recursion, as plugin-originated
198  * changes will trigger this signal again after calling Score::endCmd()
199  *
200  * Example:
201  * \code
202  * import QtQuick 2.0
203  * import MuseScore 3.0
204  *
205  * MuseScore {
206  * menuPath: "Plugins.selectionChangeExample"
207  * pluginType: "dock"
208  * dockArea: "left"
209  * implicitHeight: 75 // necessary for dock widget to appear with nonzero height
210  * implicitWidth: 150
211  *
212  * Text {
213  * // A label which will display pitch of the currently selected note
214  * id: pitchLabel
215  * anchors.fill: parent
216  * }
217  *
218  * onScoreStateChanged: {
219  * if (state.selectionChanged) {
220  * var el = curScore ? curScore.selection.elements[0] : null;
221  * if (el && el.type == Element.NOTE)
222  * pitchLabel.text = el.pitch;
223  * else
224  * pitchLabel.text = "no note selected";
225  * }
226  * }
227  * }
228  * \endcode
229  * \warning This functionality is considered experimental.
230  * This API may change in future versions of MuseScore.
231  * \since MuseScore 3.3
232  */
233  void scoreStateChanged(const QMap<QString, QVariant>& state);
