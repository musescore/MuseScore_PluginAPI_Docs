24 void RadioButtonGroupBox::paintEvent(QPaintEvent*)
25  {
26  QStylePainter painter(this);
27  QStyleOptionGroupBox styleOption;
28  initStyleOption(&styleOption);
29 
30  // Paint the default QGroupBox-style control, which includes an unwanted checkbox that we'll cover up afterwards.
31  painter.drawComplexControl(QStyle::CC_GroupBox, styleOption);
32 
33  // Calculate the background color the same way Qt does in QFusionStylePrivate::tabFrameColor().
34  const QColor& buttonColor = styleOption.palette.button().color();
35  QColor bgColor = buttonColor.lighter(100 + std::max(1, (180 - qGray(buttonColor.rgb())) / 6));
36  bgColor.setHsv(bgColor.hue(), 3 * bgColor.saturation() / 4, bgColor.value());
37  bgColor = bgColor.lighter(104);
38 
39  // Adjust the style options to use the checkbox's rectangle.
40  styleOption.rect = style()->subControlRect(QStyle::CC_GroupBox, &styleOption, QStyle::SC_GroupBoxCheckBox, this);
41 
42  // Cover up the checkbox, making sure to enlarge the rectangle a bit to cover up any anti-aliasing around the edges.
43  painter.fillRect(styleOption.rect.adjusted(-2, -2, 2, 2), bgColor);
44 
45  // Paint the radio button.
46  painter.drawPrimitive(QStyle::PE_IndicatorRadioButton, styleOption);
47  }
