87 void TrackList::combineTuplet(Tuplet* dst, Tuplet* src)
88  {
89  dst->setTicks(dst->ticks() * 2);
90  dst->setBaseLen(dst->baseLen().shift(-1));
91 
92  // try to combine tie'd notes
93  unsigned idx = 0;
94  if (dst->elements().back()->isChord() && src->elements().front()->isChord()) {
95  Chord* chord = toChord(src->elements().front());
96  bool akkumulateChord = true;
97  for (Note* n : chord->notes()) {
98  if (!n->tieBack() || !n->tieBack()->generated()) {
99  akkumulateChord = false;
100  break;
101  }
102  }
103  if (akkumulateChord) {
104  Chord* bc = toChord(dst->elements().back());
105  bc->setTicks(bc->ticks() + chord->ticks());
106 
107  // forward ties
108  int i = 0;
109  for (Note* n : bc->notes()) {
110  n->setTieFor(chord->notes()[i]->tieFor());
111  ++i;
112  }
113  idx = 1; // skip first src element
114  }
115  }
116 
117  for (; idx < src->elements().size(); ++idx) {
118  DurationElement* de = src->elements()[idx];
119  DurationElement* e = toDurationElement(de->clone());
120  dst->add(e);
121  if (de->isTuplet()) {
122  Tuplet* st = toTuplet(de);
123  Tuplet* dt = toTuplet(e);
124  appendTuplet(st, dt);
125  }
126  }
127  }

133 void TrackList::append(Element* e)
134  {
135  if (e->isDurationElement()) {
136  _duration += toDurationElement(e)->ticks();
137 
138  bool accumulateRest = e->isRest() && !empty() && back()->isRest();
139  Segment* s = accumulateRest ? toRest(e)->segment() : 0;
140 
141  if (s && !s->score()->isSpannerStartEnd(s->tick(), e->track()) && !s->annotations().size()) {
142  // akkumulate rests
143  Rest* rest = toRest(back());
144  Fraction du = rest->ticks();
145  du += toRest(e)->ticks();
146  rest->setTicks(du);
147  }
148  else {
149  Element* element = 0;
150  if (e->isTuplet()) {
151  Tuplet* src = toTuplet(e);
152  if (src->generated() && !empty() && back()->isTuplet()) {
153  Tuplet* b = toTuplet(back());
154  combineTuplet(b, src);
155  }
156  else {
157  element = e->clone();
158  Tuplet* dst = toTuplet(element);
159  appendTuplet(src, dst);
160  }
161  }
162  else {
163  element = e->clone();
164  ChordRest* src = toChordRest(e);
165  Segment* s1 = src->segment();
166  for (Element* ee : s1->annotations()) {
167  if (ee->track() == e->track())
168  _range->annotations.push_back({ s1->tick(), ee->clone() });
169  }
170  if (e->isChord()) {
171  Chord* chord = toChord(e);
172  bool akkumulateChord = true;
173  for (Note* n : chord->notes()) {
174  if (!n->tieBack() || !n->tieBack()->generated()) {
175  akkumulateChord = false;
176  break;
177  }
178  }
179  if (akkumulateChord && !empty() && back()->isChord()) {
180  Chord* bc = toChord(back());
181  const Fraction du = bc->ticks() + chord->ticks();
182  bc->setTicks(du);
183 
184  // forward ties
185  int idx = 0;
186  for (Note* n : bc->notes()) {
187  n->setTieFor(chord->notes()[idx]->tieFor());
188  ++idx;
189  }
190  delete element;
191  element = 0;
192  }
193  }
194  }
195  if (element) {
196  element->setSelected(false);
197  QList<Element*>::append(element);
198  }
199  }
200  }
201  else {
202  Element* c = e->clone();
203  c->setParent(0);
204  QList<Element*>::append(c);
205  }
206  }

212 void TrackList::appendGap(const Fraction& du)
213  {
214  if (du.isZero())
215  return;
216  Element* e = empty() ? 0 : back();
217  if (e && e->isRest()) {
218  Rest* rest = toRest(back());
219  Fraction dd = rest->ticks();
220  dd += du;

222  rest->setTicks(dd);
223  }
224  else {
225  Rest* rest = new Rest(0);
226  rest->setTicks(du);
227  QList<Element*>::append(rest);
228  _duration += du;

237 bool TrackList::truncate(const Fraction& f)
238  {
239  if (empty())
240  return true;

242  if (!e->isRest())
243  return false;

245  if (r->ticks() < f)
246  return false;

261 void TrackList::read(const Segment* fs, const Segment* es)
262  {
263  Fraction tick = fs->tick();
264 
265  const Segment* s;
266  for (s = fs; s && (s != es); s = s->next1()) {
267  if (!s->enabled())
268  continue;
269  Element* e = s->element(_track);
270  if (!e || e->generated()) {
271  for (Element* ee : s->annotations()) {
272  if (ee->track() == _track)
273  _range->annotations.push_back({ s->tick(), ee->clone() });
274  }
275  continue;
276  }
277  if (e->isRepeatMeasure()) {
278  // TODO: copy previous measure contents?
279  RepeatMeasure* rm = toRepeatMeasure(e);
280  Rest r(*rm);
281  append(&r);
282  tick += r.ticks();
283  }
284  else if (e->isChordRest()) {
285  DurationElement* de = toDurationElement(e);
286  Fraction gap = s->tick() - tick;
287  if (de->tuplet()) {
288  Tuplet* t = de->topTuplet();
289  s = skipTuplet(t); // continue with first chord/rest after tuplet
290  de = t;
291  }
292 
293  if (gap.isNotZero()) {
294  appendGap(gap);
295  tick += gap;
296  }
297  append(de);
298  tick += de->ticks();
299  }
300  else if (e->isBarLine()) {
301  BarLine* bl = toBarLine(e);
302  if (bl->barLineType() != BarLineType::NORMAL)
303  append(e);
304  }
305  else
306  append(e);
307  }
308  Fraction gap = es->tick() - tick;
309  if (gap.isNotZero())
310  appendGap(gap);
311 
312  //
313  // connect ties
314  //
315  int n = size();
316  for (int i = 0; i < n; ++i) {
317  Element* e = at(i);
318  if (!e->isChord())
319  continue;
320  Chord* chord = toChord(e);
321  for (Note* n1 : chord->notes()) {
322  Tie* tie = n1->tieFor();
323  if (!tie)
324  continue;
325  for (int k = i+1; k < n; ++k) {
326  Element* ee = at(k);
327  if (!ee->isChord())
328  continue;
329  Chord* c2 = toChord(ee);
330  bool found = false;
331  for (Note* n2 : c2->notes()) {
332  if (n1->pitch() == n2->pitch()) {
333  tie->setEndNote(n2);
334  n2->setTieBack(tie);
335  found = true;
336  break;
337  }
338  }
339  if (!found)
340  qDebug("Tied note not found");
341  break;
342  }
343  }
344  }
345  }

351 static bool checkRest(Fraction& rest, Measure*& m, const Fraction& d)
352  {
353  if (rest.isZero()) {
354  if (m->nextMeasure()) {
355  m = m->nextMeasure();
356  rest = m->ticks();
357  }
358  else {
359  qWarning("premature end of measure list, rest %d/%d", d.numerator(), d.denominator());
360  return false;
361  }
362  }
363  return true;
364  }

372 Tuplet* TrackList::writeTuplet(Tuplet* parent, Tuplet* tuplet, Measure*& measure, Fraction& rest) const
373  {
374  Score* score = measure->score();
375  Tuplet* dt = tuplet->clone();
376  dt->setParent(measure);
377  Fraction du = tuplet->ticks();
378  if (du > rest) {
379  // we must split the tuplet
380  dt->setTicks(du * Fraction(1, 2));
381  dt->setBaseLen(tuplet->baseLen().shift(1));
382  }
383  if (parent)
384  parent->add(dt);
385 
386  for (DurationElement* e : tuplet->elements()) {
387  Fraction duration = e->globalTicks();
388  Tuplet* tt = dt;
389  Fraction ratio = Fraction(1, 1);
390  while (tt) {
391  ratio *= tt->ratio();
392  tt = tt->tuplet();
393  }
394 
395  bool firstpart = true;
396  while (duration > Fraction(0,1)) {
397  if (rest.isZero()) {
398  if (measure->nextMeasure()) {
399  measure = measure->nextMeasure();
400  rest = measure->ticks();
401  if (e != tuplet->elements().back()) {
402  // create second part of split tuplet
403  dt = dt->clone();
404  dt->setGenerated(true);
405  dt->setParent(measure);
406  Tuplet* pt = dt;
407  while (parent) {
408  Tuplet* tt1 = parent->clone();
409  tt1->setGenerated(true);
410  tt1->setParent(measure);
411  tt1->add(pt);
412  pt = tt1;
413  parent = parent->tuplet();
414  }
415  }
416  }
417  else {
418  qFatal("premature end of measure list in track %d, rest %d/%d",
419  _track, duration.numerator(), duration.denominator());
420  }
421  }
422  if (e->isChordRest()) {
423  Fraction dd = qMin(rest, duration) * ratio;
424  std::vector<TDuration> dl = toDurationList(dd, false);
425  for (const TDuration& k : dl) {
426  Segment* segment = measure->undoGetSegmentR(SegmentType::ChordRest, measure->ticks() - rest);
427  Fraction gd = k.fraction() / ratio;
428  ChordRest* cr = toChordRest(e->clone());
429  if (!firstpart)
430  cr->removeMarkings(true);
431  cr->setScore(score);
432  cr->setTrack(_track);
433  segment->add(cr);
434  cr->setTicks(k.fraction());
435  cr->setDurationType(k);
436  rest -= gd;
437  duration -= gd;
438 
439  if (cr->isChord()) {
440  for (Note* note : toChord(cr)->notes()) {
441  if (!duration.isZero() && !note->tieFor()) {
442  Tie* tie = new Tie(score);
443  tie->setGenerated(true);
444  note->add(tie);
445  }
446  }
447  }
448  dt->add(cr);
449  firstpart = false;
450  }
451  }
452  else if (e->isTuplet()) {
453  Tuplet* tt1 = toTuplet(e);
454  Tuplet* ttt = writeTuplet(dt, tt1, measure, rest);
455  dt = ttt->tuplet();
456  parent = dt->tuplet();
457  duration = Fraction();
458  }
459  firstpart = false;
460  }
461  }
462  return dt;
463  }

470 bool TrackList::write(Score* score, const Fraction& tick) const
471  {
472  if ((_track % VOICES) && size() == 1 && at(0)->isRest()) // don’t write rests in voice > 0
473  return true;
474  Measure* measure = score->tick2measure(tick);
475  Measure* m = measure;
476  Fraction remains = m->endTick() - tick;
477  Segment* segment = 0;
478 
479  for (Element* e : *this) {
480  if (e->isDurationElement()) {
481  Fraction duration = toDurationElement(e)->ticks();
482  if (!checkRest(remains, m, duration)) { // go to next measure, if necessary
483  MScore::setError(CORRUPTED_MEASURE);
484  return false;
485  }
486  if (duration > remains && e->isTuplet()) {
487  // experimental: allow tuplet split in the middle
488  if (duration != remains * 2) {
489  MScore::setError(CANNOT_SPLIT_TUPLET);
490  return false;
491  }
492  }
493  bool firstpart = true;
494  while (duration > Fraction(0,1)) {
495  if ((e->isRest() || e->isRepeatMeasure()) && (duration >= remains || e == back()) && (remains == m->ticks())) {
496  //
497  // handle full measure rest
498  //
499  Segment* seg = m->getSegmentR(SegmentType::ChordRest, m->ticks() - remains);
500  if ((_track % VOICES) == 0) {
501  // write only for voice 1
502  Rest* r = new Rest(score, TDuration::DurationType::V_MEASURE);
503  // ideally we should be using stretchedLen
504  // but this is not valid during rewrite when adding time signatures
505  // since the time signature has not been added yet
506  //Fraction stretchedLen = m->stretchedLen(staff);
507  //r->setTicks(stretchedLen);
508  r->setTicks(m->ticks());
509  r->setTrack(_track);
510  seg->add(r);
511  }
512  duration -= m->ticks();
513  remains.set(0, 1);
514  }
515  else if (e->isChordRest()) {
516  Fraction du = qMin(remains, duration);
517  std::vector<TDuration> dl = toDurationList(du, e->isChord());
518  if (dl.empty()) {
519  MScore::setError(CORRUPTED_MEASURE);
520  return false;
521  }
522  for (const TDuration& k : dl) {
523  segment = m->undoGetSegmentR(SegmentType::ChordRest, m->ticks() - remains);
524  ChordRest* cr = toChordRest(e->clone());
525  if (!firstpart)
526  cr->removeMarkings(true);
527  cr->setTrack(_track);
528  cr->setScore(score);
529  Fraction gd = k.fraction();
530  cr->setTicks(gd);
531  cr->setDurationType(k);
532 
533  segment->add(cr);
534  duration -= gd;
535  remains -= gd;
536 
537  if (cr->isChord()) {
538  if (!firstpart && toChord(cr)->tremolo() && toChord(cr)->tremolo()->twoNotes()) { // remove partial two-note tremolo
539  if (toChord(e)->tremolo()->chord1() == toChord(e))
540  toChord(cr)->tremolo()->setChords(toChord(cr),nullptr);
541  else
542  toChord(cr)->tremolo()->setChords(nullptr,toChord(cr));
543  Tremolo* tremoloPointer = toChord(cr)->tremolo();
544  toChord(cr)->setTremolo(nullptr);
545  delete tremoloPointer;
546  }
547  for (Note* note : toChord(cr)->notes()) {
548  if (!duration.isZero() && !note->tieFor()) {
549  Tie* tie = new Tie(score);
550  tie->setGenerated(true);
551  note->add(tie);
552  }
553  }
554  }
555  }
556  }
557  else if (e->isTuplet()) {
558  writeTuplet(0, toTuplet(e), m, remains);
559  duration = Fraction();
560  }
561  firstpart = false;
562  if (duration > Fraction(0,1)) {
563  if (!checkRest(remains, m, duration)) { // go to next measure, if necessary
564  MScore::setError(CORRUPTED_MEASURE);
565  return false;
566  }
567  }
568  }
569  }
570  else if (e->isBarLine()) {
571 // if (pos.numerator() == 0 && m) {
572 // BarLineType t = toBarLine(e)->barLineType();
573 // Measure* pm = m->prevMeasure();
574 //TODO if (pm)
575 // pm->setEndBarLineType(t,0);
576 // }
577  }
578  else if (e->isClef()) {
579  Segment* seg;
580  if (remains == m->ticks() && m->tick() > Fraction(0,1)) {
581  Measure* pm = m->prevMeasure();
582  seg = pm->getSegmentR(SegmentType::Clef, pm->ticks());
583  }
584  else if (remains != m->ticks())
585  seg = m->getSegmentR(SegmentType::Clef, m->ticks() - remains);
586  else
587  seg = m->getSegmentR(SegmentType::HeaderClef, Fraction(0,1));
588  Element* ne = e->clone();
589  ne->setScore(score);
590  ne->setTrack(_track);
591  seg->add(ne);
592  }
593  else {
594  if (!m)
595  break;
596  // add the element in its own segment;
597  // but KeySig has to be at start of (current) measure
598 
599  Segment* seg = m->getSegmentR(Segment::segmentType(e->type()), e->isKeySig() ? Fraction() : m->ticks() - remains);
600  Element* ne = e->clone();
601  ne->setScore(score);
602  ne->setTrack(_track);
603  seg->add(ne);
604  }
605  }
606  //
607  // connect ties from measure->first() to segment
608  //
609 
610  for (Segment* s = measure->first(); s; s = s->next1()) {
611  Element* e = s->element(_track);
612  if (!e || !e->isChord())
613  continue;
614  Chord* chord = toChord(e);
615  for (Note* n : chord->notes()) {
616  Tie* tie = n->tieFor();
617  if (!tie)
618  continue;
619  Note* nn = searchTieNote(n);
620  if (nn) {
621  tie->setEndNote(nn);
622  nn->setTieBack(tie);
623  }
624  }
625  if (s == segment)
626  break;
627  }
628  return true;
629  }

662  Spanner* ns = toSpanner(s->clone());

669  }

672  int sTrack = staffIdx * VOICES;

674  for (int track = sTrack; track < eTrack; ++track) {
675  TrackList* dl = new TrackList(this);
676  dl->setTrack(track);

678  tracks.append(dl);
679  }

687 bool ScoreRange::write(Score* score, const Fraction& tick) const
688  {
689  for (TrackList* dl : tracks) {
690  int track = dl->track();
691  if (!dl->write(score, tick))
692  return false;
693  if ((track % VOICES) == VOICES - 1) {
694  // clone staff if appropriate after all voices have been copied
695  int staffIdx = track / VOICES;
696  Staff* ostaff = score->staff(staffIdx);
697  const LinkedElements* linkedStaves = ostaff->links();
698  if (linkedStaves) {
699  for (auto le : *linkedStaves) {
700  Staff* nstaff = toStaff(le);
701  if (nstaff == ostaff)
702  continue;
703  Excerpt::cloneStaff2(ostaff, nstaff, tick, (tick + dl->ticks()));
704  }
705  }
706  }
707  ++track;
708  }
709  for (Spanner* s : spanner) {
710  s->setTick(s->tick() + tick);
711  if (s->isSlur()) {
712  Slur* slur = toSlur(s);
713  if (slur->startCR()->isGrace()) {
714  Chord* sc = slur->startChord();
715  int idx = sc->graceIndex();
716  Chord* dc = toChord(score->findCR(s->tick(), s->track()));
717  s->setStartElement(dc->graceNotes()[idx]);
718  }
719  else
720  s->setStartElement(0);
721  if (slur->endCR()->isGrace()) {
722  Chord* sc = slur->endChord();
723  int idx = sc->graceIndex();
724  Chord* dc = toChord(score->findCR(s->tick2(), s->track2()));
725  s->setEndElement(dc->graceNotes()[idx]);
726  }
727  else
728  s->setEndElement(0);
729  }
730  score->undoAddElement(s);
731  }
