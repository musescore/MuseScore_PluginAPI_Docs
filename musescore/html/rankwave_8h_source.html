107  qDebug("Rankwave: bad key");

109  }
110  Pipewave *P = _pipes + (n - _n0);
111  P->_sbit = _sbit;
112  if (! (P->_sdel || P->_p_p || P->_p_r)) {
113  P->_sdel |= _sbit;
114  P->_link = _list;

123  P->_sdel >>= 4;
