137 QString convertFromHtml(TextBase* t, const QString& ss)
138  {
139  QTextDocument doc;
140  doc.setHtml(ss.trimmed());
141 
142  QString s;
143  qreal size = t->size(); // textStyle().size();
144  QString family = t->family(); // textStyle().family();
145 
146  for (auto b = doc.firstBlock(); b.isValid() ; b = b.next()) {
147  if (!s.isEmpty())
148  s += "\n";
149  for (auto it = b.begin(); !it.atEnd(); ++it) {
150  QTextFragment f = it.fragment();
151  if (f.isValid()) {
152  QTextCharFormat tf = f.charFormat();
153  QFont font = tf.font();
154  qreal htmlSize = font.pointSizeF();
155  // html font sizes may have spatium adjustments; need to undo this
156  if (t->sizeIsSpatiumDependent())
157  htmlSize *= SPATIUM20 / t->spatium();
158  if (fabs(size - htmlSize) > 0.1) {
159  size = htmlSize;
160  s += QString("<font size=\"%1\"/>").arg(size);
161  }
162  if (family != font.family()) {
163  family = font.family();
164  s += QString("<font face=\"%1\"/>").arg(family);
165  }
166  if (font.bold())
167  s += "<b>";
168  if (font.italic())
169  s += "<i>";
170  if (font.underline())
171  s += "<u>";
172  s += f.text().toHtmlEscaped();
173  if (font.underline())
174  s += "</u>";
175  if (font.italic())
176  s += "</i>";
177  if (font.bold())
178  s += "</b>";
179  }
180  }
181  }
182 
183  s.replace(QChar(0xe10e), QString("<sym>accidentalNatural</sym>")); //natural
184  s.replace(QChar(0xe10c), QString("<sym>accidentalSharp</sym>")); // sharp
185  s.replace(QChar(0xe10d), QString("<sym>accidentalFlat</sym>")); // flat
186  s.replace(QChar(0xe104), QString("<sym>metNoteHalfUp</sym>")), // note2_Sym
187  s.replace(QChar(0xe105), QString("<sym>metNoteQuarterUp</sym>")); // note4_Sym
188  s.replace(QChar(0xe106), QString("<sym>metNote8thUp</sym>")); // note8_Sym
189  s.replace(QChar(0xe107), QString("<sym>metNote16thUp</sym>")); // note16_Sym
190  s.replace(QChar(0xe108), QString("<sym>metNote32ndUp</sym>")); // note32_Sym

600 static void readNote(Note* note, XmlReader& e)
601  {
602  e.hasAccidental = false; // used for userAccidental backward compatibility
603 
604  note->setTpc1(Tpc::TPC_INVALID);
605  note->setTpc2(Tpc::TPC_INVALID);
606 
607  if (e.hasAttribute("pitch")) // obsolete
608  note->setPitch(e.intAttribute("pitch"));
609  if (e.hasAttribute("tpc")) // obsolete
610  note->setTpc1(e.intAttribute("tpc"));
611 
612  while (e.readNextStartElement()) {
613  const QStringRef& tag(e.name());
614  if (tag == "Accidental") {
615  // on older scores, a note could have both a <userAccidental> tag and an <Accidental> tag
616  // if a userAccidental has some other property set (like for instance offset)
617  Accidental* a;
618  if (e.hasAccidental) // if the other tag has already been read,
619  a = note->accidental(); // re-use the accidental it constructed
620  else
621  a = new Accidental(note->score());
622  // the accidental needs to know the properties of the
623  // track it belongs to (??)
624  a->setTrack(note->track());
625  readAccidental(a, e);
626  if (!e.hasAccidental) // only the new accidental, if it has been added previously
627  note->add(a);
628  e.hasAccidental = true; // we now have an accidental
629  }
630  else if (tag == "Text") {
631  Fingering* f = new Fingering(note->score());
632  readFingering114(e, f);
633  note->add(f);
634  }
635  else if (tag == "onTimeType") {
636  if (e.readElementText() == "offset")
637  note->setOnTimeType(2);
638  else
639  note->setOnTimeType(1);
640  }
641  else if (tag == "offTimeType") {
642  if (e.readElementText() == "offset")
643  note->setOffTimeType(2);
644  else
645  note->setOffTimeType(1);
