345 void Score::linkMeasures(Score* score)
346  {
347  MeasureBase *mbMaster = score->first();
348  for (MeasureBase* mb = first(); mb; mb = mb->next()) {
349  if (!mb->isMeasure())
350  continue;
351  while (mbMaster && !mbMaster->isMeasure())
352  mbMaster = mbMaster->next();
353  if (!mbMaster) {
354  qDebug("Measures in MasterScore and Score are not in sync.");
355  break;
356  }
357  mb->linkTo(mbMaster);
358  mbMaster = mbMaster->next();
359  }

398 Score::FileError MasterScore::read302(XmlReader& e)
399  {
400  bool top = true;
401  while (e.readNextStartElement()) {
402  const QStringRef& tag(e.name());
403  if (tag == "programVersion") {
404  setMscoreVersion(e.readElementText());
405  parseVersion(mscoreVersion());
406  }
407  else if (tag == "programRevision")
408  setMscoreRevision(e.readIntHex());
409  else if (tag == "Score") {
410  MasterScore* score;
411  if (top) {
412  score = this;
413  top = false;
414  }
415  else {
416  score = new MasterScore();
417  score->setMscVersion(mscVersion());
418  addMovement(score);
419  }
420  if (!score->read(e)) {
421  if (e.error() == QXmlStreamReader::CustomError)
422  return FileError::FILE_CRITICALLY_CORRUPTED;
423  return FileError::FILE_BAD_FORMAT;
424  }
425  }
426  else if (tag == "Revision") {
427  Revision* revision = new Revision;
428  revision->read(e);
429  revisions()->add(revision);
430  }
431  }
432  return FileError::FILE_NO_ERROR;
433  }
