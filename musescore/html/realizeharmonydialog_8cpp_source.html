59 void RealizeHarmonyDialog::setChordList(QList<Harmony*> hlist)
60  {
61  static const QStringList header = { "ID", "Name", "Intervals", "Notes" };
62  QString s; //chord label string
63  int rows = hlist.size();
64 
65  chordTable->setRowCount(rows);
66  chordTable->setColumnCount(header.size());
67  chordTable->setHorizontalHeaderLabels(header);
68  for (int i = 0; i < rows; ++i) {
69  Harmony* h = hlist.at(i);
70  if (!h->isRealizable())
71  continue;
72 
73  s += h->harmonyName() + " ";
74  QString intervals;
75  QString noteNames;
76  int rootTpc;
77 
78  //adjust for nashville function
79  if (h->harmonyType() == HarmonyType::NASHVILLE)
80  rootTpc = function2Tpc(h->hFunction(), h->staff()->key(h->tick()));
81  else
82  rootTpc = h->rootTpc();
83 
84  noteNames = tpc2name(rootTpc, NoteSpellingType::STANDARD, NoteCaseType::AUTO);
85  RealizedHarmony::PitchMap map = h->getRealizedHarmony().notes();
86  for (int pitch : map.keys()) {
87  intervals += QString::number((pitch - tpc2pitch(rootTpc)) % 128 % 12) + " ";
88  }
89  for (int tpc : map.values()) {
90  noteNames += ", " + tpc2name(tpc, NoteSpellingType::STANDARD, NoteCaseType::AUTO);
91  }
92 
93  chordTable->setItem(i, 0, new QTableWidgetItem(QString::number(h->id())));
94  chordTable->setItem(i, 1, new QTableWidgetItem(h->harmonyName()));
95  chordTable->setItem(i, 2, new QTableWidgetItem(intervals));
96  chordTable->setItem(i, 3, new QTableWidgetItem(noteNames));
97  }
98  chordLabel->setText(s);
99 
100  //set table uneditable again
101  chordTable->setEditTriggers(QAbstractItemView::NoEditTriggers);
102  }
