59 QString RecognitionProccessDialog::formatStep(const IAvsOmrRecognizer::Step& step) const
60  {
61  switch (step.type) {
62  case IAvsOmrRecognizer::Step::Undefined:
63  return QString();
64  case IAvsOmrRecognizer::Step::PrepareStart:
65  return QObject::tr("Preparing…");
66  case IAvsOmrRecognizer::Step::PrepareFinish:
67  if (step.success())
68  return QObject::tr("Successfully prepared");
69 
70  return QObject::tr("Failed to prepare")
71  + "\n" + step.error.formatedText()
72  + "\n" + step.error.supportHint();
73  case IAvsOmrRecognizer::Step::ProcessingStart:
74  return QObject::tr("Processing…");
75  case IAvsOmrRecognizer::Step::ProcessingFinish:
76  if (step.success())
77  return QObject::tr("Successfully processed");
78 
79  return QObject::tr("Failed to process")
80  + "\n" + step.error.formatedText()
81  + "\n" + step.error.supportHint();
82  case IAvsOmrRecognizer::Step::LoadStart:
83  return QObject::tr("Loading…");
84  case IAvsOmrRecognizer::Step::LoadFinish:
85  if (step.success())
86  return QObject::tr("Successfully loaded");
87 
88  return QObject::tr("Failed to load")
89  + "\n" + step.error.formatedText()
90  + "\n" + step.error.supportHint();
91  }
92  return QString();
93  }
