116 void ScriptRecorderWidget::on_recordButton_clicked()
117  {
118  const bool startRecord = !_recorder.isRecording();
119  _ui->changeFolderButton->setEnabled(!startRecord);
120  _ui->scriptNameEdit->setEnabled(!startRecord);
121  _ui->replayButton->setEnabled(false);
122  _ui->recordButton->setText(startRecord ? "Stop recording" : "Start recording");
123 
124  if (startRecord) {
125  _recorder.setFile(scriptFileName());
126  _recorder.context().setCwd(scriptDir());
127  _recorder.setRecording(true);
128  _recorder.recordInitState();
129  }
130  else {
131  QFileInfo scriptInfo(scriptFileName());
132  QString scoreName(scriptInfo.completeBaseName() + ".mscx");
133  _recorder.recordScoreTest(scoreName);
134  _recorder.close();
135  }
136 
137  if (!_recorder.isRecording())
138  updateActions();
139  }
