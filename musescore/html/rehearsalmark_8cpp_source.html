44 void RehearsalMark::layout()
45  {
46  TextBase::layout();
47 
48  Segment* s = segment();
49  if (s) {
50  if (s->rtick().isZero()) {
51  // first CR of measure, alignment is hcenter or right (the usual cases)
52  // align with barline, point just after header, or start of measure depending on context
53 
54  Measure* m = s->measure();
55  Segment* header = s->prev(); // possibly just a start repeat
56  qreal measureX = -s->x();
57  Segment* repeat = m->findSegmentR(SegmentType::StartRepeatBarLine, Fraction(0, 1));
58  qreal barlineX = repeat ? repeat->x() - s->x() : measureX;
59  System* sys = m->system();
60  bool systemFirst = (sys && m->isFirstInSystem());
61 
62  if (!header || repeat || !systemFirst) {
63  // no header, or header with repeat, or header mid-system - align with barline
64  rxpos() = barlineX;
65  }
66  else {
67  // header at start of system
68  // align to a point just after the header
69  Element* e = header->element(track()); // TODO: firstVisibleStaff
70  qreal w = e ? e->width() : header->width();
71  rxpos() = header->x() + w - s->x();
72 
73  // special case for right aligned rehearsal marks at start of system
74  // left align with start of measure if that is further left
75  if (align() & Align::RIGHT)
76  rxpos() = qMin(rpos().x(), measureX + width());
77  }
78  }
79  autoplaceSegmentElement();
80  }
81  }
