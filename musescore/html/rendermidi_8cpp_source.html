170  for (Staff* staff : *e->part()->staves()) {

175  }

179  for (int voice = 0; voice < VOICES; ++voice) {

181  if (an.isEmpty())

184  int a = staff->part()->instrument(s->tick())->channelIdx(an);
185  if (a != -1)

187  }
188  }
189  }

191  for (auto it = spanner().cbegin(); it != spanner().cend(); ++it) {
192  Spanner* spanner = (*it).second;
193  if (!spanner->isVolta())
194  continue;

250 static void playNote(EventMap* events, const Note* note, int channel, int pitch,
251  int velo, int onTime, int offTime, int staffIdx)
252  {
253  if (!note->play())
254  return;
255  velo = note->customizeVelocity(velo);
256  NPlayEvent ev(ME_NOTEON, channel, pitch, velo);
257  ev.setOriginatingStaff(staffIdx);
258  ev.setTuning(note->tuning());
259  ev.setNote(note);
260  if (offTime < onTime)
261  offTime = onTime;
262  events->insert(std::pair<int, NPlayEvent>(onTime, ev));
263  // adds portamento for continuous glissando
264  for (Spanner* spanner : note->spannerFor()) {
265  if (spanner->type() == ElementType::GLISSANDO) {
266  Glissando *glissando = toGlissando(spanner);
267  if (glissando->glissandoStyle() == GlissandoStyle::PORTAMENTO) {
268  Note* nextNote = toNote(spanner->endElement());
269  double pitchDelta = (static_cast<double>(nextNote->pitch()) - pitch) * 50.0;
270  double timeDelta = static_cast<double>(offTime - onTime);
271  if (pitchDelta != 0.0 && timeDelta != 0.0) {
272  double timeStep = std::abs(timeDelta / pitchDelta * 20.0);
273  double t = 0.0;
274  QList<int> onTimes;
275  EaseInOut easeInOut(static_cast<qreal>(glissando->easeIn()) / 100.0,
276  static_cast<qreal>(glissando->easeOut()) / 100.0);
277  easeInOut.timeList(static_cast<int>((timeDelta + timeStep * 0.5) / timeStep), int(timeDelta), &onTimes);
278  double nTimes = static_cast<double>(onTimes.size() - 1);
279  for (double time : onTimes) {
280  int p = static_cast<int>((t / nTimes) * pitchDelta);
281  int timeStamp = std::min(onTime + int(time), offTime - 1);
282  int midiPitch = (p * 16384) / 1200 + 8192;
283  NPlayEvent evb(ME_PITCHBEND, channel, midiPitch % 128, midiPitch / 128);
284  evb.setOriginatingStaff(staffIdx);
285  events->insert(std::pair<int, NPlayEvent>(timeStamp, evb));
286  t += 1.0;
287  }
288  ev.setVelo(0);
289  events->insert(std::pair<int, NPlayEvent>(offTime, ev));
290  NPlayEvent evb(ME_PITCHBEND, channel, 0, 64); // 0:64 is 8192 - no pitch bend
291  evb.setOriginatingStaff(staffIdx);
292  events->insert(std::pair<int, NPlayEvent>(offTime, evb));
293  return;
294  }
295  }
296  }
297  }
298 
299  ev.setVelo(0);
300  events->insert(std::pair<int, NPlayEvent>(offTime, ev));
301  }

307 static void collectNote(EventMap* events, int channel, const Note* note, qreal velocityMultiplier, int tickOffset, Staff* staff, SndConfig config)
308  {
309  if (!note->play() || note->hidden()) // do not play overlapping notes
310  return;
311  Chord* chord = note->chord();
312 
313  int staffIdx = staff->idx();
314  int ticks;
315  int tieLen = 0;
316  if (chord->isGrace()) {
317  Q_ASSERT( !graceNotesMerged(chord)); // this function should not be called on a grace note if grace notes are merged
318  chord = toChord(chord->parent());
319  }
320 
321  ticks = chord->actualTicks().ticks(); // ticks of the actual note
322  // calculate additional length due to ties forward
323  // taking NoteEvent length adjustments into account
324  // but stopping at any note with multiple NoteEvents
325  // and processing those notes recursively
326  if (note->tieFor()) {
327  Note* n = note->tieFor()->endNote();
328  while (n) {
329  NoteEventList nel = n->playEvents();
330  if (nel.size() == 1 && !isGlissandoFor(n)) {
331  // add value of this note to main note
332  // if we wish to suppress first note of ornament,
333  // then do this regardless of number of NoteEvents
334  tieLen += (n->chord()->actualTicks().ticks() * (nel[0].len())) / 1000;
335  }
336  else {
337  // recurse
338  collectNote(events, channel, n, velocityMultiplier, tickOffset, staff, config);
339  break;
340  }
341  if (n->tieFor() && n != n->tieFor()->endNote())
342  n = n->tieFor()->endNote();
343  else
344  break;
345  }
346  }
347 
348  int tick1 = chord->tick().ticks() + tickOffset;
349  bool tieFor = note->tieFor();
350  bool tieBack = note->tieBack();
351 
352  NoteEventList nel = note->playEvents();
353  int nels = nel.size();
354  for (int i = 0, pitch = note->ppitch(); i < nels; ++i) {
355  const NoteEvent& e = nel[i]; // we make an explicit const ref, not a const copy. no need to copy as we won't change the original object.
356 
357  // skip if note has a tie into it and only one NoteEvent
358  // its length was already added to previous note
359  // if we wish to suppress first note of ornament
360  // then change "nels == 1" to "i == 0", and change "break" to "continue"
361  if (tieBack && nels == 1 && !isGlissandoFor(note))
362  break;
363  int p = pitch + e.pitch();
364  if (p < 0)
365  p = 0;
366  else if (p > 127)
367  p = 127;
368  int on = tick1 + (ticks * e.ontime())/1000;
369  int off = on + (ticks * e.len())/1000 - 1;
370  if (tieFor && i == nels - 1)
371  off += tieLen;
372 
373  // Get the velocity used for this note from the staff
374  // This allows correct playback of tremolos even without SND enabled.
375  int velo;
376  Fraction nonUnwoundTick = Fraction::fromTicks(on - tickOffset);
377  if (config.useSND) {
378  switch (config.method) {
379  case DynamicsRenderMethod::FIXED_MAX:
380  velo = 127;
381  break;
382  case DynamicsRenderMethod::SEG_START:
383  default:
384  velo = staff->velocities().val(nonUnwoundTick);
385  break;
386  }
387  }
388  else {
389  velo = staff->velocities().val(nonUnwoundTick);
390  }
391 
392  velo *= velocityMultiplier;
393  playNote(events, note, channel, p, qBound(1, velo, 127), on, off, staffIdx);
394  }
395 
396  // Single-note dynamics
397  // Find any changes, and apply events
398  if (config.useSND) {
399  ChangeMap& veloEvents = staff->velocities();
400  ChangeMap& multEvents = staff->velocityMultiplications();
401  Fraction stick = chord->tick();
402  Fraction etick = stick + chord->ticks();
403  auto changes = veloEvents.changesInRange(stick, etick);
404  auto multChanges = multEvents.changesInRange(stick, etick);
405 
406  std::map<int, int> velocityMap;
407  for (auto& change : changes) {
408  int lastVal = -1;
409  int endPoint = change.second.ticks();
410  for (int t = change.first.ticks(); t <= endPoint; t++) {
411  int velo = veloEvents.val(Fraction::fromTicks(t));
412  if (velo == lastVal)
413  continue;
414  lastVal = velo;
415 
416  velocityMap[t] = velo;
417  }
418  }
419 
420 
421  qreal CONVERSION_FACTOR = MidiRenderer::ARTICULATION_CONV_FACTOR;
422  for (auto& change : multChanges) {
423  // Ignore fix events: they are available as cached ramp starts
424  // and considering them ends up with multiplying twice effectively
425  if (change.first == change.second)
426  continue;
427 
428  int lastVal = MidiRenderer::ARTICULATION_CONV_FACTOR;
429  int endPoint = change.second.ticks();
430  int lastVelocity = velocityMap.upper_bound(change.first.ticks())->second;
431  for (int t = change.first.ticks(); t <= endPoint; t++) {
432  int mult = multEvents.val(Fraction::fromTicks(t));
433  if (mult == lastVal || mult == CONVERSION_FACTOR)
434  continue;
435  lastVal = mult;
436 
437  qreal realMult = mult / CONVERSION_FACTOR;
438  if (velocityMap.find(t) != velocityMap.end()) {
439  lastVelocity = velocityMap[t];
440  velocityMap[t] *= realMult;
441  }
442  else {
443  velocityMap[t] = lastVelocity * realMult;
444  }
445  }
446  }
447 
448 
449  for (auto point = velocityMap.cbegin(); point != velocityMap.cend(); ++point) {
450  // NOTE:JT if we ever want to use poly aftertouch instead of CC, this is where we want to
451  // be using it. Instead of ME_CONTROLLER, use ME_POLYAFTER (but duplicate for each note in chord)
452  NPlayEvent event = NPlayEvent(ME_CONTROLLER, channel, config.controller, qBound(0, point->second, 127));
453  event.setOriginatingStaff(staffIdx);
454  events->insert(std::make_pair(point->first + tickOffset, event));
455  }
456  }
457 
458  // Bends
459  for (Element* e : note->el()) {
460  if (e == 0 || e->type() != ElementType::BEND)
461  continue;
462  Bend* bend = toBend(e);
463  if (!bend->playBend())
464  break;
465  const QList<PitchValue>& points = bend->points();
466  int pitchSize = points.size();
467 
468  double noteLen = note->playTicks();
469  int lastPointTick = tick1;
470  for (int pitchIndex = 0; pitchIndex < pitchSize-1; pitchIndex++) {
471  PitchValue pitchValue = points[pitchIndex];
472  PitchValue nextPitch = points[pitchIndex+1];
473  int nextPointTick = tick1 + nextPitch.time / 60.0 * noteLen;
474  int pitch = pitchValue.pitch;
475 
476  if (pitchIndex == 0 && (pitch == nextPitch.pitch)) {
477  int midiPitch = (pitch * 16384) / 1200 + 8192;
478  int msb = midiPitch / 128;
479  int lsb = midiPitch % 128;
480  NPlayEvent ev(ME_PITCHBEND, channel, lsb, msb);
481  ev.setOriginatingStaff(staffIdx);
482  events->insert(std::pair<int, NPlayEvent>(lastPointTick, ev));
483  lastPointTick = nextPointTick;
484  continue;
485  }
486  if (pitch == nextPitch.pitch && !(pitchIndex == 0 && pitch != 0)) {
487  lastPointTick = nextPointTick;
488  continue;
489  }
490 
491  double pitchDelta = nextPitch.pitch - pitch;
492  double tickDelta = nextPitch.time - pitchValue.time;
493  /* B
494  /. pitch is 1/100 semitones
495  bend / . pitchDelta time is in noteDuration/60
496  / . midi pitch is 12/16384 semitones
497  A....
498  tickDelta */
499  for (int i = lastPointTick; i <= nextPointTick; i += 16) {
500  double dx = ((i-lastPointTick) * 60) / noteLen;
501  int p = pitch + dx * pitchDelta / tickDelta;
502 
503  // We don't support negative pitch, but Midi does. Let's center by adding 8192.
504  int midiPitch = (p * 16384) / 1200 + 8192;
505  // Representing pitch as two bytes
506  int msb = midiPitch / 128;
507  int lsb = midiPitch % 128;
508  NPlayEvent ev(ME_PITCHBEND, channel, lsb, msb);
509  ev.setOriginatingStaff(staffIdx);
510  events->insert(std::pair<int, NPlayEvent>(i, ev));
511  }
512  lastPointTick = nextPointTick;
513  }
514  NPlayEvent ev(ME_PITCHBEND, channel, 0, 64); // 0:64 is 8192 - no pitch bend
515  ev.setOriginatingStaff(staffIdx);
516  events->insert(std::pair<int, NPlayEvent>(tick1+int(noteLen), ev));
517  }
518  }

548  {

566  NamedEventList* nel = instr->midiAction(ma, ca.channel);

570  event.setChannel(channel);

572  e1.setOriginatingStaff(firstStaffIdx);
573  if (e1.dataA() == CTRL_PROGRAM)
574  events->insert(std::pair<int, NPlayEvent>(tick.ticks()-1, e1));

898 //---------------------------------------------------------
899 // updateVelo
900 // calculate velocity for all notes
901 //---------------------------------------------------------
902 
903 void Score::updateVelo()
904  {
905  //
906  // collect Dynamics
907  //
908  if (!firstMeasure())
909  return;
910 
911  for (Staff* st : qAsConst(_staves)) {
912  st->velocities().clear();
913  st->velocityMultiplications().clear();
914  }
915  for (int staffIdx = 0; staffIdx < nstaves(); ++staffIdx) {
916  Staff* st = staff(staffIdx);
917  ChangeMap& velo = st->velocities();
918  ChangeMap& mult = st->velocityMultiplications();
919  Part* prt = st->part();
920  int partStaves = prt->nstaves();
921  int partStaff = Score::staffIdx(prt);
922 
923  for (Segment* s = firstMeasure()->first(); s; s = s->next1()) {
924  Fraction tick = s->tick();
925  for (const Element* e : s->annotations()) {
926  if (e->staffIdx() != staffIdx)
927  continue;
928  if (e->type() != ElementType::DYNAMIC)
929  continue;
930  const Dynamic* d = toDynamic(e);
931  int v = d->velocity();
932 
933  // treat an invalid dynamic as no change, i.e. a dynamic set to 0
934  if (v < 1)
935  continue;
936 
937  v = qBound(1, v, 127); // illegal values
938 
939  // If a dynamic has 'velocity change' update its ending
940  int change = d->changeInVelocity();
941  ChangeDirection direction = ChangeDirection::INCREASING;
942  if (change < 0) {
943  direction = ChangeDirection::DECREASING;
944  }
945 
946  int dStaffIdx = d->staffIdx();
947  switch(d->dynRange()) {
948  case Dynamic::Range::STAFF:
949  if (dStaffIdx == staffIdx) {
950  velo.addFixed(tick, v);
951  if (change != 0) {
952  Fraction etick = tick + d->velocityChangeLength();
953  ChangeMethod method = ChangeMethod::NORMAL;
954  velo.addRamp(tick, etick, change, method, direction);
955  }
956  }
957  break;
958  case Dynamic::Range::PART:
959  if (dStaffIdx >= partStaff && dStaffIdx < partStaff+partStaves) {
960  for (int i = partStaff; i < partStaff+partStaves; ++i) {
961  ChangeMap& stVelo = staff(i)->velocities();
962  stVelo.addFixed(tick, v);
963  if (change != 0) {
964  Fraction etick = tick + d->velocityChangeLength();
965  ChangeMethod method = ChangeMethod::NORMAL;
966  stVelo.addRamp(tick, etick, change, method, direction);
967  }
968  }
969  }
970  break;
971  case Dynamic::Range::SYSTEM:
972  for (int i = 0; i < nstaves(); ++i) {
973  ChangeMap& stVelo = staff(i)->velocities();
974  stVelo.addFixed(tick, v);
975  if (change != 0) {
976  Fraction etick = tick + d->velocityChangeLength();
977  ChangeMethod method = ChangeMethod::NORMAL;
978  stVelo.addRamp(tick, etick, change, method, direction);
979  }
980  }
981  break;
982  }
983  }
984  if (s->isChordRestType()) {
985  for (int i = staffIdx * VOICES; i < (staffIdx + 1) * VOICES; ++i) {
986  Element* el = s->element(i);
987  if (!el || !el->isChord())
988  continue;
989 
990  Chord* chord = toChord(el);
991  Instrument* instr = chord->part()->instrument();
992 
993  qreal veloMultiplier = 1;
994  for (Articulation* a : chord->articulations()) {
995  if (a->playArticulation()) {
996  veloMultiplier *= instr->getVelocityMultiplier(a->articulationName());
997  }
998  }
999 
1000  if (veloMultiplier == 1.0)
1001  continue;
1002 
1003  // TODO this should be a (configurable?) constant somewhere
1004  static Fraction ARTICULATION_CHANGE_TIME_MAX = Fraction(1, 16);
1005  Fraction ARTICULATION_CHANGE_TIME = qMin(s->ticks(), ARTICULATION_CHANGE_TIME_MAX);
1006  int start = veloMultiplier * MidiRenderer::ARTICULATION_CONV_FACTOR;
1007  int change = (veloMultiplier - 1) * MidiRenderer::ARTICULATION_CONV_FACTOR;
1008  mult.addFixed(chord->tick(), start);
1009  mult.addRamp(chord->tick(), chord->tick() + ARTICULATION_CHANGE_TIME, change, ChangeMethod::NORMAL, ChangeDirection::DECREASING);
1010  }
1011  }
1012  }
1013  for (const auto& sp : _spanner.map()) {
1014  Spanner* s = sp.second;
1015  if (s->type() != ElementType::HAIRPIN || sp.second->staffIdx() != staffIdx)
1016  continue;
1017  Hairpin* h = toHairpin(s);
1018  updateHairpin(h);
1019  }
1020  }
1021 
1022  for (Staff* st : qAsConst(_staves)) {
1023  st->velocities().cleanup();
1024  st->velocityMultiplications().cleanup();
1025  }
1026 
1027  for (auto it = spanner().cbegin(); it != spanner().cend(); ++it) {
1028  Spanner* spanner = (*it).second;
1029  if (!spanner->isVolta())
1030  continue;
1031  Volta* volta = toVolta(spanner);
1032  volta->setVelocity();
1033  }
1034  }

1064 void MidiRenderer::renderSpanners(const Chunk& chunk, EventMap* events)
1065  {

1067  const int tick1 = chunk.tick1();
1068  const int tick2 = chunk.tick2();
1069 
1070  std::map<int, std::vector<std::pair<int, std::pair<bool, int> > > > channelPedalEvents;
1071  for (const auto& sp : score->spannerMap().map()) {
1072  Spanner* s = sp.second;
1073 
1074  int staff = s->staffIdx();
1075  int idx = s->staff()->channel(s->tick(), 0);
1076  int channel = s->part()->instrument(s->tick())->channel(idx)->channel();
1077 
1078  if (s->isPedal() || s->isLetRing()) {
1079  channelPedalEvents.insert({channel, std::vector<std::pair<int, std::pair<bool, int> > >()});
1080  std::vector<std::pair<int, std::pair<bool, int> > > pedalEventList = channelPedalEvents.at(channel);
1081  std::pair<int, std::pair<bool, int> > lastEvent;
1082 
1083  if (!pedalEventList.empty())
1084  lastEvent = pedalEventList.back();
1085  else
1086  lastEvent = std::pair<int, std::pair<bool, int> >(0, std::pair<bool, int>(true, staff));
1087 
1088  int st = s->tick().ticks();
1089  if (st >= tick1 && st < tick2) {
1090  // Handle "overlapping" pedal segments (usual case for connected pedal line)
1091  if (lastEvent.second.first == false && lastEvent.first >= (st + tickOffset + 2)) {
1092  channelPedalEvents.at(channel).pop_back();
1093  channelPedalEvents.at(channel).push_back(std::pair<int, std::pair<bool, int> >(st + tickOffset + (2 - MScore::pedalEventsMinTicks), std::pair<bool, int>(false, staff)));
1094  }
1095  int a = st + tickOffset + 2;
1096  channelPedalEvents.at(channel).push_back(std::pair<int, std::pair<bool, int> >(a, std::pair<bool, int>(true, staff)));
1097  }
1098  if (s->tick2().ticks() >= tick1 && s->tick2().ticks() <= tick2) {
1099  int t = s->tick2().ticks() + tickOffset + (2 - MScore::pedalEventsMinTicks);
1100  const RepeatSegment& lastRepeat = *score->repeatList().back();
1101  if (t > lastRepeat.utick + lastRepeat.len())
1102  t = lastRepeat.utick + lastRepeat.len();
1103  channelPedalEvents.at(channel).push_back(std::pair<int, std::pair<bool, int> >(t, std::pair<bool, int>(false, staff)));
1104  }
1105  }
1106  else if (s->isVibrato()) {
1107  int stick = s->tick().ticks();
1108  int etick = s->tick2().ticks();
1109  if (stick >= tick2 || etick < tick1)
1110  continue;
1111 
1112  if (stick < tick1)
1113  stick = tick1;
1114  if (etick > tick2)
1115  etick = tick2;
1116 
1117  // from start to end of trill, send bend events at regular interval
1118  Vibrato* t = toVibrato(s);
1119  // guitar vibrato, up only
1120  int spitch = 0; // 1/8 (100 is a semitone)
1121  int epitch = 12;
1122  if (t->vibratoType() == Vibrato::Type::GUITAR_VIBRATO_WIDE) {
1123  spitch = 0; // 1/4
1124  epitch = 25;
1125  }
1126  // vibrato with whammy bar up and down
1127  else if (t->vibratoType() == Vibrato::Type::VIBRATO_SAWTOOTH_WIDE) {
1128  spitch = 25; // 1/16
1129  epitch = -25;
1130  }
1131  else if (t->vibratoType() == Vibrato::Type::VIBRATO_SAWTOOTH) {
1132  spitch = 12;
1133  epitch = -12;
1134  }
1135 
1136  int j = 0;
1137  int delta = MScore::division / 8; // 1/8 note
1138  int lastPointTick = stick;
1139  while (lastPointTick < etick) {
1140  int pitch = (j % 4 < 2) ? spitch : epitch;
1141  int nextPitch = ((j+1) % 4 < 2) ? spitch : epitch;
1142  int nextPointTick = lastPointTick + delta;
1143  for (int i = lastPointTick; i <= nextPointTick; i += 16) {
1144  double dx = ((i - lastPointTick) * 60) / delta;
1145  int p = pitch + dx * (nextPitch - pitch) / delta;
1146  int midiPitch = (p * 16384) / 1200 + 8192;
1147  int msb = midiPitch / 128;
1148  int lsb = midiPitch % 128;
1149  NPlayEvent ev(ME_PITCHBEND, channel, lsb, msb);
1150  ev.setOriginatingStaff(staff);
1151  events->insert(std::pair<int, NPlayEvent>(i + tickOffset, ev));
1152  }
1153  lastPointTick = nextPointTick;
1154  j++;
1155  }
1156  NPlayEvent ev(ME_PITCHBEND, channel, 0, 64); // no pitch bend
1157  ev.setOriginatingStaff(staff);
1158  events->insert(std::pair<int, NPlayEvent>(etick + tickOffset, ev));
1159  }
1160  else
1161  continue;
1162  }
1163 
1164  for (const auto& pedalEvents : channelPedalEvents) {
1165  int channel = pedalEvents.first;
1166  for (const auto& pe : pedalEvents.second) {
1167  NPlayEvent event;
1168  if (pe.second.first == true)
1169  event = NPlayEvent(ME_CONTROLLER, channel, CTRL_SUSTAIN, 127);
1170  else
1171  event = NPlayEvent(ME_CONTROLLER, channel, CTRL_SUSTAIN, 0);
1172  event.setOriginatingStaff(pe.second.second);
1173  events->insert(std::pair<int,NPlayEvent>(pe.first, event));
1174  }
1175  }
1176  }

1421 //---------------------------------------------------------
1422 // articulationExcursion -- an articulation such as a trill, or modant consists of several notes
1423 // played in succession. The pitch offsets of each such note in the sequence can be represented either
1424 // as a number of steps in the diatonic scale, or in half steps as on a piano keyboard.
1425 // this function, articulationExcursion, takes deltastep indicating the number of steps in the
1426 // diatonic scale, and calculates (and returns) the number of half steps, taking several things into account.
1427 // E.g., the key signature, a trill from e to f, is to be understood as a trill between E and F# if we are
1428 // in the key of G.
1429 // E.g., if previously (looking backward in time) in the same measure there is another note on the same
1430 // staff line/space, and that note has an accidental (sharp,flat,natural,etc), then we want to match that
1431 // tone exactly.
1432 // E.g., If there are multiple notes on the same line/space, then we only consider the most
1433 // recent one, but avoid looking forward in time after the current note.
1434 // E.g., Also if there is an accidental // on a note one (or more) octaves above or below we
1435 // observe its accidental as well.
1436 // E.g., Still another case is that if two staves are involved (such as a glissando between two
1437 // notes on different staves) then we have to search both staves for the most recent accidental.
1438 //
1439 // noteL is the note to measure the deltastep from, i.e., ornaments are w.r.t. this note
1440 // noteR is the note to search backward from to find accidentals.
1441 // for ornament calculation noteL and noteR are the same, but for glissando they are
1442 // the start end end note of glissando.
1443 // deltastep is the desired number of diatonic steps between the base note and this articulation step.
1444 //---------------------------------------------------------
1445 
1446 int articulationExcursion(Note *noteL, Note *noteR, int deltastep)
1447  {
1448  if (0 == deltastep)
1449  return 0;
1450  Chord *chordL = noteL->chord();
1451  Chord *chordR = noteR->chord();
1452  int epitchL = noteL->epitch();
1453  Fraction tickL = chordL->tick();
1454  // we cannot use staffL = chord->staff() because that won't correspond to the noteL->line()
1455  // in the case the user has pressed Shift-Cmd->Up or Shift-Cmd-Down.
1456  // Therefore we have to take staffMove() into account using vStaffIdx().
1457  Staff * staffL = noteL->score()->staff(chordL->vStaffIdx());
1458  ClefType clefL = staffL->clef(tickL);
1459  // line represents the ledger line of the staff. 0 is the top line, 1, is the space between the top 2 lines,
1460  // ... 8 is the bottom line.
1461  int lineL = noteL->line();
1462  // we use line - deltastep, because lines are oriented from top to bottom, while step is oriented from bottom to top.
1463  int lineL2 = lineL - deltastep;
1464  Measure* measureR = chordR->segment()->measure();
1465 
1466  Segment* segment = noteL->chord()->segment();
1467  int lineR2 = convertLine(lineL2, noteL, noteR);
1468  // is there another note in this segment on the same line?
1469  // if so, use its pitch exactly.
1470  int halfsteps = 0;
1471  int staffIdx = noteL->chord()->staff()->idx(); // cannot use staffL->idx() because of staffMove()
1472  int startTrack = staffIdx * VOICES;
1473  int endTrack = startTrack + VOICES;
1474  bool done = false;
1475  for (int track = startTrack; track < endTrack; ++track) {
1476  Element *e = segment->element(track);
1477  if (!e || e->type() != ElementType::CHORD)
1478  continue;
1479  Chord* chord = toChord(e);
1480  if (chord->vStaffIdx() != chordL->vStaffIdx())
1481  continue;
1482  for (Note* note : chord->notes()) {
1483  if (note->tieBack())
1484  continue;
1485  int pc = (note->line() + 700) % 7;
1486  int pc2 = (lineL2 + 700) % 7;
1487  if (pc2 == pc) {
1488  // e.g., if there is an F# note at this staff/tick, then force every F to be F#.
1489  int octaves = (note->line() - lineL2) / 7;
1490  halfsteps = note->epitch() + 12 * octaves - epitchL;
1491  done = true;
1492  break;
1493  }
1494  }
1495  if (!done) {
1496  if (staffL->isPitchedStaff(segment->tick())) {
1497  bool error = false;
1498  AccidentalVal acciv2 = measureR->findAccidental(chordR->segment(), chordR->vStaffIdx(), lineR2, error);
1499  int acci2 = int(acciv2);
1500  // epitch (effective pitch) is a visible pitch so line2pitch returns exactly that.
1501  halfsteps = line2pitch(lineL-deltastep, clefL, Key::C) + acci2 - epitchL;
1502  }
1503  else {
1504  // cannot rely on accidentals or key signatures
1505  halfsteps = deltastep;
1506  }
1507  }
1508  }
1509  return halfsteps;
1510  }

1529 // renderNoteArticulation
1530 // prefix, vector of int, normally something like {0,-1,0,1} modeling the prefix of tremblement relative to the base note
1531 // body, vector of int, normally something like {0,-1,0,1} modeling the possibly repeated tremblement relative to the base note
1532 // tickspernote, number of ticks, either _16h or _32nd, i.e., MScore::division/4 or MScore::division/8
1533 // repeatp, true means repeat the body as many times as possible to fill the time slice.
1534 // sustainp, true means the last note of the body is sustained to fill remaining time slice
1535 //---------------------------------------------------------
1536 
1537 bool renderNoteArticulation(NoteEventList* events, Note* note, bool chromatic, int requestedTicksPerNote,
1538  const std::vector<int>& prefix, const std::vector<int>& body,
1539  bool repeatp, bool sustainp, const std::vector<int>& suffix,
1540  int fastestFreq=64, int slowestFreq=8 // 64 Hz and 8 Hz
1541  )
1542  {
1543  events->clear();
1544  Chord *chord = note->chord();
1545  int maxticks = totalTiedNoteTicks(note);
1546  int space = 1000 * maxticks;
1547  int numrepeat = 1;
1548  int sustain = 0;
1549  int ontime = 0;
1550 
1551  int gnb = note->chord()->graceNotesBefore().size();
1552  int p = int(prefix.size());
1553  int b = int(body.size());
1554  int s = int(suffix.size());
1555  int gna = note->chord()->graceNotesAfter().size();
1556 
1557  int ticksPerNote = 0;
1558 
1559  if (gnb + p + b + s + gna <= 0 )
1560  return false;
1561 
1562  Fraction tick = chord->tick();
1563  qreal tempo = chord->score()->tempo(tick);
1564  int ticksPerSecond = tempo * MScore::division;
1565 
1566  int minTicksPerNote = int(ticksPerSecond / fastestFreq);
1567  int maxTicksPerNote = (0 == slowestFreq) ? 0 : int(ticksPerSecond / slowestFreq);
1568 
1569  // for fast tempos, we have to slow down the tremblement frequency, i.e., increase the ticks per note
1570  if (requestedTicksPerNote >= minTicksPerNote)
1571  ;
1572  else { // try to divide the requested frequency by a power of 2 if possible, if not, use the maximum frequency, ie., minTicksPerNote
1573  ticksPerNote = requestedTicksPerNote;
1574  while (ticksPerNote < minTicksPerNote) {
1575  ticksPerNote *= 2; // decrease the tremblement frequency
1576  }
1577  if (ticksPerNote > maxTicksPerNote)
1578  ticksPerNote = minTicksPerNote;
1579  }
1580 
1581  ticksPerNote = std::max(requestedTicksPerNote, minTicksPerNote);
1582 
1583  if (slowestFreq <= 0) // no slowest freq given such as something silly like glissando with 4 notes over 8 counts.
1584  ;
1585  else if (ticksPerNote <= maxTicksPerNote) // in a good range, so we don't need to adjust ticksPerNote
1586  ;
1587  else {
1588  // for slow tempos, such as adagio, we may need to speed up the tremblement frequency, i.e., decrease the ticks per note, to make it sound reasonable.
1589  ticksPerNote = requestedTicksPerNote;
1590  while (ticksPerNote > maxTicksPerNote) {
1591  ticksPerNote /= 2;
1592  }
1593  if (ticksPerNote < minTicksPerNote)
1594  ticksPerNote = minTicksPerNote;
1595  }
1596  // calculate whether to shorten the duration value.
1597  if ( ticksPerNote*(gnb + p + b + s + gna) <= maxticks )
1598  ; // plenty of space to play the notes without changing the requested trill note duration
1599  else if ( ticksPerNote == minTicksPerNote )
1600  return false; // the ornament is impossible to implement respecting the minimum duration and all the notes it contains
1601  else {
1602  ticksPerNote = maxticks / (gnb + p + b + s + gna); // integer division ignoring remainder
1603  if ( slowestFreq <= 0 )
1604  ;
1605  else if ( ticksPerNote < minTicksPerNote )
1606  return false;
1607  }
1608 
1609  int millespernote = space * ticksPerNote / maxticks; // rescale duration into per mille
1610 
1611  // local function:
1612  // look ahead in the given vector to see if the current note is the same pitch as the next note or next several notes.
1613  // If so, increment the duration by the appropriate note duration, and increment the index, j, to the next note index
1614  // of a different pitch.
1615  // The total duration of the tied note is returned, and the index is modified.
1616  auto tieForward = [millespernote] (int & j, const std::vector<int> & vec) {
1617  int size = int(vec.size());
1618  int duration = millespernote;
1619  while ( j < size-1 && vec[j] == vec[j+1] ) {
1620  duration += millespernote;
1621  j++;
1622  }
1623  return duration;
1624  };

1626  // local function:
1627  // append a NoteEvent either by calculating an articulationExcursion or by
1628  // the given chromatic relative pitch.
1629  // RETURNS the new ontime value. The caller is expected to assign this value.
1630  auto makeEvent = [note,chord,chromatic,events] (int pitch, int ontime, int duration) {
1631  events->append( NoteEvent(chromatic ? pitch : articulationExcursion(note,note,pitch),
1632  ontime/chord->actualTicks().ticks(),
1633  duration/chord->actualTicks().ticks()));
1634  return ontime + duration;

1637  // local function:
1638  // Given a chord from a grace note, (normally the chord contains a single note) and create
1639  // a NoteEvent as if the grace note were part of the articulation (such as trill). This
1640  // local function works for the graceNotesBefore() and also graceNotesAfter().
1641  // If the grace note has play=false, then it will sound as a rest, but the other grace
1642  // notes will still play. This means graceExtend simply omits the call to append( NoteEvent(...))

1649  // The pitch is relative to the pitch of the note, the event is rendering

1651  events->append( NoteEvent(n->pitch() - notePitch,

1654  }

1656  }
1657  return ontime;

1692  }

1695  for (int r = 0; r < numrepeat - 1; r++) {
1696  for (int j = 0; j < b; j++)

1698  }

1701  ontime = makeEvent(body[j], ontime, millespernote);
1702  // render the final note of the final repeat of body
1703  ontime = makeEvent(body[b - 1], ontime, millespernote + sustain);
1704  }
1705  }

1708  ontime = makeEvent(suffix[j], ontime, tieForward(j,suffix));
1709  // render graceNotesAfter
1710  graceExtend(note->pitch(), note->chord()->graceNotesAfter(), ontime);

1714 // This struct specifies how to render an articulation.
1715 // atype - the articulation type to implement, such as SymId::ornamentTurn
1716 // ostyles - the actual ornament has a property called ornamentStyle whose value is
1717 // a value of type MScore::OrnamentStyle. This ostyles field indicates the
1718 // the set of ornamentStyles which apply to this rendition.
1719 // duration - the default duration for each note in the rendition, the final duration
1720 // rendered might be less than this if an articulation is attached to a note of
1721 // short duration.
1722 // prefix - vector of integers. indicating which notes to play at the beginning of rendering the
1723 // articulation. 0 represents the principle note, 1==> the note diatonically 1 above
1724 // -1 ==> the note diatonically 1 below. E.g., in the key of G, if a turn articulation
1725 // occurs above the note F#, then 0==>F#, 1==>G, -1==>E.
1726 // These integers indicate which notes actual notes to play when rendering the ornamented
1727 // note. However, if the same integer appears several times adjacently such as {0,0,0,1}
1728 // That means play the notes tied. e.g., F# followed by G, but the duration of F# is 3x the
1729 // duration of the G.
1730 // body - notes to play comprising the body of the rendered ornament.
1731 // The body differs from the prefix and suffix in several ways.
1732 // * body does not support tied notes: {0,0,0,1} means play 4 distinct notes (not tied).
1733 // * if there is sufficient duration in the principle note, AND repeatep is true, then body
1734 // will be rendered multiple times, as the duration allows.
1735 // * to avoid a time gap (or rest) in rendering the articulation, if sustainp is true,
1736 // then the final note of the body will be sustained to fill the left-over time.
1737 // suffix - similar to prefix but played once at the end of the rendered ornament.
1738 // repeatp - whether the body is repeatable in its entirety.
1739 // sustainp - whether the final note of the body should be sustained to fill the remaining duration.

1750  };

2057 // as a side effect of createGraceNotesPlayEvents, ontime and trailtime (passed by ref)
2058 // are modified. ontime reflects the time needed to play the grace-notes-before, and
2059 // trailtime reflects the time for the grace-notes-after. These are used by the caller

2062 
2063 void Score::createGraceNotesPlayEvents(const Fraction& tick, Chord* chord, int& ontime, int& trailtime)
2064  {
2065  QVector<Chord*> gnb = chord->graceNotesBefore();
2066  QVector<Chord*> gna = chord->graceNotesAfter();
2067  int nb = gnb.size();
2068  int na = gna.size();
2069  if (0 == nb + na) {
2070  return; // return immediately if no grace notes to deal with
2071  }
2072  // return immediately if the chord has a trill or articulation which effectively plays the graces notes.
2073  if (graceNotesMerged(chord)) {
2074  return;
2075  }
2076  // if there are graceNotesBefore and also graceNotesAfter, and the before grace notes are
2077  // not ACCIACCATURA, then the total time of all of them will be 50% of the time of the main note.
2078  // if the before grace notes are ACCIACCATURA then the grace notes after (if there are any).
2079  // get 50% of the time of the main note.
2080  // this is achieved by the two floating point weights: weighta and weightb whose total is 1.0
2081  // assuring that all the grace notes get the same duration, and their total is 50%.
2082  // exception is if the note is dotted or double-dotted; see below.
2083  float weighta = float(na) / (nb+na);
2084  float weightb = float(nb) / (nb+na);
2085

2087  bool drumset = (getDrumset(chord) != nullptr);
2088  const qreal ticksPerSecond = tempo(tick) * MScore::division;

2090  if (drumset) {
2091  int flamDuration = 15; //ms
2092  graceDuration = flamDuration / chordTimeMS * 1000; //ratio 1/1000 from the main note length
2093  ontime = graceDuration * nb;
2094  }
2095  else if (nb) {
2096  //
2097  // render grace notes:
2098  // simplified implementation:
2099  // - grace notes start on the beat of the main note
2100  // - duration: appoggiatura: 0.5 * duration of main note (2/3 for dotted notes, 4/7 for double-dotted)
2101  // acciacatura: min of 0.5 * duration or 65ms fixed (independent of duration or tempo)
2102  // - for appoggiaturas, the duration is divided by the number of grace notes
2103  // - the grace note duration as notated does not matter
2104  //
2105  Chord* graceChord = gnb[0];
2106  if (graceChord->noteType() == NoteType::ACCIACCATURA || nb > 1) { // treat multiple subsequent grace notes as acciaccaturas

2108  // 1000 occurs below as a unit for ontime
2109  ontime = qMin(500, static_cast<int>((graceTimeMS / chordTimeMS) * 1000));
2110  weightb = 0.0;
2111  weighta = 1.0;
2112  }
2113  else if (chord->dots() == 1)
2114  ontime = floor(667 * weightb);
2115  else if (chord->dots() == 2)
2116  ontime = floor(571 * weightb);
2117  else
2118  ontime = floor(500 * weightb);
2119 
2120  graceDuration = ontime / nb;
2121  }
2122 
2123  for (int i = 0, on = 0; i < nb; ++i) {
2124  QList<NoteEventList> el;
2125  Chord* gc = gnb.at(i);
2126  size_t nn = gc->notes().size();

2128  NoteEventList nel;
2129  nel.append(NoteEvent(0, on, graceDuration));
2130  el.append(nel);
2131  }
2132 
2133  if (gc->playEventType() == PlayEventType::Auto)
2134  gc->setNoteEventLists(el);
2135  on += graceDuration;
2136  }
2137  if (na) {
2138  if (chord->dots() == 1)
2139  trailtime = floor(667 * weighta);
2140  else if (chord->dots() == 2)
2141  trailtime = floor(571 * weighta);
2142  else
2143  trailtime = floor(500 * weighta);
2144  int graceDuration1 = trailtime / na;
2145  int on = 1000 - trailtime;
2146  for (int i = 0; i < na; ++i) {
2147  QList<NoteEventList> el;
2148  Chord* gc = gna.at(i);
2149  size_t nn = gc->notes().size();
2150  for (size_t ii = 0; ii < nn; ++ii) {
2151  NoteEventList nel;
2152  nel.append(NoteEvent(0, on, graceDuration1)); // NoteEvent(pitch,ontime,len)
