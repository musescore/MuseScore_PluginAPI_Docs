47 void RepeatMeasure::layout()
48  {
49  for (Element* e : el())
50  e->layout();
51 
52  Staff* st = staff();
53  qreal ld = st ? st->lineDistance(tick()) : 1.0;
54  qreal sp = spatium();
55 
56  qreal y = sp * 1.0 * ld;
57  qreal w = sp * 2.4 * ld;
58  qreal h = sp * 2.0 * ld;
59  qreal lw = sp * .50 * ld; // line width
60  qreal r = sp * .20 * ld; // dot radius
61 
62  setPos(0.0, (st ? (st->height() - h) / 2.0 : y) - y);
63 
64  path = QPainterPath();
65 
66  path.moveTo(w - lw, y);
67  path.lineTo(w, y);
68  path.lineTo(lw, h+y);
69  path.lineTo(0.0, h+y);
70  path.closeSubpath();
71  path.addEllipse(QRectF(w * .25 - r, y+h * .25 - r, r * 2.0, r * 2.0 ));
72  path.addEllipse(QRectF(w * .75 - r, y+h * .75 - r, r * 2.0, r * 2.0 ));
73 
74  setbbox(path.boundingRect());
75 // _space.setRw(width());
76  }
