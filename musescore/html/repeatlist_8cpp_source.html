175 int RepeatList::utick2tick(int tick) const
176  {
177  unsigned n = size();
178  if (n == 0)
179  return tick;
180  if (tick < 0)
181  return 0;
182  unsigned ii = (idx1 < n) && (tick >= at(idx1)->utick) ? idx1 : 0;
183  for (unsigned i = ii; i < n; ++i) {
184  if ((tick >= at(i)->utick) && ((i + 1 == n) || (tick < at(i+1)->utick))) {
185  idx1 = i;
186  return tick - (at(i)->utick - at(i)->tick);
187  }
188  }
189  if (MScore::debugMode) {
190  qFatal("tick %d not found in RepeatList", tick);
191  }
192  return 0;
193  }

327 ///---------------------------------------------------------
328 /// \brief Collects all RepeatListElementType elements in the score
329 /// \details Those are the only type of elements that actually influence playback order
330 ///---------------------------------------------------------
331 void RepeatList::collectRepeatListElements()
332  {
333  QList<RepeatListElement*>* sectionRLElements = new QList<RepeatListElement*>();
334 
335  // Clear out previous listing
336  for (QList<RepeatListElement*>* srle : qAsConst(_rlElements)) {
337  qDeleteAll(*srle);
338  }
339  qDeleteAll(_rlElements);
340  _rlElements.clear();
341 
342  RepeatListElement* startFromRepeatMeasure = nullptr;
343 
344  // Section breaks may occur on non-Measure frames, so must search list of all MeasureBases
345  // unwinding itself will only use actual Measures
346  MeasureBase * mb = _score->firstMeasure();
347  // First measure of a section/score is always used as a reference REPEAT_START point
348  // even if it doesn't have a start repeat
349  startFromRepeatMeasure = new RepeatListElement(RepeatListElementType::REPEAT_START, mb, toMeasure(mb));
350  sectionRLElements->push_back(startFromRepeatMeasure);
351  // Also trace down the final actual measure of this section (in case a frame follows)
352  MeasureBase * sectionEndMeasureBase = nullptr;
353  // Used to track voltas; overlappings and real endings
354  Volta * volta;
355  std::list<Volta *> preProcessedVoltas;
356 
357  // Voltas might overlap (duplicate entries on multiple staves or "real" overlaps)
358  // so we will pre-process them into cloned versions that handle those overlaps.
359  // This assumes that spanners are ordered from first to last tick-wise
360  for (const auto & spannerEntry : _score->spanner()) {
361  if ((spannerEntry.second)->isVolta()) {
362  volta = toVolta(spannerEntry.second)->clone();
363  if (preProcessedVoltas.empty()) { // First entry
364  preProcessedVoltas.push_back(volta);
365  }
366  else { // Compare
367  std::list<Volta *> voltasToMerge;
368  // List all overlapping voltas
369  while ( (!preProcessedVoltas.empty())
370  && (volta->startMeasure()->tick() <= preProcessedVoltas.back()->endMeasure()->tick())
371  ){
372  voltasToMerge.push_back(preProcessedVoltas.back());
373  preProcessedVoltas.pop_back();
374  }
375 
376  while (!voltasToMerge.empty()) {
377  // We'll have to shorten the already stored volta and split its remainder for merging
378  Volta * remainder = voltasToMerge.back()->clone();
379  if (volta->startMeasure() != remainder->startMeasure()) {
380  // First part is not empty
381  voltasToMerge.back()->setEndElement(volta->startMeasure()->prevMeasure());
382  remainder->setStartElement(volta->startMeasure());
383  // Store it
384  preProcessedVoltas.push_back(voltasToMerge.back());
385  }
386  //else { New volta and existing one start at the same moment, there is no first part, only a remainder }
387  voltasToMerge.pop_back();
388 
389  // remainder and volta now have the same start point
390  // Compare the end points and make remainder end first
391  if (volta->endMeasure()->tick() < remainder->endMeasure()->tick()) {
392  Volta * swap = volta;
393  volta = remainder;
394  remainder = swap;
395  }
396  // Cross-section of the repeatList
397  std::list<int> endings = remainder->endings().toStdList();
398  endings.remove_if([&volta](const int & ending) {
399  return (!(volta->hasEnding(ending)));
400  });
401  remainder->setEndings(QList<int>::fromStdList(endings));
402  // Split and merge done
403  preProcessedVoltas.push_back(remainder);
404  if (volta->endMeasure() != remainder->endMeasure()) {
405  // volta extends past the end of remainder -> move its startpoint after remainder
406  volta->setStartElement(remainder->endMeasure()->nextMeasure());
407  }
408  else { // volta matched remainder endpoint, nothing left to merge from
409  preProcessedVoltas.splice(preProcessedVoltas.cend(), voltasToMerge);
410  delete volta;
411  volta = nullptr;
412  }
413  } // !voltasToMerge.empty()
414 
415  if (volta != nullptr) {
416  preProcessedVoltas.push_back(volta);
417  }
418  }
419  } // spanner->isVolta
420  }
421 
422  volta = nullptr;
423  for (; mb; mb = mb->next()) {
424  if (mb->isMeasure()) {
425  sectionEndMeasureBase = mb; // ending measure of section is the most recently encountered actual Measure
426 
427  // Volta ?
428  if ((!preProcessedVoltas.empty()) && (preProcessedVoltas.front()->startMeasure() == mb)) {
429  if (volta != nullptr) {
430  //if (volta->endMeasure()->tick() < mb->tick()) {
431  // The previous volta was supposed to end before us (open volta case) -> insert the end
432  sectionRLElements->push_back(new RepeatListElement(RepeatListElementType::VOLTA_END, volta, toMeasure(mb->prevMeasure())));
433  //volta = nullptr; // No need, replaced immediately further down
434  // }
435  //else { // Overlapping voltas; this should not happen as preProcessedVoltas should've dealt with this already }
436  }
437  // Now insert the start of the current volta
438  volta = preProcessedVoltas.front();
439  sectionRLElements->push_back(new RepeatListElement(RepeatListElementType::VOLTA_START, volta, toMeasure(mb)));
440  // Look for start of next volta
441  preProcessedVoltas.pop_front();
442  }
443  // Start
444  if (mb->repeatStart()) {
445  if (volta != nullptr) {
446  if (volta->startMeasure() != toMeasure(mb)) {
447  // Volta and Start repeat are not on the same measure
448  // assume the previous volta was supposed to end before us (open volta case) -> insert the end
449  // Warning: This might "break" a volta prematurely if its explicit notated end is later than this point
450  // Consider splitting the volta or ignoring this repeat all together
451  sectionRLElements->push_back(new RepeatListElement(RepeatListElementType::VOLTA_END, volta, toMeasure(mb->prevMeasure())));
452  volta = nullptr;
453  }
454  //else { // Volta and Start Repeat coincide on the same measure, see test::repeat56.mscx }
455  }
456  startFromRepeatMeasure = new RepeatListElement(RepeatListElementType::REPEAT_START, mb, toMeasure(mb));
457  sectionRLElements->push_back(startFromRepeatMeasure);
458  }
459  // Jumps and Markers
460  for (Element* e : mb->el()) {
461  if (e->isJump()) {
462  sectionRLElements->push_back(new RepeatListElement(RepeatListElementType::JUMP, e, toMeasure(mb)));
463  if (volta != nullptr) {
464  if ((volta->endMeasure()->tick() < mb->tick())
465  || ((volta->endMeasure()->tick() == mb->tick())
466  && (volta->getProperty(Pid::END_HOOK_TYPE).value<HookType>() == HookType::NONE)
467  )
468  ) {
469  // The previous volta was supposed to end before us
470  // or open volta ends together with us -> insert the end
471  sectionRLElements->push_back(new RepeatListElement(RepeatListElementType::VOLTA_END, volta, toMeasure(mb)));
472  volta = nullptr;
473  }
474  //else { // Volta is spanning past this jump instruction }
475  }
476  }
477  else if (e->isMarker()) {
478  RepeatListElement * markerRLE = new RepeatListElement(RepeatListElementType::MARKER, e, toMeasure(mb));
479  // There may be multiple markers in the same measure and there is no guarantee we're reading
480  // them from left to right. The only way available to guess their order is to look at their
481  // text alignment and order them left to right
482  // At the same time, we should ensure Markers are evaluated before Jumps
483  Align markerRLEalignmentH = static_cast<Align>(static_cast<char>(toMarker(e)->align()) & static_cast<char>(Align::HMASK));
484  auto insertionIt = sectionRLElements->end() - 1;
485  while ((*insertionIt)->measure == markerRLE->measure) {
486  bool markerShouldGoBefore = false;
487  if ( ((*insertionIt)->repeatListElementType == RepeatListElementType::MARKER)
488  && (markerRLEalignmentH != Align::RIGHT) // We can be the end when right aligned
489  ) {
490  Align storedMarkerAlignmentH = static_cast<Align>(static_cast<char>(toMarker((*insertionIt)->element)->align()) & static_cast<char>(Align::HMASK));
491  if (markerRLEalignmentH == Align::HCENTER) {
492  markerShouldGoBefore = (storedMarkerAlignmentH == Align::RIGHT);
493  }
494  else { //(markerRLEalignmentH == Align::LEFT)
495  markerShouldGoBefore = (storedMarkerAlignmentH != Align::LEFT);
496  }
497  }
498  if ( markerShouldGoBefore
499  || ((*insertionIt)->repeatListElementType == RepeatListElementType::JUMP)
500  ) {
501  // Decrease position - this should always be possible as the list always starts with a REPEAT_START element
502  Q_ASSERT(insertionIt != sectionRLElements->begin());
503  --insertionIt;
504  }
505  else { // Found location after which we should go
506  break;
507  }
508  }
509  // We should be inserted right after insertionIt
510  sectionRLElements->insert(insertionIt + 1, markerRLE);
511  }
512  }
513  // End
514  if (mb->repeatEnd()) {
515  sectionRLElements->push_back(new RepeatListElement(RepeatListElementType::REPEAT_END, mb, toMeasure(mb)));
516  if (startFromRepeatMeasure != nullptr) {
517  startFromRepeatMeasure->addToRepeatCount(toMeasure(mb)->repeatCount() - 1);
518  }
519  if (volta != nullptr) {
520  //if (volta->endMeasure()->tick() < mb->tick()) {
521  // The previous volta was supposed to end before us (open volta case) -> insert the end
522  sectionRLElements->push_back(new RepeatListElement(RepeatListElementType::VOLTA_END, volta, toMeasure(mb)));
523  volta = nullptr;
524  // }
525  //else { // Volta is spanning over this end repeat, consider splitting the volta or ignoring this repeat all together }
526  }
527  }
528  // Volta end
529  if ( (volta != nullptr)
530  && (volta->endMeasure()->tick() == mb->tick())
531  && (volta->getProperty(Pid::END_HOOK_TYPE).value<HookType>() != HookType::NONE)) {
532  // end of closed volta
533  sectionRLElements->push_back(new RepeatListElement(RepeatListElementType::VOLTA_END, volta, toMeasure(mb)));
534  volta = nullptr;
535  }
536  }
537  // Section break (or end of score)
538  if (mb->sectionBreak() || !mb->nextMeasure()) {
539  if (sectionEndMeasureBase != nullptr) {
540  if (volta != nullptr) {
541  //if (volta->endMeasure()->tick() < mb->tick()) {
542  // The previous volta was supposed to end before us (open volta case) -> insert the end
543  sectionRLElements->push_back(new RepeatListElement(RepeatListElementType::VOLTA_END, volta, toMeasure(mb)));
544  volta = nullptr;
545  // }
546  //else { // Volta is spanning over this section break, consider splitting the volta and adding it again at the start of the next section }
547  }
548  sectionRLElements->push_back(new RepeatListElement(RepeatListElementType::SECTION_BREAK, mb, toMeasure(sectionEndMeasureBase)));
549  sectionEndMeasureBase = nullptr; // reset to indicate not having found the end for the next section
550  // store section
551  _rlElements.push_back(sectionRLElements);
552  }
553  // prepare for new section
554  if (mb->nextMeasure()) {
555  sectionRLElements = new QList<RepeatListElement*>();
556  // First measure of a section/score is always used as a reference REPEAT_START point
557  // even if it doesn't have a start repeat
558  startFromRepeatMeasure = new RepeatListElement(RepeatListElementType::REPEAT_START, mb->nextMeasure(), toMeasure(mb->nextMeasure()));
559  sectionRLElements->push_back(startFromRepeatMeasure);
560  // Loop will forward one measureBase, so return one now
561  // this logic aids in skipping multiple frames between sections
562  mb = mb->nextMeasure()->prev();
563  }
564  else {
565  break; // no more measures -> done
566  }
567  }
568  }
569  }

571 ///
572 /// \brief Finds a matching marker in the score (section first)
573 /// \remark Special case labels:
574 /// "start" will result in start of current section
575 /// "end" will result in end of current section
576 ///
577 std::pair<QList<QList<RepeatListElement*>*>::const_iterator, QList<RepeatListElement*>::const_iterator> RepeatList::findMarker(
578  QString label, QList<QList<RepeatListElement*>*>::const_iterator referenceSectionIt, QList<RepeatListElement*>::const_iterator referenceRepeatListElementIt) const
579  {
580  bool found = false;
581  QList<QList<RepeatListElement*>*>::const_iterator foundSectionIt;
582  QList<RepeatListElement*>::const_iterator foundRepeatListElementIt;
583 
584  // Start in the current section
585  foundSectionIt = referenceSectionIt;
586  // Handle special label casing first
587  if (label == "start") {
588  foundRepeatListElementIt = (*referenceSectionIt)->cbegin();
589  found = true;
590  }
591  else if (label == "end") {
592  foundRepeatListElementIt = (*referenceSectionIt)->cend() - 1;
593  found = true;
594  }
595  else {
596  foundRepeatListElementIt = referenceRepeatListElementIt;
597  }
598 
599  // Search backwards in this section
600  while (!found && (foundRepeatListElementIt != (*referenceSectionIt)->cbegin())) {
601  --foundRepeatListElementIt;
602  if ( ((*foundRepeatListElementIt)->repeatListElementType == RepeatListElementType::MARKER)
603  && ((toMarker((*foundRepeatListElementIt)->element))->label() == label)
604  ) {
605  found = true;
606  }
607  }
608 
609  // Search forwards in this section
610  if (!found) {
611  foundRepeatListElementIt = referenceRepeatListElementIt + 1;
612  while (!found && (foundRepeatListElementIt != (*referenceSectionIt)->cend())) {
613  if ( ((*foundRepeatListElementIt)->repeatListElementType == RepeatListElementType::MARKER)
614  && ((toMarker((*foundRepeatListElementIt)->element))->label() == label)
615  ) {
616  found = true;
617  break;
618  }
619  ++foundRepeatListElementIt;
620  }
621  }
622 
623  // Search backwards through all previous sections
624  if (!found) {
625  while (!found && (foundSectionIt != _rlElements.cbegin())) {
626  --foundSectionIt;
627  foundRepeatListElementIt = (*foundSectionIt)->cend();
628  while (!found && (foundRepeatListElementIt != (*foundSectionIt)->cbegin())) {
629  --foundRepeatListElementIt;
630  if ( ((*foundRepeatListElementIt)->repeatListElementType == RepeatListElementType::MARKER)
631  && ((toMarker((*foundRepeatListElementIt)->element))->label() == label)
632  ) {
633  found = true;
634  }
635  }
636  }
637  }
638 
639  // Search forwards through all following sections
640  if (!found) {
641  foundSectionIt = referenceSectionIt + 1;
642  while (foundSectionIt != _rlElements.cend()) {
643  foundRepeatListElementIt = (*foundSectionIt)->cbegin();
644  while (!found && (foundRepeatListElementIt != (*foundSectionIt)->cend())) {
645  if ( ((*foundRepeatListElementIt)->repeatListElementType == RepeatListElementType::MARKER)
646  && ((toMarker((*foundRepeatListElementIt)->element))->label() == label)
647  ) {
648  found = true;
649  break;
650  }
651  ++foundRepeatListElementIt;
652  }
653  if (found) {
654  break;
655  }
656  ++foundSectionIt;
657  }
658  }
659 
660  return std::make_pair(foundSectionIt, foundRepeatListElementIt);
661  }

666 /// \param repeatListElementTargetIt [in] RepeatListElement of the jump target within the section
667 /// \param withRepeats [in] Whether first or last playtrough of the target is the actual target, influences playbackCount
668 /// \param playbackCount [out] Will contain the resulting playbackCount value

672 void RepeatList::performJump(QList<QList<RepeatListElement*>*>::const_iterator sectionIt, QList<RepeatListElement*>::const_iterator repeatListElementTargetIt,

674  Volta const * * const activeVolta, RepeatListElement const * * const startRepeatReference) const

678  *activeVolta = nullptr;
679  *startRepeatReference = *((*sectionIt)->cbegin());

684  } break;
685  case RepeatListElementType::VOLTA_END: {

687  } break;
688  case RepeatListElementType::REPEAT_START: {
689  *startRepeatReference = *repeatListElementIt;

693  } break;
694  }
695  }

699  *playbackCount = (*activeVolta)->firstEnding();
700  if (*playbackCount == 0) {
701  *playbackCount = 1;
702  }
703  }
704  else {
705  *playbackCount = 1;
706  }
707  }
708  else { // Final repeat after jump
709  if (*activeVolta != nullptr) {
710  *playbackCount = (*activeVolta)->lastEnding();
711  if (*playbackCount == 0) {
712  *playbackCount = (*startRepeatReference)->getRepeatCount();
713  }
714  }
715  else {
716  *playbackCount = (*startRepeatReference)->getRepeatCount();
717  }
718  }
719  }

723 ///
724 void RepeatList::unwind()
725  {
726  qDeleteAll(*this);
727  clear();
728  _jumpsTaken.clear();
729 
730  if (!_score->firstMeasure())
731  return;
732 
733  collectRepeatListElements();
734 
735  // Following variables are used during unwinding, but may be altered when following jumps
736  // Therefor they are declared outside of the loop
737  RepeatSegment * rs = nullptr;
738  int playbackCount;
739  Volta const * activeVolta = nullptr;
740  std::pair<QList<QList<RepeatListElement*>*>::const_iterator, QList<RepeatListElement*>::const_iterator> playUntil = std::make_pair(_rlElements.cend(), _rlElements[0]->cend());
741  std::pair<QList<QList<RepeatListElement*>*>::const_iterator, QList<RepeatListElement*>::const_iterator> continueAt = std::make_pair(_rlElements.cend(), _rlElements[0]->cend());
742  Jump const * activeJump = nullptr;
743  bool forceFinalRepeat; // Used during jump processing
744  QList<RepeatListElement*>::const_iterator repeatListElementIt;
745 
746  for (QList<QList<RepeatListElement*>*>::const_iterator sectionIt = _rlElements.cbegin(); sectionIt != _rlElements.cend(); ++sectionIt)
747  {
748  // Unwind this section
749  RepeatListElement const * startRepeatReference;
750  playbackCount = 1;
751  repeatListElementIt = (*sectionIt)->cbegin(); // Should always be a REPEAT_START indicator
752  startRepeatReference = *repeatListElementIt;
753  activeVolta = nullptr;
754  playUntil.first = _rlElements.cend();
755  continueAt.first = _rlElements.cend();
756  forceFinalRepeat = false;
757 
758  rs = new RepeatSegment(playbackCount);
759  rs->addMeasure((*repeatListElementIt)->measure);
760  ++repeatListElementIt;
761 
762  while (repeatListElementIt != (*sectionIt)->cend()) {
763  if (rs) {
764  rs->addMeasures((*repeatListElementIt)->measure);
765  }
766  switch ((*repeatListElementIt)->repeatListElementType) {
767  case RepeatListElementType::SECTION_BREAK: {
768  if ((rs != nullptr) && (!rs->isEmpty())) {
769  push_back(rs);
770  }
771  } break;
772  case RepeatListElementType::VOLTA_START: {
773  activeVolta = toVolta((*repeatListElementIt)->element);
774  if (!(activeVolta->hasEnding(playbackCount))) {
775  // Should be skipped, remove our measure from rs
776  rs->popMeasure();
777  if (!rs->isEmpty()) {
778  push_back(rs);
779  }
780  // Skip the volta
781  do {
782  ++repeatListElementIt;
783  } while ((*repeatListElementIt)->repeatListElementType != RepeatListElementType::VOLTA_END);
784  activeVolta = nullptr;
785  // Start next rs on the following measure
786  Measure const * const possibleNextMeasure = (*repeatListElementIt)->measure->nextMeasure();
787  if (possibleNextMeasure == nullptr) {
788  rs = nullptr; // end of score, but will still encounter section break, notify it
789  }
790  else {
791  rs = new RepeatSegment(playbackCount);
792  rs->addMeasure(possibleNextMeasure);
793  }
794  }
795  // else { take the volta, it's already set as activeVolta, so just continue }
796  } break;
797  case RepeatListElementType::VOLTA_END: {
798  activeVolta = nullptr;
799  } break;
800  case RepeatListElementType::REPEAT_START: {
801  if (rs == nullptr) { // Sent here by an end-repeat
802  rs = new RepeatSegment(playbackCount);
803  rs->addMeasure((*repeatListElementIt)->measure);
804  }
805  else {
806  int desiredPlaybackCount = (forceFinalRepeat)? (*repeatListElementIt)->getRepeatCount() : 1;
807  if (rs->playbackCount != desiredPlaybackCount) {
808  // Restart a segment as we have a new playbackCount reference
809  rs->popMeasure();
810  if (!rs->isEmpty()) {
811  push_back(rs);
812  }
813  playbackCount = desiredPlaybackCount;
814  rs = new RepeatSegment(playbackCount);
815  rs->addMeasure((*repeatListElementIt)->measure);
816  }
817  startRepeatReference = *repeatListElementIt;
818  }
819  } break;
820  case RepeatListElementType::REPEAT_END: {
821  (*repeatListElementIt)->addToRepeatCount(1);
822  if ( (playbackCount < startRepeatReference->getRepeatCount())
823  && ((*repeatListElementIt)->getRepeatCount() < (*repeatListElementIt)->measure->repeatCount())
824  ) {
825  // Honor the repeat
826  push_back(rs);
827  rs = nullptr;
828  do { // rewind
829  --repeatListElementIt;
830  if ((*repeatListElementIt)->repeatListElementType == RepeatListElementType::VOLTA_START) {
831  activeVolta = nullptr;
832  }
833  else if ((*repeatListElementIt)->repeatListElementType == RepeatListElementType::VOLTA_END) {
834  activeVolta = toVolta((*repeatListElementIt)->element);
835  }
836  } while ((*repeatListElementIt) != startRepeatReference);
837  ++playbackCount;
838  continue; // Force evaluation of the start repeat
839  }
840  //else { // Exhausted repeats, just continue }
841  } break;
842  case RepeatListElementType::JUMP: {
843  if (activeJump == toJump((*repeatListElementIt)->element)) {
844  // We've reached this jump again after just having taken it
845  // If this jump forced final repeat playthrough of measures
846  // it only forces so up to this point - see test::repeat23
847  forceFinalRepeat = false;
848  }
849  // TODO future improvement, use the repeatList property from the Jump to verify if it should be honored
850  // For now (or in case of an empty repeatList property) a Jump is only honored on final playthrough
851  // of the measure, and only taken once
852  if ( (playbackCount >= startRepeatReference->getRepeatCount())
853  || ((activeVolta != nullptr) && (playbackCount == activeVolta->lastEnding()))
854  ) {
855  std::pair<Jump const * const, int> jumpOccurence = std::make_pair(toJump((*repeatListElementIt)->element), playbackCount);
856  if (_jumpsTaken.find(jumpOccurence) == _jumpsTaken.end()) { // Not yet processed
857  // Processing it now
858  _jumpsTaken.insert(jumpOccurence);
859  // Find the jump targets
860  std::pair<QList<QList<RepeatListElement*>*>::const_iterator, QList<RepeatListElement*>::const_iterator> jumpTo = findMarker(jumpOccurence.first->jumpTo(), sectionIt, repeatListElementIt);
861  playUntil = findMarker(jumpOccurence.first->playUntil(), sectionIt, repeatListElementIt);
862  continueAt = findMarker(jumpOccurence.first->continueAt(), sectionIt, repeatListElementIt);
863 
864  // Execute
865  if (jumpTo.first != _rlElements.cend()) {
866  push_back(rs);
867  rs = nullptr;
868 
869  activeJump = jumpOccurence.first;
870  performJump(jumpTo.first, jumpTo.second, activeJump->playRepeats(), &playbackCount, &activeVolta, &startRepeatReference);
871  sectionIt = jumpTo.first;
872  repeatListElementIt = jumpTo.second;
873  forceFinalRepeat = !(activeJump->playRepeats());
874 
875  // Re-evaluate our repeat count for end repeat measures
876  if (playUntil.first != _rlElements.cend()) { // Only required if we have an end target
877  for (auto rleIt = repeatListElementIt + 1; rleIt != (*sectionIt)->cend(); ++rleIt) {
878  if (((*rleIt)->repeatListElementType == RepeatListElementType::REPEAT_END)
879  && ((*rleIt)->getRepeatCount() != 0)
880  ) { // We've been played before - see test::repeat22 for why only then
881  // Clear out our current repeat count
882  (*rleIt)->addToRepeatCount(-((*rleIt)->getRepeatCount()));
883  if (forceFinalRepeat) {
884  // Set repeat count to the maximum number for this repeat
885  (*rleIt)->addToRepeatCount((*rleIt)->measure->repeatCount());
886  }
887  }
888  }
889  }
890  if (activeVolta != nullptr) {
891  // We jumped into a volta, which might not be the only one of this startRepeatReference
892  // If our jump didn't take us to the last playthrough, we must re-evaluate
893  // all voltas related to this startRepeatReference
894  if (playbackCount < startRepeatReference->getRepeatCount()) {
895  // test::repeat52 and test::repeat53
896  QList<RepeatListElement*>::const_iterator findRepeatIt = repeatListElementIt;
897  do {
898  --findRepeatIt;
899  }
900  while ((*findRepeatIt) != startRepeatReference);
901  ++findRepeatIt; // Start volta analysis past this start repeat
902  Volta const * voltaReference = nullptr;
903  int processedRepeatCount = 1;
904  while ( (findRepeatIt != (*sectionIt)->cend())
905  && ((*findRepeatIt)->repeatListElementType != RepeatListElementType::REPEAT_START)
906  && (processedRepeatCount < startRepeatReference->getRepeatCount())
907  ) {
908  if ((*findRepeatIt)->repeatListElementType == RepeatListElementType::VOLTA_START) {
909  voltaReference = toVolta((*findRepeatIt)->element);
910  if (voltaReference->lastEnding() < playbackCount) {
911  // We won't pass here anymore - no need to adjust end repeat to this volta
912  voltaReference = nullptr;
913  }
914  }
915  else if ((*findRepeatIt)->repeatListElementType == RepeatListElementType::REPEAT_END) {
916  if (voltaReference != nullptr) {
917  int remainingRepeatsCount = 0;
918  for (int ending : voltaReference->endings()) {
919  if (ending >= playbackCount) {
920  ++remainingRepeatsCount;
921  }
922  }
923  // Apply new remainingRepeatsCount
924  (*findRepeatIt)->addToRepeatCount(-((*findRepeatIt)->getRepeatCount()));
925  (*findRepeatIt)->addToRepeatCount((*findRepeatIt)->measure->repeatCount() - remainingRepeatsCount - 1);
926  }
927  processedRepeatCount += (*findRepeatIt)->measure->repeatCount() - 1;
928  }
929  ++findRepeatIt;
930  }
931  }
932  }
933 
934  // We've arrived at the target marker
935  rs = new RepeatSegment(playbackCount);
936  rs->addMeasure((*repeatListElementIt)->measure);
937  } // End of jump execution
938  }
939  }
940  } break;
941  case RepeatListElementType::MARKER: {
942  if ( ((sectionIt == playUntil.first) && (repeatListElementIt == playUntil.second))
943  && ( (playbackCount == startRepeatReference->getRepeatCount())
944  || ((activeVolta != nullptr) && (playbackCount == activeVolta->lastEnding()))
945  )
946  ) { // Found final playThrough of this Marker
947  push_back(rs);
948  rs = nullptr;
949  playUntil.first = _rlElements.cend(); // Clear this reference - processed
950  forceFinalRepeat = false;
951 
952  if (continueAt.first != _rlElements.cend()) {
953  performJump(continueAt.first, continueAt.second, true, &playbackCount, &activeVolta, &startRepeatReference);
954  sectionIt = continueAt.first;
955  repeatListElementIt = continueAt.second;
956  if (startRepeatReference->measure != (*(continueAt.second))->measure) {
957  // Use the continueAt target as repeat reference if it wasn't this measure already
958  // Check for measure is required as REPEAT_START is processed before a marker
959  // see test::repeat23 m12
960  startRepeatReference = *(continueAt.second);
961  }
962 
963  // We've arrived at the target marker
964  rs = new RepeatSegment(playbackCount);
965  rs->addMeasure((*repeatListElementIt)->measure);
966 
967  continueAt.first = _rlElements.cend(); // Clear this reference - processed
968  }
969  else { // Nowhere to go to, break out of this section loop and onto the next section
970  repeatListElementIt = (*sectionIt)->cend();
971  continue;
972  }
973  }
974  } break;
975  }
976  ++repeatListElementIt;
977  }
978 
979  // Reached the end of this section
980  // Inform the last RepeatSegment that the Section Break pause property should be honored now
981  rs = this->back();
982  repeatListElementIt = (*sectionIt)->cend() - 1;
983  Q_ASSERT((*repeatListElementIt)->repeatListElementType == RepeatListElementType::SECTION_BREAK);
984 
985  LayoutBreak const * const sectionBreak = toMeasureBase((*repeatListElementIt)->element)->sectionBreakElement();
986  if (sectionBreak != nullptr) {
987  rs->pause = sectionBreak->pause();
988  }
989  }
990 
991  updateTempo();
992  _expanded = true;
993  }
