33 void RawScoreDiffModel::update() {
34  beginResetModel();
35  _textDiffs.clear();
36  for (const TextDiff& diff : _diff->textDiffs()) {
37  if (_skipEqual && diff.type == DiffType::EQUAL)
38  continue;
39  if (diff.type == DiffType::REPLACE) {
40  // Push this item twice. First will be shown as DELETE,
41  // second as INSERT.
42  _textDiffs.push_back(&diff);
43  }
44  _textDiffs.push_back(&diff);
45  }
46  endResetModel();
47  }

64 QVariant RawScoreDiffModel::data(const QModelIndex& index, int role) const
65  {
66  const int row = index.row();
67  if (row < 0 || row >= int(_textDiffs.size()))
68  return QVariant();
69 
70  const TextDiff& diff = *_textDiffs[row];
71  switch(role) {
72  case Qt::DisplayRole:
73  {
74  DiffType type = diff.type;
75  if (diff.type == DiffType::REPLACE)
76  type = (row == 0 || _textDiffs[row - 1] != &diff) ? DiffType::DELETE : DiffType::INSERT;
