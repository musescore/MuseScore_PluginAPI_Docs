116  break;

118  val = QVariant::fromValue(Spatium(val.toReal()));
119  break;

121  break;
122  }
123 
124  const PropertyFlags f = e->propertyFlags(pid);
125  const PropertyFlags newFlags = (f == PropertyFlags::NOSTYLE) ? f : PropertyFlags::UNSTYLED;
126 
127  if (_ownership == Ownership::SCORE) {
128  e->undoChangeProperty(pid, val, newFlags);
129  }
130  else { // not added to a score so no need (and dangerous) to deal with undo stack
131  e->setProperty(pid, val);
132  e->setPropertyFlags(pid, newFlags);
133  }
