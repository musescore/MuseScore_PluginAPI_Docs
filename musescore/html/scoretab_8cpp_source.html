146 QSplitter* ScoreTab::viewSplitter(int n) const
147  {
148  const TabScoreView* tsv = tabScoreView(n);
149  if(!tsv) {
150  return nullptr;
151  }
152 
153  Score* score = tsv->score;
154  if (tsv->part) {
155  QList<Excerpt*>& excerpts = score->excerpts();
156  if (!excerpts.isEmpty() && ((tsv->part - 1) < excerpts.size()))
157  score = excerpts.at(tsv->part - 1)->partScore();
158  }
159 
160  int nn = stack->count();
161  for (int i = 0; i < nn; ++i) {
162  QSplitter* sp = static_cast<QSplitter*>(stack->widget(i));
163  if (sp->count() == 0)
164  return 0;
165  ScoreView* v = static_cast<ScoreView*>(sp->widget(0));
166  if (v->score() == score)
167  return sp;
168  }
169  return 0;
170  }

245  {

263  tab2->setCurrentIndex(0);
264  tab2->blockSignals(false);
265  vs->addWidget(v);
266  v->setScore(scoreList->value(n));
267  stack->addWidget(vs);
268 #ifdef AVSOMR

293  sv->setScale(scale);
294  sv->setOffset(_matrix.dx(), _matrix.dy());

297  vs->setSizes(sizes);

305  }

323  tab2->setCurrentIndex(tsv->part);
324  tab2->blockSignals(false);
325  tab2->setVisible(true);
326  }
327  else {
328  tab2->setVisible(false);
329  }
330  }
331  else {

333  }

344 void ScoreTab::updateExcerpts()
345  {
346  int idx = currentIndex();
347  if (idx == -1)
348  return;
349  TabScoreView* tsv = tabScoreView(idx);
350  IF_ASSERT_FAILED(tsv) {
351  return;
352  }
353 
354  MasterScore* score = tsv->score;
355  if (!score->excerptsChanged())
356  return;
357 
358  clearTab2();
359  //delete all scoreviews for parts, especially for the deleted ones
360  int n = stack->count() - 1;
361  for (int i = n; i >= 0; --i) {
362  QSplitter* vs = static_cast<QSplitter*>(stack->widget(i));
363  ScoreView* sview = static_cast<ScoreView*>(vs->widget(0));
364 
365  if (sview->score() != score && sview->score()->masterScore() == score) {
366  stack->takeAt(i);
367  sview->deleteLater();
368  }
369  }
370 
371  QList<Excerpt*>& excerpts = score->excerpts();
372  if (!excerpts.isEmpty()) {
373  tab2->blockSignals(true);
374  tab2->addTab(score->fileInfo()->completeBaseName().replace("&","&&") + (score->dirty() ? "*" : ""));
375  for (const Excerpt* excerpt : excerpts)
376  tab2->addTab(excerpt->title().replace("&","&&"));
377  tab2->blockSignals(false);
378  tab2->setVisible(true);
379  }
380  else {
381  tab2->setVisible(false);
382  }
383  setExcerpt(0);
384  }

514 void ScoreTab::removeTab(int idx, bool noCurrentChangedSignal)
515  {
516  TabScoreView* tsv = tabScoreView(idx);
517  IF_ASSERT_FAILED(tsv) {
518  return;

520  Score* score = tsv->score;
521 
522  for (int i = 0; i < stack->count(); ++i) {
523  QSplitter* vs = static_cast<QSplitter*>(stack->widget(i));
524  ScoreView* v = static_cast<ScoreView*>(vs->widget(0));
525  if (v->score() == score) {
526  stack->takeAt(i);
527  delete v;
528  break;
529  }
530  }
531  foreach (Excerpt* excerpt, score->excerpts()) {
532  Score* sc = excerpt->partScore();
533  for (int i = 0; i < stack->count(); ++i) {
534  QSplitter* vs = static_cast<QSplitter*>(stack->widget(i));
535  ScoreView* v = static_cast<ScoreView*>(vs->widget(0));
536  if (v->score() == sc) {
537  stack->takeAt(i);
538  delete v;
539  break;
540  }
541  }
542  }
543 
544  bool blocked = false;
545  if (noCurrentChangedSignal)
546  blocked = blockSignals(true);
547 
548  tab->removeTab(idx); // Will call setCurrent via a signal-slot connection
549 
550  if (noCurrentChangedSignal)
551  blockSignals(blocked);
552 
553  emit tabRemoved(idx);
554  }
