37 std::unique_ptr<ScriptEntry> ScriptEntry::deserialize(const QString& line)
38  {
39  if (line.isEmpty() || line.startsWith('#')) // comment or empty line
40  return nullptr;
41  QStringList tokens = line.split(' ');
42  const QString& type = tokens[0];
43  if (type == SCRIPT_CMD) {
44  if (tokens.size() != 2) {
45  qWarning("Unexpected number of tokens in command: %d", tokens.size());
46  return nullptr;
47  }
48  return std::unique_ptr<ScriptEntry>(new CommandScriptEntry(tokens[1]));
49  }
50  if (type == SCRIPT_INIT)
51  return InitScriptEntry::deserialize(tokens);
52  if (type == SCRIPT_TEST)
53  return TestScriptEntry::deserialize(tokens);
54  if (type == SCRIPT_PALETTE)
55  return PaletteElementScriptEntry::deserialize(tokens);
56  if (type == SCRIPT_INSPECTOR)
57  return InspectorScriptEntry::deserialize(tokens);
58  if (type == SCRIPT_EXCERPT_CHANGE)
59  return ExcerptChangeScriptEntry::deserialize(tokens);
60  qWarning() << "Unsupported action type:" << type;
61  return nullptr;
62  }

206 static std::vector<Element*> getPaletteElements(ElementType elType, ScriptContext& ctx)
207  {
208  std::vector<Element*> elements;
209 
210  PaletteWorkspace* pw = ctx.mscore()->getPaletteWorkspace();
211  if (!pw)
212  return elements;
213 
214  const PaletteTree* tree = pw->userPaletteModel()->paletteTree();
215 
216  for (auto& p : tree->palettes) {
217  const int n = p->ncells();
218  for (int i = 0; i < n; ++i) {
219  Element* e = p->cell(i)->element.get();
220  if (e && e->type() == elType)
221  elements.push_back(e);
222  }
223  }
224  return elements;
225  }

231 bool PaletteElementScriptEntry::execute(ScriptContext& ctx) const
232  {
233  const auto elements(getPaletteElements(_type, ctx));
234 
235  for (Element* e : elements) {
236  bool match = true;
237  for (const auto& p : _props) {
238  const Pid pid = p.first;
239  const QVariant pVal = e->getProperty(pid);
240  if (serializePropertyValue(pid, pVal) != p.second) {
241  match = false;
242  break;
243  }
244  }
245 
246  if (match) {
247  Palette::applyPaletteElement(e);
248  return true;
249  }
250  }
251 
252  return false;
253  }

258 
259 std::unique_ptr<ScriptEntry> PaletteElementScriptEntry::fromContext(const Element* e, ScriptContext& ctx)
260  {
261  const ElementType type = e->type();
262  std::vector<std::pair<Pid, QString>> props;
263  auto paletteElements(getPaletteElements(type, ctx));
264 
265  if (paletteElements.empty())
266  return nullptr;
267 
268  if (paletteElements.size() > 1) {
269  for (Pid pid : PaletteElementScriptEntry::_pids) {
270  const QVariant val = e->getProperty(pid);
271  bool sameValue = false;
272  if (!val.isValid())
273  continue;
274  for (auto i = paletteElements.begin(); i != paletteElements.end(); ++i) {
275  const Element* pe = *i;
276  if (pe == e || !pe)
277  continue;
278  if (pe->getProperty(pid) == val)
279  sameValue = true;
280  else
281  (*i) = nullptr; // exclude it from further comparisons
282  }
283 
284  props.emplace_back(pid, serializePropertyValue(pid, val));
285  if (!sameValue)
286  break;
287  }
288  }
289 
290  return std::unique_ptr<ScriptEntry>(new PaletteElementScriptEntry(type, std::move(props)));
291  }

309 std::unique_ptr<ScriptEntry> PaletteElementScriptEntry::deserialize(const QStringList& tokens)
310  {
311  const int ntokens = tokens.size();
312  if (ntokens < 2) {
313  qWarning("palette: unexpected number of tokens: %d", ntokens);
314  return nullptr;
315  }
316  const ElementType type = ScoreElement::name2type(QStringRef(&tokens[1]));
317  std::vector<std::pair<Pid, QString>> props;
318  const int propTokenIdx = 2;
319  if (ntokens > propTokenIdx) {
320  if ((ntokens - propTokenIdx) % 2) {
321  qWarning("palette: unexpected number of tokens: %d", ntokens);
322  return nullptr;
323  }
324  for (int i = propTokenIdx; i < ntokens; i += 2) {
325  const Pid pid = deserializePropertyId(type, tokens[i]);
326  const QString value = tokens[i+1];
327  props.emplace_back(pid, value);
328  }
329  }
330  return std::unique_ptr<ScriptEntry>(new PaletteElementScriptEntry(type, std::move(props)));
331  }

338 bool InspectorScriptEntry::execute(ScriptContext& ctx) const
339  {
340  Inspector* i = ctx.mscore()->inspector();
341  if (!i)
342  return false;
343  InspectorBase* ib = i->ie;
344  if (!ib)
345  return false;
346 
347  const auto iiList = ib->iList;
348  for (unsigned idx = 0; idx < iiList.size(); ++idx) {
349  const InspectorItem& ii = iiList[idx];
350  if (ii.t == _pid && ii.parent == _parentLevel) {
351  const Score* score = ctx.mscore()->currentScore();
352  if (!score) {
353  qWarning() << "InspectorScriptEntry: no score!";
354  return false;
355  }
356  const auto scoreState = score->state();
357  ib->setValue(ii, _val);
358  if (scoreState == score->state()) // probably valueChanged wasn't triggered
359  ib->valueChanged(idx);
360  return true;
361  }
362  }
363  return false;
364  }
