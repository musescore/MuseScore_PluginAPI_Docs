41 void SegmentList::check()
42  {

44  Segment* f = 0;
45  Segment* l = 0;
46  for (Segment* s = _first; s; s = s->next()) {
47  if (f == 0)
48  f = s;

50  ++n;
51  }
52  for (Segment* s = _first; s; s = s->next()) {
53  switch (s->segmentType()) {
54  case SegmentType::Invalid:
55  case SegmentType::BeginBarLine:
56  case SegmentType::HeaderClef:
57  case SegmentType::Clef:
58  case SegmentType::KeySig:
59  case SegmentType::Ambitus:
60  case SegmentType::TimeSig:
61  case SegmentType::StartRepeatBarLine:
62  case SegmentType::BarLine:
63  case SegmentType::ChordRest:
64  case SegmentType::Breath:
65  case SegmentType::EndBarLine:
66  case SegmentType::TimeSigAnnounce:
67  case SegmentType::KeySigAnnounce:
68  break;
69  default:
70  qFatal("SegmentList::check: invalid segment type 0x%x", int(s->segmentType()));
71  break;
72  }
73  Segment* ss = s->next();
74  while (ss) {
75  if (s == ss) {
76  qFatal("SegmentList::check: segment twice in list");
77  }
78  ss = ss->next();
79  }
80  }
81  if (f != _first) {
82  qFatal("SegmentList::check: bad first");
83  }
84  if (l != _last) {
85  qFatal("SegmentList::check: bad last");
86  }
87  if (f && f->prev())
88  qFatal("SegmentList::check: first has prev");
89  if (l && l->next())
90  qFatal("SegmentList::check: last has next");
91  if (n != _size) {
92  qFatal("SegmentList::check: counted %d but _size is %d", n, _size);
93  _size = n;
94  }
95  }

124  {

152  e->prev()->setNext(e->next());
