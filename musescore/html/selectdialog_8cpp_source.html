41 SelectDialog::SelectDialog(const Element* _e, QWidget* parent)
42  : QDialog(parent)
43  {
44  setObjectName("SelectDialog");
45  setupUi(this);
46  setWindowFlags(this->windowFlags() & ~Qt::WindowContextHelpButtonHint);
47  e = _e;
48  type->setText(qApp->translate("elementName", e->userName().toUtf8()));
49 
50  switch (e->type()) {
51  case ElementType::ACCIDENTAL:
52  subtype->setText(qApp->translate("accidental", e->subtypeName().toUtf8()));
53  break;
54  case ElementType::SLUR_SEGMENT:
55  subtype->setText(qApp->translate("elementName", e->subtypeName().toUtf8()));
56  break;
57  case ElementType::FINGERING:
58  case ElementType::STAFF_TEXT:
59  subtype->setText(qApp->translate("TextStyle", e->subtypeName().toUtf8()));
60  break;
61  case ElementType::ARTICULATION: // comes translated, but from a different method
62  subtype->setText(toArticulation(e)->userName());
63  break;
64  // other come translated or don't need any or are too difficult to implement
65  default: subtype->setText(e->subtypeName());
66  }
67  sameSubtype->setEnabled(e->subtype() != -1);
68  subtype->setEnabled(e->subtype() != -1);
69  inSelection->setEnabled(e->score()->selection().isRange());
70  sameDuration->setEnabled(e->isRest());
71 
72  MuseScore::restoreGeometry(this);
73  }

79 void SelectDialog::setPattern(ElementPattern* p)
80  {
81  p->type = int(e->type());
82  p->subtype = e->subtype();
83  if (e->isSlurSegment())
84  p->subtype = int(toSlurSegment(e)->spanner()->type());
85 
86  if (sameStaff->isChecked()) {
87  p->staffStart = e->staffIdx();
88  p->staffEnd = e->staffIdx() + 1;
89  }
90  else if (inSelection->isChecked()) {
91  p->staffStart = e->score()->selection().staffStart();
92  p->staffEnd = e->score()->selection().staffEnd();
93  }
94  else {
95  p->staffStart = -1;
96  p->staffEnd = -1;
97  }
98 
99  if (sameDuration->isChecked() && e->isRest()) {
100  const Rest* r = toRest(e);
101  p->durationTicks = r->actualTicks();
102  }
103  else
104  p->durationTicks = Fraction(-1,1);
105 
106  if (sameBeat->isChecked())
107  p->beat = e->beat();
108  else
109  p->beat = Fraction(0,0);
110 
111  if (sameMeasure->isChecked())
112  p->measure = e->findMeasure();
113  else
114  p->measure = nullptr;
115 
116  p->voice = sameVoice->isChecked() ? e->voice() : -1;
117  p->subtypeValid = sameSubtype->isChecked();
118  if (sameSystem->isChecked()) {
119  do {
120  if (e->type() == ElementType::SYSTEM) {
121  p->system = static_cast<const System*>(e);
122  break;
123  }
124  e = e->parent();
125  } while (e);
126  }
