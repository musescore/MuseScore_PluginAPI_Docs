126 void SelectionWindow::updateFilteredElements()
127  {
128  if (!_score)
129  return;
130  int filter = _score->selectionFilter().filtered();
131  bool all = true;
132  bool none = true;
133  _listWidget->blockSignals(true);
134  for (int row = 1; row < _listWidget->count(); row++) {
135  QListWidgetItem *item = _listWidget->item(row);
136  if (filter & 1 << (row - 1)) {
137  if (item->checkState() != Qt::Checked)
138  item->setCheckState(Qt::Checked);
139  none = false;
140  }
141  else {
142  if (item->checkState() != Qt::Unchecked)
143  item->setCheckState(Qt::Unchecked);
144  all = false;
145  }
146  }
147  QListWidgetItem *item = _listWidget->item(0);
148  Qt::CheckState state = all ? Qt::Checked : (none ? Qt::Unchecked : Qt::PartiallyChecked);
149  if (item->checkState() != state)
150  item->setCheckState(state);
151  _listWidget->blockSignals(false);
152  }

158 void SelectionWindow::changeCheckbox(QListWidgetItem* item)
159  {
160  if (!_score)
161  return;
162  int type = item->data(Qt::UserRole).toInt();
163  bool set = false;
164  item->checkState() == Qt::Checked ? set = true : set = false;
165  if (type > 0) {
166  _score->selectionFilter().setFiltered(static_cast<SelectionFilterType>(type), set);
167  }
168  else {
169  for (size_t row = 1; row < numLabels; row++)
170  _score->selectionFilter().setFiltered(static_cast<SelectionFilterType>(1 << (row - 1)), set);
171  }
172  _score->startCmd();
173  if (_score->selection().isRange())
174  _score->selection().updateSelectedElements();
175  updateFilteredElements();
176  _score->setUpdateAll();
177  _score->endCmd();
178  ScoreAccessibility::instance()->updateAccessibilityInfo();
179  }
