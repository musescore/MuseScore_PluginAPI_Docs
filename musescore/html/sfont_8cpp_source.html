186 bool Preset::noteon(Fluid* synth, unsigned id, int chan, int key, int vel, double nt)
187  {
188  Mod* mod;
189  Mod* mod_list[FLUID_NUM_MOD]; /* list for 'sorting' preset modulators */
190 
191  Zone* global_preset_zone = global_zone();
192 
193  /* run thru all the zones of this preset */
194  for (Zone* preset_zone : qAsConst(zones)) {
195  /* check if the note falls into the key and velocity range of this
196  preset */
197  if (preset_zone->inside_range(key, vel)) {
198 
199  Instrument* inst = preset_zone->get_inst();
200  Zone* global_inst_zone = inst->get_global_zone();
201 
202  /* set portamento attributes*/
203  synth->setFromKeyPortamento(chan, key);
204 
205  /* run thru all the zones of this instrument */
206  for(Zone* inst_zone : inst->get_zone()) {
207  /* make sure this instrument zone has a valid sample */
208  Sample* sample = inst_zone->get_sample();
209  if (sample == 0 || sample->inRom())
210  continue;
211  /* check if the note falls into the key and velocity range of this
212  instrument */
213  if (inst_zone->inside_range(key, vel) && (sample != 0)) {
214 
215  /* this is a good zone. allocate a new synthesis process and
216  initialize it */
217 
218  Voice* voice = synth->alloc_voice(id, sample, chan, key, vel, nt);
219  if (voice == 0)
220  return false;
221 
222  /* Instrumentrument level, generators */
223 
224  for (int i = 0; i < GEN_LAST; i++) {
225  /* SF 2.01 section 9.4 'bullet' 4:
226  *
227  * A generator in a local instrument zone supersedes a
228  * global instrument zone generator. Both cases supersede
229  * the default generator -> voice_gen_set */
230 
231  if (inst_zone->genlist[i].flags)
232  voice->gen_set(i, inst_zone->genlist[i].val);
233  else if ((global_inst_zone != 0) && (global_inst_zone->genlist[i].flags))
234  voice->gen_set(i, global_inst_zone->genlist[i].val);
235  else {
236  /* The generator has not been defined in this instrument.
237  * Do nothing, leave it at the default.
238  */
239  }
240 
241  } /* for all generators */
242 
243  /* global instrument zone, modulators: Put them all into a
244  * list. */
245 
246  int mod_list_count = 0;
247 
248  if (global_inst_zone){
249  for(Mod* mod1 : qAsConst(global_inst_zone->modlist))
250  mod_list[mod_list_count++] = mod1;
251  }
252 
253  /* local instrument zone, modulators.
254  * Replace modulators with the same definition in the list:
255  * SF 2.01 page 69, 'bullet' 8
256  */
257  for(Mod* mod1 : qAsConst(inst_zone->modlist)) {
258  /* 'Identical' modulators will be deleted by setting their
259  * list entry to 0. The list length is known, 0
260  * entries will be ignored later. SF2.01 section 9.5.1
261  * page 69, 'bullet' 3 defines 'identical'. */
262 
263  for (int i = 0; i < mod_list_count; i++){
264  if (mod_list[i] && test_identity(mod1, mod_list[i])){
265  mod_list[i] = 0;
266  }
267  }
268 
269  /* Finally add the new modulator to to the list. */
270  mod_list[mod_list_count++] = mod1;
271  }
272 
273  /* Add instrument modulators (global / local) to the voice. */
274  for (int i = 0; i < mod_list_count; i++){
275  mod = mod_list[i];
276  if (mod) { // disabled modulators CANNOT be skipped.
277  /* Instrumentrument modulators -supersede- existing (default)
278  * modulators. SF 2.01 page 69, 'bullet' 6 */
279  voice->add_mod(mod, FLUID_VOICE_OVERWRITE);
280  }
281  }
282 
283  /* Preset level, generators */
284 
285  for (int i = 0; i < GEN_LAST; i++) {
286  /* SF 2.01 section 8.5 page 58: If some generators are
287  * encountered at preset level, they should be ignored */
288  if ((i != GEN_STARTADDROFS)
289  && (i != GEN_ENDADDROFS)
290  && (i != GEN_STARTLOOPADDROFS)
291  && (i != GEN_ENDLOOPADDROFS)
292  && (i != GEN_STARTADDRCOARSEOFS)
293  && (i != GEN_ENDADDRCOARSEOFS)
294  && (i != GEN_STARTLOOPADDRCOARSEOFS)
295  && (i != GEN_KEYNUM)
296  && (i != GEN_VELOCITY)
297  && (i != GEN_ENDLOOPADDRCOARSEOFS)
298  && (i != GEN_SAMPLEMODE)
299  && (i != GEN_EXCLUSIVECLASS)
300  && (i != GEN_OVERRIDEROOTKEY)) {
301 
302  /* SF 2.01 section 9.4 'bullet' 9: A generator in a
303  * local preset zone supersedes a global preset zone
304  * generator. The effect is -added- to the destination
305  * summing node -> voice_gen_incr */
306 
307  if (preset_zone->genlist[i].flags) {
308  voice->gen_incr(i, preset_zone->genlist[i].val);
309  }
310  else if ((global_preset_zone != 0) && global_preset_zone->genlist[i].flags) {
311  voice->gen_incr(i, global_preset_zone->genlist[i].val);
312  }
313  else {
314  /* The generator has not been defined in this preset
315  * Do nothing, leave it unchanged.
316  */
317  }
318  } /* if available at preset level */
319  } /* for all generators */
320 
321 
322  /* Global preset zone, modulators: put them all into a
323  * list. */
324  mod_list_count = 0;
325  if (global_preset_zone){
326  for(Mod* mod1 : qAsConst(global_preset_zone->modlist))
327  mod_list[mod_list_count++] = mod1;
328  }
329 
330  /* Process the modulators of the local preset zone. Kick
331  * out all identical modulators from the global preset zone
332  * (SF 2.01 page 69, second-last bullet) */
333 
334  for(Mod* mod1 : qAsConst(preset_zone->modlist)) {
335  for (int i = 0; i < mod_list_count; i++) {
336  if (mod_list[i] && test_identity(mod1,mod_list[i]))
337  mod_list[i] = 0;
338  }
339  /* Finally add the new modulator to the list. */
340  mod_list[mod_list_count++] = mod1;
341  }
342 
343  /* Add preset modulators (global / local) to the voice. */
344  for (int i = 0; i < mod_list_count; i++){
345  mod = mod_list[i];
346  if ((mod != 0) && (mod->amount != 0)) { /* disabled modulators can be skipped. */
347  /* Preset modulators -add- to existing instrument /
348  * default modulators. SF2.01 page 70 first bullet on
349  * page */
350  voice->add_mod(mod, FLUID_VOICE_ADD);
351  }
352  }
353 
354  /* add the synthesis process to the synthesis loop. */
355  synth->start_voice(voice);
356  synth->setLastNote(key);
357  /* Store the ID of the first voice that was created by this noteon event.
358  * Exclusive class may only terminate older voices.
359  * That avoids killing voices, which have just been created.
360  * (a noteon event can create several voice processes with the same exclusive
361  * class - for example when using stereo samples)
362  */
363  }
364  }
365  }
366  }
367  return true;
368  }

469 bool Zone::importZone()
470  {
471  for (SFGen* sfgen : qAsConst(gen)) {
472  switch (sfgen->id) {
473  case GEN_KEYRANGE:
474  keylo = sfgen->amount.range.lo;
475  keyhi = sfgen->amount.range.hi;
476  break;
477  case GEN_VELRANGE:
478  vello = sfgen->amount.range.lo;
479  velhi = sfgen->amount.range.hi;
480  break;
481  default:
482  /* FIXME: some generators have an unsigned word amount value but i don't know which ones
483  */
484  genlist[sfgen->id].val = (float) sfgen->amount.sword;
485  genlist[sfgen->id].flags = GEN_SET;
486  break;
487  }
488  }
489 
490  // Import the modulators (only SF2.1 and higher)
491  for(SFMod* mod_src : qAsConst(mod)) {
492  Mod* mod_dest = new Mod;
493  int type;
494  // mod_dest->next = 0; /* pointer to next modulator, this is the end of the list now.*/
495 
496  /* *** Amount *** */
497  mod_dest->amount = mod_src->amount;
498 
499  /* *** Source *** */
500  mod_dest->src1 = mod_src->src & 127; /* index of source 1, seven-bit value, SF2.01 section 8.2, page 50 */
501  mod_dest->flags1 = 0;
502 
503  /* Bit 7: CC flag SF 2.01 section 8.2.1 page 50*/
504  if (mod_src->src & (1<<7))
505  mod_dest->flags1 |= FLUID_MOD_CC;
506  else
507  mod_dest->flags1 |= FLUID_MOD_GC;
508 
509  /* Bit 8: D flag SF 2.01 section 8.2.2 page 51*/
510  if (mod_src->src & (1<<8))
511  mod_dest->flags1 |= FLUID_MOD_NEGATIVE;
512  else
513  mod_dest->flags1 |= FLUID_MOD_POSITIVE;
514 
515  /* Bit 9: P flag SF 2.01 section 8.2.3 page 51*/
516  if (mod_src->src & (1<<9))
517  mod_dest->flags1 |= FLUID_MOD_BIPOLAR;
518  else
519  mod_dest->flags1 |= FLUID_MOD_UNIPOLAR;
520 
521  /* modulator source types: SF2.01 section 8.2.1 page 52 */
522  type = (mod_src->src) >> 10;
523  type &= 63; /* type is a 6-bit value */
524  if (type == 0)
525  mod_dest->flags1 |= FLUID_MOD_LINEAR;
526  else if (type == 1)
527  mod_dest->flags1 |= FLUID_MOD_CONCAVE;
528  else if (type == 2)
529  mod_dest->flags1 |= FLUID_MOD_CONVEX;
530  else if (type == 3)
531  mod_dest->flags1 |= FLUID_MOD_SWITCH;
532  else {
533  /* This shouldn't happen - unknown type!
534  * Deactivate the modulator by setting the amount to 0.
535  */
536  mod_dest->amount=0;
537  }
538 
539  /* *** Dest *** */
540  mod_dest->dest = mod_src->dest; /* index of controlled generator */
541 
542  /* *** Amount source *** */
543  mod_dest->src2 = mod_src->amtsrc & 127; /* index of source 2, seven-bit value, SF2.01 section 8.2, p.50 */
544  mod_dest->flags2 = 0;
545 
546  /* Bit 7: CC flag SF 2.01 section 8.2.1 page 50*/
547  if (mod_src->amtsrc & (1<<7))
548  mod_dest->flags2 |= FLUID_MOD_CC;
549  else
550  mod_dest->flags2 |= FLUID_MOD_GC;
551 
552  /* Bit 8: D flag SF 2.01 section 8.2.2 page 51*/
553  if (mod_src->amtsrc & (1<<8))
554  mod_dest->flags2 |= FLUID_MOD_NEGATIVE;
555  else
556  mod_dest->flags2 |= FLUID_MOD_POSITIVE;
557 
558  /* Bit 9: P flag SF 2.01 section 8.2.3 page 51*/
559  if (mod_src->amtsrc & (1<<9))
560  mod_dest->flags2 |= FLUID_MOD_BIPOLAR;
561  else
562  mod_dest->flags2 |= FLUID_MOD_UNIPOLAR;
563 
564  /* modulator source types: SF2.01 section 8.2.1 page 52 */
565  type = (mod_src->amtsrc) >> 10;
566  type &= 63; /* type is a 6-bit value */
567  if (type == 0)
568  mod_dest->flags2 |= FLUID_MOD_LINEAR;
569  else if (type == 1)
570  mod_dest->flags2 |= FLUID_MOD_CONCAVE;
571  else if (type == 2)
572  mod_dest->flags2 |= FLUID_MOD_CONVEX;
573  else if (type == 3)
574  mod_dest->flags2 |= FLUID_MOD_SWITCH;
575  else {
576  /* This shouldn't happen - unknown type!
577  * Deactivate the modulator by setting the amount to 0. */
578  mod_dest->amount=0;
579  }
580 
581  /* *** Transform *** */
582  /* SF2.01 only uses the 'linear' transform (0).
583  * Deactivate the modulator by setting the amount to 0 in any other case.
584  */
585  if (mod_src->trans !=0)
586  mod_dest->amount = 0;
587 
588  /* Store the new modulator in the zone The order of modulators
589  * will make a difference, at least in an instrument context: The
590  * second modulator overwrites the first one, if they only differ
591  * in amount.
592  */
593  modlist.append(mod_dest);
594  }
595  return true;
596  }

632 void Sample::load()
633  {
634  if (!_valid || data)
635  return;
636  QFile fd(sf->get_name());
637  if (!fd.open(QIODevice::ReadOnly))
638  return;
639  if (sampletype & FLUID_SAMPLETYPE_OGG_VORBIS) {
640  if (!fd.seek(sf->samplePos() + start))
641  return;
642  }
643  else {
644  if (!fd.seek(sf->samplePos() + start * sizeof(short)))
645  return;
646  }
647  unsigned int size = end - start;
648 
649  if (sampletype & FLUID_SAMPLETYPE_OGG_VORBIS) {
650 #ifdef SOUNDFONT3
651  std::vector<char> p;
652  p.resize(size);
653  if (fd.read(p.data(), size) != size) {
654  qDebug("read %d failed", size);
655  return;
656  }
657  decompressOggVorbis(p.data(), size);
658 #endif
659  }
660  else {
661  data = new short[size];
662  size *= sizeof(short);
663 
664  if (fd.read((char*)data, size) != size)
665  return;
666 
667  if (QSysInfo::ByteOrder == QSysInfo::BigEndian) {
668  unsigned char hi, lo;
669  unsigned int i, j;
670  short s;
671  uchar* cbuf = (uchar*) data;
672  for (i = 0, j = 0; j < size; i++) {
673  lo = cbuf[j++];
674  hi = cbuf[j++];
675  s = (hi << 8) | lo;
676  data[i] = s;
677  }
678  }
679  end -= (start + 1); // marks last sample, contrary to SF spec.
680  loopstart -= start;
681  loopend -= start;
682  start = 0;
683  }
684  optimize();
685  }
