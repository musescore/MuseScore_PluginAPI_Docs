249 int SfzRegion::readKey(const QString& s, SfzControl c) const
250  {
251  bool ok;
252  int i = s.toInt(&ok);
253  if (ok) {
254  i += c.octave_offset * 12;
255  i += c.note_offset;
256  return i;
257  }
258  switch (s[0].toLower().toLatin1()) {
259  case 'c': i = 0; break;
260  case 'd': i = 2; break;
261  case 'e': i = 4; break;
262  case 'f': i = 5; break;
263  case 'g': i = 7; break;
264  case 'a': i = 9; break;
265  case 'b': i = 11; break;
266  default:
267  qDebug("SfzRegion: Not a note: %s", qPrintable(s));
268  return 0;
269  }
270  int idx = 1;
271  if (s[idx].toLatin1() == '#') {
272  i++;
273  ++idx;
274  }
275  else if (s[idx].toLower().toLatin1() == 'b') {
276  i--;
277  ++idx;
278  }
279  int octave = s.midRef(idx).toInt(&ok);
280  if (!ok) {
281  qDebug("SfzRegion: Not a note: %s", qPrintable(s));
282  return 0;
283  }
284  i += (octave + 1) * 12;
285  i += c.octave_offset * 12;
286  i += c.note_offset;
287  return i;
288  }

294 void ZInstrument::addRegion(SfzRegion& r)
295  {
296  for (int i = 0; i < 128; ++i) {
297  if (r.on_locc[i] != -1 || r.on_hicc[i] != -1) {
298  r.trigger = Trigger::CC;
299  break;
300  }
301  }
302  Zone* z = new Zone;
303  z->sample = readSample(r.sample, 0);
304  if (z->sample) {
305  //qDebug("Sample Loop - start %ll, end %ll, mode %d", z->sample->loopStart(), z->sample->loopEnd(), z->sample->loopMode());
306  // if there is no opcode defining loop ranges, use sample definitions as fallback (according to spec)
307  if (r.loopStart == -1)
308  r.loopStart = z->sample->loopStart();
309  if (r.loopEnd == -1)
310  r.loopEnd = z->sample->loopEnd();
311  }
312  r.setZone(z);
313  if (z->sample)
314  addZone(z);
315  }

592 bool ZInstrument::loadSfz(const QString& s)
593  {
594  _program = 0;
595  QFileInfo fi(s);
596  QString path = fi.absolutePath();
597 
598  QStringList fileContents = readFile(s);
599 
600  if (fileContents.empty()) {
601  return false;
602  }
603 
604  SfzControl c;
605  c.init();
606  c.defines.clear();
607  for (int i = 0;i < 128; i++)
608  c.set_cc[i] = -1;
609 
610  int idx0 = 0;
611  bool inBlockComment = false;
612  // preprocessor
613  while(idx0 < fileContents.size()) {
614  QRegularExpression findWithSpaces("\"(.+)\"");
615  QRegularExpression comment("//.*$");
616  QRegularExpression trailingSpacesOrTab("^[\\s\\t]*");
617  QRegularExpressionMatch foundWithSpaces;
618  QString curLine = fileContents[idx0];
619  QString curLineCopy = curLine;
620  bool nextIsImportant = false;
621  int idxBlockComment = 0;
622  int from = 0;
623 
624  for (QChar chr : curLineCopy) {
625  bool terminated = false;
626 
627  if (nextIsImportant) {
628  nextIsImportant = false;
629  if (inBlockComment && chr == '/') { // found block end
630  inBlockComment = false;
631  terminated = true;
632  curLine.remove(from, idxBlockComment - from + 1);
633  idxBlockComment = from - 1;
634  }
635  else if (!inBlockComment && chr == '*') { // found block start
636  inBlockComment = true;
637  terminated = true;
638  from = idxBlockComment - 1;
639  }
640  }
641 
642  if (!terminated && inBlockComment && chr == '*')
643  nextIsImportant = true;
644  else if (!terminated && !inBlockComment && chr == '/')
645  nextIsImportant = true;
646 
647  idxBlockComment++;
648  }
649 
650  if (inBlockComment)
651  curLine.remove(from, curLine.size() - from);
652 
653  curLine = curLine.remove(comment);
654  curLine.remove(trailingSpacesOrTab);
655  fileContents[idx0] = curLine;
656 
657  if (curLine.startsWith("#define")) {
658  QStringList define = curLine.split(" ");
659  foundWithSpaces = findWithSpaces.match(curLine);
660  if (define.size() == 3)
