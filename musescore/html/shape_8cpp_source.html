20 // Currently implemented by adding rectangles of zero
21 // height to the Y position corresponding to the type.
22 // This is a simple solution but has its drawbacks too.

25 void Shape::addHorizontalSpacing(HorizontalSpacingType type, qreal leftEdge, qreal rightEdge)
26  {
27  constexpr qreal eps = 100 * std::numeric_limits<qreal>::epsilon();
28  const qreal y = eps * int(type);
29  if (leftEdge == rightEdge) // HACK zero-width shapes collide with everything currently.
30  rightEdge += eps;
31  add(QRectF(leftEdge, y, rightEdge - leftEdge, 0));
32  }

76 // minHorizontalDistance
77 // a is located right of this shape.
78 // Calculates the minimum horizontal distance between the two shapes
79 // so they donâ€™t touch.
80 //-------------------------------------------------------------------
81 
82 qreal Shape::minHorizontalDistance(const Shape& a) const
83  {
84  qreal dist = -1000000.0; // min real
85  for (const QRectF& r2 : a) {
86  qreal by1 = r2.top();
87  qreal by2 = r2.bottom();
88  for (const QRectF& r1 : *this) {
89  qreal ay1 = r1.top();
90  qreal ay2 = r1.bottom();
91  if (Ms::intersects(ay1, ay2, by1, by2)
92  || ((r1.height() == 0.0) && (r2.height() == 0.0) && (ay1 == by1))
93  || ((r1.width() == 0.0) || (r2.width() == 0.0)))
94  dist = qMax(dist, r1.right() - r2.left());
95  }
96  }
97  return dist;
98  }

101 // minVerticalDistance
102 // a is located below this shape.
103 // Calculates the minimum distance between two shapes.
104 //-------------------------------------------------------------------
105 
106 qreal Shape::minVerticalDistance(const Shape& a) const
107  {
108  qreal dist = -1000000.0; // min real
109  for (const QRectF& r2 : a) {
110  if (r2.height() <= 0.0)
111  continue;
112  qreal bx1 = r2.left();
113  qreal bx2 = r2.right();
114  for (const QRectF& r1 : *this) {
115  if (r1.height() <= 0.0)
116  continue;
117  qreal ax1 = r1.left();
118  qreal ax2 = r1.right();
119  if (Ms::intersects(ax1, ax2, bx1, bx2))
120  dist = qMax(dist, r1.bottom() - r2.top());
121  }
122  }
123  return dist;
124  }
