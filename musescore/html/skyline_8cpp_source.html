104 void SkylineLine::add(qreal x, qreal y, qreal w)
105  {
106 // Q_ASSERT(w >= 0.0);
107  if (x < 0.0) {
108  w -= -x;
109  x = 0.0;
110  if (w <= 0.0)
111  return;
112  }
113 
114  DP("===add %f %f %f\n", x, y, w);
115 
116  SegIter i = find(x);
117  qreal cx = seg.empty() ? 0.0 : i->x;
118  for (; i != seg.end(); ++i) {
119  qreal cy = i->y;
120  if ((x + w) <= cx) // A
121  return; // break;
122  if (x > (cx + i->w)) { // B
123  cx += i->w;
124  continue;
125  }
126  if ((north && (cy <= y)) || (!north && (cy >= y))) {
127  cx += i->w;
128  continue;
129  }
130  if ((x >= cx) && ((x+w) < (cx+i->w))) { // (E) insert segment
131  DP(" insert at %f %f x:%f w:%f\n", cx, i->w, x, w);
132  qreal w1 = x - cx;
133  qreal w2 = w;
134  qreal w3 = i->w - (w1 + w2);
135  if (w1 > 0.0000001) {
136  i->w = w1;
137  ++i;
138  i = insert(i, x, y, w2);
139  DP(" A w1 %f w2 %f\n", w1, w2);
140  }
141  else {
142  i->w = w2;
143  i->y = y;
144  DP(" B w2 %f\n", w2);
145  }
146  if (w3 > 0.0000001) {
147  ++i;
148  DP(" C w3 %f\n", w3);
149  insert(i, x + w2, cy, w3);
150  }
151  return;
152  }
153  else if ((x <= cx) && ((x + w) >= (cx + i->w))) { // F
154  DP(" change(F) cx %f y %f\n", cx, y);
155  i->y = y;
156  }
157  else if (x < cx) { // C
158  qreal w1 = x + w - cx;
159  i->w -= w1;
160  DP(" add(C) cx %f y %f w %f w1 %f\n", cx, y, w1, i->w);
161  insert(i, cx, y, w1);
162  return;
163  }
164  else { // D
165  qreal w1 = x - cx;
166  qreal w2 = i->w - w1;
167  if (w2 > 0.0000001) {
168  i->w = w1;
169  cx += w1;
170  DP(" add(D) %f %f\n", y, w2);
171  ++i;
172  i = insert(i, cx, y, w2);
173  }
174  }
175  cx += i->w;
176  }
177  if (x >= cx) {
178  if (x > cx) {
179  qreal cy = north ? MAXIMUM_Y : MINIMUM_Y;
180  DP(" append1 %f %f\n", cy, x - cx);
181  append(cx, cy, x - cx);
182  }
183  DP(" append2 %f %f\n", y, w);
184  append(x, y, w);
185  }
186  else if (x + w > cx)
187  append(cx, y, x + w - cx);
188  }

211 qreal SkylineLine::minDistance(const SkylineLine& sl) const
212  {
213  qreal dist = MINIMUM_Y;
214 
215  qreal x1 = 0.0;
216  qreal x2 = 0.0;
217  auto k = sl.begin();
218  for (auto i = begin(); i != end(); ++i) {
219  while (k != sl.end() && (x2 + k->w) < x1) {
220  x2 += k->w;
221  ++k;
222  }
223  if (k == sl.end())
224  break;
225  for (;;) {
226  if ((x1 + i->w > x2) && (x1 < x2 + k->w))
227  dist = qMax(dist, i->y - k->y);
228  if (x2 + k->w < x1 + i->w) {
229  x2 += k->w;
230  ++k;
231  if (k == sl.end())
232  break;
233  }
234  else
235  break;
236  }
237  if (k == sl.end())
238  break;
239  x1 += i->w;
240  }
241  return dist;
242  }

261 void SkylineLine::paint(QPainter& p) const
262  {
263  qreal x1 = 0.0;
264  qreal x2;
265  qreal y = 0.0;
266 
267  bool pvalid = false;
