169 void Slider::mouseMoveEvent(QMouseEvent* ev)
170  {
171  if (!dragMode)
172  return;
173  int delta = orient == Qt::Horizontal ? (startDrag.x() - ev->x()) : (startDrag.y() - ev->y());
174 
175 // if (_invert)
176 // delta = -delta;
177  if (orient == Qt::Horizontal)
178  delta = -delta;
179  int ppos = dragppos + delta;
180  if (ppos < 0)
181  ppos = 0;
182 
183  int pixel = (orient == Qt::Vertical) ? height() - _sliderSize.height() : width() - _sliderSize.width();
184  if (ppos > pixel)
185  ppos = pixel;
186  int pos = _invert ? (pixel - ppos) : ppos;
187  _value = (pos * (maxValue() - minValue()) / pixel) + minValue() - 0.000001;
188  valueChange();
189  }
