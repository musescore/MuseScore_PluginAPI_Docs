239 void SlurSegment::computeBezier(QPointF p6o)
240  {
241  qreal _spatium = spatium();
242  qreal shoulderW; // height as fraction of slur-length
243  qreal shoulderH;
244  //
245  // pp1 and pp2 are the end points of the slur
246  //
247  QPointF pp1 = ups(Grip::START).p + ups(Grip::START).off;
248  QPointF pp2 = ups(Grip::END).p + ups(Grip::END).off;
249 
250  QPointF p2 = pp2 - pp1;
251  if ((p2.x() == 0.0) && (p2.y() == 0.0)) {
252  Measure* m1 = slur()->startCR()->segment()->measure();
253  Measure* m2 = slur()->endCR()->segment()->measure();
254  qDebug("zero slur at tick %d(%d) track %d in measure %d-%d tick %d ticks %d",
255  m1->tick().ticks(), tick().ticks(), track(), m1->no(), m2->no(), slur()->tick().ticks(), slur()->ticks().ticks());
256  slur()->setBroken(true);
257  return;
258  }
259 
260  qreal sinb = atan(p2.y() / p2.x());
261  QTransform t;
262  t.rotateRadians(-sinb);
263  p2 = t.map(p2);
264  p6o = t.map(p6o);
265 
266  double smallH = 0.5;
267  qreal d = p2.x() / _spatium;
268  if (d <= 2.0) {
269  shoulderH = d * 0.5 * smallH * _spatium;
270  shoulderW = .6;
271  }
272  else {
273  qreal dd = log10(1.0 + (d - 2.0) * .5) * 2.0;
274  if (dd > 3.0)
275  dd = 3.0;
276  shoulderH = (dd + smallH) * _spatium + _extraHeight;
277  if (d > 18.0)
278  shoulderW = 0.7; // 0.8;
279  else if (d > 10)
280  shoulderW = 0.6; // 0.7;
281  else
282  shoulderW = 0.5; // 0.6;
283  }
284 
285  shoulderH -= p6o.y();
286 
287  if (!slur()->up())
288  shoulderH = -shoulderH;
289 
290  qreal c = p2.x();
291  qreal c1 = (c - c * shoulderW) * .5 + p6o.x();
292  qreal c2 = c1 + c * shoulderW + p6o.x();
293 
294  QPointF p5 = QPointF(c * .5, 0.0);
295 
296  QPointF p3(c1, -shoulderH);
297  QPointF p4(c2, -shoulderH);
298 
299  qreal w = score()->styleP(Sid::SlurMidWidth) - score()->styleP(Sid::SlurEndWidth);
300  if (staff())
301  w *= staff()->mag(slur()->tick());
302  if ((c2 - c1) <= _spatium)
303  w *= .5;
304  QPointF th(0.0, w); // thickness of slur
305 
306  QPointF p3o = p6o + t.map(ups(Grip::BEZIER1).off);
307  QPointF p4o = p6o + t.map(ups(Grip::BEZIER2).off);
308 
309  if (!p6o.isNull()) {
310  QPointF p6i = t.inverted().map(p6o);
311  ups(Grip::BEZIER1).off += p6i ;
312  ups(Grip::BEZIER2).off += p6i;
313  }
314 
315  //-----------------------------------calculate p6
316  QPointF pp3 = p3 + p3o;
317  QPointF pp4 = p4 + p4o;
318  QPointF ppp4 = pp4 - pp3;
319 
320  qreal r2 = atan(ppp4.y() / ppp4.x());
321  t.reset();
322  t.rotateRadians(-r2);
323  QPointF p6 = QPointF(t.map(ppp4).x() * .5, 0.0);
324 
325  t.rotateRadians(2 * r2);
326  p6 = t.map(p6) + pp3 - p6o;
327  //-----------------------------------
328 
329  path = QPainterPath();
330  path.moveTo(QPointF());
331  path.cubicTo(p3 + p3o - th, p4 + p4o - th, p2);
332  if (slur()->lineType() == 0)
333  path.cubicTo(p4 +p4o + th, p3 + p3o + th, QPointF());
334 
335  th = QPointF(0.0, 3.0 * w);
336  shapePath = QPainterPath();
337  shapePath.moveTo(QPointF());
338  shapePath.cubicTo(p3 + p3o - th, p4 + p4o - th, p2);
339  shapePath.cubicTo(p4 +p4o + th, p3 + p3o + th, QPointF());
340 
341  // translate back
342  t.reset();
343  t.translate(pp1.x(), pp1.y());
344  t.rotateRadians(sinb);
345  path = t.map(path);
346  shapePath = t.map(shapePath);
347  ups(Grip::BEZIER1).p = t.map(p3);
348  ups(Grip::BEZIER2).p = t.map(p4);
349  ups(Grip::END).p = t.map(p2) - ups(Grip::END).off;
350  ups(Grip::DRAG).p = t.map(p5);
351  ups(Grip::SHOULDER).p = t.map(p6);
352 
353  _shape.clear();
354  QPointF start = pp1;
355  int nbShapes = 32; // (pp2.x() - pp1.x()) / _spatium;
356  qreal minH = qAbs(3 * w);
357  const CubicBezier b(pp1, ups(Grip::BEZIER1).pos(), ups(Grip::BEZIER2).pos(), ups(Grip::END).pos());
358  for (int i = 1; i <= nbShapes; i++) {
359  const QPointF point = b.pointAtPercent(i/float(nbShapes));
360  QRectF re = QRectF(start, point).normalized();
361  if (re.height() < minH) {
362  qreal d1 = (minH - re.height()) * .5;
363  re.adjust(0.0, -d1, 0.0, d1);
364  }
365  _shape.add(re);
366  start = point;
367  }
368  }

374 void SlurSegment::layoutSegment(const QPointF& p1, const QPointF& p2)
375  {
376  setPos(QPointF());
377  ups(Grip::START).p = p1;
378  ups(Grip::END).p = p2;
379  _extraHeight = 0.0;
380 
381  //Adjust Y pos to staff type yOffset before other calculations
382  if (staffType())
383  rypos() += staffType()->yoffset().val() * spatium();
384 
385  computeBezier();
386 
387  if (autoplace() && system()) {
388  bool up = slur()->up();
389  Segment* ls = system()->lastMeasure()->last();
390  Segment* fs = system()->firstMeasure()->first();
391  Segment* ss = slur()->startSegment();
392  Segment* es = slur()->endSegment();
393  QPointF pp1 = ups(Grip::START).p;
394  QPointF pp2 = ups(Grip::END).p;
395  qreal slurMaxMove = spatium();
396  bool intersection = false;
397  qreal gdist = 0.0;
398  qreal minDistance = score()->styleS(Sid::SlurMinDistance).val() * spatium();
399  for (int tries = 1; true; ++tries) {
400  for (Segment* s = fs; s && s != ls; s = s->next1()) {
401  if (!s->enabled())
402  continue;
403  // skip start and end segments on assumption start and end points were placed well already
404  // this avoids overcorrection on collision with own ledger lines and accidentals
405  // it also avoids issues where slur appears to be attached to a note in a different voice
406  if (s == ss || s == es)
407  continue;
408  // allow slurs to cross barlines
409  if (s->segmentType() & SegmentType::BarLineType)
410  continue;
411  qreal x1 = s->x() + s->measure()->x();
412  qreal x2 = x1 + s->width();
413  if (pp1.x() > x2)
414  continue;
415  if (pp2.x() < x1)
416  break;
417  const Shape& segShape = s->staffShape(staffIdx()).translated(s->pos() + s->measure()->pos());
418  if (!intersection)
419  intersection = segShape.intersects(_shape);
420  if (up) {
421  //QPointF pt = QPointF(s->x() + s->measure()->x(), s->staffShape(staffIdx()).top() + s->y() + s->measure()->y());
422  qreal dist = _shape.minVerticalDistance(segShape);
423  if (dist > 0.0)
424  gdist = qMax(gdist, dist);
425  }
426  else {
427  //QPointF pt = QPointF(s->x() + s->measure()->x(), s->staffShape(staffIdx()).bottom() + s->y() + s->measure()->y());
428  qreal dist = segShape.minVerticalDistance(_shape);
429  if (dist > 0.0)
430  gdist = qMax(gdist, dist);
431  }
432  }
433  if (!intersection || gdist <= slurMaxMove || tries >= 2)
434  break;
435  // slur would be moved too far
436  // try again with a steeper curve
437  _extraHeight += gdist;
438  computeBezier();
439  intersection = false;
440  gdist = 0.0;
441  }
442  if (intersection && gdist > 0.0) {
443  qreal min = minDistance + gdist;
444  rypos() += up ? -min : min;
445  }
446  }
447  setbbox(path.boundingRect());
448  }

613  sp->system2 = ecr->measure()->system();

624  if (scr->vStaffIdx() != vStaffIdx() && sp->system1) {
625  qreal diff = sp->system1->staff(scr->vStaffIdx())->y() - sp->system1->staff(vStaffIdx())->y();
626  sp->p1.ry() += diff;
627  }
628  if (ecr->vStaffIdx() != vStaffIdx() && sp->system2) {
629  qreal diff = sp->system2->staff(ecr->vStaffIdx())->y() - sp->system2->staff(vStaffIdx())->y();
630  sp->p2.ry() += diff;
631  }
632 
633  // account for centering or other adjustments (other than mirroring)
634  if (note1 && !note1->mirror())
635  sp->p1.rx() += note1->x();
636  if (note2 && !note2->mirror())
637  sp->p2.rx() += note2->x();
638 
639  qreal xo, yo;
640 
641  Stem* stem1 = sc && staffHasStems ? sc->stem() : 0;
642  Stem* stem2 = ec && staffHasStems ? ec->stem() : 0;
643 
644  enum class SlurAnchor : char {
645  NONE, STEM
646  };
647  SlurAnchor sa1 = SlurAnchor::NONE;
648  SlurAnchor sa2 = SlurAnchor::NONE;
649  // if slur is 'embedded' between either stem or both (as it might happen with voices)
650  // link corresponding slur end to stem position
651  if ((scr->up() == ecr->up()) && !scr->beam() && !ecr->beam() && (_up == scr->up())) {
652  // both chords are facing same direction and slur is also in same direction
653  // and no beams
654  if (stem1)
655  sa1 = SlurAnchor::STEM;
656  if (stem2)
657  sa2 = SlurAnchor::STEM;
658  }
659  // also link start of slur to stem if start chord & slur are in same direction and there is a hook
660  if (scr->up() == _up && stem1 && sc->hook()) {
661  sa1 = SlurAnchor::STEM;
662  // if end chord is in same direction, link end of slur to stem too
663  if (ecr->up() == scr->up() && stem2 && (!ecr->beam() || !ecr->beam()->cross()))
664  sa2 = SlurAnchor::STEM;
665  }
666 
667  qreal __up = _up ? -1.0 : 1.0;
668  qreal hw1 = note1 ? note1->tabHeadWidth(stt) : scr->width(); // if stt == 0, tabHeadWidth()
669  qreal hw2 = note2 ? note2->tabHeadWidth(stt) : ecr->width(); // defaults to headWidth()
670  QPointF pt;
671  switch (sa1) {
672  case SlurAnchor::STEM: //sc can't be null
673  {
674  // place slur starting point at stem end point
675  pt = sc->stemPos() - sc->pagePos() + sc->stem()->p2();
676  if (useTablature) // in tabs, stems are centred on note:
677  pt.rx() = hw1 * 0.5 + (note1 ? note1->bboxXShift() : 0.0); // skip half notehead to touch stem, anatoly-os: incorrect. half notehead width is not always the stem position
678  // clear the stem (x)
679  // allow slight overlap (y) as per Gould
680  // don't allow overlap with hook if not disabling the autoplace checks against start/end segments in SlurSegment::layoutSegment()
681  qreal yadj = -0.25; // sc->hook() ? 0.25 : -0.25;
682  yadj *= _spatium * __up;
683  pt += QPointF(0.35 * _spatium, yadj);
684  // account for articulations
685  pt.ry() = fixArticulations(pt.y(), sc, __up, true);
686  sp->p1 += pt;
687  }
688  break;
689  case SlurAnchor::NONE:
690  break;
691  }
692  switch (sa2) {
693  case SlurAnchor::STEM: //ec can't be null
694  {
695  pt = ec->stemPos() - ec->pagePos() + ec->stem()->p2();
696  if (useTablature)
697  pt.rx() = hw2 * 0.5;
698  // don't allow overlap with beam
699  qreal yadj = ec->beam() ? 0.75 : -0.25;
700  yadj *= _spatium * __up;
701  pt += QPointF(-0.35 * _spatium, yadj);
702  // account for articulations
703  pt.ry() = fixArticulations(pt.y(), ec, __up, true);
704  sp->p2 += pt;
705  }
706  break;
707  case SlurAnchor::NONE:
708  break;
709  }
710 
711  //
712  // default position:
713  // horizontal: middle of notehead
714  // vertical: _spatium * .4 above/below notehead
715  //
716  //------p1
717  // Compute x0, y0 and stemPos
718  if (sa1 == SlurAnchor::NONE || sa2 == SlurAnchor::NONE) { // need stemPos if sa2 == SlurAnchor::NONE
719  bool stemPos = false; // p1 starts at chord stem side
720 
721  // default positions
722  xo = hw1 * .5 + (note1 ? note1->bboxXShift() : 0.0);
723  if (note1)
724  yo = note1->pos().y();
725  else if (_up)
726  yo = scr->bbox().top();
727  else
728  yo = scr->bbox().top() + scr->height();
729  yo += _spatium * .9 * __up;
730 
731  // adjustments for stem and/or beam
732 
733  if (stem1) { //sc not null
734  Beam* beam1 = sc->beam();
735  if (beam1 && beam1->cross()) {
736  // TODO: stem direction is not finalized, so we cannot use it here
737  yo = fixArticulations(yo, sc, __up, false);
738  }
739  else if (beam1 && (beam1->elements().back() != sc) && (sc->up() == _up)) {
740  // start chord is beamed but not the last chord of beam group
741  // and slur direction is same as start chord (stem side)
742 
743  // in these cases, layout start of slur to stem
744 
745  qreal sh = stem1->height() + _spatium;
746  if (_up)
747  yo = sc->downNote()->pos().y() - sh;
748  else
749  yo = sc->upNote()->pos().y() + sh;
750  xo = stem1->pos().x();
751 
752  // account for articulations
753  yo = fixArticulations(yo, sc, __up, true);
754 
755  // force end of slur to layout to stem as well,
756  // if start and end chords have same stem direction
757  stemPos = true;
758  }
759  else {
760  // start chord is not beamed or is last chord of beam group
761  // or slur direction is opposite that of start chord
762 
763  // at this point slur is in default position relative to note on slur side
764  // but we may need to make further adjustments
765 
766  // if stem and slur are both up
767  // we need to clear stem horizontally
768  if (sc->up() && _up)
769  xo = hw1 + _spatium * .3;
770 
771  //
772  // handle case: stem up - stem down
773  // stem down - stem up
774  //
775  if ((sc->up() != ecr->up()) && (sc->up() == _up)) {
776  // start and end chord have opposite direction
777  // and slur direction is same as start chord
778  // (so slur starts on stem side)
779 
780  // float the start point along the stem to follow direction of movement
781  // see for example Gould p. 111
782 
783  // get position of note on slur side for start & end chords
784  Note* n1 = sc->up() ? sc->upNote() : sc->downNote();
785  Note* n2 = 0;
786  if (ec)
787  n2 = ec->up() ? ec->upNote() : ec->downNote();
788 
789  // differential in note positions
790  qreal yd = (n2 ? n2->pos().y() : ecr->pos().y()) - n1->pos().y();
791  yd *= .5;
792 
793  // float along stem according to differential
794  qreal sh = stem1->height();
795  if (_up && yd < 0.0)
796  yo = qMax(yo + yd, sc->downNote()->pos().y() - sh - _spatium);
797  else if (!_up && yd > 0.0)
798  yo = qMin(yo + yd, sc->upNote()->pos().y() + sh + _spatium);
799 
800  // account for articulations
801  yo = fixArticulations(yo, sc, __up, true);
802 
803  // we may wish to force end to align to stem as well,
804  // if it is in same direction
805  // (but it won't be, so this assignment should have no effect)
806  stemPos = true;
807  }
808  else {
809  // avoid articulations
810  yo = fixArticulations(yo, sc, __up, sc->up() == _up);
811  }
812  }
813  }
814  else if (sc) {
815  // avoid articulations
816  yo = fixArticulations(yo, sc, __up, sc->up() == _up);
817  }
818 
819  if (sa1 == SlurAnchor::NONE)
820  sp->p1 += QPointF(xo, yo);
821 
822  //------p2
823  if (sa2 == SlurAnchor::NONE) {
824 
825  // default positions
826  xo = hw2 * .5 + (note2 ? note2->bboxXShift() : 0.0);
827  if (note2)
828  yo = note2->pos().y();
829  else if (_up)
830  yo = endCR()->bbox().top();
831  else
832  yo = endCR()->bbox().top() + endCR()->height();
833  yo += _spatium * .9 * __up;
834 
835  // adjustments for stem and/or beam
836 
837  if (stem2) { //ec can't be null
838  Beam* beam2 = ec->beam();
839  if (beam2 && beam2->cross()) {
840  // TODO: stem direction is not finalized, so we cannot use it here
841  yo = fixArticulations(yo, ec, __up, false);
842  }
843  else if ((stemPos && (scr->up() == ec->up()))
844  || (beam2
845  && (!beam2->elements().empty())
846  && (beam2->elements().front() != ec)
847  && (ec->up() == _up)
848  && sc && (sc->noteType() == NoteType::NORMAL)
849  )
850  ) {
851 
852  // slur start was laid out to stem and start and end have same direction
853  // OR
854  // end chord is beamed but not the first chord of beam group
855  // and slur direction is same as end chord (stem side)
856  // and start chordrest is not a grace chord
857 
858  // in these cases, layout end of slur to stem
859 
860  qreal sh = stem2->height() + _spatium;
861  if (_up)
862  yo = ec->downNote()->pos().y() - sh;
863  else
864  yo = ec->upNote()->pos().y() + sh;
865  xo = stem2->pos().x();
866 
867  // account for articulations
868  yo = fixArticulations(yo, ec, __up, true);
869  }
870  else
871  {
872  // slur was not aligned to stem or start and end have different direction
873  // AND
874  // end chord is not beamed or is first chord of beam group
875  // or slur direction is opposite that of end chord
876 
877  // if stem and slur are both down,
878  // we need to clear stem horizontally
879  if (!ec->up() && !_up)
880  xo = -_spatium * .3 + note2->x();
881 
882  //
883  // handle case: stem up - stem down
884  // stem down - stem up
885  //
886  if ((scr->up() != ec->up()) && (ec->up() == _up)) {
887  // start and end chord have opposite direction
888  // and slur direction is same as end chord
889  // (so slur end on stem side)
890 
891  // float the end point along the stem to follow direction of movement
892  // see for example Gould p. 111
893 
894  Note* n1 = 0;
895  if (sc)
896  n1 = sc->up() ? sc->upNote() : sc->downNote();
897  Note* n2 = ec->up() ? ec->upNote() : ec->downNote();
898 
899  qreal yd = n2->pos().y() - (n1 ? n1->pos().y() : startCR()->pos().y());
900  yd *= .5;
901 
902  qreal mh = stem2->height();
903  if (_up && yd > 0.0)
904  yo = qMax(yo - yd, ec->downNote()->pos().y() - mh - _spatium);
905  else if (!_up && yd < 0.0)
906  yo = qMin(yo - yd, ec->upNote()->pos().y() + mh + _spatium);
907 
908  // account for articulations
909  yo = fixArticulations(yo, ec, __up, true);
910  }
911  else {
912  // avoid articulations
913  yo = fixArticulations(yo, ec, __up, ec->up() == _up);
914  }
915 
916  }
917  }
918  else if (ec) {
919  // avoid articulations
920  yo = fixArticulations(yo, ec, __up, ec->up() == _up);
921  }
922 
923  sp->p2 += QPointF(xo, yo);
924  }
925  }
926  }

999 SpannerSegment* Slur::layoutSystem(System* system)
1000  {
1001  Fraction stick = system->firstMeasure()->tick();
1002  Fraction etick = system->lastMeasure()->endTick();
1003 
1004  SlurSegment* slurSegment = toSlurSegment(getNextLayoutSystemSegment(system, [this]() { return new SlurSegment(score()); }));
1005 
1006  SpannerSegmentType sst;
1007  if (tick() >= stick) {
1008  //
1009  // this is the first call to layoutSystem,
1010  // processing the first line segment
1011  //
1012  if (track2() == -1)
1013  setTrack2(track());
1014  if (startCR() == 0 || startCR()->measure() == 0) {
1015  qDebug("Slur::layout(): track %d-%d %p - %p tick %d-%d null start anchor",
1016  track(), track2(), startCR(), endCR(), tick().ticks(), tick2().ticks());
1017  return slurSegment;
1018  }
1019  if (endCR() == 0) { // sanity check
1020  setEndElement(startCR());
1021  setTick2(tick());
1022  }
1023  switch (_slurDirection) {
1024  case Direction::UP:
1025  _up = true;
1026  break;
1027  case Direction::DOWN:
1028  _up = false;
1029  break;
1030  case Direction::AUTO:
1031  {
1032  //
1033  // assumption:
1034  // slurs have only chords or rests as start/end elements
1035  //
1036  if (startCR() == 0 || endCR() == 0) {
1037  _up = true;
1038  break;
1039  }
1040  Chord* c1 = startCR()->isChord() ? toChord(startCR()) : 0;
1041  Chord* c2 = endCR()->isChord() ? toChord(endCR()) : 0;
1042 
1043  if (c1 && c1->beam() && c1->beam()->cross()) {
1044  // TODO: stem direction is not finalized, so we cannot use it here
1045  _up = true;
1046  break;
1047  }
1048 
1049  _up = !(startCR()->up());
1050 
1051  Measure* m1 = startCR()->measure();
1052  if ((endCR()->tick() - startCR()->tick()) > m1->ticks()) // long slurs are always above
1053  _up = true;
1054  else
1055  _up = !startCR()->up();
1056 
1057  if (c1 && c2 && isDirectionMixture(c1, c2) && !c1->isGrace()) {
1058  // slurs go above if start and end note have different stem directions,
1059  // but grace notes are exceptions
1060  _up = true;
1061  }
1062  else if (m1->hasVoices(startCR()->staffIdx(), tick(), ticks()) && c1 && !c1->isGrace()) {
1063  // in polyphonic passage, slurs go on the stem side
1064  _up = startCR()->up();
1065  }
1066  else if (c1 && c2 && chordsHaveTie(c1, c2)) {
1067  // could confuse slur with tie, put slur on stem side
1068  _up = startCR()->up();
1069  }
1070  }
1071  break;
1072  }
1073  sst = tick2() < etick ? SpannerSegmentType::SINGLE : SpannerSegmentType::BEGIN;
1074  }
1075  else if (tick() < stick && tick2() >= etick)
1076  sst = SpannerSegmentType::MIDDLE;
1077  else
1078  sst = SpannerSegmentType::END;
1079  slurSegment->setSpannerSegmentType(sst);
1080 
1081  SlurPos sPos;
1082  slurPos(&sPos);
1083 
1084  switch (sst) {
1085  case SpannerSegmentType::SINGLE:
1086  slurSegment->layoutSegment(sPos.p1, sPos.p2);
1087  break;
1088  case SpannerSegmentType::BEGIN:
1089  slurSegment->layoutSegment(sPos.p1, QPointF(system->bbox().width(), sPos.p1.y()));
1090  break;
1091  case SpannerSegmentType::MIDDLE: {
1092  qreal x1 = system->firstNoteRestSegmentX(true);
1093  qreal x2 = system->bbox().width();
1094  qreal y = staffIdx() > system->staves()->size() ? system->y() : system->staff(staffIdx())->y();
1095  slurSegment->layoutSegment(QPointF(x1, y), QPointF(x2, y));
1096  }
1097  break;
1098  case SpannerSegmentType::END:
1099  slurSegment->layoutSegment(QPointF(system->firstNoteRestSegmentX(true), sPos.p2.y()), sPos.p2);
1100  break;
1101  }
1102 
1103  return slurSegment;
1104  }

1113  setTrack2(track());

1118  //
1119  // when used in a palette, slur has no parent and
1120  // tick and tick2 has no meaning so no layout is
1121  // possible and needed
1122  //

1124  if (spannerSegments().empty()) {
1125  s = new SlurSegment(score());
1126  s->setTrack(track());
1127  add(s);
1128  }
1129  else {
1130  s = frontSegment();
1131  }

1136  }
1137 
1138  if (startCR() == 0 || startCR()->measure() == 0) {
1139  qDebug("track %d-%d %p - %p tick %d-%d null start anchor",
1140  track(), track2(), startCR(), endCR(), tick().ticks(), tick2().ticks());
1141  return;
1142  }

1144  qDebug("no end CR for %d", (tick()+ticks()).ticks());
1145  setEndElement(startCR());
1146  setTick2(tick());
1147  }

1149  case Direction::UP:
1150  _up = true;
1151  break;
1152  case Direction::DOWN:
1153  _up = false;
1154  break;
1155  case Direction::AUTO:

1158  // assumption:
1159  // slurs have only chords or rests as start/end elements

1162  _up = true;
1163  break;
1164  }

1168  Chord* c2 = endCR()->isChord() ? toChord(endCR()) : 0;
1169 
1170  _up = !(startCR()->up());
1171 
1172  if ((endCR()->tick() - startCR()->tick()) > m1->ticks()) {
1173  // long slurs are always above
1174  _up = true;
1175  }
1176  else
1177  _up = !(startCR()->up());
1178 
1179  if (c1 && c2 && isDirectionMixture(c1, c2) && (c1->noteType() == NoteType::NORMAL)) {
1180  // slurs go above if start and end note have different stem directions,
1181  // but grace notes are exceptions
1182  _up = true;
1183  }
1184  else if (m1->hasVoices(startCR()->staffIdx(), tick(), ticks()) && c1 && c1->noteType() == NoteType::NORMAL) {
1185  // in polyphonic passage, slurs go on the stem side
1186  _up = startCR()->up();
1187  }
1188  else if (c1 && c2 && chordsHaveTie(c1, c2)) {
1189  // could confuse slur with tie, put slur on stem side
1190  _up = startCR()->up();
1191  }
1192  }
1193  break;

1197  slurPos(&sPos);

1200  ciSystem is = sl.begin();
1201  while (is != sl.end()) {
1202  if (*is == sPos.system1)
1203  break;
1204  ++is;
1205  }

1207  qDebug("Slur::layout first system not found");

1211  // count number of segments, if no change, all
1212  // user offsets (drags) are retained
1213  //---------------------------------------------------------

1217  if ((*iis)->vbox())
1218  continue;

1220  break;

1222  }

1227  System* system = *is;
1228  if (system->vbox()) {
1229  --i;
1230  continue;
1231  }
1232  SlurSegment* segment = segmentAt(i);
1233  segment->setSystem(system);
1234 
1235  // case 1: one segment
1236  if (sPos.system1 == sPos.system2) {
1237  segment->setSpannerSegmentType(SpannerSegmentType::SINGLE);
1238  segment->layoutSegment(sPos.p1, sPos.p2);
1239  }
1240  // case 2: start segment
1241  else if (i == 0) {
1242  segment->setSpannerSegmentType(SpannerSegmentType::BEGIN);
1243  qreal x = system->bbox().width();
1244  segment->layoutSegment(sPos.p1, QPointF(x, sPos.p1.y()));
1245  }

1249  qreal x1 = system->firstNoteRestSegmentX(true);
1250  qreal x2 = system->bbox().width();
1251  qreal y = staffIdx() > system->staves()->size() ? system->y() : system->staff(staffIdx())->y();
1252  segment->layoutSegment(QPointF(x1, y), QPointF(x2, y));
1253  }
1254  // case 4: end segment
1255  else {
