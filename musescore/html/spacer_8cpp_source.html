58 void Spacer::layout0()
59  {
60  qreal _spatium = spatium();
61 
62  path = QPainterPath();
63  qreal w = _spatium;
64  qreal b = w * .5;
65  qreal h = parent() ? _gap : qMin(_gap, spatium() * 4.0); // limit length for palette
66 
67  switch (spacerType()) {
68  case SpacerType::DOWN:
69  path.lineTo(w, 0.0);
70  path.moveTo(b, 0.0);
71  path.lineTo(b, h);
72  path.lineTo(0.0, h-b);
73  path.moveTo(b, h);
74  path.lineTo(w, h-b);
75  break;
76  case SpacerType::UP:
77  path.moveTo(b, 0.0);
78  path.lineTo(0.0, b);
79  path.moveTo(b, 0.0);
80  path.lineTo(w, b);
81  path.moveTo(b, 0.0);
82  path.lineTo(b, h);
83  path.moveTo(0.0, h);
84  path.lineTo(w, h);
85  break;
86  case SpacerType::FIXED:
87  path.lineTo(w, 0.0);
88  path.moveTo(b, 0.0);
89  path.lineTo(b, h);
90  path.moveTo(0.0, h);
91  path.lineTo(w, h);
92  break;
93  }
94  qreal lw = _spatium * 0.4;
95  QRectF bb(0, 0, w, h);
96  bb.adjust(-lw, -lw, lw, lw);
97  setbbox(bb);
98  }

135  {

137 
138  switch (spacerType()) {
139  case SpacerType::DOWN:

142  break;
143  case SpacerType::UP:

191  {
192  while (e.readNextStartElement()) {
193  const QStringRef& tag(e.name());
