96 bool SyntiParameterData::operator==(const SyntiParameterData& sp) const
97  {
98  if (_id == -1 ? (_name != sp._name) : (_id != sp._id))
99  return false;
100  switch(_type) {
101  case SP_FLOAT:
102  return qAbs(_fval - sp._fval) < 0.000001;
103  case SP_STRING:
104  return _sval == sp._sval;
105  }
106  return false;
