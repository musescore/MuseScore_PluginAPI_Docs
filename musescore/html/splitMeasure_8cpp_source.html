42 void Score::splitMeasure(Segment* segment)
43  {
44  if (segment->rtick().isZero()) {
45  MScore::setError(CANNOT_SPLIT_MEASURE_FIRST_BEAT);
46  return;
47  }
48  if (segment->splitsTuplet()) {
49  MScore::setError(CANNOT_SPLIT_MEASURE_TUPLET);
50  return;
51  }
52  Measure* measure = segment->measure();
53 
54  ScoreRange range;
55  range.read(measure->first(), measure->last());
56 
57  Fraction stick = measure->tick();
58  Fraction etick = measure->endTick();
59 
60  std::list<std::tuple<Spanner*, Fraction, Fraction>> sl;
61  for (auto i : spanner()) {
62  Spanner* s = i.second;
63  Element* start = s->startElement();
64  Element* end = s->endElement();
65  if (s->tick() >= stick && s->tick() < etick)
66  start = nullptr;
67  if (s->tick2() >= stick && s->tick2() < etick)
68  end = nullptr;
69  if (start != s->startElement() || end != s->endElement()) {
70  sl.push_back(std::make_tuple(s, s->tick(), s->ticks()));
71  undo(new ChangeStartEndSpanner(s, start, end));
72  }
73  if (s->tick() < stick && s->tick2() > stick)
74  sl.push_back(std::make_tuple(s, s->tick(), s->ticks()));
75  }
76 
77  // Make sure ties are the beginning the split measure are restored.
78  std::vector<Tie*> ties;
79  for (int track = 0; track < ntracks(); track++) {
80  Chord* chord = measure->findChord(stick, track);
81  if (chord)
82  for (Note* note : chord->notes()) {
83  Tie* tie = note->tieBack();
84  if (tie)
85  ties.push_back(tie->clone());
86  }
87  }
88 
89  MeasureBase* nm = measure->next();
90 
91  undoRemoveMeasures(measure, measure, true);
92  undoInsertTime(measure->tick(), -measure->ticks());
93 
94  // create empty measures:
95  insertMeasure(ElementType::MEASURE, nm, true, false);
96  Measure* m2 = toMeasure(nm ? nm->prev() : lastMeasure());
97  insertMeasure(ElementType::MEASURE, m2, true, false);
98  Measure* m1 = toMeasure(m2->prev());
99 
100  Fraction tick = segment->tick();
101  m1->setTick(measure->tick());
102  m2->setTick(tick);
103  Fraction ticks1 = segment->tick() - measure->tick();
104  Fraction ticks2 = measure->ticks() - ticks1;
105  m1->setTimesig(measure->timesig());
106  m2->setTimesig(measure->timesig());
107  ticks1.reduce();
108  ticks2.reduce();
109  // Now make sure this reduction doesn't go 'beyond' the original measure's
110  // actual denominator for both resultant measures.
111  if (ticks1.denominator() < measure->ticks().denominator()) {
112  if (measure->ticks().denominator() % m1->timesig().denominator() == 0) {
113  int mult = measure->ticks().denominator() / ticks1.denominator();
114  // *= operator audomatically reduces via GCD, so rather do literal multiplication:
115  ticks1.setDenominator(ticks1.denominator() * mult);
116  ticks1.setNumerator(ticks1.numerator() * mult);
117  }
118  }
119  if (ticks2.denominator() < measure->ticks().denominator()) {
120  if (measure->ticks().denominator() % m2->timesig().denominator() == 0) {
121  int mult = measure->ticks().denominator() / ticks2.denominator();
122  ticks2.setDenominator(ticks2.denominator() * mult);
123  ticks2.setNumerator(ticks2.numerator() * mult);
124  }
125  }
126  m1->adjustToLen(ticks1, false);
127  m2->adjustToLen(ticks2, false);
128  range.write(this, m1->tick());
129 
130  // Restore ties the the beginning of the split measure.
131  for (auto tie : ties) {
132  tie->setEndNote(searchTieNote(tie->startNote()));
133  undoAddElement(tie);
134  }
135 
136  for (auto i : sl) {
137  Spanner* s = std::get<0>(i);
138  Fraction t = std::get<1>(i);
139  Fraction ticks = std::get<2>(i);
140  if (s->tick() != t)
141  s->undoChangeProperty(Pid::SPANNER_TICK, t);
142  if (s->ticks() != ticks)
143  s->undoChangeProperty(Pid::SPANNER_TICKS, ticks);
144  }
145  }
146 }
