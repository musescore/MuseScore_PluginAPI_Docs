96 void StaffState::layout()
97  {
98  qreal _spatium = spatium();
99  path = QPainterPath();
100  lw = _spatium * 0.3;
101  qreal h = _spatium * 4;
102  qreal w = _spatium * 2.5;
103 // qreal w1 = w * .6;
104 
105  switch(staffStateType()) {
106  case StaffStateType::INSTRUMENT:
107  path.lineTo(w, 0.0);
108  path.lineTo(w, h);
109  path.lineTo(0.0, h);
110  path.lineTo(0.0, 0.0);
111  path.moveTo(w * .5, h - _spatium * .5);
112  path.lineTo(w * .5, _spatium * 2);
113  path.moveTo(w * .5, _spatium * .8);
114  path.lineTo(w * .5, _spatium * 1.0);
115  break;
116 
117  case StaffStateType::TYPE:
118  path.lineTo(w, 0.0);
119  path.lineTo(w, h);
120  path.lineTo(0.0, h);
121  path.lineTo(0.0, 0.0);
122  break;
123 
124  case StaffStateType::VISIBLE:
125  path.lineTo(w, 0.0);
126  path.lineTo(w, h);
127  path.lineTo(0.0, h);
128  path.lineTo(0.0, 0.0);
129  break;
130 
131  case StaffStateType::INVISIBLE:
132  path.lineTo(w, 0.0);
133  path.lineTo(w, h);
134  path.lineTo(0.0, h);
135  path.lineTo(0.0, 0.0);
136  break;
137 
138  default:
139  qDebug("unknown layout break symbol");
140  break;
141  }
142  QRectF bb(0, 0, w, h);
143  bb.adjust(-lw, -lw, lw, lw);
144  setbbox(bb);
145  setPos(0.0, _spatium * -6.0);
146  }
