36 void StaffTextBase::write(XmlWriter& xml) const
37  {
38  if (!xml.canWrite(this))
39  return;
40  xml.stag(this);
41 
42  for (const ChannelActions &s : _channelActions) {
43  int channel = s.channel;
44  for (const QString &name : qAsConst(s.midiActionNames))
45  xml.tagE(QString("MidiAction channel=\"%1\" name=\"%2\"").arg(channel).arg(name));
46  }
47  for (int voice = 0; voice < VOICES; ++voice) {
48  if (!_channelNames[voice].isEmpty())
49  xml.tagE(QString("channelSwitch voice=\"%1\" name=\"%2\"").arg(voice).arg(_channelNames[voice]));
50  }
51  if (_setAeolusStops) {
52  for (int i = 0; i < 4; ++i)
53  xml.tag(QString("aeolus group=\"%1\"").arg(i), aeolusStops[i]);
54  }
55  if (swing()) {
56  QString swingUnit;
57  if (swingParameters()->swingUnit == MScore::division / 2)
58  swingUnit = TDuration(TDuration::DurationType::V_EIGHTH).name();
59  else if (swingParameters()->swingUnit == MScore::division / 4)
60  swingUnit = TDuration(TDuration::DurationType::V_16TH).name();
61  else
62  swingUnit = TDuration(TDuration::DurationType::V_ZERO).name();
63  int swingRatio = swingParameters()->swingRatio;
64  xml.tagE(QString("swing unit=\"%1\" ratio= \"%2\"").arg(swingUnit).arg(swingRatio));
65  }
66  if (capo() != 0)
67  xml.tagE(QString("capo fretId=\"%1\"").arg(capo()));
68  TextBase::writeProperties(xml);
69 
70  xml.etag();
71  }

92 bool StaffTextBase::readProperties(XmlReader& e)
93  {
94  const QStringRef& tag(e.name());
95 
96  if (tag == "MidiAction") {
97  int channel = e.intAttribute("channel", 0);
98  QString name = e.attribute("name");
99  bool found = false;
100  int n = _channelActions.size();
101  for (int i = 0; i < n; ++i) {
102  ChannelActions* a = &_channelActions[i];
103  if (a->channel == channel) {
104  a->midiActionNames.append(name);
105  found = true;
106  break;
107  }
108  }
109  if (!found) {
110  ChannelActions a;
111  a.channel = channel;
112  a.midiActionNames.append(name);
113  _channelActions.append(a);
114  }
115  e.readNext();
116  }
117  else if (tag == "channelSwitch" || tag == "articulationChange") {
118  int voice = e.intAttribute("voice", -1);
119  if (voice >= 0 && voice < VOICES)
120  _channelNames[voice] = e.attribute("name");
121  else if (voice == -1) {
122  // no voice applies channel to all voices for
123  // compatibility
124  for (int i = 0; i < VOICES; ++i)
125  _channelNames[i] = e.attribute("name");
126  }
127  e.readNext();
128  }
129  else if (tag == "aeolus") {
130  int group = e.intAttribute("group", -1);
131  if (group >= 0 && group < 4)
132  aeolusStops[group] = e.readInt();
133  else
134  e.readNext();
135  _setAeolusStops = true;
136  }
137  else if (tag == "swing") {
138  QString swingUnit = e.attribute("unit","");
139  int unit = 0;
140  if (swingUnit == TDuration(TDuration::DurationType::V_EIGHTH).name())
141  unit = MScore::division / 2;
142  else if (swingUnit == TDuration(TDuration::DurationType::V_16TH).name())
143  unit = MScore:: division / 4;
144  else if (swingUnit == TDuration(TDuration::DurationType::V_ZERO).name())
145  unit = 0;
146  int ratio = e.intAttribute("ratio", 60);
147  setSwing(true);
148  setSwingParameters(unit, ratio);
149  e.readNext();
150  }
151  else if (tag == "capo") {
152  int fretId = e.intAttribute("fretId", 0);
153  setCapo(fretId);
154  e.readNext();
155  }
156  else if (!TextBase::readProperties(e))
157  return false;
158  return true;
159  }
