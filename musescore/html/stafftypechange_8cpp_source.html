84 void StaffTypeChange::layout()
85  {
86  qreal _spatium = score()->spatium();
87  setbbox(QRectF(-lw*.5, -lw*.5, _spatium * 2.5 + lw, _spatium*2.5 + lw));
88  if (measure()) {
89  qreal y = -1.5 * _spatium - height() + measure()->system()->staff(staffIdx())->y();
90  setPos(_spatium * .8, y);
91  }
92  else
93  setPos(0.0, 0.0);
94  }

100 void StaffTypeChange::draw(QPainter* painter) const
101  {
102  if (score()->printing() || !score()->showUnprintable())
103  return;
104  qreal _spatium = score()->spatium();
105  qreal h = _spatium * 2.5;
106  qreal w = _spatium * 2.5;
107  qreal lineDist = 0.35; // line distance for the icon 'staff lines'
108  // draw icon rectangle
109  painter->setPen(QPen(selected() ? MScore::selectColor[0] : MScore::layoutBreakColor,
110  lw, Qt::SolidLine, Qt::SquareCap, Qt::MiterJoin));
111  painter->setBrush(Qt::NoBrush);
112  painter->drawRect(0, 0, w, h);
113 
114  // draw icon contents
115  int lines = 5;
116  if (staffType()) {
117  if (staffType()->stemless()) // a single notehead represents a stemless staff
118  drawSymbol(SymId::noteheadBlack, painter, QPoint(w * 0.5 - 0.33 * _spatium, h * 0.5), 0.5);
119  if (staffType()->invisible()) // no lines needed. It's done.
120  return;
121  // show up to 6 lines
122  lines = qMin(staffType()->lines(),6);
123  }
124  // calculate starting point Y for the lines from half the icon height (2.5) so staff lines appear vertically centered
125  qreal startY = 1.25 - (lines - 1) * lineDist * 0.5;
126  painter->setPen(QPen(selected() ? MScore::selectColor[0] : MScore::layoutBreakColor,
127  2.5, Qt::SolidLine, Qt::SquareCap, Qt::MiterJoin));
128  for (int i=0; i < lines; i++) {
129  int y = (startY + i * lineDist) * _spatium;
130  painter->drawLine(0, y, w, y);
131  }
132  }

180 bool StaffTypeChange::setProperty(Pid propertyId, const QVariant& v)
181  {

191  break;
192  case Pid::STAFF_SHOW_BARLINES:
193  _staffType->setShowBarlines(v.toBool());
194  break;
195  case Pid::STAFF_SHOW_LEDGERLINES:
196  _staffType->setShowLedgerLines(v.toBool());
197  break;
198  case Pid::STAFF_STEMLESS:
199  _staffType->setStemless(v.toBool());
200  break;
201  case Pid::HEAD_SCHEME:
202  _staffType->setNoteHeadScheme(NoteHead::Scheme(v.toInt()));
203  break;
204  case Pid::STAFF_GEN_CLEF:
205  _staffType->setGenClef(v.toBool());
206  break;
207  case Pid::STAFF_GEN_TIMESIG:
208  _staffType->setGenTimesig(v.toBool());
209  break;
210  case Pid::STAFF_GEN_KEYSIG:
211  _staffType->setGenKeysig(v.toBool());
212  break;
213  case Pid::MAG: {
214  qreal _spatium = spatium();
215  _staffType->setUserMag(v.toDouble());
216  Staff* _staff = staff();
217  if (_staff)
218  _staff->localSpatiumChanged(_spatium, spatium(), tick());
219  }
220  break;
221  case Pid::SMALL: {
222  qreal _spatium = spatium();
223  _staffType->setSmall(v.toBool());
224  Staff* _staff = staff();
225  if (_staff)
226  _staff->localSpatiumChanged(_spatium, spatium(), tick());
227  }
228  break;
229  case Pid::STAFF_INVISIBLE:
230  _staffType->setInvisible(v.toBool());
231  break;
232  case Pid::STAFF_COLOR:
233  _staffType->setColor(v.value<QColor>());
234  break;
235  case Pid::STAFF_YOFFSET:
236  _staffType->setYoffset(v.value<Spatium>());
237  break;
238  default:
239  if (!Element::setProperty(propertyId, v))
240  return false;
241  break;
242  }
243  if (parent())
244  staff()->staffTypeListChanged(measure()->tick());
245  return true;
246  }
