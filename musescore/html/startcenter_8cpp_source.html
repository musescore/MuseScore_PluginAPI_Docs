252 bool MyWebEnginePage::acceptNavigationRequest(const QUrl & url, QWebEnginePage::NavigationType type, bool isMainFrame)
253  {
254  qDebug() << "acceptNavigationRequest(" << url << "," << type << "," << isMainFrame << ")";
255 
256  if (type == QWebEnginePage::NavigationTypeLinkClicked)
257  {
258  QString path(url.path());
259  QFileInfo fi(path);
260  if (fi.suffix() == "mscz" || fi.suffix() == "xml"
261  || fi.suffix() == "musicxml" || fi.suffix() == "mxl") {
262  mscore->loadFile(url);
263  QAction* a = getAction("startcenter");
264  a->setChecked(false);
265  mscore->showStartcenter(false);
266  }
267  else
268  QDesktopServices::openUrl(url);
269 
270  return false;
271  }
272  return true;
273  }
