72 //-------------------------------------------------------------------
73 // layout
74 // For beamed notes this is called twice. The final stem
75 // length can only be calculated after stretching of the measure.
76 // We need a guessed stem shape to calculate the minimal distance
77 // between segments. The guessed stem must have at least the
78 // right direction.
79 //-------------------------------------------------------------------
80 
81 void Stem::layout()
82  {
83  qreal l = _len + _userLen;
84  qreal _up = up() ? -1.0 : 1.0;
85  l *= _up;
86 
87  qreal y1 = 0.0; // vertical displacement to match note attach point
88  const Staff* stf = staff();
89  if (chord()) {
90  setMag(chord()->mag());
91  Fraction tick = chord()->tick();
92  const StaffType* st = stf ? stf->staffType(tick) : 0;
93  if (st && st->isTabStaff() ) { // TAB staves
94  if (st->stemThrough()) {
95  // if stems through staves, gets Y pos. of stem-side note relative to chord other side
96  qreal lineDist = st->lineDistance().val() * spatium();
97  y1 = (chord()->downString() - chord()->upString()) * _up * lineDist;
98  // if fret marks above lines, raise stem beginning by 1/2 line distance
99  if (!st->onLines())
100  y1 -= lineDist * 0.5;
101  // shorten stem by 1/2 lineDist to clear the note and a little more to keep 'air' between stem and note
102  lineDist *= 0.7 * mag();
103  y1 += _up * lineDist;
104  }
105  // in other TAB types, no correction
106  }
107  else { // non-TAB
108  // move stem start to note attach point
109  Note* n = up() ? chord()->downNote() : chord()->upNote();
110  if ((up() && !n->mirror()) || (!up() && n->mirror()))
111  y1 += n->stemUpSE().y();
112  else
113  y1 += n->stemDownNW().y();
114  rypos() = n->rypos();
115  }
116  }
117 
118  qreal lw5 = 0.5 * lineWidthMag();
119 
120  line.setLine(0.0, y1, 0.0, l);
121 
122  // compute bounding rectangle
123  QRectF r(line.p1(), line.p2());
124  setbbox(r.normalized().adjusted(-lw5, -lw5, lw5, lw5));
125  }

151 void Stem::draw(QPainter* painter) const
152  {
153  if (!chord()) // may be need assert?
154  return;
155 
156  // hide if second chord of a cross-measure pair
157  if (chord()->crossMeasure() == CrossMeasure::SECOND)
158  return;
159 
160  const Staff* st = staff();
161  const StaffType* stt = st ? st->staffType(chord()->tick()) : 0;
162  bool useTab = stt && stt->isTabStaff();
163 
164  painter->setPen(QPen(curColor(), lineWidthMag(), Qt::SolidLine, Qt::FlatCap));
165  painter->drawLine(line);
166 
167  if (!useTab)
168  return;
169 
170  // TODO: adjust bounding rectangle in layout() for dots and for slash
171  qreal sp = spatium();
172  bool isUp = up();
173 
174  // slashed half note stem
175  if (chord()->durationType().type() == TDuration::DurationType::V_HALF && stt->minimStyle() == TablatureMinimStyle::SLASHED) {
176  // position slashes onto stem
177  qreal y = isUp ? -(_len+_userLen) + STAFFTYPE_TAB_SLASH_2STARTY_UP*sp : (_len+_userLen) - STAFFTYPE_TAB_SLASH_2STARTY_DN*sp;
178  // if stems through, try to align slashes within or across lines
179  if (stt->stemThrough()) {
180  qreal halfLineDist = stt->lineDistance().val() * sp * 0.5;
181  qreal halfSlashHgt = STAFFTYPE_TAB_SLASH_2TOTHEIGHT * sp * 0.5;
182  y = lrint( (y + halfSlashHgt) / halfLineDist) * halfLineDist - halfSlashHgt;
183  }
184  // draw slashes
185  qreal hlfWdt= sp * STAFFTYPE_TAB_SLASH_WIDTH * 0.5;
186  qreal sln = sp * STAFFTYPE_TAB_SLASH_SLANTY;
187  qreal thk = sp * STAFFTYPE_TAB_SLASH_THICK;
188  qreal displ = sp * STAFFTYPE_TAB_SLASH_DISPL;
189  QPainterPath path;
190  for (int i = 0; i < 2; ++i) {
191  path.moveTo( hlfWdt, y); // top-right corner
192  path.lineTo( hlfWdt, y+thk); // bottom-right corner
193  path.lineTo(-hlfWdt, y+thk+sln); // bottom-left corner
194  path.lineTo(-hlfWdt, y+sln); // top-left corner
195  path.closeSubpath();
196  y += displ;
197  }
198  painter->setBrush(QBrush(curColor()));
199  painter->setPen(Qt::NoPen);
200  painter->drawPath(path);
201  }
202 
203  // dots
204  // NOT THE BEST PLACE FOR THIS?
205  // with tablatures and stems beside staves, dots are not drawn near 'notes', but near stems
206  int nDots = chord()->dots();
207  if (nDots > 0 && !stt->stemThrough()) {
208  qreal x = chord()->dotPosX();
209  qreal y = ( (STAFFTYPE_TAB_DEFAULTSTEMLEN_DN * 0.2) * sp) * (isUp ? -1.0 : 1.0);
210  qreal step = score()->styleS(Sid::dotDotDistance).val() * sp;
211  for (int dot = 0; dot < nDots; dot++, x += step)
212  drawSymbol(SymId::augmentationDot, painter, QPointF(x, y));
213  }
214  }
