142 //---------------------------------------------------------
143 // fretChords
144 // Assigns fretting to all the notes of each chord in the same segment of chord
145 // re-using existing fretting wherever possible
146 //
147 // Minimizes fret conflicts (multiple notes on the same string)
148 // but marks as fretConflict notes which cannot be fretted
149 // (outside tablature range) or which cannot be assigned
150 // a separate string
151 //---------------------------------------------------------
152 
153 void StringData::fretChords(Chord * chord) const
154  {
155  int nFret, minFret, maxFret, nNewFret, nTempFret;
156  int nString, nNewString, nTempString;
157 
158  if(bFretting)
159  return;
160  bFretting = true;
161 
162  // we need to keep track of string allocation
163 #if (!defined (_MSCVER) && !defined (_MSC_VER))
164  int bUsed[strings()]; // initially all strings are available
165 #else
166  // MSVC does not support VLA. Replace with std::vector. If profiling determines that the
167  // heap allocation is slow, an optimization might be used.
168  std::vector<int> bUsed(strings());
169 #endif
170  for(nString=0; nString<strings(); nString++)
171  bUsed[nString] = 0;
172  // we also need the notes sorted in order of string (from highest to lowest) and then pitch
173  QMap<int, Note *> sortedNotes;
174  int count = 0;
175  // store staff pitch offset at this tick, to speed up actual note pitch calculations
176  // (ottavas not implemented yet)
177  int transp = chord->staff() ? chord->part()->instrument(chord->tick())->transpose().chromatic : 0; // TODO: tick?
178  int pitchOffset = /*chord->staff()->pitchOffset(chord->segment()->tick())*/ - transp;
179  // if chord parent is not a segment, the chord is special (usually a grace chord):
180  // fret it by itself, ignoring the segment
181  if (chord->parent()->type() != ElementType::SEGMENT)
182  sortChordNotes(sortedNotes, chord, pitchOffset, &count);
183  else {
184  // scan each chord of seg from same staff as 'chord', inserting each of its notes in sortedNotes
185  Segment* seg = chord->segment();
186  int trk;
187  int trkFrom = (chord->track() / VOICES) * VOICES;
188  int trkTo = trkFrom + VOICES;
189  for(trk = trkFrom; trk < trkTo; ++trk) {
190  Element* ch = seg->elist().at(trk);
191  if (ch && ch->type() == ElementType::CHORD)
192  sortChordNotes(sortedNotes, toChord(ch), pitchOffset, &count);
193  }
194  }
195  // determine used range of frets
196  minFret = INT32_MAX;
197  maxFret = INT32_MIN;
198  foreach(Note* note, sortedNotes) {
199  if (note->string() != STRING_NONE)
200  bUsed[note->string()]++;
201  if (note->fret() != FRET_NONE && note->fret() < minFret)
202  minFret = note->fret();
203  if (note->fret() != FRET_NONE && note->fret() > maxFret)
204  maxFret = note->fret();
205  }
206 
207  // scan chord notes from highest, matching with strings from the highest
208  foreach(Note * note, sortedNotes) {
209  nString = nNewString = note->string();
210  nFret = nNewFret = note->fret();
211  note->setFretConflict(false); // assume no conflicts on this note
212  // if no fretting (any invalid fretting has been erased by sortChordNotes() )
213  if (nString == STRING_NONE /*|| nFret == FRET_NONE || getPitch(nString, nFret) != note->pitch()*/) {
214  // get a new fretting
215  if (!convertPitch(note->pitch(), pitchOffset, &nNewString, &nNewFret) ) {
216  // no way to fit this note in this tab:
217  // mark as fretting conflict
218  note->setFretConflict(true);
219  // store fretting change without affecting chord context
220  if (nFret != nNewFret)
221  note->undoChangeProperty(Pid::FRET, nNewFret);
222  if (nString != nNewString)
223  note->undoChangeProperty(Pid::STRING, nNewString);
224  continue;
225  }
226  // note can be fretted: use string
227  else {
228  bUsed[nNewString]++;
229  }
230  }
231 
232  // if the note string (either original or newly assigned) is also used by another note
233  if (bUsed[nNewString] > 1) {
234  // attempt to find a suitable string, from topmost
235  for (nTempString=0; nTempString < strings(); nTempString++) {
236  if (bUsed[nTempString] < 1
237  && (nTempFret=fret(note->pitch(), nTempString, pitchOffset)) != FRET_NONE) {
238  bUsed[nNewString]--; // free previous string
239  bUsed[nTempString]++; // and occupy new string
240  nNewFret = nTempFret;
241  nNewString = nTempString;
242  break;
243  }
244  }
245  }
246 
247  // TODO : try to optimize used fret range, avoiding eccessively open positions
248 
249  // if fretting did change, store as a fret change
250  if (nFret != nNewFret)
251  note->undoChangeProperty(Pid::FRET, nNewFret);
252  if (nString != nNewString)
253  note->undoChangeProperty(Pid::STRING, nNewString);
254  }
255 
256  // check for any remaining fret conflict
257  for (Note* note : sortedNotes)
258  if (note->string() == -1 || bUsed[note->string()] > 1)
259  note->setFretConflict(true);
260 
261  bFretting = false;
262  }

301 //---------------------------------------------------------
302 // convertPitch
303 // Finds string and fret for a note.
304 //
305 // Fills *string and *fret with suitable values for pitch / pitchOffset,
306 // using the highest possible string.
307 // If note cannot be fretted, uses fret 0 on nearest string and returns false
308 //
309 // Note: Strings are stored internally from lowest (0) to highest (strings()-1),
310 // but the returned *string value references strings in reversed, 'visual', order:
311 // from highest (0) to lowest (strings()-1)
312 //---------------------------------------------------------
313 
314 bool StringData::convertPitch(int pitch, int pitchOffset, int* string, int* fret) const
315  {
316  int strings = stringTable.size();
317  if (strings < 1)
318  return false;
319 
320  pitch += pitchOffset;
321 
322  // if above max fret on highest string, fret on first string, but return failure
323  if(pitch > stringTable.at(strings-1).pitch + _frets) {
324  *string = 0;
325  *fret = 0;
326  return false;
327  }
328 
329  if (isFiveStringBanjo()) {
330  // special case: open banjo 5th string
331  if (pitch == stringTable.at(0).pitch) {
332  *string = 4;
333  *fret = 0;
334  return true;
335  }
336  // test remaining 4 strings from highest to lowest
337  for (int i = 4; i > 0; i--) {
338  instrString strg = stringTable.at(i);
339  if (pitch >= strg.pitch) {
340  *string = strings - i - 1;
341  *fret = pitch - strg.pitch;
342  return true;
343  }
344  }
345  }
346  else {
347  // look for a suitable string, starting from the highest
348  // NOTE: this assumes there are always enough frets to fill
349  // the interval between any fretted string and the next
350  for (int i = strings-1; i >=0; i--) {
351  instrString strg = stringTable.at(i);
352  if(pitch >= strg.pitch) {
353  if (pitch == strg.pitch || !strg.open)
354  *string = strings - i - 1;
355  *fret = pitch - strg.pitch;
356  return true;
357  }
358  }
359  }
360 
361  // if no string found, pitch is below lowest string:
362  // fret on last string, but return failure
363  *string = strings-1;
364  *fret = 0;
365  return false;
366  }

375 int StringData::getPitch(int string, int fret, int pitchOffset) const
376  {
377  int strings = stringTable.size();
378  if (string < 0 || string >= strings)
379  return INVALID_PITCH;
380  instrString strg = stringTable.at(strings - string - 1);
381  int pitch = strg.pitch - pitchOffset + (strg.open ? 0 : fret);
382  if (strg.startFret > 0 && fret >= strg.startFret)
383  pitch -= strg.startFret; // banjo 5th string adjustment
384  return pitch;
385  }

388 // fret
389 // Returns the fret corresponding to the pitch / string / pitchOffset combination.
390 // returns FRET_NONE if not possible

393 int StringData::fret(int pitch, int string, int pitchOffset) const
394  {
395  int strings = stringTable.size();
396  if (strings < 1) // no strings at all!
397  return FRET_NONE;
398 
399  if (string < 0 || string >= strings) // no such a string
400  return FRET_NONE;
401 
402  pitch += pitchOffset;
403 
404  const instrString& strg = stringTable[strings - string - 1];
405  int fret = pitch - strg.pitch;
406  if (fret > 0 && strg.startFret > 0)
407  fret += strg.startFret; // banjo 5th string adjustment
408 
409  // fret number is invalid or string cannot be fretted
410  if (fret < 0 || fret >= _frets || (fret > 0 && strg.open))
411  return FRET_NONE;
412  return fret;
413  }

415 //---------------------------------------------------------
416 // sortChordNotes
417 // Adds to sortedNotes the notes of Chord in string/pitch order
418 // Note: notes are sorted first by string (top string being 0),
419 // then by negated pitch (higher pitches resulting in lower key),
420 // then by order of submission to disambiguate notes with the same pitch.
421 // Everything else being equal, this makes notes in higher-numbered voices
422 // to be sorted after notes in lower-numbered voices (voice 2 after voice 1 and so on)
423 // Notes without a string assigned yet, are sorted according to the lowest string which can accommodate them.
424 //---------------------------------------------------------
425 
426 void StringData::sortChordNotes(QMap<int, Note *>& sortedNotes, const Chord *chord, int pitchOffset, int* count) const
427 {
428  int key, string, fret;
429 
430  foreach(Note * note, chord->notes()) {
431  string = note->string();
432  fret = note->fret();
433  // if note not fretted yet or current fretting no longer valid,
434  // use most convenient string as key
435  if (string <= STRING_NONE || fret <= FRET_NONE
436  || getPitch(string, fret, pitchOffset) != note->pitch()) {
437  note->setString(STRING_NONE);
438  note->setFret(FRET_NONE);
439  convertPitch(note->pitch(), pitchOffset, &string, &fret);
440  }
441  key = string * 100000;
442  key += -(note->pitch()+pitchOffset) * 100 + *count; // disambiguate notes of equal pitch
443  sortedNotes.insert(key, note);
444  (*count)++;
445  }
446 }
