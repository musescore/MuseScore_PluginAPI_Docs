27 QString stringutils::removeLigatures(const QString& pre)
28  {
29  QString result = pre;
30 
31  // Characters with above dots (Ae, ae, Oe, oe, Ue, ue
32  result = result.replace(QRegularExpression("[\\x{00C4}]"), QString("Ae"));
33  result = result.replace(QRegularExpression("[\\x{00E4}]"), QString("ae"));
34  result = result.replace(QRegularExpression("[\\x{00D6}]"), QString("Oe"));
35  result = result.replace(QRegularExpression("[\\x{00F6}]"), QString("oe"));
36  result = result.replace(QRegularExpression("[\\x{00DC}]"), QString("Ue"));
37  result = result.replace(QRegularExpression("[\\x{00FC}]"), QString("ue"));
38 
39  // Latin Big Letter AA (&#42802) and Latin Small Letter aa (&#42803)
40  result = result.replace(QRegularExpression("[\\x{A732}]"), QString("Aa"));
41  result = result.replace(QRegularExpression("[\\x{A733}]"), QString("aa"));
42 
43  // Latin Big Letter AE (&#198) and Latin Small Letter ae (&#230)
44  result = result.replace(QRegularExpression("[\\x{00C6}]"), QString("Ae"));
45  result = result.replace(QRegularExpression("[\\x{00E6}]"), QString("ae"));
46 
47  // Latin Big Letter AO (&#42804) and Latin Small Letter ao (&#42805)
48  result = result.replace(QRegularExpression("[\\x{A734}]"), QString("Ao"));
49  result = result.replace(QRegularExpression("[\\x{A735}]"), QString("ao"));
50 
51  // Latin Big Letter AU (&#42806) and Latin Small Letter au (&#42807)
52  result = result.replace(QRegularExpression("[\\x{A736}]"), QString("Au"));
53  result = result.replace(QRegularExpression("[\\x{A737}]"), QString("au"));
54 
55  // IJ (&#306) and ij (&#307)
56  result = result.replace(QRegularExpression("[\\x{0132}]"), QString("IJ"));
57  result = result.replace(QRegularExpression("[\\x{0133}]"), QString("ij"));
58 
59  // Eszett SS (&#7838) and ss (&#223)
60  result = result.replace(QRegularExpression("[\\x{1E9E}]"), QString("SS"));
61  result = result.replace(QRegularExpression("[\\x{00DF}]"), QString("ss"));
62 
63  // O with stroke (&#216) and o with stroke (&#248)
64  result = result.replace(QRegularExpression("[\\x{00D8}]"), QChar('O'));
65  result = result.replace(QRegularExpression("[\\x{00F8}]"), QChar('o'));
66 
67  // Big Letter OE (&#338) and small letter oe (&#339)
68  result = result.replace(QRegularExpression("[\\x{0152}]"), QString("Oe"));
69  result = result.replace(QRegularExpression("[\\x{0153}]"), QString("oe"));
70 
71  // Big Letter OO (&#42830) and small letter oo (&#42831)
72  result = result.replace(QRegularExpression("[\\x{A74E}]"), QString("Oo"));
73  result = result.replace(QRegularExpression("[\\x{A74F}]"), QString("oo"));
74 
75  // ue (&#7531)
76  result = result.replace(QRegularExpression("[\\x{1D6B}]"), QString("ue"));
77 
78  return result;
79  }

85 QString stringutils::removeDiacritics(const QString& pre)
86  {
87  QString text = pre.normalized(QString::NormalizationForm_KD);
88  QString result;
89  for (int i = 0; i < text.length(); ++i) {
90  if (text.at(i).category() != QChar::Mark_NonSpacing) {
91  result.append(text.at(i));
92  }
93  }
94  result = result.normalized(QString::NormalizationForm_C);
95  return result;

102 QString stringutils::convertFileSizeToHumanReadable(const qlonglong& bytes)
103  {
104  // If less than 1 KB, report in B
105  if (bytes < 0x400) {
106  return tr("%1 B").arg(bytes);
107  }
108  // If less than 1 MB, report in KB, unless rounded result is 1024 KB, then report in MB
109  else if (bytes >= 0x400 && bytes < 0x100000) {
110  qlonglong result = (bytes + (0x400 / 2)) / 0x400;
111  if (result < 0x400) {
112  return tr("%1 KB").arg(result);
113  }
114  else {
115  result = (bytes + (0x100000 / 2)) / 0x100000;
116  return tr("%1 MB").arg(result);
117  }
118  }
119  // If less than 1 GB, report in MB, unless rounded result is 1024 MB, then report in GB
120  else if (bytes >= 0x100000 && bytes < 0x40000000) {
121  qlonglong result = (bytes + (0x100000 / 2)) / 0x100000;
122  if (result < 0x100000) {
123  return tr("%1 MB").arg(result);
124  }
125  else {
126  result = (bytes + (0x40000000 / 2)) / 0x40000000;
127  return tr("%1 GB").arg(result);
128  }
129  }
130  // If less than 1 TB, report in GB, unless rounded result is 1024 GB, then report in TB
131  else if (bytes >= 0x40000000 && bytes < 0x10000000000) {
132  qlonglong result = (bytes + (0x40000000 / 2)) / 0x40000000;
133  if (result < 0x40000000) {
134  return tr("%1 GB").arg(result);
135  }
136  else {
137  result = (bytes + (0x10000000000 / 2)) / 0x10000000000;
138  return tr("%1 TB").arg(result);
139  }
140  }
