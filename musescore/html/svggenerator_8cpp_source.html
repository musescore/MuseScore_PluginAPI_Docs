115 static void translate_color(const QColor &color, QString *color_string,
116  QString *opacity_string)
117 {
118  Q_ASSERT(color_string);
119  Q_ASSERT(opacity_string);
120 
121  *color_string =
122  QString::fromLatin1("#%1%2%3")
123  .arg(color.red(), 2, 16, QLatin1Char('0'))
124  .arg(color.green(), 2, 16, QLatin1Char('0'))
125  .arg(color.blue(), 2, 16, QLatin1Char('0'));
126  *opacity_string = QString::number(color.alphaF());
127 }

129 static void translate_dashPattern(QVector<qreal> pattern, const qreal& width, QString *pattern_string)
130 {
131  Q_ASSERT(pattern_string);
132 
133  // Note that SVG operates in absolute lengths, whereas Qt uses a length/width ratio.
134  foreach (qreal entry, pattern)
135  *pattern_string += QString::fromLatin1("%1,").arg(entry * width);
136 
137  pattern_string->chop(1);
138 }

480  const QString qpenToSvg(const QPen &spen)
481  {
482  QString qs;
483  QTextStream qts(&qs);
484 
485  QString color, colorOpacity;
486 
487  // Set stroke, stroke-dasharray, stroke-dashoffset attributes
488  switch (spen.style()) {
489  case Qt::NoPen:
490  return qs; // Default value for stroke = "none" = Qt::NoPen = NOOP;
491  break;
492 
493  case Qt::SolidLine:
494  case Qt::DashLine:
495  case Qt::DotLine:
496  case Qt::DashDotLine:
497  case Qt::DashDotDotLine:
498  case Qt::CustomDashLine: {
499  // These values are class variables because they are needed by
500  // drawTextItem(). This is the fill color/opacity for text.
501  translate_color(spen.color(), &color, &colorOpacity);
502 
503  // default stroke="none" is handled by case Qt::NoPen above
504  qts << SVG_STROKE << color << SVG_QUOTE;
505 
506  // stroke-opacity is seldom used, usually set to default 1
507  if (colorOpacity != SVG_ONE)
508  qts << SVG_STROKE_OPACITY << colorOpacity << SVG_QUOTE;
509 
510  // If it's a solid line, were done for now
511  if (spen.style() == Qt::SolidLine)
512  break;
513 
514  // It's a dashed line
515  qreal penWidth = spen.width() == 0 ? qreal(1) : spen.widthF();
516 
517  QString dashPattern, dashOffset;
518  translate_dashPattern(spen.dashPattern(), penWidth, &dashPattern);
519  dashOffset = QString::number(spen.dashOffset() * penWidth); // SVG uses absolute offset
520 
521  qts << SVG_STROKE_DASHARRAY << dashPattern << SVG_QUOTE;
522  qts << SVG_STROKE_DASHOFFSET << dashOffset << SVG_QUOTE;
523  break; }
524  default:
525  qWarning("Unsupported pen style");
526  break;
527  }
528  // Set stroke-width attribute, unless it's zero or 1 (default is 1)
529  if (spen.widthF() > 0 && spen.widthF() != 1) {
530  // Formatting for vertical alignment of elements
531  qts.setRealNumberPrecision(2); // with DPI=72 only 2 decimals necessary
532  qts.setRealNumberNotation(QTextStream::FixedNotation);
533  qts << SVG_STROKE_WIDTH << spen.widthF() << SVG_QUOTE;
534  qts.setRealNumberNotation(QTextStream::SmartNotation);
535  }
536  // Set stroke-linecap attribute
537  switch (spen.capStyle()) {
538  case Qt::FlatCap:
539  // This is the default stroke-linecap value
540  //qts << SVG_STROKE_LINECAP << SVG_BUTT << SVG_QUOTE;
541  break;
542  case Qt::SquareCap:
543  qts << SVG_STROKE_LINECAP << SVG_SQUARE << SVG_QUOTE;
544  break;
545  case Qt::RoundCap:
546  qts << SVG_STROKE_LINECAP << SVG_ROUND << SVG_QUOTE;
547  break;
548  default:
549  qWarning("Unhandled cap style");
550  break;
551  }
552  // Set stroke-linejoin, stroke-miterlimit attributes
553  switch (spen.joinStyle()) {
554  case Qt::MiterJoin:
555  case Qt::SvgMiterJoin:
556  qts << SVG_STROKE_LINEJOIN << SVG_MITER << SVG_QUOTE
557  << SVG_STROKE_MITERLIMIT << spen.miterLimit() << SVG_QUOTE;
558  break;
559  case Qt::BevelJoin:
560  qts << SVG_STROKE_LINEJOIN << SVG_BEVEL << SVG_QUOTE;
561  break;
562  case Qt::RoundJoin:
563  qts << SVG_STROKE_LINEJOIN << SVG_ROUND << SVG_QUOTE;
564  break;
565  default:
566  qWarning("Unhandled join style");
567  break;
568  }
569  // An uncommon, possibly non-existent in MuseScore, effect
570  if (spen.isCosmetic())
571  qts << SVG_VECTOR_EFFECT;
572 
573  return qs;
574  }

1147 {

1149  stateString.clear();
1150 
1151  // stateString = Attribute Settings
1152 
1153  // SVG class attribute, based on Ms::ElementType
1154  stateStream << SVG_CLASS << getClass(_element) << SVG_QUOTE;
1155 
1156  // Brush and Pen attributes
1157  stateStream << qbrushToSvg(s.brush());
1158  stateStream << qpenToSvg(s.pen());
1159 
1160 // TBD: "opacity" attribute: Is it ever used?
1161 // Or is opacity determined by fill-opacity & stroke-opacity instead?
1162 // PLUS: qFuzzyIsNull() is not officially supported in Qt.
1163 // Should probably use QFuzzyCompare() instead.
1164  if (!qFuzzyIsNull(s.opacity() - 1))
1165  stateStream << SVG_OPACITY << s.opacity() << SVG_QUOTE;
1166 
1167  // Translations, SVG transform="translate()", are handled separately from
1168  // other transformations such as rotation. Qt translates everything, but
1169  // other transformations do occur, and must be handled here.
1170  QTransform t = s.transform();
1171 
1172  // Tablature Note Text:
1173  // m11 and m22 have floating point flotsam, for example: 1.000000629
1174  // Both values should be == integer 1, because no scaling is intended.
1175  // So round them to three decimal places, as MuseScore does elsewhere.
1176  const qreal m11 = qRound(t.m11() * 1000) / 1000.0;
1177  const qreal m22 = qRound(t.m22() * 1000) / 1000.0;
1178 
1179  if (m11 == 1 && m22 == 1 // No scaling
1180  && t.m12() == t.m21()) { // No rotation, etc.
1181  // No transformation except translation
1182  _dx = t.m31();
1183  _dy = t.m32();
1184  }
1185  else {
1186  // Other transformations are more straightforward with a full matrix
1187  _dx = 0;
1188  _dy = 0;
1189  stateStream << SVG_MATRIX << t.m11() << SVG_COMMA
1190  << t.m12() << SVG_COMMA
1191  << t.m21() << SVG_COMMA
1192  << t.m22() << SVG_COMMA
1193  << t.m31() << SVG_COMMA
1194  << t.m32() << SVG_RPAREN_QUOTE;
1195  }

1202

1204  // Majority of <path>s use the default value: fill-rule="nonzero"

1213  qreal y = e.y + _dy;

1218  case QPainterPath::LineToElement:

1220  break;
1221  case QPainterPath::CurveToElement:

1223  ++i;

1225  const QPainterPath::Element &ee = p.elementAt(i);
1226  if (ee.type == QPainterPath::CurveToDataElement) {
1227  stream() << SVG_SPACE << ee.x + _dx
