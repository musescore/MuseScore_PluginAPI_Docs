57 void Symbol::layout()
58  {
59  // foreach(Element* e, leafs()) done in BSymbol::layout() ?
60  // e->layout();
61  setbbox(_scoreFont ? _scoreFont->bbox(_sym, magS()) : symBbox(_sym));
62  qreal w = width();
63  QPointF p;
64  if (align() & Align::BOTTOM)
65  p.setY(- height());
66  else if (align() & Align::VCENTER)
67  p.setY((- height()) * .5);
68  else if (align() & Align::BASELINE)
69  p.setY(-baseLine());
70  if (align() & Align::RIGHT)
71  p.setX(-w);
72  else if (align() & Align::HCENTER)
73  p.setX(-(w * .5));
74  setPos(p);
75  BSymbol::layout();
76  }

108 // Symbol::read
109 //---------------------------------------------------------
110 
111 void Symbol::read(XmlReader& e)
112  {
113  QPointF pos;
114  while (e.readNextStartElement()) {
115  const QStringRef& tag(e.name());
116  if (tag == "name") {
117  QString val(e.readElementText());
118  SymId symId = Sym::name2id(val);
119  if (val != "noSym") {
120  if (symId == SymId::noSym) {
121  // if symbol name not found, fall back to user names
122  // TODO : does it make sense? user names are probably localized
123  symId = Sym::userName2id(val);
124  if (symId == SymId::noSym) {
125  qDebug("unknown symbol <%s>, falling back to no symbol", qPrintable(val));
126  // set a default symbol, or layout() will crash
127  symId = SymId::noSym;
128  }
129  }
130  }
131  setSym(symId);
132  }
133  else if (tag == "font")
134  _scoreFont = ScoreFont::fontFactory(e.readElementText());
135  else if (tag == "Symbol") {
136  Symbol* s = new Symbol(score());
137  s->read(e);
138  add(s);
139  }
140  else if (tag == "Image") {
141  if (MScore::noImages)
142  e.skipCurrentElement();
143  else {
144  Image* image = new Image(score());
145  image->read(e);
146  add(image);
147  }
148  }
149  else if (tag == "small" || tag == "subtype") // obsolete
150  e.skipCurrentElement();
151  else if (!BSymbol::readProperties(e))
152  e.unknown();
153  }
154  setPos(pos);
155  }
