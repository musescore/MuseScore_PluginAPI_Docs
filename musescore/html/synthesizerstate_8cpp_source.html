23  {

43 void SynthesizerState::read(XmlReader& e)
44  {
45  std::list<SynthesizerGroup> tempGroups;
46  while (e.readNextStartElement()) {
47  SynthesizerGroup group;
48  group.setName(e.name().toString());
49 
50  while (e.readNextStartElement()) {
51  if (e.name() == "val") {
52  int id = e.intAttribute("id");
53  group.push_back(IdValue(id, e.readElementText()));
54  }
55  else
56  e.unknown();
57  }
58  tempGroups.push_back(group);
59  }
60 
61  if (!tempGroups.empty()) {
62  // Replace any previously set state if we have read a new state
63  swap(tempGroups);
64  setIsDefault(false);
65  }
66  }

105  {

107 
108  int method = 1;
109  int cc = -1;
110 
111  for (const IdValue &idVal : g) {
112  if (idVal.id == 4)
113  method = idVal.data.toInt();
114  else if (idVal.id == 5) {
115  switch (idVal.data.toInt()) {
116  case 0:
117  cc = 1;
118  break;
119  case 1:
120  cc = 2;
121  break;
122  case 2:
123  cc = 4;
124  break;
125  case 3:
126  cc = 11;
127  break;
128  default:
129  qWarning("Unrecognised CCToUse index from synthesizer: %d", idVal.data.toInt());
130  }
131  }
132  }
133 
134  if (method == 0) // velocity only
135  return -1;
136 
137  return cc;
