100 void TempoMap::normalize()
101  {
102  qreal time = 0;
103  int tick = 0;
104  qreal tempo = 2.0;
105  for (auto e = begin(); e != end(); ++e) {
106  // entries that represent a pause *only* (not tempo change also)
107  // need to be corrected to continue previous tempo
108  if (!(e->second.type & (TempoType::FIX|TempoType::RAMP)))
109  e->second.tempo = tempo;
110  int delta = e->first - tick;
111  time += qreal(delta) / (MScore::division * tempo * _relTempo);
112  time += e->second.pause;
113  e->second.time = time;
114  tick = e->first;
115  tempo = e->second.tempo;
116  }
117  ++_tempoSN;
118  }

162 qreal TempoMap::tempo(int tick) const
163  {
164  if (empty())
165  return 2.0;
166  auto i = lower_bound(tick);
167  if (i == end()) {

242 qreal TempoMap::tick2time(int tick, int* sn) const
243  {
244  qreal time = 0.0;
245  qreal delta = qreal(tick);
246  qreal tempo = 2.0;
247 
248  if (!empty()) {
249  int ptick = 0;
250  auto e = lower_bound(tick);
251  if (e == end()) {
252  auto pe = e;
253  --pe;
254  ptick = pe->first;
255  tempo = pe->second.tempo;
256  time = pe->second.time;
257  }
258  else if (e->first == tick) {
259  ptick = tick;
260  tempo = e->second.tempo;
261  time = e->second.time;
262  }
263  else if (e != begin()) {
264  auto pe = e;
265  --pe;
266  ptick = pe->first;
267  tempo = pe->second.tempo;
268  time = pe->second.time;
269  }
270  delta = qreal(tick - ptick);
271  }
272  else
273  qDebug("TempoMap: empty");
274  if (sn)
275  *sn = _tempoSN;
276  time += delta / (MScore::division * tempo * _relTempo);
277  return time;
278  }

284 int TempoMap::time2tick(qreal time, int* sn) const
285  {
286  int tick = 0;
287  qreal delta = time;
288  qreal tempo = _tempo;
289 
290  delta = 0.0;
291  tempo = 2.0;
292  for (auto e = begin(); e != end(); ++e) {
293  // if in a pause period, wait on previous tick
294  if ((time <= e->second.time) && (time > e->second.time - e->second.pause)) {
295  delta = (time - (e->second.time - e->second.pause) + delta);
296  break;
297  }
298  if (e->second.time >= time)
299  break;
300  delta = e->second.time;
301  tick = e->first;
302  tempo = e->second.tempo;
303  }
304  delta = time - delta;
305  tick += lrint(delta * _relTempo * MScore::division * tempo);
306  if (sn)
307  *sn = _tempoSN;
308  return tick;
309  }
