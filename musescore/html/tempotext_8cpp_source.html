134 // findTempoDuration
135 // find the duration part (note + dot) of a tempo text in string s
136 // return the match position or -1 if not found
137 // set len to the match length and dur to the duration value

139 
140 int TempoText::findTempoDuration(const QString& s, int& len, TDuration& dur)
141  {
142  len = 0;
143  dur = TDuration();
144  for (const auto& i : tp) {
145  QRegExp re(i.pattern);
146  int pos = re.indexIn(s);
147  if (pos != -1) {
148  len = re.matchedLength();
149  dur = i.d;
150  return pos;
151  }
152  }
153  return -1;
154  }

263 //---------------------------------------------------------
264 // updateTempo
265 //---------------------------------------------------------
266 
267 void TempoText::updateTempo()
268  {
269  // cache regexp, they are costly to create
270  static QHash<QString, QRegExp> regexps;
271  static QHash<QString, QRegExp> regexps2;
272  QString s = plainText();
273  s.replace(",", ".");
274  s.replace("<sym>space</sym>"," ");
275  for (const TempoPattern& pa : tp) {
276  QRegExp re;
277  if (!regexps.contains(pa.pattern)) {
278  re = QRegExp(QString("%1\\s*=\\s*(\\d+[.]{0,1}\\d*)\\s*").arg(pa.pattern));
279  regexps[pa.pattern] = re;
280  }
281  re = regexps.value(pa.pattern);
282  if (re.indexIn(s) != -1) {
283  QStringList sl = re.capturedTexts();
284  if (sl.size() == 2) {
285  qreal nt = qreal(sl[1].toDouble()) * pa.f;
286  if (nt != _tempo) {
287  undoChangeProperty(Pid::TEMPO, QVariant(qreal(sl[1].toDouble()) * pa.f), propertyFlags(Pid::TEMPO));
288  _relative = 1.0;
289  _isRelative = false;
290  updateScore();
291  }
292  break;
293  }
294  }
295  else {
296  for (const TempoPattern& pa2 : tp) {
297  QString key = QString("%1_%2").arg(pa.pattern, pa2.pattern);
298  QRegExp re2;
299  if (!regexps2.contains(key)) {
300  re2 = QRegExp(QString("%1\\s*=\\s*%2\\s*").arg(pa.pattern, pa2.pattern));
301  regexps2[key] = re2;
302  }
303  re2 = regexps2.value(key);
304  if (re2.indexIn(s) != -1) {
305  _relative = pa2.f / pa.f;
306  _isRelative = true;
307  updateRelative();
308  updateScore();
309  return;
310  }
311  }
312  }
313  }
