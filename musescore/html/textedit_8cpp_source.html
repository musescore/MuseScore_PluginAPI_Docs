84 void TextBase::endEdit(EditData& ed)
85  {
86  TextEditData* ted = static_cast<TextEditData*>(ed.getData(this));
87  IF_ASSERT_FAILED(ted) {
88  return;
89  }
90 
91  UndoStack* undo = score()->undoStack();
92  IF_ASSERT_FAILED(undo) {
93  return;
94  }
95 
96  const QString actualXmlText = xmlText();
97  const QString actualPlainText = plainText();
98 
99  // replace all undo/redo records collected during text editing with
100  // one property change
101 
102  using Filter = UndoCommand::Filter;
103  const bool textWasEdited = (undo->getCurIdx() != ted->startUndoIdx);
104 
105  if (textWasEdited) {
106  undo->mergeCommands(ted->startUndoIdx);
107  undo->last()->filterChildren(Filter::TextEdit, this);
108  }
109  else {
110  // No text changes in "undo" part of undo stack,
111  // hence nothing to merge and filter.
112  undo->cleanRedoStack(); // prevent text editing commands from remaining in undo stack
113  }
114 
115  bool newlyAdded = false;
116 
117  if (ted->oldXmlText.isEmpty()) {
118  UndoCommand* ucmd = textWasEdited ? undo->prev() : undo->last();
119  if (ucmd && ucmd->hasFilteredChildren(Filter::AddElement, this)) {
120  // We have just added this element to a score.
121  // Combine undo records of text creation with text editing.
122  newlyAdded = true;
123  undo->mergeCommands(ted->startUndoIdx - 1);
124  }
125  }
126 
127  // TBox'es manage their Text themselves and are not removed if text is empty
128  const bool removeTextIfEmpty = !(parent() && parent()->isTBox());
129 
130  if (actualPlainText.isEmpty() && removeTextIfEmpty) {
131  qDebug("actual text is empty");
132 
133  // If this assertion fails, no undo command relevant to this text
134  // resides on undo stack and reopen() would corrupt the previous
135  // command. Text shouldn't happen to be empty in other cases though.
136  Q_ASSERT(newlyAdded || textWasEdited);
137 
138  undo->reopen();
139  score()->undoRemoveElement(this);
140  ed.element = 0;
141 
142  static const std::vector<Filter> filters {
143  Filter::AddElementLinked,
144  Filter::RemoveElementLinked,
145  Filter::ChangePropertyLinked,
146  Filter::Link,
147  };
148 
149  if (newlyAdded && !undo->current()->hasUnfilteredChildren(filters, this)) {
150  for (Filter f : filters)
151  undo->current()->filterChildren(f, this);
152 
153  score()->endCmd();
154  ted->setDeleteText(true); // mark this text element for deletion
155  }
156  else {
157  score()->endCmd();
158  }
159 
160  return;
161  }
162 
163  if (textWasEdited) {
164  setXmlText(ted->oldXmlText); // reset text to value before editing
165  undo->reopen();
166  undoChangeProperty(Pid::TEXT, actualXmlText); // change property to set text to actual value again
167  // this also changes text of linked elements
168  layout1();
169  triggerLayout(); // force relayout even if text did not change
170  score()->endCmd();
171  }
172  else {
173  triggerLayout();
174  }
175 
176  static const qreal w = 2.0;
177  score()->addRefresh(canvasBoundingRect().adjusted(-w, -w, w, w));
178  }

215 bool TextBase::edit(EditData& ed)
216  {
217  TextEditData* ted = static_cast<TextEditData*>(ed.getData(this));
218  TextCursor* _cursor = &ted->cursor;
219 
220  // do nothing on Shift, it messes up IME on Windows. See #64046
221  if (ed.key == Qt::Key_Shift)
222  return false;
223  QString s = ed.s;
224  bool ctrlPressed = ed.modifiers & Qt::ControlModifier;
225  bool shiftPressed = ed.modifiers & Qt::ShiftModifier;
226  bool altPressed = ed.modifiers & Qt::AltModifier;
227 
228  QTextCursor::MoveMode mm = shiftPressed ? QTextCursor::KeepAnchor : QTextCursor::MoveAnchor;
229 
230  bool wasHex = false;
231  if (hexState >= 0) {
232  if (ed.modifiers == (Qt::ControlModifier | Qt::ShiftModifier | Qt::KeypadModifier)) {
233  switch (ed.key) {
234  case Qt::Key_0:
235  case Qt::Key_1:
236  case Qt::Key_2:
237  case Qt::Key_3:
238  case Qt::Key_4:
239  case Qt::Key_5:
240  case Qt::Key_6:
241  case Qt::Key_7:
242  case Qt::Key_8:
243  case Qt::Key_9:
244  s = QChar::fromLatin1(ed.key);
245  ++hexState;
246  wasHex = true;
247  break;
248  default:
249  break;
250  }
251  }
252  else if (ed.modifiers == (Qt::ControlModifier | Qt::ShiftModifier)) {
253  switch (ed.key) {
254  case Qt::Key_A:
255  case Qt::Key_B:
256  case Qt::Key_C:
257  case Qt::Key_D:
258  case Qt::Key_E:
259  case Qt::Key_F:
260  s = QChar::fromLatin1(ed.key);
261  ++hexState;
262  wasHex = true;
263  break;
264  default:
265  break;
266  }
267  }
268  }
269 
270  if (!wasHex) {
271 //printf("======%x\n", s.isEmpty() ? -1 : s[0].unicode());
272 
273  switch (ed.key) {
274  case Qt::Key_Z: // happens when the undo stack is empty
275  if (ed.modifiers == Qt::ControlModifier)
276  return true;
277  break;
278 
279  case Qt::Key_Enter:
280  case Qt::Key_Return:
281  deleteSelectedText(ed);
282  score()->undo(new SplitText(_cursor), &ed);
283  return true;
284 
285  case Qt::Key_Delete:
286  if (!deleteSelectedText(ed)) {
287  // check for move down
288  if (_cursor->column() == _cursor->columns()) { // if you are on the end of the line, delete the newline char
289  int cursorRow = _cursor->row();
290  _cursor->movePosition(QTextCursor::Down);
291  if (_cursor->row() != cursorRow) {
292  _cursor->movePosition(QTextCursor::StartOfLine);
293  score()->undo(new JoinText(_cursor), &ed);
294  }
295  }
296  else {
297  score()->undo(new RemoveText(_cursor, QString(_cursor->currentCharacter())), &ed);
298  }
299  }
300  return true;
301 
302  case Qt::Key_Backspace:
303  if (ctrlPressed) {
304  // delete last word
305  _cursor->movePosition(QTextCursor::WordLeft, QTextCursor::MoveMode::KeepAnchor);
306  s.clear();
307  deleteSelectedText(ed);
308  }
309  else {
310  if (!deleteSelectedText(ed)) {
311  if (_cursor->column() == 0 && _cursor->row() != 0)
312  score()->undo(new JoinText(_cursor), &ed);
313  else {
314  if (!_cursor->movePosition(QTextCursor::Left))
315  return false;
316  score()->undo(new RemoveText(_cursor, QString(_cursor->currentCharacter())), &ed);
317  }
318  }
319  }
320  return true;
321 
322  case Qt::Key_Left:
323  if (!_cursor->movePosition(ctrlPressed ? QTextCursor::WordLeft : QTextCursor::Left, mm) && type() == ElementType::LYRICS)
324  return false;
325  s.clear();
326  break;
327 
328  case Qt::Key_Right:
329  if (!_cursor->movePosition(ctrlPressed ? QTextCursor::NextWord : QTextCursor::Right, mm) && type() == ElementType::LYRICS)
330  return false;
331  s.clear();
332  break;
333 
334  case Qt::Key_Up:
335 #if defined(Q_OS_MAC)
336  if (!_cursor->movePosition(QTextCursor::Up, mm))
337  _cursor->movePosition(QTextCursor::StartOfLine, mm);
338 #else
339  _cursor->movePosition(QTextCursor::Up, mm);
340 #endif
341  s.clear();
342  break;
343 
344  case Qt::Key_Down:
345 #if defined(Q_OS_MAC)
346  if (!_cursor->movePosition(QTextCursor::Down, mm))
347  _cursor->movePosition(QTextCursor::EndOfLine, mm);
348 #else
349  _cursor->movePosition(QTextCursor::Down, mm);
350 #endif
351  s.clear();
352  break;
353 
354  case Qt::Key_Home:
355  if (ctrlPressed)
356  _cursor->movePosition(QTextCursor::Start, mm);
357  else
358  _cursor->movePosition(QTextCursor::StartOfLine, mm);
359 
360  s.clear();
361  break;
362 
363  case Qt::Key_End:
364  if (ctrlPressed)
365  _cursor->movePosition(QTextCursor::End, mm);
366  else
367  _cursor->movePosition(QTextCursor::EndOfLine, mm);
368 
369  s.clear();
370  break;
371 
372  case Qt::Key_Tab:
373  s = " ";
374  ed.modifiers = {};
375  break;
376 
377  case Qt::Key_Space:
378  if (ed.modifiers & CONTROL_MODIFIER) {
379  s = QString(QChar(0xa0)); // non-breaking space
380  }
381  else {
382  if (isFingering() && ed.view) {
383  score()->endCmd();
384  ed.view->textTab(ed.modifiers & Qt::ShiftModifier);
385  return true;
386  }
387  s = " ";
388  }
389  ed.modifiers = {};
390  break;
391 
392  case Qt::Key_Minus:
393  if (ed.modifiers == 0)
394  s = "-";
395  break;
396 
397  case Qt::Key_Underscore:
398  if (ed.modifiers == 0)
399  s = "_";
400  break;
401 
402  case Qt::Key_A:
403  if (ctrlPressed) {
404  _cursor->movePosition(QTextCursor::Start, QTextCursor::MoveMode::MoveAnchor);
405  _cursor->movePosition(QTextCursor::End, QTextCursor::MoveMode::KeepAnchor);
406  s.clear();
407  }
408  break;
409  default:
410  break;
411  }
412  if (ctrlPressed && shiftPressed) {
413  switch (ed.key) {
414  case Qt::Key_U:
415  if (hexState == -1) {
416  hexState = 0;
417  s = "u";
418  }
419  break;
420  case Qt::Key_B:
421  s = "\u266d"; // Unicode flat
422  break;
423  case Qt::Key_NumberSign: // e.g. QWERTY (US)
424  case Qt::Key_AsciiTilde: // e.g. QWERTY (GB)
425  case Qt::Key_Apostrophe: // e.g. QWERTZ (DE)
426  case Qt::Key_periodcentered: // e.g. QWERTY (ES)
427  case Qt::Key_3: // e.g. AZERTY (FR, BE)
428  s = "\u266f"; // Unicode sharp
429  break;
430  case Qt::Key_H:
431  s = "\u266e"; // Unicode natural
432  break;
433  case Qt::Key_Space:
434  insertSym(ed, SymId::space);
435  return true;
436  case Qt::Key_F:
437  insertSym(ed, SymId::dynamicForte);
438  return true;
439  case Qt::Key_M:
440  insertSym(ed, SymId::dynamicMezzo);
441  return true;
442  case Qt::Key_N:
443  insertSym(ed, SymId::dynamicNiente);
444  return true;
445  case Qt::Key_P:
446  insertSym(ed, SymId::dynamicPiano);
447  return true;
448  case Qt::Key_S:
449  insertSym(ed, SymId::dynamicSforzando);
450  return true;
451  case Qt::Key_R:
452  insertSym(ed, SymId::dynamicRinforzando);
453  return true;
454  case Qt::Key_Z:
455  // Ctrl+Z is normally "undo"
456  // but this code gets hit even if you are also holding Shift
457  // so Shift+Ctrl+Z works
458  insertSym(ed, SymId::dynamicZ);
459  return true;
460  }
461  }
462  if (ctrlPressed && altPressed) {
463  if (ed.key == Qt::Key_Minus) {
464  insertSym(ed, SymId::lyricsElision);
465  return true;
466  }
467  }
468  }
469  if (!s.isEmpty()) {
470  deleteSelectedText(ed);
471  score()->undo(new InsertText(_cursor, s), &ed);
472  }
473  return true;
474  }

618 void TextBase::paste(EditData& ed)
619  {
620  QString txt = QApplication::clipboard()->text(QClipboard::Clipboard);
621  if (MScore::debugMode)
622  qDebug("<%s>", qPrintable(txt));
623 
624  int state = 0;
625  QString token;
626  QString sym;
627  bool symState = false;
628 
629  score()->startCmd();
630  for (int i = 0; i < txt.length(); i++ ) {
631  QChar c = txt[i];
632  if (state == 0) {
633  if (c == '<') {
634  state = 1;
635  token.clear();
636  }
637  else if (c == '&') {
638  state = 2;
639  token.clear();
640  }
641  else {
642  if (symState)
643  sym += c;
644  else {
645  deleteSelectedText(ed);
646  if (c.isHighSurrogate()) {
647  QChar highSurrogate = c;
648  Q_ASSERT(i + 1 < txt.length());
649  i++;
650  QChar lowSurrogate = txt[i];
651  insertText(ed, QString(QChar::surrogateToUcs4(highSurrogate, lowSurrogate)));
652  }
653  else {
654  insertText(ed, QString(QChar(c.unicode())));
655  }
656  }
657  }
658  }
659  else if (state == 1) {
660  if (c == '>') {
661  state = 0;
662  if (token == "sym") {
663  symState = true;
664  sym.clear();
665  }
666  else if (token == "/sym") {
667  symState = false;
668  insertSym(ed, Sym::name2id(sym));
669  }
670  }
671  else
672  token += c;
673  }
674  else if (state == 2) {
675  if (c == ';') {
676  state = 0;
677  if (token == "lt")
678  insertText(ed, "<");
679  else if (token == "gt")
680  insertText(ed, ">");
681  else if (token == "amp")
682  insertText(ed, "&");
683  else if (token == "quot")
684  insertText(ed, "\"");
685  else
686  insertSym(ed, Sym::name2id(token));
687  }
688  else if (!c.isLetter()) {
689  state = 0;
690  insertText(ed, "&");
691  insertText(ed, token);
692  insertText(ed, c);
693  }
694  else
695  token += c;
696  }
697  }
698  if (state == 2) {
699  insertText(ed, "&");
700  insertText(ed, token);
701  }
702  score()->endCmd();
703  }
