73 void TextLineBaseSegment::draw(QPainter* painter) const
74  {
75  TextLineBase* tl = textLineBase();
76 
77  if (!_text->empty()) {
78  painter->translate(_text->pos());
79  _text->setVisible(tl->visible());
80  _text->draw(painter);
81  painter->translate(-_text->pos());
82  }
83 
84  if (!_endText->empty()) {
85  painter->translate(_endText->pos());
86  _endText->setVisible(tl->visible());
87  _endText->draw(painter);
88  painter->translate(-_endText->pos());
89  }
90 
91  if ((npoints == 0) || (score() && (score()->printing() || !score()->showInvisible()) && !tl->lineVisible()))
92  return;
93 
94  // color for line (text color comes from the text properties)
95 #if 0
96  QColor color;
97  if ((selected() && !(score() && score()->printing())) || !tl->visible() || !tl->lineVisible())
98  color = curColor(tl->visible() && tl->lineVisible());
99  else
100  color = tl->lineColor();
101 #endif
102  QColor color = curColor(tl->visible() && tl->lineVisible(), tl->lineColor());
103 
104  qreal textlineLineWidth = tl->lineWidth();
105  if (staff())
106  textlineLineWidth *= mag();
107  QPen pen(color, textlineLineWidth, tl->lineStyle());
108  QPen solidPen(color, textlineLineWidth, Qt::SolidLine);
109 
110  //Replace generic Qt dash patterns with improved equivalents to show true dots
111  QVector<qreal> dotted = { 0.01, 1.99 }; // 0.01 for cap dots. tighter than default Qt Dotline (would be { 0.01, 2.99 }).
112  QVector<qreal> dashed = { 3.0, 3.0 }; // Compensating for caps. Qt default DashLine is { 4.0, 2.0 }
113  QVector<qreal> dashDotted = { 3.0, 3.0, 0.01, 2.99 };
114  QVector<qreal> dashDotDotted = { 3.0, 3.0, 0.01, 2.99, 0.01, 2.99 };
115  QVector<qreal> customDashes = { tl->dashLineLen(), tl->dashGapLen() };
116 
117  switch (tl->lineStyle()) {
118  case Qt::DashLine:
119  pen.setDashPattern(dashed);
120  break;
121  case Qt::DotLine:
122  pen.setDashPattern(dotted);
123  pen.setCapStyle(Qt::RoundCap); // round dots
124  break;
125  case Qt::DashDotLine:
126  pen.setDashPattern(dashDotted);
127  break;
128  case Qt::DashDotDotLine:
129  pen.setDashPattern(dashDotDotted);
130  break;
131  case Qt::CustomDashLine:
132  pen.setDashPattern(customDashes);
133  break;
134  default:
135  break;
136  }
137 
138  //Draw lines
139  if (twoLines) { // hairpins
140  painter->setPen(pen);
141  painter->drawLines(&points[0], 1);
142  painter->drawLines(&points[2], 1);
143  }
144  else {
145  int start = 0;
146  int end = npoints;
147  //draw centered hooks as solid
148  painter->setPen(solidPen);
149  if (tl->beginHookType() == HookType::HOOK_90T) {
150  painter->drawLines(&points[0], 1);
151  start++;
152  }
153  if (tl->endHookType() == HookType::HOOK_90T) {
154  painter->drawLines(&points[npoints-1], 1);
155  end--;
156  }
157  //draw rest of line as regular
158  //calculate new gap
159  if (tl->lineStyle() == Qt::CustomDashLine) {
160  qreal adjustedLineLength = lineLength / textlineLineWidth;
161  qreal dash = tl->dashLineLen();
162  qreal gap = tl->dashGapLen();
163  int numPairs;
164  qreal newGap = 0;
165  QVector<qreal> nDashes { dash, newGap };
166  if (tl->beginHookType() == HookType::HOOK_45 || tl->beginHookType() == HookType::HOOK_90) {
167  qreal absD = sqrt(QPointF::dotProduct(points[start+1]-points[start], points[start+1]-points[start])) / textlineLineWidth;
168  numPairs = std::max(qreal(1), absD / (dash + gap));
169  nDashes[1] = (absD - dash * (numPairs + 1)) / numPairs;
170  pen.setDashPattern(nDashes);
171  painter->setPen(pen);
172  painter->drawLine(points[start+1], points[start]);
173  start++;
174  }
175  if (tl->endHookType() == HookType::HOOK_45 || tl->endHookType() == HookType::HOOK_90) {
176  qreal absD = sqrt(QPointF::dotProduct(points[end]-points[end-1], points[end]-points[end-1])) / textlineLineWidth;
177  numPairs = std::max(qreal(1), absD / (dash + gap));
178  nDashes[1] = (absD - dash * (numPairs + 1)) / numPairs;
179  pen.setDashPattern(nDashes);
180  painter->setPen(pen);
181  painter->drawLines(&points[end-1], 1);
182  end--;
183  }
184  numPairs = std::max(qreal(1), adjustedLineLength / (dash + gap));
185  nDashes[1] = (adjustedLineLength - dash * (numPairs + 1)) / numPairs;
186  pen.setDashPattern(nDashes);
187  }
188  painter->setPen(pen);
189  for (int i = start; i < end; ++i)
190  painter->drawLines(&points[i], 1);
191  }
192  }

198 Shape TextLineBaseSegment::shape() const
199  {
200  Shape shape;
201  if (!_text->empty())
202  shape.add(_text->bbox().translated(_text->pos()));
203  if (!_endText->empty())
204  shape.add(_endText->bbox().translated(_endText->pos()));
205  qreal lw = textLineBase()->lineWidth();
206  qreal lw2 = lw * .5;
207  if (twoLines) { // hairpins
208  shape.add(QRectF(points[0].x(), points[0].y() - lw2,
209  points[1].x() - points[0].x(), points[1].y() - points[0].y() + lw));
210  shape.add(QRectF(points[2].x(), points[2].y() - lw2,
211  points[3].x() - points[2].x(), points[3].y() - points[2].y() + lw));
212  }
213  else if (textLineBase()->lineVisible()) {
214  for (int i = 0; i < npoints; ++i) {
215  shape.add(QRectF(points[i].x() - lw2, points[i].y() - lw2,
216  points[i+1].x() - points[i].x() + lw, points[i+1].y() - points[i].y() + lw));
217  }
218  }
219  return shape;
220  }

226 void TextLineBaseSegment::layout()
227  {
228  npoints = 0;
229  TextLineBase* tl = textLineBase();
230  qreal _spatium = tl->spatium();
231 
232  if (spanner()->placeBelow())
233  rypos() = staff() ? staff()->height() : 0.0;
234 
235  // adjust Y pos to staffType offset
236  if (staffType())
237  rypos() += staffType()->yoffset().val() * spatium();
238 
239  if (!tl->diagonal())
240  _offset2.setY(0);
241 
242  switch (spannerSegmentType()) {
243  case SpannerSegmentType::SINGLE:
244  case SpannerSegmentType::BEGIN:
245  _text->setXmlText(tl->beginText());
246  _text->setFamily(tl->beginFontFamily());
247  _text->setSize(tl->beginFontSize());
248  _text->setOffset(tl->beginTextOffset() * mag());
249  _text->setAlign(tl->beginTextAlign());
250  _text->setBold(tl->beginFontStyle() & FontStyle::Bold);
251  _text->setItalic(tl->beginFontStyle() & FontStyle::Italic);
252  _text->setUnderline(tl->beginFontStyle() & FontStyle::Underline);
253  break;
254  case SpannerSegmentType::MIDDLE:
255  case SpannerSegmentType::END:
256  _text->setXmlText(tl->continueText());
257  _text->setFamily(tl->continueFontFamily());
258  _text->setSize(tl->continueFontSize());
259  _text->setOffset(tl->continueTextOffset() * mag());
260  _text->setAlign(tl->continueTextAlign());
261  _text->setBold(tl->continueFontStyle() & FontStyle::Bold);
262  _text->setItalic(tl->continueFontStyle() & FontStyle::Italic);
263  _text->setUnderline(tl->continueFontStyle() & FontStyle::Underline);
264 
265  break;
266  }
267  _text->setPlacement(Placement::ABOVE);
268  _text->setTrack(track());
269  _text->layout();
270 
271  if ((isSingleType() || isEndType())) {
272  _endText->setXmlText(tl->endText());
273  _endText->setFamily(tl->endFontFamily());
274  _endText->setSize(tl->endFontSize());
275  _endText->setOffset(tl->endTextOffset());
276  _endText->setAlign(tl->endTextAlign());
277  _endText->setBold(tl->endFontStyle() & FontStyle::Bold);
278  _endText->setItalic(tl->endFontStyle() & FontStyle::Italic);
279  _endText->setUnderline(tl->endFontStyle() & FontStyle::Underline);
280  _endText->setPlacement(Placement::ABOVE);
281  _endText->setTrack(track());
282  _endText->layout();
283  }
284  else {
285  _endText->setXmlText("");
286  }
287 
288  QPointF pp1;
289  QPointF pp2(pos2());
290 
291  // diagonal line with no text or hooks - just use the basic rectangle for line
292  if (_text->empty() && _endText->empty() && pp2.y() != 0
293  && textLineBase()->beginHookType() == HookType::NONE
294  && textLineBase()->endHookType() == HookType::NONE) {
295  npoints = 1; // 2 points, but only one line must be drawn
296  points[0] = pp1;
297  points[1] = pp2;
298  lineLength = sqrt(QPointF::dotProduct(pp2-pp1, pp2-pp1));
299 
300  setbbox(QRectF(pp1, pp2).normalized());
301  return;
302  }
303 
304  // line has text or hooks or is not diagonal - calculate reasonable bbox
305 
306  qreal x1 = qMin(0.0, pp2.x());
307  qreal x2 = qMax(0.0, pp2.x());
308  qreal y0 = -textLineBase()->lineWidth();
309  qreal y1 = qMin(0.0, pp2.y()) + y0;
310  qreal y2 = qMax(0.0, pp2.y()) - y0;
311 
312  qreal l = 0.0;
313  if (!_text->empty()) {
314  qreal textlineTextDistance = _spatium * .5;
315  if (((isSingleType() || isBeginType())
316  && (tl->beginTextPlace() == PlaceText::LEFT || tl->beginTextPlace() == PlaceText::AUTO))
317  || ((isMiddleType() || isEndType()) && (tl->continueTextPlace() == PlaceText::LEFT))) {
318  l = _text->pos().x() + _text->bbox().width() + textlineTextDistance;
319  }
320  qreal h = _text->height();
321  if (textLineBase()->beginTextPlace() == PlaceText::ABOVE)
322  y1 = qMin(y1, -h);
323  else if (textLineBase()->beginTextPlace() == PlaceText::BELOW)
324  y2 = qMax(y2, h);
325  else {
326  y1 = qMin(y1, -h * .5);
327  y2 = qMax(y2, h * .5);
328  }
329  x2 = qMax(x2, _text->width());
330  }
331 
332  if (textLineBase()->endHookType() != HookType::NONE) {
333  qreal h = pp2.y() + textLineBase()->endHookHeight().val() * _spatium;
334  if (h > y2)
335  y2 = h;
336  else if (h < y1)
337  y1 = h;
338  }
339 
340  if (textLineBase()->beginHookType() != HookType::NONE) {
341  qreal h = textLineBase()->beginHookHeight().val() * _spatium;
342  if (h > y2)
343  y2 = h;
344  else if (h < y1)
345  y1 = h;
346  }
347  bbox().setRect(x1, y1, x2 - x1, y2 - y1);
348  if (!_text->empty())
349  bbox() |= _text->bbox().translated(_text->pos()); // DEBUG
350  // set end text position and extend bbox
351  if (!_endText->empty()) {
352  _endText->setPos(bbox().right(), 0);
353  bbox() |= _endText->bbox().translated(_endText->pos());
354  }
355 
356  if (!(tl->lineVisible() || score()->showInvisible()))
357  return;
358 
359  if (tl->lineVisible() || !score()->printing()) {
360  pp1 = QPointF(l, 0.0);
361 
362  qreal beginHookWidth;
363  qreal endHookWidth;
364 
365  if (tl->beginHookType() == HookType::HOOK_45) {
366  beginHookWidth = fabs(tl->beginHookHeight().val() * _spatium * .4);
367  pp1.rx() += beginHookWidth;
368  }
369  else
370  beginHookWidth = 0;
371 
372  if (tl->endHookType() == HookType::HOOK_45) {
373  endHookWidth = fabs(tl->endHookHeight().val() * _spatium * .4);
374  pp2.rx() -= endHookWidth;
375  }
376  else
377  endHookWidth = 0;
378 
379  // don't draw backwards lines (or hooks) if text is longer than nominal line length
380  bool backwards = !_text->empty() && pp1.x() > pp2.x() && !tl->diagonal();
381 
382  if ((tl->beginHookType() != HookType::NONE) && (isSingleType() || isBeginType())) {
383  qreal hh = tl->beginHookHeight().val() * _spatium;
384  if (tl->beginHookType() == HookType::HOOK_90T)
385  points[npoints++] = QPointF(pp1.x() - beginHookWidth, pp1.y() - hh);
386  points[npoints] = QPointF(pp1.x() - beginHookWidth, pp1.y() + hh);
387  ++npoints;
388  points[npoints] = pp1;
389  }
390  if (!backwards) {
391  points[npoints] = pp1;
392  ++npoints;
393  points[npoints] = pp2;
394  lineLength = sqrt(QPointF::dotProduct(pp2-pp1, pp2-pp1));
395  // painter->drawLine(QLineF(pp1.x(), pp1.y(), pp2.x(), pp2.y()));
396 
397  if ((tl->endHookType() != HookType::NONE) && (isSingleType() || isEndType())) {
398  ++npoints;
399  qreal hh = tl->endHookHeight().val() * _spatium;
400  // painter->drawLine(QLineF(pp2.x(), pp2.y(), pp2.x() + endHookWidth, pp2.y() + hh));
401  points[npoints] = QPointF(pp2.x() + endHookWidth, pp2.y() + hh);
402  if (tl->endHookType() == HookType::HOOK_90T)
403  points[++npoints] = QPointF(pp2.x() + endHookWidth, pp2.y() - hh);
404  }
405  }
406  }
407  }
