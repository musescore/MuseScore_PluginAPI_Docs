84 void TimeDialog::addClicked()
85  {
86  TimeSig* ts = new TimeSig(gscore);
87  ts->setSig(Fraction(zNominal->value(), denominator()));
88  ts->setGroups(groups->groups());
89 
90  // check for special text
91  if ((QString("%1").arg(zNominal->value()) != zText->text())
92  || (QString("%1").arg(denominator()) != nText->text())) {
93  ts->setNumeratorString(zText->text());
94  ts->setDenominatorString(nText->text());
95  }
96  // extend palette:
97  sp->append(ts, "");
98  sp->setSelected(sp->size() - 1);
99  _dirty = true;
100  emit timeSigAdded(ts);

202  zText->setEnabled(true);
203  nText->setEnabled(true);
204  groups->setEnabled(true);
205  addButton->setEnabled(true);
206 
207  Fraction sig(e->sig());
208  Groups g = e->groups();
209  if (g.empty())
210  g = Groups::endings(sig);
211  zNominal->setValue(sig.numerator());
212  nNominal->setCurrentIndex(denominator2Idx(sig.denominator()));
213  zText->setText(e->numeratorString());
214  nText->setText(e->denominatorString());
215  groups->setSig(sig, g, zText->text(), nText->text());
216  }
