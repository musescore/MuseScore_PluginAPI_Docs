151 
152 void TimeSig::read(XmlReader& e)
153  {
154  int n=0, z1=0, z2=0, z3=0, z4=0;
155  bool old = false;
156 
157  while (e.readNextStartElement()) {
158  const QStringRef& tag(e.name());
159 
160  if (tag == "den") {
161  old = true;
162  n = e.readInt();
163  }
164  else if (tag == "nom1") {
165  old = true;
166  z1 = e.readInt();
167  }
168  else if (tag == "nom2") {
169  old = true;
170  z2 = e.readInt();
171  }
172  else if (tag == "nom3") {
173  old = true;
174  z3 = e.readInt();
175  }
176  else if (tag == "nom4") {
177  old = true;
178  z4 = e.readInt();
179  }
180  else if (tag == "subtype") {
181  int i = e.readInt();
182  if (score()->mscVersion() <= 114) {
183  if (i == 0x40000104)
184  _timeSigType = TimeSigType::FOUR_FOUR;
185  else if (i == 0x40002084)
186  _timeSigType = TimeSigType::ALLA_BREVE;
187  else
188  _timeSigType = TimeSigType::NORMAL;
189  }
190  else
191  _timeSigType = TimeSigType(i);
192  }
193  else if (tag == "showCourtesySig")
194  _showCourtesySig = e.readInt();
195  else if (tag == "sigN")
196  _sig.setNumerator(e.readInt());
197  else if (tag == "sigD")
198  _sig.setDenominator(e.readInt());
199  else if (tag == "stretchN")
200  _stretch.setNumerator(e.readInt());
201  else if (tag == "stretchD")
202  _stretch.setDenominator(e.readInt());
203  else if (tag == "textN")
204  setNumeratorString(e.readElementText());
205  else if (tag == "textD")
206  setDenominatorString(e.readElementText());
207  else if (tag == "Groups")
208  _groups.read(e);
209  else if (readStyledProperty(e, tag))
210  ;
211  else if (!Element::readProperties(e))
212  e.unknown();
213  }
214  if (old) {
215  _sig.set(z1+z2+z3+z4, n);
216  }
217  _stretch.reduce();
218 
219  // HACK: handle time signatures from scores before 3.5 differently on some special occasions.
220  // See https://musescore.org/node/308139.
221  QString version = masterScore()->mscoreVersion();
222  if (!version.isEmpty() && (version >= "3.0") && (version < "3.5")) {
223  if ((_timeSigType == TimeSigType::NORMAL) && !_numeratorString.isEmpty() && _denominatorString.isEmpty()) {
224  if (_numeratorString == QString::number(_sig.numerator()))
225  _numeratorString.clear();
226  else
227  setDenominatorString(QString::number(_sig.denominator()));
228  }
229  }
230  }

253 void TimeSig::layout()
254  {
255  setPos(0.0, 0.0);
256  qreal _spatium = spatium();
257 
258  setbbox(QRectF()); // prepare for an empty time signature
259  pointLargeLeftParen = QPointF();
260  pz = QPointF();
261  pn = QPointF();
262  pointLargeRightParen = QPointF();
263 
264  qreal lineDist;
265  int numOfLines;
266  TimeSigType sigType = timeSigType();
267  const Staff* _staff = staff();
268 
269  if (_staff) {
270  // if staff is without time sig, format as if no text at all
271  if (!_staff->staffTypeForElement(this)->genTimesig()) {
272  // reset position and box sizes to 0
273  // qDebug("staff: no time sig");
274  pointLargeLeftParen.rx() = 0.0;
275  pn.rx() = 0.0;
276  pz.rx() = 0.0;
277  pointLargeRightParen.rx() = 0.0;
278  setbbox(QRectF());
279  // leave everything else as it is:
280  // draw() will anyway skip any drawing if staff type has no time sigs
281  return;
282  }
283  numOfLines = _staff->lines(tick());
284  lineDist = _staff->lineDistance(tick());
285  }
286  else {
287  // assume dimensions of a standard staff
288  lineDist = 1.0;
289  numOfLines = 5;
290  }
291 
292  // if some symbol
293  // compute vert. displacement to center in the staff height
294  // determine middle staff position:
295 
296  qreal yoff = _spatium * (numOfLines-1) *.5 * lineDist;
297 
298  // C and Ccut are placed at the middle of the staff: use yoff directly
299  if (sigType == TimeSigType::FOUR_FOUR) {
300  pz = QPointF(0.0, yoff);
301  setbbox(symBbox(SymId::timeSigCommon).translated(pz));
302  ns.clear();
303  ns.push_back(SymId::timeSigCommon);
304  ds.clear();
305  }
306  else if (sigType == TimeSigType::ALLA_BREVE) {
307  pz = QPointF(0.0, yoff);
308  setbbox(symBbox(SymId::timeSigCutCommon).translated(pz));
309  ns.clear();
310  ns.push_back(SymId::timeSigCutCommon);
311  ds.clear();
312  }
313  else if (sigType == TimeSigType::CUT_BACH) {
314  pz = QPointF(0.0, yoff);
315  setbbox(symBbox(SymId::timeSigCut2).translated(pz));
316  ns.clear();
317  ns.push_back(SymId::timeSigCut2);
318  ds.clear();
319  }
320  else if (sigType == TimeSigType::CUT_TRIPLE) {
321  pz = QPointF(0.0, yoff);
322  setbbox(symBbox(SymId::timeSigCut3).translated(pz));
323  ns.clear();
324  ns.push_back(SymId::timeSigCut3);
325  ds.clear();
326  }
327  else {
328  if (_numeratorString.isEmpty()) {
329  ns = toTimeSigString(_numeratorString.isEmpty() ? QString::number(_sig.numerator()) : _numeratorString);
330  ds = toTimeSigString(_denominatorString.isEmpty() ? QString::number(_sig.denominator()) : _denominatorString);
331  }
332  else {
333  ns = toTimeSigString(_numeratorString);
334  ds = toTimeSigString(_denominatorString);
335  }
336 
337  ScoreFont* font = score()->scoreFont();
338  QSizeF mag(magS() * _scale);
339 
340  QRectF numRect = font->bbox(ns, mag);
341  QRectF denRect = font->bbox(ds, mag);
342 
343  // position numerator and denominator; vertical displacement:
344  // number of lines is odd: 0.0 (strings are directly above and below the middle line)
345  // number of lines even: 0.05 (strings are moved up/down to leave 1/10sp between them)
346 
347  qreal displ = (numOfLines & 1) ? 0.0 : (0.05 * _spatium);
348 
349  //align on the wider
350  qreal pzY = yoff - (denRect.width() < 0.01 ? 0.0 : (displ + numRect.height() * .5));
351  qreal pnY = yoff + displ + denRect.height() * .5;
352 
353  if (numRect.width() >= denRect.width()) {
354  // numerator: one space above centre line, unless denomin. is empty (if so, directly centre in the middle)
355  pz = QPointF(0.0, pzY);
356  // denominator: horiz: centred around centre of numerator | vert: one space below centre line
357  pn = QPointF((numRect.width() - denRect.width())*.5, pnY);
358  }
359  else {
360  // numerator: one space above centre line, unless denomin. is empty (if so, directly centre in the middle)
361  pz = QPointF((denRect.width() - numRect.width())*.5, pzY);
362  // denominator: horiz: centred around centre of numerator | vert: one space below centre line
363  pn = QPointF(0.0, pnY);
364  }
365 
366  // centering of parenthesis so the middle of the parenthesis is at the divisor marking level
367  int centerY = yoff/2 + _spatium;
368  int widestPortion = numRect.width() > denRect.width() ? numRect.width() : denRect.width();
369  pointLargeLeftParen = QPointF(-_spatium, centerY);
370  pointLargeRightParen = QPointF(widestPortion + _spatium, centerY);
371 
372  setbbox(numRect.translated(pz)); // translate bounding boxes to actual string positions
373  addbbox(denRect.translated(pn));
374  if (_largeParentheses) {
375  addbbox(QRect(pointLargeLeftParen.x(), pointLargeLeftParen.y() - denRect.height(), _spatium / 2, numRect.height() + denRect.height()));
376  addbbox(QRect(pointLargeRightParen.x(), pointLargeRightParen.y() - denRect.height(), _spatium / 2, numRect.height() + denRect.height()));
377  }
378  }
379  }

385 Shape TimeSig::shape() const
386  {
387  QRectF box(bbox());
388  const Staff* st = staff();
389  if (st && addToSkyline()) {
390  // Extend time signature shape up and down to
391  // the first ledger line height to ensure that
392  // no notes will be too close to the timesig.
393  const qreal sp = spatium();
394  const qreal y = pos().y();
395  box.setTop(std::min(-sp - y, box.top()));
396  box.setBottom(std::max(st->height() - y + sp, box.bottom()));
397  }
398  return Shape(box);
399  }
