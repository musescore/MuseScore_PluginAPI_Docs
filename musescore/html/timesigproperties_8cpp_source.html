144 void TimeSigProperties::accept()
145  {
146  TimeSigType ts = TimeSigType::NORMAL;
147  if (textButton->isChecked() || otherButton->isChecked())
148  ts = TimeSigType::NORMAL;
149  else if (fourfourButton->isChecked())
150  ts = TimeSigType::FOUR_FOUR;
151  else if (allaBreveButton->isChecked())
152  ts = TimeSigType::ALLA_BREVE;
153 
154  Fraction actual(zActual->value(), nActual->value());
155  Fraction nominal(zNominal->value(), nNominal->value());
156  timesig->setSig(actual, ts);
157  timesig->setStretch(nominal / actual);
158 
159  if (zText->text() != timesig->numeratorString())
160  timesig->setNumeratorString(zText->text());
161  if (nText->text() != timesig->denominatorString())
162  timesig->setDenominatorString(nText->text());
163 
164  if (otherButton->isChecked()) {
165  ScoreFont* scoreFont = timesig->score()->scoreFont();
166  SymId symId = (SymId)( otherCombo->itemData(otherCombo->currentIndex()).toInt() );
167  // ...and set numerator to font string for symbol and denominator to empty string
168  timesig->setNumeratorString(scoreFont->toString(symId));
169  timesig->setDenominatorString(QString());
170  }
171 
172  Groups g = groups->groups();
173  timesig->setGroups(g);
174  QDialog::accept();
175  }
