119 TransposeDirection TransposeDialog::direction() const
120  {
121  switch (mode())
122  {
123  case TransposeMode::TO_KEY:
124  if (closestKey->isChecked())
125  return TransposeDirection::CLOSEST;
126  return upKey->isChecked() ? TransposeDirection::UP : TransposeDirection::DOWN;
127  case TransposeMode::BY_INTERVAL:
128  return upInterval->isChecked() ? TransposeDirection::UP : TransposeDirection::DOWN;
129  case TransposeMode::DIATONICALLY:
130  return upDiatonic->isChecked() ? TransposeDirection::UP : TransposeDirection::DOWN;
131  }
132  return TransposeDirection::UP;
133  }
