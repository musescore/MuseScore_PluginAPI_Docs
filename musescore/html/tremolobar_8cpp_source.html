45 void TremoloBar::layout()
46  {
47  qreal _spatium = spatium();
48  if (parent())
49  setPos(0.0, -_spatium * 3.0);
50  else
51  setPos(QPointF());
52 
53  /* we place the tremolo bars starting slightly before the
54  * notehead, and end it slightly after, drawing above the
55  * note. The values specified in Guitar Pro are very large, too
56  * large for the scale used in Musescore. We used the
57  * timeFactor and pitchFactor below to reduce these values down
58  * consistently to values that make sense to draw with the
59  * Musescore scale. */
60 
61  qreal timeFactor = _userMag / 1.0;
62  qreal pitchFactor = -_spatium * .02;
63 
64  polygon.clear();
65  for (auto v : qAsConst(_points))
66  polygon << QPointF(v.time * timeFactor, v.pitch * pitchFactor);
67 
68  qreal w = _lw.val();
69  setbbox(polygon.boundingRect().adjusted(-w, -w, w, w));
70  }

87 void TremoloBar::write(XmlWriter& xml) const
88  {
89  xml.stag(this);
90  writeProperty(xml, Pid::MAG);

104 void TremoloBar::read(XmlReader& e)
105  {
106  while (e.readNextStartElement()) {
107  auto tag = e.name();
108  if (tag == "point") {
109  PitchValue pv;
110  pv.time = e.intAttribute("time");
111  pv.pitch = e.intAttribute("pitch");
112  pv.vibrato = e.intAttribute("vibrato");
113  _points.append(pv);
114  e.readNext();
115  }
116  else if (tag == "mag")
117  _userMag = e.readDouble(0.1, 10.0);
118  else if (readStyledProperty(e, tag))
119  ;
120  else if (tag == "play")
121  setPlay(e.readInt());
122  else if (readProperty(tag, e, Pid::LINE_WIDTH))
123  ;
124  else
125  e.unknown();
126  }
127  }
