169  if (_elements.empty()) {
170  qDebug("Tuplet::layout(): tuplet is empty");
171  return;
172  }
173  // is in a TAB without stems, skip any format: tuplets are not shown
174  const StaffType* stt = staffType();

176  return;
177 
178  //
179  // create tuplet number if necessary
180  //
181  qreal _spatium = spatium();
182  if (_numberType != TupletNumberType::NO_TEXT) {
183  if (_number == 0) {
184  _number = new Text(score(), Tid::TUPLET);
185  _number->setComposition(true);
186  _number->setTrack(track());
187  _number->setParent(this);
188  _number->setVisible(visible());

190  }
191  if (_numberType == TupletNumberType::SHOW_NUMBER)
192  _number->setXmlText(QString("%1").arg(_ratio.numerator()));
193  else
194  _number->setXmlText(QString("%1:%2").arg(_ratio.numerator()).arg(_ratio.denominator()));
195 
196  bool small = true;
197  for (const DurationElement* e : _elements) {
198  if (e->isChordRest() && !toChordRest(e)->small()) {
199  small = false;
200  break;
201  }
202  }
203  _number->setMag(small ? score()->styleD(Sid::smallNoteMag) : 1.0);
204 
205  }
206  else {
207  if (_number) {
208  if (_number->selected())
209  score()->deselect(_number);
210  delete _number;
211  _number = 0;
212  }
213  }
214  //
215  // find out main direction
216  //
217  if (_direction == Direction::AUTO) {
218  int up = 1;
219  for (const DurationElement* e : _elements) {
220  if (e->isChord()) {
221  const Chord* c = toChord(e);
222  if (c->stemDirection() != Direction::AUTO)
223  up += c->stemDirection() == Direction::UP ? 1000 : -1000;
224  else {
225  up += c->up() ? 1 : -1;
226  }
227  }
228  else if (e->isTuplet()) {
229  // TODO
230  }
231  }
232  _isUp = up > 0;
233  }
234  else
235  _isUp = _direction == Direction::UP;
236 
237  //
238  // find first and last chord of tuplet
239  // (tuplets can be nested)
240  //
241  bool nested = false;
242  const DurationElement* cr1 = _elements.front();
243  while (cr1->isTuplet()) {
244  const Tuplet* t = toTuplet(cr1);
245  if (t->elements().empty())
246  break;
247  nested = true;
248  cr1 = t->elements().front();
249  }
250  const DurationElement* cr2 = _elements.back();
251  while (cr2->isTuplet()) {
252  const Tuplet* t = toTuplet(cr2);
253  if (t->elements().empty())
254  break;
255  nested = true;
256  cr2 = t->elements().back();
257  }
258 
259  //
260  // shall we draw a bracket?
261  //
262  if (_bracketType == TupletBracketType::AUTO_BRACKET) {
263  _hasBracket = false;
264  for (DurationElement* e : _elements) {
265  if (e->isTuplet() || e->isRest()) {
266  _hasBracket = true;
267  break;
268  }
269  else if (e->isChordRest()) {
270  ChordRest* cr = toChordRest(e);
271  //
272  // maybe we should check for more than one beam
273  //
274  if (cr->beam() == 0) {
275  _hasBracket = true;
276  break;
277  }
278  }
279  }
280  }
281  else
282  _hasBracket = _bracketType != TupletBracketType::SHOW_NO_BRACKET;
283 
284 
285  //
286  // calculate bracket start and end point p1 p2
287  //
288  qreal maxSlope = score()->styleD(Sid::tupletMaxSlope);
289  bool outOfStaff = score()->styleB(Sid::tupletOufOfStaff);
290  qreal vHeadDistance = score()->styleP(Sid::tupletVHeadDistance);
291  qreal vStemDistance = score()->styleP(Sid::tupletVStemDistance);
292  qreal stemLeft = score()->styleP(Sid::tupletStemLeftDistance);
293  qreal stemRight = score()->styleP(Sid::tupletStemRightDistance);
294  qreal noteLeft = score()->styleP(Sid::tupletNoteLeftDistance);
295  qreal noteRight = score()->styleP(Sid::tupletNoteRightDistance);
296 
297  int move = 0;
298  setTrack(cr1->staffIdx() * VOICES + voice());
299  if (outOfStaff && cr1->isChordRest() && cr2->isChordRest()) {
300  // account for staff move when adjusting bracket to avoid staff
301  // but don't attempt adjustment unless both endpoints are in same staff
302  // and not a nested tuplet
303  if (toChordRest(cr1)->staffMove() == toChordRest(cr2)->staffMove() && !tuplet() && !nested) {
304  move = toChordRest(cr1)->staffMove();
305  if (move == 1)
306  setTrack(cr1->vStaffIdx() * VOICES + voice());
307  }
308  else
309  outOfStaff = false;
310  }
311 
312  qreal l1 = score()->styleP(Sid::tupletBracketHookHeight);
313  qreal l2l = vHeadDistance; // left bracket vertical distance
314  qreal l2r = vHeadDistance; // right bracket vertical distance right
315 
316  if (_isUp)
317  vHeadDistance = -vHeadDistance;
318 
319  p1 = cr1->pagePos();
320  p2 = cr2->pagePos();
321 
322  p1.rx() -= noteLeft;
323  p2.rx() += score()->noteHeadWidth() + noteRight;
324  p1.ry() += vHeadDistance; // TODO: Direction ?
325  p2.ry() += vHeadDistance;
326 
327  qreal xx1 = p1.x(); // use to center the number on the beam
328 
329  // follow beam angle if one beam extends over entire tuplet
330  bool followBeam = false;
331  qreal beamAdjust = 0.0;
332  if (cr1->beam() && cr1->beam() == cr2->beam()) {
333  followBeam = true;
334  beamAdjust = point(score()->styleS(Sid::beamWidth)) * 0.5 * mag();
335  }
336 
337  if (_isUp) {
338  if (cr1->isChord()) {
339  const Chord* chord1 = toChord(cr1);
340  Stem* stem = chord1->stem();
341  if (stem)
342  xx1 = stem->abbox().x();
343  if (chord1->up()) {
344  if (stem) {
345  if (followBeam)
346  p1.ry() = stem->abbox().y() - beamAdjust;
347  else if (chord1->beam())
348  p1.ry() = chord1->beam()->abbox().y();
349  else
350  p1.ry() = stem->abbox().y();
351  l2l = vStemDistance;
352  }
353  else {
354  p1.ry() = chord1->upNote()->abbox().top(); // whole note
355  }
356  }
357  else if (!chord1->up()) {
358  p1.ry() = chord1->upNote()->abbox().top();
359  if (stem)
360  p1.rx() = cr1->pagePos().x() - stemLeft;
361  }
362  }
363 
364  if (cr2->isChord()) {
365  const Chord* chord2 = toChord(cr2);
366  Stem* stem = chord2->stem();
367  if (stem && chord2->up()) {
368  if (followBeam)
369  p2.ry() = stem->abbox().top() - beamAdjust;
370  else if (chord2->beam() && !chord2->staffMove() && !chord2->beam()->cross())
371  p2.ry() = chord2->beam()->abbox().top();
372  else
373  p2.ry() = stem->abbox().top();
374  l2r = vStemDistance;
375  p2.rx() = chord2->pagePos().x() + chord2->maxHeadWidth() + stemRight;
376  }
377  else {
378  p2.ry() = chord2->upNote()->abbox().top();
379  }
380  }
381  //
382  // special case: one of the bracket endpoints is
383  // a rest
384  //
385  if (cr1->isChord() && cr2->isChord()) {
386  if (p2.y() < p1.y())
387  p1.setY(p2.y());
388  else
389  p2.setY(p1.y());
390  }
391  else if (cr1->isChord() && !cr2->isChord()) {
392  if (p1.y() < p2.y())
393  p2.setY(p1.y());
394  else
395  p1.setY(p2.y());
396  }
397 
398  // outOfStaff
399  if (outOfStaff) {
400  qreal min = cr1->measure()->staffabbox(cr1->staffIdx() + move).y();
401  if (min < p1.y()) {
402  p1.ry() = min;
403  l2l = vStemDistance;
404  }
405  min = cr2->measure()->staffabbox(cr2->staffIdx() + move).y();
406  if (min < p2.y()) {
407  p2.ry() = min;
408  l2r = vStemDistance;
409  }
410  }
411 
412  // check that slope is no more than max
413  qreal d = (p2.y() - p1.y())/(p2.x() - p1.x());
414  if (d < -maxSlope) {
415  // move p1 y up
416  p1.ry() = p2.y() + maxSlope * (p2.x() - p1.x());
417  }
418  else if (d > maxSlope) {
419  // move p2 y up
420  p2.ry() = p1.ry() + maxSlope * (p2.x() - p1.x());
421  }
422 
423  // check for collisions
424  size_t n = _elements.size();
425  if (n >= 3) {
426  d = (p2.y() - p1.y())/(p2.x() - p1.x());
427  for (size_t i = 1; i < (n-1); ++i) {
428  Element* e = _elements[i];
429  if (e->isChord()) {
430  const Chord* chord = toChord(e);
431  const Stem* stem = chord->stem();
432  if (stem) {
433  QRectF r(chord->up() ? stem->abbox() : chord->upNote()->abbox());
434  qreal y3 = r.top();
435  qreal x3 = r.x() + r.width() * .5;
436  qreal y0 = p1.y() + (x3 - p1.x()) * d;
437  qreal c = y0 - y3;
438  if (c > 0) {
439  p1.ry() -= c;
440  p2.ry() -= c;
441  }
442  }
443  }
444  }
445  }
446  }
447  else {
448  if (cr1->isChord()) {
449  const Chord* chord1 = toChord(cr1);
450  Stem* stem = chord1->stem();
451  if (stem)
452  xx1 = stem->abbox().x();
453  if (!chord1->up()) {
454  if (stem) {
455  if (followBeam)
456  p1.ry() = stem->abbox().bottom() + beamAdjust;
457  else if (chord1->beam())
458  p1.ry() = chord1->beam()->abbox().bottom();
459  else
460  p1.ry() = stem->abbox().bottom();
461  l2l = vStemDistance;
462  p1.rx() = cr1->pagePos().x() - stemLeft;
463  }
464  else {
465  p1.ry() = chord1->downNote()->abbox().bottom(); // whole note
466  }
467  }
468  else if (chord1->up()) {
469  p1.ry() = chord1->downNote()->abbox().bottom();
470  }
471  }
472 
473  if (cr2->isChord()) {
474  const Chord* chord2 = toChord(cr2);
475  Stem* stem = chord2->stem();
476  if (stem && !chord2->up()) {
477  // if (chord2->beam())
478  // p2.setX(stem->abbox().x());
479  if (followBeam) //??
480  p2.ry() = stem->abbox().bottom() + beamAdjust; //??
481  if (chord2->beam() && !chord2->staffMove() && !chord2->beam()->cross())
482  p2.ry() = chord2->beam()->abbox().bottom();
483  else
484  p2.ry() = stem->abbox().bottom();
485  l2r = vStemDistance;
486  }
487  else {
488  p2.ry() = chord2->downNote()->abbox().bottom();
489  if (stem)
490  p2.rx() = chord2->pagePos().x() + chord2->maxHeadWidth() + stemRight;
491  }
492  }
493  //
494  // special case: one of the bracket endpoints is
495  // a rest
496  //
497  if (!cr1->isChord() && cr2->isChord()) {
498  if (p2.y() > p1.y())
499  p1.setY(p2.y());
500  else
501  p2.setY(p1.y());
502  }
503  else if (cr1->isChord() && !cr2->isChord()) {
504  if (p1.y() > p2.y())
505  p2.setY(p1.y());
506  else
507  p1.setY(p2.y());
508  }

510  if (outOfStaff) {
511  qreal max = cr1->measure()->staffabbox(cr1->staffIdx() + move).bottom();
512  if (max > p1.y()) {
513  p1.ry() = max;
514  l2l = vStemDistance;
515  }
516  max = cr2->measure()->staffabbox(cr2->staffIdx() + move).bottom();
517  if (max > p2.y()) {
518  p2.ry() = max;

1201 // sanitizeTuplet
1202 /// Check validity of tuplets and coherence between duration
1203 /// and baselength. Needed for importing old files due to a bug
1204 /// in the released version for corner-case tuplets.
1205 /// See issue #136406 and Pull request #2881
1206 //---------------------------------------------------------
1207 
1208 void Tuplet::sanitizeTuplet()
1209  {
1210  if (ratio().numerator() == ratio().reduced().numerator()) // return if the ratio is an irreducible fraction
1211  return;
1212  Fraction baseLenDuration = (Fraction(ratio().denominator(),1) * baseLen().fraction()).reduced();
1213 
1214  // Due to a bug present in 2.1 (and before), a tuplet with non-reduced ratio could be
1215  // in a corrupted state (mismatch between duration and base length).
1216  // A tentative will now be made to retrieve the correct duration by summing up all the
1217  // durations of the elements constituting the tuplet. This does not work for
1218  // not-completely filled tuplets, such as tuplets in voices > 0 with
1219  // gaps (for example, a tuplet in second voice with a deleted chordrest element)
1220 
1221  Fraction testDuration(0,1);
1222  for (DurationElement* de : elements()) {
1223  if (de == 0)
1224  continue;
1225  Fraction elementDuration(0,1);
1226  if (de->isTuplet()){
1227  Tuplet* t = toTuplet(de);
1228  t->sanitizeTuplet();
1229  elementDuration = t->ticks();
1230  }
1231  else {
1232  elementDuration = de->ticks();
1233  }
1234  testDuration += elementDuration;
1235  }
1236  testDuration = testDuration / ratio();
1237  testDuration.reduce();
1238  if (elements().back()->tick() + elements().back()->actualTicks() - elements().front()->tick() > testDuration)
1239  return; // this tuplet has missing elements; do not sanitize
1240  if (!(testDuration == baseLenDuration && baseLenDuration == ticks())) {
1241  Fraction f = testDuration * Fraction(1, ratio().denominator());
1242  f.reduce();
1243  Fraction fbl(1, f.denominator());
1244  if (TDuration::isValid(fbl)) {
1245  setTicks(testDuration);
1246  setBaseLen(fbl);
1247  qDebug("Tuplet %p sanitized duration %d/%d baseLen %d/%d",this,
1248  testDuration.numerator(), testDuration.denominator(),
1249  1, fbl.denominator());
1250  }
1251  else {
1252  qDebug("Impossible to sanitize the tuplet");
1253  }
1254  }

1285 // addMissingElements
1286 // Make this tuplet complete by filling in holes where
1287 // there ought to be rests. Needed for importing files
1288 // that saved incomplete tuplets.
1289 //---------------------------------------------------------
1290 
1291 void Tuplet::addMissingElements()
1292  {
1293  if (tuplet())
1294  return; // do not correct nested tuplets
1295  if (voice() == 0)
1296  return; // nothing to do for tuplets in voice 1
1297  Fraction missingElementsDuration = ticks() * ratio() - elementsDuration();
1298  if (missingElementsDuration.isZero())
1299  return;

1301  Fraction expectedTick = elements().front()->tick();
1302  for (DurationElement* de : elements()) {
1303  if (de->tick() != expectedTick) {
1304  missingElementsDuration -= addMissingElement(expectedTick, de->tick());
1305  if (missingElementsDuration.isZero())
1306  return;
1307  }
1308  expectedTick += de->actualTicks();
1309  }
1310  // calculate the tick where we would expect a tuplet of this duration to start
1311  // TODO: check:
1312  expectedTick = elements().front()->tick() - Fraction::fromTicks(elements().front()->tick().ticks() % ticks().ticks());
1313  if (expectedTick != elements().front()->tick()) {

1315  Fraction firstAvailableTick = measure()->tick();
1316  Segment* segment = measure()->findSegment(SegmentType::ChordRest, elements().front()->tick());
1317  ChordRest* prevChordRest = segment && segment->prev() ? segment->prev()->nextChordRest(track(), true) : nullptr;
1318  if (prevChordRest && prevChordRest->measure() == measure())
1319  firstAvailableTick = prevChordRest->tick() + prevChordRest->actualTicks();

1321  Fraction f = missingElementsDuration / ratio();
1322  Fraction ticksRequired = f;
1323  Fraction endTick = elements().front()->tick();
1324  Fraction startTick = std::max(firstAvailableTick, endTick - ticksRequired);
1325  if (expectedTick > startTick)
1326  startTick = expectedTick;
1327  missingElementsDuration -= addMissingElement(startTick, endTick);
1328  if (missingElementsDuration.isZero())
1329  return;
1330  }
1331  }
1332  // now fill a hole at the end of the tuplet
1333  Fraction startTick = elements().back()->tick() + elements().back()->actualTicks();
1334  Fraction endTick = elements().front()->tick() + ticks();

1336  Segment* segment = measure()->findSegment(SegmentType::ChordRest, elements().back()->tick());

1338  if (nextChordRest && nextChordRest->tick() < endTick)
1339  endTick = nextChordRest->tick();
