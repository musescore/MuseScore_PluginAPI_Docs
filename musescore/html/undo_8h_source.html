204  bool empty() const { return !canUndo() && !canRedo(); }
205  UndoMacro* current() const { return curCmd; }
206  UndoMacro* last() const { return curIdx > 0 ? list[curIdx-1] : 0; }
207  UndoMacro* prev() const { return curIdx > 1 ? list[curIdx-2] : 0; }
208  void undo(EditData*);
209  void redo(EditData*);
210  void rollback();
211  void reopen();
