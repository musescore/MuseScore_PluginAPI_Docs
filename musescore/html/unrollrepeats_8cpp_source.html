68 static void removeRepeatMarkings(Score* score)
69  {
70 
71  // remove bar-level repeats
72  for (Measure* m = score->firstMeasure(); m; m = m->nextMeasure()) {
73  m->setRepeatStart(false);
74  m->setRepeatEnd(false);
75  }
76 
77  // remove volta markers
78  auto smap = score->spannerMap().map();
79  for (auto it = smap.cbegin(); it != smap.cend(); ++it) {
80  Spanner* s = (*it).second;
81  if (!s || !s->isVolta()) continue;
82  score->removeSpanner(s);
83  }
84 
85  // remove coda/fine labels and jumps
86  QList<Element*> elems;
87  score->scanElements(&elems, collectElements, false);
88  for (auto e : elems) {
89  if (e->isMarker() || e->isJump())
90  score->deleteItem(e);
91  else if (e->isBarLine())
92  toBarLine(e)->setBarLineType(BarLineType::NORMAL);
93  }
94 
95  // set the last bar line to end symbol
96  score->lastMeasure()->setEndBarLineType(BarLineType::END, false);
97  Segment* last = score->lastMeasure()->segments().last();
98  if (last->segmentType() == SegmentType::EndBarLine) {
99  auto els = last->elist();
100  for (uint i = 0; i < els.size(); i++) {
101  if (!els[i]) continue;
102  toBarLine(els[i])->setBarLineType(BarLineType::END);
103  }
104  }
105  }

142 MasterScore* MasterScore::unrollRepeats()
143  {
144 
145  MasterScore* original = this;
146 
147  // create a copy of the original score to play with
148  MasterScore* score = original->clone();
149 
150  // Give it an appropriate name
151  score->setName(original->title()+"_unrolled");
152 
153  // figure out repeat structure
154  original->setExpandRepeats(true);
155 
156  // if no repeats, just return the score as-is
157  if (original->repeatList().size() == 1)
158  return score;
159 
160  // remove excerpts for now (they are re-created after unrolling master score)
161  QList<Excerpt *> excerpts;
162  for (Excerpt* e : score->excerpts()) {
163  excerpts.append(new Excerpt(*e,false));
164  score->masterScore()->deleteExcerpt(e);
165  }
166 
167  // follow along with the repeatList
168  bool first = true;
169  for (const RepeatSegment* rs: original->repeatList()) {
170  Fraction startTick = Fraction::fromTicks(rs->tick);
171  Fraction endTick = Fraction::fromTicks(rs->tick + rs->len());
172 
173  // first segment left from clone, everything past that removed
174  if (first) {
175  if (endTick <= score->lastMeasure()->tick()) // check if we actually need to remove any measures
176  score->deleteMeasures(score->tick2measure(endTick), score->lastMeasure());
177  first = false;
178  }
179  else {// just append this section from the original to the new score
180  score->appendMeasuresFromScore(original, startTick, endTick);
181  }
182  }
183 
184  removeRepeatMarkings(score);
185 
186  score->fixTicks();
187 
188  score->setLayoutAll();
189  score->doLayout();
190 
191  // re-create excerpt parts
192  if (!excerpts.isEmpty())
193  createExcerpts(score, excerpts);
194 
195  return score;
196  }
