45  msgBox.exec();
46  return;
47  }
48  if (uploadScoreDialog == nullptr) {
49  uploadScoreDialog = new UploadScoreDialog(_loginManager);

123 void UploadScoreDialog::uploadSuccess(const QString& url, const QString& nid, const QString& vid)
124  {
125  setVisible(false);
126  _url = url;
127  const int oldScoreID = _nid;
128 
129  bool ok;
130  _nid = nid.toInt(&ok);
131  if (!ok)
132  _nid = 0;
133 
134  _newScore = !(oldScoreID && oldScoreID == _nid);
135 
136  MasterScore* score = mscore->currentScore()->masterScore();
137  QMap<QString, QString> metatags = score->metaTags();
138  if (metatags.value("source") != url) {
139  metatags.insert("source", url);
140  score->startCmd();
141  score->undo(new ChangeMetaTags(score, metatags));
142  score->endCmd();
143 
144  //!Note Automatically save score file with received web-link if file already exists
145  if (!score->created()) {
146  mscore->saveFile();
147  }
148  }
149  if (uploadAudio->isChecked())
150  _loginManager->getMediaUrl(nid, vid, "mp3");
151  else
152  updateScoreData(nid, _newScore);
153  }

210 void UploadScoreDialog::display()
211  {
212  DownloadUtils* avatarDownload = new DownloadUtils(this);
213  const QSize avatarSize = userAvatarLabel->maximumSize();
214  const QString avatarUrl = _loginManager->avatar().toString().replace(QRegExp("\\@[0-9]+x[0-9]+"), QString("@%1x%2").arg(avatarSize.width()).arg(avatarSize.height()));
215  avatarDownload->setTarget(avatarUrl);
216  connect(avatarDownload, &DownloadUtils::done, this, [this, avatarSize, avatarDownload]() {
217  QPixmap pm;
218  pm.loadFromData(avatarDownload->returnData());
219  if (pm.size() != avatarSize)
220  pm = pm.scaled(avatarSize);
221 
222  userAvatarLabel->setPixmap(pm);
223 
224  avatarDownload->deleteLater();
225  });
226  avatarDownload->download();
227 
228  lblUsername->setText(_loginManager->userName());
229  QString source = mscore->currentScore()->masterScore()->metaTag("source");
230  if (!source.isEmpty()) {
231  QStringList sl = source.split("/");
232  if (sl.length() > 0) {
233  QString nidString = sl.last();
234  bool ok;
235  int nid = nidString.toInt(&ok);
236  if (ok) {
237  _nid = nid;
238  _loginManager->getScoreInfo(nid);
239  return;
240  }
241  }
242  }
243  showOrHideUploadAudio();
244  clear();
245  setVisible(true);
246  }
