54 QString FileIO::read()
55  {
56  if (mSource.isEmpty()) {
57  emit error("source is empty");
58  return QString();
59  }
60  QUrl url(mSource);
61  QString source(mSource);
62  if(url.isValid() && url.isLocalFile()) {
63  source = url.toLocalFile();
64  }
65  QFile file(source);
66  QString fileContent;
67  if ( file.open(QIODevice::ReadOnly) ) {
68  QString line;
69  QTextStream t( &file );
70  do {
71  line = t.readLine();
72  fileContent += line + "\n";
73  } while (!line.isNull());
74  file.close();
75  }
76  else {
77  emit error("Unable to open the file");
78  return QString();
79  }
80  return fileContent;
