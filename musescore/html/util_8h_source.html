29 /// \class FileIO
30 /// Provides a simple API to perform file reading and
31 /// writing operations. To use this class in a plugin put
32 /// \code import FileIO 3.0 \endcode
33 /// to the top part of the .qml plugin file.
34 /// After that FileIO object can be declared and used
35 /// in QML code:
36 /// \code
37 /// import MuseScore 3.0
38 /// import FileIO 3.0
39 /// MuseScore {
40 /// FileIO {
41 /// id: exampleFile
42 /// source: "/tmp/example.txt"
43 /// }
44 /// onRun: {
45 /// var test = exampleFile.read();
46 /// console.log(test); // will print the file content
47 /// }
48 /// }
49 /// \endcode
50 //---------------------------------------------------------
51
