137 void Volta::read(XmlReader& e)
138  {
139  eraseSpannerSegments();
140 
141  while (e.readNextStartElement()) {
142  const QStringRef& tag(e.name());
143  if (tag == "endings") {
144  QString s = e.readElementText();
145  QStringList sl = s.split(",", QString::SkipEmptyParts);
146  _endings.clear();
147  for (const QString& l : qAsConst(sl)) {
148  int i = l.simplified().toInt();
149  _endings.append(i);
150  }
151  }
152  else if (!readProperties(e))
153  e.unknown();
154  }
155  }

332 void Volta::setVelocity() const
333  {
334  Measure* startMeasure = Spanner::startMeasure();
335  Measure* endMeasure = Spanner::endMeasure();
336 
337  if (startMeasure && endMeasure) {
338  if (!endMeasure->repeatEnd())
339  return;
340 
341  Fraction startTick = Fraction::fromTicks(startMeasure->tick().ticks() - 1);
342  Fraction endTick = Fraction::fromTicks((endMeasure->tick() + endMeasure->ticks()).ticks() - 1);
343  Staff* st = staff();
344  ChangeMap& velo = st->velocities();
345  auto prevVelo = velo.val(startTick);
346  velo.addFixed(endTick, prevVelo);
347  }
348  }

350 //---------------------------------------------------------
351 // setChannel
352 //---------------------------------------------------------
353 
354 void Volta::setChannel() const
355  {
356  Measure* startMeasure = Spanner::startMeasure();
357  Measure* endMeasure = Spanner::endMeasure();
358 
359  if (startMeasure && endMeasure) {
360  if (!endMeasure->repeatEnd())
361  return;
362 
363  Fraction startTick = startMeasure->tick() - Fraction::fromTicks(1);
364  Fraction endTick = endMeasure->endTick() - Fraction::fromTicks(1);
365  Staff* st = staff();
366  for (int voice = 0; voice < VOICES; ++voice) {
367  int channel = st->channel(startTick, voice);
368  st->insertIntoChannelList(voice, endTick, channel);
369  }
370  }
371  }
