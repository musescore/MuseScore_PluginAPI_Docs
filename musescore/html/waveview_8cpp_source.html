47 static size_t ovRead(void* ptr, size_t size, size_t nmemb, void* datasource)
48  {
49  VorbisData* vd = (VorbisData*)datasource;
50  size_t n = size * nmemb;
51  if (vd->data.size() < int(vd->pos + n))
52  n = vd->data.size() - vd->pos;
53  if (n) {
54  const char* src = vd->data.data() + vd->pos;
55  memcpy(ptr, src, n);
56  vd->pos += int(n);
57  }
58  return n;
59  }

115 void WaveView::setAudio(Audio* audio)
116  {
117  waves.clear();
118  OggVorbis_File vf;
119  vorbisData.pos = 0;
120  vorbisData.data = audio->data();
121  int rv = ov_open_callbacks(&vorbisData, &vf, 0, 0, ovCallbacks);
122  if (rv < 0) {
123  qDebug("ogg open failed: %d", rv);
124  return;
125  }
126  int rn = 0;
127  const int n = 10000 / WSCALE;
128  uchar dst[n];
129  float* r = 0;
130  float* l = 0;
131  for (;;) {
132  for (int i = 0; i < n; ++i) {
133  float val = 0;
134  for (int k = 0; k < WSCALE; ++k) {
135  if (rn == 0) {
136  float** pcm;
137  int section;
138  rn = ov_read_float(&vf, &pcm, 1000, &section);
139  if (rn == 0) {
140  for (; i < n; ++i)
141  dst[i] = 0;
142  waves.append((char*)dst, n);
143  ov_clear(&vf);
144  return;
145  }
146  r = pcm[0];
147  l = pcm[1];
148  }
149  val += fabsf(*r++) + fabsf(*l++);
150  --rn;
151  }
152  unsigned int v = lrint(val * 128);
153  if (v > 255)
154  v = 255;
155  dst[i] = v;
156  }
157  waves.append((char*)dst, n);
158  }
159  }

165 int WaveView::pegel(int frame1, int frame2)
166  {
167  if (frame1 < 0)
168  frame1 = 0;
169  if (frame1 > frame2)
170  return 0;
171  frame1 = (frame1 + WSCALE/2) / WSCALE;
172  frame2 = (frame2 + WSCALE/2) / WSCALE;
173  if (frame2 >= waves.size())
174  frame2 = waves.size()-1;
175  int p = 0;
176  for (int i = frame1; i <= frame2; ++i)
177  p += uchar(waves[i]);
178  p /= (frame2 - frame1 + 1);
179  return p;
180  }
