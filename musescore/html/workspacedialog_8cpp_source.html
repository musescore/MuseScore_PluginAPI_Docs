102 void WorkspaceDialog::accepted()
103  {
104  QString s = nameLineEdit->text();
105  if (s.isEmpty())
106  return;
107  s = s.replace( QRegExp( "[" + QRegExp::escape( "\\/:*?\"<>|" ) + "]" ), "_" ); //FAT/NTFS special chars
108 
109  for (;;) {
110  if (editMode && s == WorkspacesManager::currentWorkspace()->name())
111  break;
112  bool notFound = true;
113  for (Workspace* p : WorkspacesManager::workspaces()) {
114  if ((qApp->translate("Ms::Workspace", p->name().toUtf8()).toLower() == s.toLower())) {
115  notFound = false;
116  break;
117  }
118  }
119  if (!notFound) {
120  s = QInputDialog::getText(this,
121  tr("Read Workspace Name"),
122  tr("'%1' does already exist,\nplease choose a different name:").arg(s)
123  );
124  if (s.isEmpty())
125  return;
126  s = s.replace( QRegExp( "[" + QRegExp::escape( "\\/:*?\"<>|" ) + "]" ), "_" ); //FAT/NTFS special chars
127  }
128  else
129  break;
130  }
131 
132  Workspace* newWorkspace = editMode ? WorkspacesManager::currentWorkspace() : WorkspacesManager::createNewWorkspace(s);
133  if (!editMode) {
134  //save current workspace
135  if (WorkspacesManager::currentWorkspace()->dirty())
136  WorkspacesManager::currentWorkspace()->save();
137  }
138 
139  //update workspace properties with the dialog values
140  newWorkspace->setSaveComponents(componentsCheck->isChecked());
141  newWorkspace->setSaveToolbars(toolbarsCheck->isChecked());
142  newWorkspace->setSaveMenuBar(menubarCheck->isChecked());
143  preferences.setUseLocalPreferences(prefsCheck->isChecked());
144 
145  //save newly created/edited workspace
146  newWorkspace->save();
147 
148  //rename if we edit name of the existing workspace
149  if (editMode && newWorkspace->name() != s)
150  newWorkspace->rename(s);
151 
152  if (!editMode) {
153  mscore->changeWorkspace(newWorkspace);
154  preferences.updateLocalPreferences();
155  }
156 
157  preferences.setPreference(PREF_APP_WORKSPACE, WorkspacesManager::currentWorkspace()->id());
158  emit mscore->workspacesChanged();
159  close();
160  }
