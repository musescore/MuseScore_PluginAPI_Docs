70 void Voice::init()
71  {
72  // Initialize the coefficients for the interpolation. The math comes
73  // from a mail, posted by Olli Niemitalo to the music-dsp mailing
74  // list (I found it in the music-dsp archives
75  // http://www.smartelectronix.com/musicdsp/).
76 
77  static const float MIN_GAIN = -80.0;
78  static const float dbStep = MIN_GAIN / float(EG_SIZE);
79 
80  for (int i = 0; i < EG_SIZE; ++i) {
81  Envelope::egPow[EG_SIZE-i-1] = pow(10.0, (dbStep * i)/20.0);
82  Envelope::egLin[i] = 1.0 - (double(i) / double(EG_SIZE));
83  }
84 
85  }

91 void Voice::start(Channel* c, int key, int v, const Zone* zone, double durSinceNoteOn)
92  {
93  z = zone;
94  _state = VoiceState::ATTACK;
95  //_state = VoiceState::PLAYING;
96  _channel = c;
97  _key = key;
98  _velocity = v;
99  Sample* s = z->sample;
100  audioChan = s->channel();
101  data = s->data() + z->offset * audioChan;
102  //avoid processing sample if offset is bigger than sample length
103  eidx = std::max((s->frames() - z->offset - 1) * audioChan, 0ll);
104  _loopMode = z->loopMode;
105  _loopStart = z->loopStart;
106  _loopEnd = z->loopEnd;
107  _samplesSinceStart = 0;
108 
109  _offMode = z->offMode;
110  _offBy = z->offBy;
111 
112  trigger = z->trigger;
113 
114  float offset = -z->ampVeltrack;
115  if (offset <= 0)
116  offset += 100;
117  if (trigger == Trigger::CC)
118  _velocity = 127;
119  float curve = _velocity * _velocity / (127.0 * 127.0);
120 
121  double rt_decay_value = 1.0;
122  if (trigger == Trigger::RELEASE)
123  rt_decay_value = pow(10, (-z->rtDecay * durSinceNoteOn)/20);
124  // the .005 in this calculation is made up like this:
125  // -> (offset + z->ampVeltrack*curve) being a percent value so
126  // this should be divided by 100 or multiplied by 0.01
127  // -> afterwards 0.5 (-6dB) is applied to compensate possible coherent
128  // signals in a stereo output see http://www.sengpielaudio.com/calculator-coherentsources.htm
129  // -> 0.005 = 0.01 * 0.5
130  gain = (z->volume * z->group_volume * z->global_volume) * (offset + z->ampVeltrack * curve)
131  * .005 * c->gain() * rt_decay_value *
132  pow(10.0, 4.5 / 20.0); //attenuated volume between Fluid and Zerberus on 4.5dB
133 
134  phase.set(0);
135  float sr = float(s->sampleRate()) / _zerberus->sampleRate();
136  double targetcents = ((((key - z->keyBase) * z->pitchKeytrack) + z->keyBase) * 100.0) + z->tune;
137  if (trigger == Trigger::CC)
138  targetcents = z->keyBase * 100;
139  phaseIncr.set(_zerberus->ct2hz(targetcents) * sr/_zerberus->ct2hz(z->keyBase * 100.0));
140 
141  filter.initialize(_zerberus, z, _velocity);
142 
143  currentEnvelope = V1Envelopes::DELAY;
144 
145  float velPercent = _velocity / 127.0;
146 
147  envelopes[V1Envelopes::DELAY].setTable(Envelope::egLin);
148  envelopes[V1Envelopes::DELAY].setTime(z->ampegDelay + (z->ampegVel2Delay * velPercent) + z->delay, _zerberus->sampleRate());
149  envelopes[V1Envelopes::DELAY].setConstant(0.0);
150 
151  envelopes[V1Envelopes::ATTACK].setTable(Envelope::egLin);
152  envelopes[V1Envelopes::ATTACK].setVariable();
153  envelopes[V1Envelopes::ATTACK].setTime(z->ampegAttack + (z->ampegVel2Attack * velPercent), _zerberus->sampleRate());
154  envelopes[V1Envelopes::ATTACK].offset = z->ampegStart;
155 
156  envelopes[V1Envelopes::HOLD].setTable(Envelope::egLin);
157  envelopes[V1Envelopes::HOLD].setTime(z->ampegHold + (z->ampegVel2Hold * velPercent), _zerberus->sampleRate());
158  envelopes[V1Envelopes::HOLD].setConstant(1.0);
159 
160  envelopes[V1Envelopes::DECAY].setTable(Envelope::egPow);
161  envelopes[V1Envelopes::DECAY].setVariable();
162  envelopes[V1Envelopes::DECAY].setTime(z->ampegDecay + (z->ampegVel2Decay * velPercent), _zerberus->sampleRate());
163  envelopes[V1Envelopes::DECAY].offset = z->ampegSustain;
164 
165  envelopes[V1Envelopes::SUSTAIN].setTable(Envelope::egLin);
166  if (trigger == Trigger::RELEASE || trigger == Trigger::CC) {
167  // Sample is played on noteoff. We need to stop the voice when it's done. Set the sustain duration accordingly.
168  //in ZInstrument::readSample we create sample data array using frames*channels
169  //so no need to devide by number of channels here, otherwise it reduces duration of samples by (Number of Channels)
170  double sampleDur = ((double) z->sample->frames() / z->sample->sampleRate()) * 1000; // in ms
171  double scaledSampleDur = sampleDur / (phaseIncr.data / 256.0);
172  double sustainDur = scaledSampleDur - (z->ampegDelay + z->ampegAttack + z->ampegHold + z->ampegDecay + z->ampegRelease + z->delay);
173  envelopes[V1Envelopes::SUSTAIN].setTime(sustainDur, _zerberus->sampleRate());
174  }
175  else
176  envelopes[V1Envelopes::SUSTAIN].setTime(std::numeric_limits<float>::infinity(), _zerberus->sampleRate());
177  envelopes[V1Envelopes::SUSTAIN].setConstant(qBound(0.0f, z->ampegSustain + (z->ampegVel2Sustain * velPercent), 1.0f));
178 
179  envelopes[V1Envelopes::RELEASE].setTable(Envelope::egPow);
180  envelopes[V1Envelopes::RELEASE].setVariable();
181  envelopes[V1Envelopes::RELEASE].setTime(z->ampegRelease + (z->ampegVel2Release * velPercent), _zerberus->sampleRate());
182  envelopes[V1Envelopes::RELEASE].max = envelopes[V1Envelopes::SUSTAIN].val;
183 
184  _looping = false;
185  }

191 void Voice::updateEnvelopes() {
192  if (_state == VoiceState::ATTACK && trigger != Trigger::RELEASE) {
193  while (envelopes[currentEnvelope].step() && currentEnvelope != V1Envelopes::SUSTAIN)
194  currentEnvelope++;
195 
196  // triggered by noteon enter virtually infinite sustain (play state)
197  if (currentEnvelope == V1Envelopes::SUSTAIN)
198  _state = VoiceState::PLAYING;
199  }
200  else if (_state == VoiceState::ATTACK && trigger == Trigger::RELEASE) {
201  while (envelopes[currentEnvelope].step() && currentEnvelope != V1Envelopes::RELEASE)
202  currentEnvelope++;
203 
204  // triggered by noteoff stop sample when entering release
205  if (currentEnvelope == V1Envelopes::RELEASE)
206  _state = VoiceState::STOP;
207  }
208  else if (_state == VoiceState::STOP) {
209  if (envelopes[V1Envelopes::RELEASE].step()) {
210  off();
211  }
212  }
213  }

219 void Voice::process(int frames, float* p)
220  {
221  filter.update();
222 
223  const float opcodePanLeftGain = 1.f - std::fmax(0.0f, z->pan / 100.0); //[0, 1]
224  const float opcodePanRightGain = 1.f + std::fmin(0.0f, z->pan / 100.0); //[0, 1]
225  const float leftChannelVol = gain * z->ccGain * _channel->panLeftGain() * opcodePanLeftGain;
226  const float rightChannelVol = gain * z->ccGain * _channel->panRightGain() * opcodePanRightGain;
227  if (audioChan == 1) {
228  while (frames--) {
229 
230  updateLoop();
231 
232  long long idx = phase.index();
233 
234  if (idx >= eidx) {
235  off();
236  break;
237  }
238 
239  float interpVal = filter.interpolate(phase.fract(),
240  getData(idx-1), getData(idx), getData(idx+1), getData(idx+2));
241  float v = filter.apply(interpVal, true);
242 
243  updateEnvelopes();
244  if (_state == VoiceState::OFF)
245  break;
246 
247  *p++ += v * envelopes[currentEnvelope].val * leftChannelVol;
248  *p++ += v * envelopes[currentEnvelope].val * rightChannelVol;
249 
250  if (V1Envelopes::DELAY != currentEnvelope)
251  phase += phaseIncr;
252 
253  _samplesSinceStart++;
254  }
255  }
256  else {
257  //
258  // handle interleaved stereo samples
259  //
260  while (frames--) {
261 
262  updateLoop();
263 
264  long long idx = phase.index() * 2;
265  if (idx >= eidx) {
266  off();
267 //printf("end of sample\n");
268  break;
269  }
270 
271  float interpValL = filter.interpolate(phase.fract(),
272  getData(idx-2), getData(idx), getData(idx+2), getData(idx+4));
273  float interpValR = filter.interpolate(phase.fract(),
274  getData(idx-1), getData(idx+1), getData(idx+3), getData(idx+5));
275  float valueL = filter.apply(interpValL, true);
276  float valueR = filter.apply(interpValR, false);
277 
278  //apply volume
279  updateEnvelopes();
280  if (_state == VoiceState::OFF)
281  break;
282 
283  *p++ += valueL * envelopes[currentEnvelope].val * leftChannelVol;
284  *p++ += valueR * envelopes[currentEnvelope].val * rightChannelVol;
285 
286  if (V1Envelopes::DELAY != currentEnvelope)
287  phase += phaseIncr;
288 
289  _samplesSinceStart++;
290  }
291  }
292  }

298 void Voice::updateLoop()
299  {
300  long long idx = phase.index();
301  int loopOffset = (audioChan * 3) - 1; // offset due to interpolation
302  bool validLoop = _loopEnd > 0 && _loopStart >= 0 && (_loopEnd <= (eidx/audioChan));
303  bool shallLoop = loopMode() == LoopMode::CONTINUOUS || (loopMode() == LoopMode::SUSTAIN && (_state < VoiceState::STOP));
304 
305  if (!(validLoop && shallLoop)) {
306  _looping = false;
307  return;
308  }
309 
310  if (idx + loopOffset > _loopEnd)
311  _looping = true;
312  if (idx > _loopEnd)
313  phase.setIndex(_loopStart+(idx-_loopEnd-1));
314  }
